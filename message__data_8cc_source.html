104std::vector<uint8_t> ParseMessageToData(std::string str) {
105 std::vector<uint8_t> bytes;
106 std::string temp_string = str;
107 int pos = 0;
108
109 while (pos < temp_string.size()) {
110 // Get next text fragment.
111 if (temp_string[pos] == '[') {
112 int next = temp_string.find(']', pos);
113 if (next == -1) {
114 break;
115 }
116
117 ParsedElement parsedElement =
118 FindMatchingElement(temp_string.substr(pos, next - pos + 1));
119
120 const auto dictionary_element =
121 TextElement(0x80, DICTIONARYTOKEN, true, "Dictionary");
122
123 if (!parsedElement.Active) {
124 util::logf("Error parsing message: %s", temp_string);
125 break;
126 } else if (parsedElement.Parent == dictionary_element) {
127 bytes.push_back(parsedElement.Value);
128 } else {
129 bytes.push_back(parsedElement.Parent.ID);
130
131 if (parsedElement.Parent.HasArgument) {
132 bytes.push_back(parsedElement.Value);
133 }
134 }
135
136 pos = next + 1;
137 continue;
138 } else {
139 uint8_t bb = FindMatchingCharacter(temp_string[pos++]);
140
141 if (bb != 0xFF) {
142 util::logf("Error parsing message: %s", temp_string);
143 bytes.push_back(bb);
144 }
145 }
146 }
147
148 return bytes;

185 MessageData message_data;

198 message_data.ID = message_data.ID + 1;
199 message_data.Address = pos;
200 message_data.RawString = current_message_raw;
201 message_data.Data = temp_bytes_raw;
202 message_data.DataParsed = temp_bytes_parsed;
203 message_data.ContentsParsed = current_message_parsed;

205 temp_bytes_raw.clear();
206 temp_bytes_parsed.clear();
207 current_message_raw.clear();
208 current_message_parsed.clear();
209

211 } else if (current_byte == 0xFF) {
212 break;
213 }
214
215 temp_bytes_raw.push_back(current_byte);
216
217 // Check for command.
218 TextElement text_element = FindMatchingCommand(current_byte);
219 if (!text_element.Empty()) {
220 current_message_raw.append(text_element.GetParamToken());
221 current_message_parsed.append(text_element.GetParamToken());
222 temp_bytes_parsed.push_back(current_byte);

224 }

231 current_message_raw.append(":");
232 current_message_raw.append(util::HexWord(dictionary));
233 current_message_raw.append("]");
234

236 uint32_t address = core::Get24LocalFromPC(
237 mutable_rom_data, kPointersDictionaries + (dictionary * 2));
238 uint32_t address_end = core::Get24LocalFromPC(
239 mutable_rom_data, kPointersDictionaries + ((dictionary + 1) * 2));
240
241 for (uint32_t i = address; i < address_end; i++) {
242 temp_bytes_parsed.push_back(rom_data[i]);
243 current_message_parsed.append(ParseTextDataByte(rom_data[i]));
244 }
245
246 continue;
247 }
248
249 // Everything else.
250 if (CharEncoder.contains(current_byte)) {
251 std::string str = "";
252 str.push_back(CharEncoder.at(current_byte));
253 current_message_raw.append(str);
254 current_message_parsed.append(str);
255 temp_bytes_parsed.push_back(current_byte);
256 }
257 }

259 *current_pos = pos;
