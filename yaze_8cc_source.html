26 RETURN_IF_EXCEPTION(parser.Parse(argc, argv));

32#ifdef __APPLE__
33 return yaze_run_cocoa_app_delegate(rom_filename.c_str());
34#endif
35
36 auto controller = std::make_unique<yaze::core::Controller>();
37 EXIT_IF_ERROR(controller->OnEntry(rom_filename))
38 while (controller->IsActive()) {
39 controller->OnInput();
40 if (auto status = controller->OnLoad(); !status.ok()) {
41 std::cerr << status.message() << std::endl;
42 break;
43 }
44 controller->DoRender();
45 }
46 controller->OnExit();
47 return EXIT_SUCCESS;

150zelda3_rom* yaze_load_rom(const char* filename) {
151 if (filename == nullptr || strlen(filename) == 0) {
152 return nullptr;
153 }
154
155 auto internal_rom = std::make_unique<yaze::Rom>();
156 if (!internal_rom->LoadFromFile(filename).ok()) {
157 return nullptr;
158 }
159
160 auto* rom = new zelda3_rom();
161 rom->filename = filename;
162 rom->impl = internal_rom.release(); // Transfer ownership
163 rom->data = const_cast<uint8_t*>(static_cast<yaze::Rom*>(rom->impl)->data());
164 rom->size = static_cast<yaze::Rom*>(rom->impl)->size();
165 rom->version = ZELDA3_VERSION_US; // Default, should be detected
166 rom->is_modified = false;
167 return rom;
168}

266yaze_status yaze_load_messages(const zelda3_rom* rom, zelda3_message** messages,
267 int* message_count) {
268 if (rom == nullptr || messages == nullptr || message_count == nullptr) {
269 return YAZE_ERROR_INVALID_ARG;
270 }
271
272 if (rom->impl == nullptr) {
273 return YAZE_ERROR_NOT_INITIALIZED;
274 }
275
276 try {
277 // Use LoadAllTextData from message_data.h
278 std::vector<yaze::editor::MessageData> message_data =
279 yaze::editor::ReadAllTextData(rom->data, 0);
280
281 *message_count = static_cast<int>(message_data.size());
282 *messages = new zelda3_message[*message_count];
283
284 for (size_t i = 0; i < message_data.size(); ++i) {
285 const auto& msg = message_data[i];
286 (*messages)[i].id = msg.ID;
287 (*messages)[i].rom_address = msg.Address;
288 (*messages)[i].length = static_cast<uint16_t>(msg.RawString.length());
289
290 // Allocate and copy string data
291 (*messages)[i].raw_data = new uint8_t[msg.Data.size()];
292 std::memcpy((*messages)[i].raw_data, msg.Data.data(), msg.Data.size());
293
294 (*messages)[i].parsed_text = new char[msg.ContentsParsed.length() + 1];
295 std::strncpy((*messages)[i].parsed_text, msg.ContentsParsed.c_str(),
296 msg.ContentsParsed.length());
297 (*messages)[i].parsed_text[msg.ContentsParsed.length()] = '\0';
298
299 (*messages)[i].is_compressed = false; // TODO: Detect compression
300 (*messages)[i].encoding_type = 0; // TODO: Detect encoding
301 }
302 } catch (const std::exception& e) {
303 return YAZE_ERROR_MEMORY;
304 }
305
306 return YAZE_OK;
307}
