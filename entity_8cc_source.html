18bool IsMouseHoveringOverEntity(const zelda3::GameEntity &entity,
19 ImVec2 canvas_p0, ImVec2 scrolling) {
20 // Get the mouse position relative to the canvas
21 const ImGuiIO &io = ImGui::GetIO();
22 const ImVec2 origin(canvas_p0.x + scrolling.x, canvas_p0.y + scrolling.y);
23 const ImVec2 mouse_pos(io.MousePos.x - origin.x, io.MousePos.y - origin.y);
24
25 // Check if the mouse is hovering over the entity
26 if (mouse_pos.x >= entity.x_ && mouse_pos.x <= entity.x_ + 16 &&
27 mouse_pos.y >= entity.y_ && mouse_pos.y <= entity.y_ + 16) {
28 return true;
29 }
30 return false;
31}

33void MoveEntityOnGrid(zelda3::GameEntity *entity, ImVec2 canvas_p0,
34 ImVec2 scrolling, bool free_movement) {
35 // Get the mouse position relative to the canvas
36 const ImGuiIO &io = ImGui::GetIO();
37 const ImVec2 origin(canvas_p0.x + scrolling.x, canvas_p0.y + scrolling.y);
38 const ImVec2 mouse_pos(io.MousePos.x - origin.x, io.MousePos.y - origin.y);
39
40 // Calculate the new position on the 16x16 grid
41 int new_x = static_cast<int>(mouse_pos.x) / 16 * 16;
42 int new_y = static_cast<int>(mouse_pos.y) / 16 * 16;
43 if (free_movement) {
44 new_x = static_cast<int>(mouse_pos.x) / 8 * 8;
45 new_y = static_cast<int>(mouse_pos.y) / 8 * 8;
46 }
47
48 // Update the entity position
49 entity->set_x(new_x);
50 entity->set_y(new_y);
51}

53void HandleEntityDragging(zelda3::GameEntity *entity, ImVec2 canvas_p0,
54 ImVec2 scrolling, bool &is_dragging_entity,
55 zelda3::GameEntity *&dragged_entity,
56 zelda3::GameEntity *&current_entity,
57 bool free_movement) {
58 std::string entity_type = "Entity";
59 if (entity->entity_type_ == zelda3::GameEntity::EntityType::kExit) {
60 entity_type = "Exit";
61 } else if (entity->entity_type_ ==
62 zelda3::GameEntity::EntityType::kEntrance) {
63 entity_type = "Entrance";
64 } else if (entity->entity_type_ == zelda3::GameEntity::EntityType::kSprite) {
65 entity_type = "Sprite";
66 } else if (entity->entity_type_ == zelda3::GameEntity::EntityType::kItem) {
67 entity_type = "Item";
68 }
69 const auto is_hovering =
70 IsMouseHoveringOverEntity(*entity, canvas_p0, scrolling);
71
72 const auto drag_or_clicked = ImGui::IsMouseDragging(ImGuiMouseButton_Left) ||
73 ImGui::IsMouseClicked(ImGuiMouseButton_Left);
74
75 if (is_hovering && drag_or_clicked && !is_dragging_entity) {
76 dragged_entity = entity;
77 is_dragging_entity = true;
78 } else if (is_hovering && ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
79 current_entity = entity;
80 ImGui::OpenPopup(absl::StrFormat("%s editor", entity_type.c_str()).c_str());
81 } else if (is_dragging_entity && dragged_entity == entity &&
82 ImGui::IsMouseReleased(ImGuiMouseButton_Left)) {
83 MoveEntityOnGrid(dragged_entity, canvas_p0, scrolling, free_movement);
84 entity->UpdateMapProperties(entity->map_id_);
85 is_dragging_entity = false;
86 dragged_entity = nullptr;
87 } else if (is_dragging_entity && dragged_entity == entity) {
88 if (ImGui::BeginDragDropSource()) {
89 ImGui::SetDragDropPayload("ENTITY_PAYLOAD", &entity,
90 sizeof(zelda3::GameEntity));
91 Text("Moving %s ID: %s", entity_type.c_str(),
92 core::UppercaseHexByte(entity->entity_id_).c_str());
93 ImGui::EndDragDropSource();
94 }
95 MoveEntityOnGrid(dragged_entity, canvas_p0, scrolling, free_movement);
96 entity->x_ = dragged_entity->x_;
97 entity->y_ = dragged_entity->y_;
98 entity->UpdateMapProperties(entity->map_id_);
99 }
100}

379void DrawSpriteTable(std::function<void(int)> onSpriteSelect) {
380 static ImGuiTextFilter filter;
381 static int selected_id = 0;
382 static std::vector<SpriteItem> items;
383
384 // Initialize items if empty
385 if (items.empty()) {
386 for (int i = 0; i < 256; ++i) {
387 items.push_back(SpriteItem{i, zelda3::kSpriteDefaultNames[i].data()});
388 }
389 }
390
391 filter.Draw("Filter", 180);
392
393 if (ImGui::BeginTable("##sprites", 2,
394 ImGuiTableFlags_Sortable | ImGuiTableFlags_Resizable)) {
395 ImGui::TableSetupColumn("ID", ImGuiTableColumnFlags_DefaultSort, 0.0f,
396 MyItemColumnID_ID);
397 ImGui::TableSetupColumn("Name", 0, 0.0f, MyItemColumnID_Name);
398 ImGui::TableHeadersRow();
399
400 // Handle sorting
401 if (ImGuiTableSortSpecs *sort_specs = ImGui::TableGetSortSpecs()) {
402 if (sort_specs->SpecsDirty) {
403 SpriteItem::SortWithSortSpecs(sort_specs, items);
404 sort_specs->SpecsDirty = false;
405 }
406 }
407
408 // Display filtered and sorted items
409 for (const auto &item : items) {
410 if (filter.PassFilter(item.name)) {
411 ImGui::TableNextRow();
412 ImGui::TableSetColumnIndex(0);
413 Text("%d", item.id);
414 ImGui::TableSetColumnIndex(1);
415
416 if (Selectable(item.name, selected_id == item.id,
417 ImGuiSelectableFlags_SpanAllColumns)) {
418 selected_id = item.id;
419 onSpriteSelect(item.id);
420 }
421 }
422 }
423 ImGui::EndTable();
424 }
425}
