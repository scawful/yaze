21void encode(uint64_t data, std::vector<uint8_t> &output) {
22 while (true) {
23 uint8_t x = data & 0x7f;
24 data >>= 7;
25 if (data == 0) {
26 output.push_back(0x80 | x);
27 break;
28 }
29 output.push_back(x);
30 data--;
31 }
32}

34uint64_t decode(const std::vector<uint8_t> &input, size_t &offset) {
35 uint64_t data = 0;
36 uint64_t shift = 1;
37 while (true) {
38 uint8_t x = input[offset++];
39 data += (x & 0x7f) * shift;
40 if (x & 0x80) break;
41 shift <<= 7;
42 data += shift;
43 }
44 return data;
45}

203void CreateBpsPatch(const std::vector<uint8_t> &source,
204 const std::vector<uint8_t> &target,
205 std::vector<uint8_t> &patch) {
206 patch.clear();
207 patch.insert(patch.end(), {'B', 'P', 'S', '1'});
208
209 encode(source.size(), patch);
210 encode(target.size(), patch);
211 encode(0, patch); // No metadata
212
213 size_t source_offset = 0;
214 size_t target_offset = 0;
215 int64_t source_rel_offset = 0;
216 int64_t target_rel_offset = 0;
217
218 while (target_offset < target.size()) {
219 if (source_offset < source.size() &&
220 source[source_offset] == target[target_offset]) {
221 size_t length = 0;
222 while (source_offset + length < source.size() &&
223 target_offset + length < target.size() &&
224 source[source_offset + length] == target[target_offset + length]) {
225 length++;
226 }
227 encode((length - 1) << 2 | 0, patch); // SourceRead
228 source_offset += length;
229 target_offset += length;
230 } else {
231 size_t length = 0;
232 while (
233 target_offset + length < target.size() &&
234 (source_offset + length >= source.size() ||
235 source[source_offset + length] != target[target_offset + length])) {
236 length++;
237 }
238 if (length > 0) {
239 encode((length - 1) << 2 | 1, patch); // TargetRead
240 for (size_t i = 0; i < length; i++) {
241 patch.push_back(target[target_offset + i]);
242 }
243 target_offset += length;
244 }
245 }
246
247 // SourceCopy
248 if (source_offset < source.size()) {
249 size_t length = 0;
250 int64_t offset = source_offset - source_rel_offset;
251 while (source_offset + length < source.size() &&
252 target_offset + length < target.size() &&
253 source[source_offset + length] == target[target_offset + length]) {
254 length++;
255 }
256 if (length > 0) {
257 encode((length - 1) << 2 | 2, patch);
258 encode((offset < 0 ? 1 : 0) | (abs(offset) << 1), patch);
259 source_offset += length;
260 target_offset += length;
261 source_rel_offset = source_offset;
262 }
263 }
264
265 // TargetCopy
266 if (target_offset > 0) {
267 size_t length = 0;
268 int64_t offset = target_offset - target_rel_offset;
269 while (target_offset + length < target.size() &&
270 target[target_offset - 1] == target[target_offset + length]) {
271 length++;
272 }
273 if (length > 0) {
274 encode((length - 1) << 2 | 3, patch);
275 encode((offset < 0 ? 1 : 0) | (abs(offset) << 1), patch);
276 target_offset += length;
277 target_rel_offset = target_offset;
278 }
279 }
280 }
281
282 patch.resize(patch.size() + 12); // Make space for the checksums
283 uint32_t source_checksum = crc32(source);
284 uint32_t target_checksum = crc32(target);
285 uint32_t patch_checksum = crc32(patch);
286
287 memcpy(patch.data() + patch.size() - 12, &source_checksum, sizeof(uint32_t));
288 memcpy(patch.data() + patch.size() - 8, &target_checksum, sizeof(uint32_t));
289 memcpy(patch.data() + patch.size() - 4, &patch_checksum, sizeof(uint32_t));
290}

292void ApplyBpsPatch(const std::vector<uint8_t> &source,
293 const std::vector<uint8_t> &patch,
294 std::vector<uint8_t> &target) {
295 if (patch.size() < 4 || patch[0] != 'B' || patch[1] != 'P' ||
296 patch[2] != 'S' || patch[3] != '1') {
297 throw std::runtime_error("Invalid patch format");
298 }
299
300 size_t patch_offset = 4;
301 uint64_t target_size = decode(patch, patch_offset);
302 uint64_t metadata_size = decode(patch, patch_offset);
303 patch_offset += metadata_size;
304
305 target.resize(target_size);
306 size_t source_offset = 0;
307 size_t target_offset = 0;
308 int64_t source_rel_offset = 0;
309 int64_t target_rel_offset = 0;
310
311 while (patch_offset < patch.size() - 12) {
312 uint64_t data = decode(patch, patch_offset);
313 uint64_t command = data & 3;
314 uint64_t length = (data >> 2) + 1;
315
316 switch (command) {
317 case 0: // SourceRead
318 while (length--) {
319 target[target_offset++] = source[source_offset++];
320 }
321 break;
322 case 1: // TargetRead
323 while (length--) {
324 target[target_offset++] = patch[patch_offset++];
325 }
326 break;
327 case 2: // SourceCopy
328 {
329 int64_t offsetData = decode(patch, patch_offset);
330 source_rel_offset += (offsetData & 1 ? -1 : +1) * (offsetData >> 1);
331 while (length--) {
332 target[target_offset++] = source[source_rel_offset++];
333 }
334 } break;
335 case 3: // TargetCopy
336 {
337 uint64_t offsetData = decode(patch, patch_offset);
338 target_rel_offset += (offsetData & 1 ? -1 : +1) * (offsetData >> 1);
339 while (length--) {
340 target[target_offset++] = target[target_rel_offset++];
341 }
342 }
343 default:
344 throw std::runtime_error("Invalid patch command");
345 }
346 }
347
348 uint32_t source_checksum;
349 uint32_t target_checksum;
350 uint32_t patch_checksum;
351 memcpy(&source_checksum, patch.data() + patch.size() - 12, sizeof(uint32_t));
352 memcpy(&target_checksum, patch.data() + patch.size() - 8, sizeof(uint32_t));
353 memcpy(&patch_checksum, patch.data() + patch.size() - 4, sizeof(uint32_t));
354
355 if (source_checksum != crc32(source) || target_checksum != crc32(target) ||
356 patch_checksum !=
357 crc32(std::vector<uint8_t>(patch.begin(), patch.end() - 4))) {
358 throw std::runtime_error("Checksum mismatch");
359 }
360}
