21void encode(uint64_t data, std::vector<uint8_t> &output) {
22 while (true) {
23 uint8_t x = data & 0x7f;
24 data >>= 7;
25 if (data == 0) {
26 output.push_back(0x80 | x);
27 break;
28 }
29 output.push_back(x);
30 data--;
31 }
32}

34uint64_t decode(const std::vector<uint8_t> &input, size_t &offset) {
35 uint64_t data = 0;
36 uint64_t shift = 1;
37 while (true) {
38 uint8_t x = input[offset++];
39 data += (x & 0x7f) * shift;
40 if (x & 0x80) break;
41 shift <<= 7;
42 data += shift;
43 }
44 return data;
45}

161void CreateBpsPatch(const std::vector<uint8_t> &source,
162 const std::vector<uint8_t> &target,
163 std::vector<uint8_t> &patch) {
164 patch.clear();
165 patch.insert(patch.end(), {'B', 'P', 'S', '1'});
166
167 encode(source.size(), patch);
168 encode(target.size(), patch);
169 encode(0, patch); // No metadata
170
171 size_t source_offset = 0;
172 size_t target_offset = 0;
173 int64_t source_rel_offset = 0;
174 int64_t target_rel_offset = 0;
175
176 while (target_offset < target.size()) {
177 if (source_offset < source.size() &&
178 source[source_offset] == target[target_offset]) {
179 size_t length = 0;
180 while (source_offset + length < source.size() &&
181 target_offset + length < target.size() &&
182 source[source_offset + length] == target[target_offset + length]) {
183 length++;
184 }
185 encode((length - 1) << 2 | 0, patch); // SourceRead
186 source_offset += length;
187 target_offset += length;
188 } else {
189 size_t length = 0;
190 while (
191 target_offset + length < target.size() &&
192 (source_offset + length >= source.size() ||
193 source[source_offset + length] != target[target_offset + length])) {
194 length++;
195 }
196 if (length > 0) {
197 encode((length - 1) << 2 | 1, patch); // TargetRead
198 for (size_t i = 0; i < length; i++) {
199 patch.push_back(target[target_offset + i]);
200 }
201 target_offset += length;
202 }
203 }
204
205 // SourceCopy
206 if (source_offset < source.size()) {
207 size_t length = 0;
208 int64_t offset = source_offset - source_rel_offset;
209 while (source_offset + length < source.size() &&
210 target_offset + length < target.size() &&
211 source[source_offset + length] == target[target_offset + length]) {
212 length++;
213 }
214 if (length > 0) {
215 encode((length - 1) << 2 | 2, patch);
216 encode((offset < 0 ? 1 : 0) | (abs(offset) << 1), patch);
217 source_offset += length;
218 target_offset += length;
219 source_rel_offset = source_offset;
220 }
221 }
222
223 // TargetCopy
224 if (target_offset > 0) {
225 size_t length = 0;
226 int64_t offset = target_offset - target_rel_offset;
227 while (target_offset + length < target.size() &&
228 target[target_offset - 1] == target[target_offset + length]) {
229 length++;
230 }
231 if (length > 0) {
232 encode((length - 1) << 2 | 3, patch);
233 encode((offset < 0 ? 1 : 0) | (abs(offset) << 1), patch);
234 target_offset += length;
235 target_rel_offset = target_offset;
236 }
237 }
238 }
239
240 patch.resize(patch.size() + 12); // Make space for the checksums
241 uint32_t source_checksum = crc32(source);
242 uint32_t target_checksum = crc32(target);
243 uint32_t patch_checksum = crc32(patch);
244
245 memcpy(patch.data() + patch.size() - 12, &source_checksum, sizeof(uint32_t));
246 memcpy(patch.data() + patch.size() - 8, &target_checksum, sizeof(uint32_t));
247 memcpy(patch.data() + patch.size() - 4, &patch_checksum, sizeof(uint32_t));
248}

250void ApplyBpsPatch(const std::vector<uint8_t> &source,
251 const std::vector<uint8_t> &patch,
252 std::vector<uint8_t> &target) {
253 if (patch.size() < 4 || patch[0] != 'B' || patch[1] != 'P' ||
254 patch[2] != 'S' || patch[3] != '1') {
255 throw std::runtime_error("Invalid patch format");
256 }
257
258 size_t patch_offset = 4;
259 uint64_t target_size = decode(patch, patch_offset);
260 uint64_t metadata_size = decode(patch, patch_offset);
261 patch_offset += metadata_size;
262
263 target.resize(target_size);
264 size_t source_offset = 0;
265 size_t target_offset = 0;
266 int64_t source_rel_offset = 0;
267 int64_t target_rel_offset = 0;
268
269 while (patch_offset < patch.size() - 12) {
270 uint64_t data = decode(patch, patch_offset);
271 uint64_t command = data & 3;
272 uint64_t length = (data >> 2) + 1;
273
274 switch (command) {
275 case 0: // SourceRead
276 while (length--) {
277 target[target_offset++] = source[source_offset++];
278 }
279 break;
280 case 1: // TargetRead
281 while (length--) {
282 target[target_offset++] = patch[patch_offset++];
283 }
284 break;
285 case 2: // SourceCopy
286 {
287 int64_t offsetData = decode(patch, patch_offset);
288 source_rel_offset += (offsetData & 1 ? -1 : +1) * (offsetData >> 1);
289 while (length--) {
290 target[target_offset++] = source[source_rel_offset++];
291 }
292 } break;
293 case 3: // TargetCopy
294 {
295 uint64_t offsetData = decode(patch, patch_offset);
296 target_rel_offset += (offsetData & 1 ? -1 : +1) * (offsetData >> 1);
297 while (length--) {
298 target[target_offset++] = target[target_rel_offset++];
299 }
300 }
301 default:
302 throw std::runtime_error("Invalid patch command");
303 }
304 }
305
306 uint32_t source_checksum;
307 uint32_t target_checksum;
308 uint32_t patch_checksum;
309 memcpy(&source_checksum, patch.data() + patch.size() - 12, sizeof(uint32_t));
310 memcpy(&target_checksum, patch.data() + patch.size() - 8, sizeof(uint32_t));
311 memcpy(&patch_checksum, patch.data() + patch.size() - 4, sizeof(uint32_t));
312
313 if (source_checksum != crc32(source) || target_checksum != crc32(target) ||
314 patch_checksum !=
315 crc32(std::vector<uint8_t>(patch.begin(), patch.end() - 4))) {
316 throw std::runtime_error("Checksum mismatch");
317 }
318}
