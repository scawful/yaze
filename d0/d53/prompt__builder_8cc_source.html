22nlohmann::json YamlToJson(const YAML::Node& node) {
23 if (!node) {
24 return nlohmann::json();
25 }
26
27 switch (node.Type()) {
28 case YAML::NodeType::Scalar:
29 return node.as<std::string>("");
30 case YAML::NodeType::Sequence: {
31 nlohmann::json array = nlohmann::json::array();
32 for (const auto& item : node) {
33 array.push_back(YamlToJson(item));
34 }
35 return array;
36 }
37 case YAML::NodeType::Map: {
38 nlohmann::json object = nlohmann::json::object();
39 for (const auto& kv : node) {
40 object[kv.first.as<std::string>()] = YamlToJson(kv.second);
41 }
42 return object;
43 }
44 default:
45 return nlohmann::json();
46 }

88absl::StatusOr<std::string> PromptBuilder::ResolveCataloguePath(
89 const std::string& yaml_path) const {
90 const auto search_paths = BuildCatalogueSearchPaths(yaml_path);
91 for (const auto& candidate : search_paths) {
92 fs::path resolved = candidate;
93 if (resolved.is_relative()) {
94 resolved = fs::absolute(resolved);
95 }
96 if (fs::exists(resolved)) {
97 return resolved.string();
98 }
99 }
100
101 return absl::NotFoundError(
102 absl::StrCat("Prompt catalogue not found. Checked paths: ",
103 absl::StrJoin(search_paths, ", ",
104 [](std::string* out, const fs::path& path) {
105 absl::StrAppend(out, path.string());
106 })));

109absl::Status PromptBuilder::LoadResourceCatalogue(const std::string& yaml_path) {
110 auto resolved_or = ResolveCataloguePath(yaml_path);
111 if (!resolved_or.ok()) {
112 ClearCatalogData();
113 return resolved_or.status();
114 }
115
116 const std::string& resolved_path = resolved_or.value();
117
118 YAML::Node root;
119 try {
120 root = YAML::LoadFile(resolved_path);
121 } catch (const YAML::BadFile& e) {
122 ClearCatalogData();
123 return absl::NotFoundError(
124 absl::StrCat("Unable to open prompt catalogue at ", resolved_path,
125 ": ", e.what()));
126 } catch (const YAML::ParserException& e) {
127 ClearCatalogData();
128 return absl::InvalidArgumentError(
129 absl::StrCat("Failed to parse prompt catalogue at ", resolved_path,
130 ": ", e.what()));
131 }
132
133 nlohmann::json catalogue = YamlToJson(root);
134 ClearCatalogData();
135
136 if (catalogue.contains("commands")) {
137 if (auto status = ParseCommands(catalogue["commands"]); !status.ok()) {
138 return status;
139 }
140 }
141
142 if (catalogue.contains("tools")) {
143 if (auto status = ParseTools(catalogue["tools"]); !status.ok()) {
144 return status;
145 }
146 }
147
148 if (catalogue.contains("examples")) {
149 if (auto status = ParseExamples(catalogue["examples"]); !status.ok()) {
150 return status;
151 }
152 }
153
154 if (catalogue.contains("tile16_reference")) {
155 ParseTileReference(catalogue["tile16_reference"]);
156 }
157
158 catalogue_loaded_ = true;
159 return absl::OkStatus();
160}

179absl::Status PromptBuilder::ParseTools(const nlohmann::json& tools) {
180 if (!tools.is_array()) {
181 return absl::InvalidArgumentError("tools section must be an array");
182 }
183
184 for (const auto& tool_json : tools) {
185 if (!tool_json.is_object()) {
186 return absl::InvalidArgumentError(
187 "Each tool entry must be an object with name/description");
188 }
189
190 ToolSpecification spec;
191 if (tool_json.contains("name") && tool_json["name"].is_string()) {
192 spec.name = tool_json["name"].get<std::string>();
193 } else {
194 return absl::InvalidArgumentError("Tool entry missing name");
195 }
196
197 if (tool_json.contains("description") && tool_json["description"].is_string()) {
198 spec.description = tool_json["description"].get<std::string>();
199 }
200
201 if (tool_json.contains("usage_notes") && tool_json["usage_notes"].is_string()) {
202 spec.usage_notes = tool_json["usage_notes"].get<std::string>();
203 }
204
205 if (tool_json.contains("arguments")) {
206 const auto& args = tool_json["arguments"];
207 if (!args.is_array()) {
208 return absl::InvalidArgumentError(
209 absl::StrCat("Tool arguments for ", spec.name, " must be an array"));
210 }
211 for (const auto& arg_json : args) {
212 if (!arg_json.is_object()) {
213 return absl::InvalidArgumentError(
214 absl::StrCat("Argument entries for ", spec.name,
215 " must be objects"));
216 }
217 ToolArgument arg;
218 if (arg_json.contains("name") && arg_json["name"].is_string()) {
219 arg.name = arg_json["name"].get<std::string>();
220 } else {
221 return absl::InvalidArgumentError(
222 absl::StrCat("Argument entry for ", spec.name,
223 " is missing a name"));
224 }
225 if (arg_json.contains("description") && arg_json["description"].is_string()) {
226 arg.description = arg_json["description"].get<std::string>();
227 }
228 if (arg_json.contains("required")) {
229 if (!arg_json["required"].is_boolean()) {
230 return absl::InvalidArgumentError(
231 absl::StrCat("Argument 'required' flag for ", spec.name,
232 "::", arg.name, " must be boolean"));

234 arg.required = arg_json["required"].get<bool>();

236 if (arg_json.contains("example") && arg_json["example"].is_string()) {
237 arg.example = arg_json["example"].get<std::string>();

239 spec.arguments.push_back(std::move(arg));
240 }
241 }
