164 return R"(
165# Critical Constraints
166
1671. **Output Format:** You MUST respond with ONLY a JSON array of strings
168 - Each string is a complete z3ed command
169 - NO explanatory text before or after
170 - NO markdown code blocks (```json)
171 - NO "z3ed" prefix in commands
172
1732. **Command Syntax:** Follow the exact syntax shown in examples
174 - Use correct flag names (--group, --id, --to, --from, etc.)
175 - Use hex format for colors (0xRRGGBB) and tile IDs (0xNNN)
176 - Coordinates are 0-based indices
177
1783. **Common Patterns:**
179 - Palette modifications: export → set-color → import
180 - Multiple tile placement: multiple overworld set-tile commands
181 - Validation: single rom validate command
182
1834. **Tile IDs Reference (ALTTP):**
184 - Tree: 0x02E
185 - House (2x2): 0x0C0, 0x0C1, 0x0D0, 0x0D1
186 - Water: 0x038
187 - Grass: 0x000
188
1895. **Error Prevention:**
190 - Always export before modifying palettes
191 - Use temporary file names (temp_*.json) for intermediate files
192 - Validate coordinates are within bounds
193)";

196std::string PromptBuilder::BuildContextSection(const RomContext& context) {
197 std::ostringstream oss;
198
199 oss << "# Current ROM Context\n\n";
200
201 if (context.rom_loaded) {
202 oss << "- **ROM Loaded:** Yes (" << context.rom_path << ")\n";
203 } else {
204 oss << "- **ROM Loaded:** No\n";
205 }
206
207 if (!context.current_editor.empty()) {
208 oss << "- **Active Editor:** " << context.current_editor << "\n";
209 }
210
211 if (!context.editor_state.empty()) {
212 oss << "- **Editor State:**\n";
213 for (const auto& [key, value] : context.editor_state) {
214 oss << " - " << key << ": " << value << "\n";
215 }
216 }
217
218 oss << "\n";
219 return oss.str();
220}

275std::vector<FewShotExample> PromptBuilder::GetExamplesForCategory(
276 const std::string& category) {
277 std::vector<FewShotExample> result;
278
279 for (const auto& example : examples_) {
280 // Simple category matching based on keywords
281 if (category == "palette" &&
282 (example.user_prompt.find("palette") != std::string::npos ||
283 example.user_prompt.find("color") != std::string::npos)) {
284 result.push_back(example);
285 } else if (category == "overworld" &&
286 (example.user_prompt.find("place") != std::string::npos ||
287 example.user_prompt.find("tree") != std::string::npos ||
288 example.user_prompt.find("house") != std::string::npos)) {
289 result.push_back(example);
290 } else if (category == "validation" &&
291 example.user_prompt.find("validate") != std::string::npos) {
292 result.push_back(example);
293 }
294 }
295
296 return result;
297}
