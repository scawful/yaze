23bool IsYamlBool(const std::string& value) {
24 const std::string lower = absl::AsciiStrToLower(value);
25 return lower == "true" || lower == "false" || lower == "yes" ||
26 lower == "no" || lower == "on" || lower == "off";
27}

29nlohmann::json YamlToJson(const YAML::Node& node) {
30 if (!node) {
31 return nlohmann::json();
32 }
33
34 switch (node.Type()) {
35 case YAML::NodeType::Scalar: {
36 const std::string scalar = node.as<std::string>("");
37
38 if (IsYamlBool(scalar)) {
39 return node.as<bool>();
40 }
41
42 if (scalar == "~" || absl::AsciiStrToLower(scalar) == "null") {
43 return nlohmann::json();
44 }
45
46 return scalar;
47 }
48 case YAML::NodeType::Sequence: {
49 nlohmann::json array = nlohmann::json::array();
50 for (const auto& item : node) {
51 array.push_back(YamlToJson(item));
52 }
53 return array;
54 }
55 case YAML::NodeType::Map: {
56 nlohmann::json object = nlohmann::json::object();
57 for (const auto& kv : node) {
58 object[kv.first.as<std::string>()] = YamlToJson(kv.second);
59 }
60 return object;
61 }
62 default:
63 return nlohmann::json();
64 }

67std::vector<fs::path> BuildCatalogueSearchPaths(const std::string& explicit_path) {
68 std::vector<fs::path> paths;
69 if (!explicit_path.empty()) {
70 paths.emplace_back(explicit_path);
71 }
72
73 if (const char* env_path = std::getenv("YAZE_AGENT_CATALOGUE")) {
74 if (*env_path != '\0') {
75 paths.emplace_back(env_path);
76 }
77 }
78
79 // Try to get executable directory for better path resolution
80 fs::path exe_dir;
81 try {
82 exe_dir = fs::current_path();
83 } catch (...) {
84 exe_dir = ".";
85 }
86
87 const std::vector<std::string> defaults = {
88 "assets/agent/prompt_catalogue.yaml",
89 "../assets/agent/prompt_catalogue.yaml",
90 "../../assets/agent/prompt_catalogue.yaml",
91 "../../../assets/agent/prompt_catalogue.yaml", // From build/bin/
92 "../../../../assets/agent/prompt_catalogue.yaml", // From build/bin/yaze.app/Contents/MacOS/
93 "../Resources/assets/agent/prompt_catalogue.yaml", // macOS app bundle
94 "assets/z3ed/prompt_catalogue.yaml",
95 "../assets/z3ed/prompt_catalogue.yaml",
96 };
97
98 for (const auto& candidate : defaults) {
99 paths.emplace_back(candidate);
100 }
101

117absl::StatusOr<std::string> PromptBuilder::ResolveCataloguePath(
118 const std::string& yaml_path) const {
119 const auto search_paths = BuildCatalogueSearchPaths(yaml_path);
120 for (const auto& candidate : search_paths) {
121 fs::path resolved = candidate;
122 if (resolved.is_relative()) {
123 try {
124 resolved = fs::absolute(resolved);
125 } catch (const std::exception& e) {
126 // If we can't resolve the absolute path (e.g., cwd doesn't exist),
127 // just try the relative path as-is
128 continue;
129 }
130 }
131
132 try {
133 if (fs::exists(resolved)) {
134 return resolved.string();
135 }
136 } catch (const std::exception& e) {
137 // If checking existence fails, just continue to next path
138 continue;
139 }
140 }
141
142 return absl::NotFoundError(
143 absl::StrCat("Prompt catalogue not found. Checked paths: ",
144 absl::StrJoin(search_paths, ", ",
145 [](std::string* out, const fs::path& path) {
146 absl::StrAppend(out, path.string());
147 })));
148}

150absl::Status PromptBuilder::LoadResourceCatalogue(const std::string& yaml_path) {
151#ifndef YAZE_WITH_JSON
152 // Gracefully degrade if JSON support not available
153 std::cerr << "⚠️ PromptBuilder requires JSON support for catalogue loading\n"
154 << " Build with -DZ3ED_AI=ON or -DYAZE_WITH_JSON=ON\n"
155 << " AI features will use basic prompts without tool definitions\n";
156 return absl::OkStatus(); // Don't fail, just skip catalogue loading
157#else
158 auto resolved_or = ResolveCataloguePath(yaml_path);
159 if (!resolved_or.ok()) {
160 ClearCatalogData();
161 return resolved_or.status();
162 }
163
164 const std::string& resolved_path = resolved_or.value();
165
166 YAML::Node root;
167 try {
168 root = YAML::LoadFile(resolved_path);
169 } catch (const YAML::BadFile& e) {
170 ClearCatalogData();
171 return absl::NotFoundError(
172 absl::StrCat("Unable to open prompt catalogue at ", resolved_path,
173 ": ", e.what()));
174 } catch (const YAML::ParserException& e) {
175 ClearCatalogData();
176 return absl::InvalidArgumentError(
177 absl::StrCat("Failed to parse prompt catalogue at ", resolved_path,
178 ": ", e.what()));
179 }
180
181 nlohmann::json catalogue = YamlToJson(root);
182 ClearCatalogData();
183
184 if (catalogue.contains("commands")) {
185 if (auto status = ParseCommands(catalogue["commands"]); !status.ok()) {
186 return status;
187 }
188 }
189
190 if (catalogue.contains("tools")) {
191 if (auto status = ParseTools(catalogue["tools"]); !status.ok()) {
192 return status;
193 }
194 }
195
196 if (catalogue.contains("examples")) {
197 if (auto status = ParseExamples(catalogue["examples"]); !status.ok()) {
198 return status;
199 }
200 }
201
202 if (catalogue.contains("tile16_reference")) {
203 ParseTileReference(catalogue["tile16_reference"]);
204 }
205
206 catalogue_loaded_ = true;
207 return absl::OkStatus();
208#endif // YAZE_WITH_JSON
209}

228absl::Status PromptBuilder::ParseTools(const nlohmann::json& tools) {
229 if (!tools.is_array()) {
230 return absl::InvalidArgumentError("tools section must be an array");
231 }
232
233 for (const auto& tool_json : tools) {
234 if (!tool_json.is_object()) {
235 return absl::InvalidArgumentError(
236 "Each tool entry must be an object with name/description");
237 }
238
239 ToolSpecification spec;
240 if (tool_json.contains("name") && tool_json["name"].is_string()) {
241 spec.name = tool_json["name"].get<std::string>();
242 } else {
243 return absl::InvalidArgumentError("Tool entry missing name");
244 }
245
246 if (tool_json.contains("description") && tool_json["description"].is_string()) {
247 spec.description = tool_json["description"].get<std::string>();
248 }
249
250 if (tool_json.contains("usage_notes") && tool_json["usage_notes"].is_string()) {
251 spec.usage_notes = tool_json["usage_notes"].get<std::string>();
252 }
253
254 if (tool_json.contains("arguments")) {
255 const auto& args = tool_json["arguments"];
256 if (!args.is_array()) {
257 return absl::InvalidArgumentError(
258 absl::StrCat("Tool arguments for ", spec.name, " must be an array"));
259 }
260 for (const auto& arg_json : args) {
261 if (!arg_json.is_object()) {
262 return absl::InvalidArgumentError(
263 absl::StrCat("Argument entries for ", spec.name,
264 " must be objects"));
265 }
266 ToolArgument arg;
267 if (arg_json.contains("name") && arg_json["name"].is_string()) {
268 arg.name = arg_json["name"].get<std::string>();
269 } else {
270 return absl::InvalidArgumentError(
271 absl::StrCat("Argument entry for ", spec.name,
272 " is missing a name"));
273 }
274 if (arg_json.contains("description") && arg_json["description"].is_string()) {
275 arg.description = arg_json["description"].get<std::string>();
276 }
277 if (arg_json.contains("required")) {
278 if (!arg_json["required"].is_boolean()) {
279 return absl::InvalidArgumentError(
280 absl::StrCat("Argument 'required' flag for ", spec.name,
281 "::", arg.name, " must be boolean"));
282 }
283 arg.required = arg_json["required"].get<bool>();
284 }
285 if (arg_json.contains("example") && arg_json["example"].is_string()) {
286 arg.example = arg_json["example"].get<std::string>();
287 }
288 spec.arguments.push_back(std::move(arg));
289 }
290 }
291
292 tool_specs_.push_back(std::move(spec));
293 }
294
295 return absl::OkStatus();
296}

298absl::Status PromptBuilder::ParseExamples(const nlohmann::json& examples) {
299 if (!examples.is_array()) {
300 return absl::InvalidArgumentError("examples section must be an array");
301 }
302
303 for (const auto& example_json : examples) {
304 if (!example_json.is_object()) {
305 return absl::InvalidArgumentError("Each example entry must be an object");
306 }
307
308 FewShotExample example;
309 if (example_json.contains("user_prompt") &&
310 example_json["user_prompt"].is_string()) {
311 example.user_prompt = example_json["user_prompt"].get<std::string>();
312 } else {
313 return absl::InvalidArgumentError("Example missing user_prompt");
314 }
315
316 if (example_json.contains("text_response") &&
317 example_json["text_response"].is_string()) {
318 example.text_response = example_json["text_response"].get<std::string>();
319 }
320
321 if (example_json.contains("reasoning") &&
322 example_json["reasoning"].is_string()) {
323 example.explanation = example_json["reasoning"].get<std::string>();
324 }
325
326 if (example_json.contains("commands")) {
327 const auto& commands = example_json["commands"];
328 if (!commands.is_array()) {
329 return absl::InvalidArgumentError(
330 absl::StrCat("Example commands for ", example.user_prompt,
331 " must be an array"));
332 }
333 for (const auto& cmd : commands) {
334 if (!cmd.is_string()) {
335 return absl::InvalidArgumentError(
336 absl::StrCat("Command entries for ", example.user_prompt,
337 " must be strings"));
338 }
339 example.expected_commands.push_back(cmd.get<std::string>());
340 }
341 }
342
343 if (example_json.contains("tool_calls")) {
344 const auto& calls = example_json["tool_calls"];
345 if (!calls.is_array()) {
346 return absl::InvalidArgumentError(
347 absl::StrCat("Tool calls for ", example.user_prompt,
348 " must be an array"));
349 }
350 for (const auto& call_json : calls) {
351 if (!call_json.is_object()) {
352 return absl::InvalidArgumentError(
353 absl::StrCat("Tool call entries for ", example.user_prompt,
354 " must be objects"));
355 }
356 ToolCall call;
357 if (call_json.contains("tool_name") && call_json["tool_name"].is_string()) {
358 call.tool_name = call_json["tool_name"].get<std::string>();
359 } else {
360 return absl::InvalidArgumentError(
361 absl::StrCat("Tool call missing tool_name in example: ",
362 example.user_prompt));
363 }
364 if (call_json.contains("args")) {
365 const auto& args = call_json["args"];
366 if (!args.is_object()) {
367 return absl::InvalidArgumentError(
368 absl::StrCat("Tool call args for ", example.user_prompt,
369 " must be an object"));
370 }
371 for (const auto& [key, value] : args.items()) {
372 if (!value.is_string()) {
373 return absl::InvalidArgumentError(
374 absl::StrCat("Tool call arg value for ", example.user_prompt,
375 " must be a string"));
376 }
377 call.args[key] = value.get<std::string>();
378 }
379 }
380 example.tool_calls.push_back(std::move(call));
381 }
382 }
383
384 example.explanation = example_json.value("explanation", example.explanation);
385 examples_.push_back(std::move(example));
386 }
387
388 return absl::OkStatus();

458std::string PromptBuilder::BuildFunctionCallSchemas() const {
459 if (tool_specs_.empty()) {
460 return "[]";
461 }
462
463 nlohmann::json tools_array = nlohmann::json::array();
464
465 for (const auto& spec : tool_specs_) {
466 nlohmann::json tool;
467 tool["type"] = "function";
468
469 nlohmann::json function;
470 function["name"] = spec.name;
471 function["description"] = spec.description;
472 if (!spec.usage_notes.empty()) {
473 function["description"] = spec.description + " " + spec.usage_notes;
474 }
475
476 nlohmann::json parameters;
477 parameters["type"] = "object";
478
479 nlohmann::json properties = nlohmann::json::object();
480 nlohmann::json required = nlohmann::json::array();
481
482 for (const auto& arg : spec.arguments) {
483 nlohmann::json arg_schema;
484 arg_schema["type"] = "string"; // All CLI args are strings
485 arg_schema["description"] = arg.description;
486 if (!arg.example.empty()) {
487 arg_schema["example"] = arg.example;
488 }
489 properties[arg.name] = arg_schema;
490
491 if (arg.required) {
492 required.push_back(arg.name);
493 }
494 }
495
496 parameters["properties"] = properties;
497 if (!required.empty()) {
498 parameters["required"] = required;
499 }
500
501 function["parameters"] = parameters;
502 tool["function"] = function;
503 tools_array.push_back(tool);
504 }
505
506 return tools_array.dump(2);
507}

712
713 // Use ResourceContextBuilder if a ROM is available
714 if (rom_ && rom_->is_loaded()) {
715 if (!resource_context_builder_) {
716 resource_context_builder_ = std::make_unique<ResourceContextBuilder>(rom_);

719 if (resource_context_or.ok()) {

721 }
