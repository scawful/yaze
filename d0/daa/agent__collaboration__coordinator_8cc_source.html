52absl::StatusOr<AgentCollaborationCoordinator::SessionInfo>
53AgentCollaborationCoordinator::HostSession(const std::string& session_name) {
54 const std::string trimmed = Trimmed(session_name);
55 if (trimmed.empty()) {
56 return absl::InvalidArgumentError("Session name cannot be empty");
57 }
58
59 RETURN_IF_ERROR(EnsureDirectory());
60
61 SessionFileData data;
62 data.session_name = trimmed;
63 data.session_code = GenerateSessionCode();
64 data.host = local_user_;
65 data.participants.push_back(local_user_);
66
67 std::filesystem::path path = SessionFilePath(data.session_code);
68
69 // Collision avoidance (extremely unlikely but cheap to guard against).
70 int attempts = 0;
71 while (std::filesystem::exists(path) && attempts++ < 5) {
72 data.session_code = GenerateSessionCode();
73 path = SessionFilePath(data.session_code);
74 }
75 if (std::filesystem::exists(path)) {
76 return absl::InternalError(
77 "Unable to allocate a new collaboration session code");
78 }
79
80 RETURN_IF_ERROR(WriteSessionFile(path, data));
81
82 active_ = true;
83 hosting_ = true;
84 session_id_ = data.session_code;
85 session_name_ = data.session_name;
86
87 SessionInfo info;
88 info.session_id = data.session_code;
89 info.session_name = data.session_name;
90 info.participants = data.participants;
91 return info;
92}

94absl::StatusOr<AgentCollaborationCoordinator::SessionInfo>
95AgentCollaborationCoordinator::JoinSession(const std::string& session_code) {
96 const std::string normalized = NormalizeSessionCode(session_code);
97 if (normalized.empty()) {
98 return absl::InvalidArgumentError("Session code cannot be empty");
99 }
100
101 RETURN_IF_ERROR(EnsureDirectory());
102
103 std::filesystem::path path = SessionFilePath(normalized);
104 ASSIGN_OR_RETURN(SessionFileData data, LoadSessionFile(path));
105
106 const auto already_joined = std::find(data.participants.begin(),
107 data.participants.end(), local_user_);
108 if (already_joined == data.participants.end()) {
109 data.participants.push_back(local_user_);
110 RETURN_IF_ERROR(WriteSessionFile(path, data));
111 }
112
113 active_ = true;
114 hosting_ = false;
115 session_id_ = data.session_code.empty() ? normalized : data.session_code;
116 session_name_ = data.session_name.empty() ? session_id_ : data.session_name;
117
118 SessionInfo info;
119 info.session_id = session_id_;
120 info.session_name = session_name_;
121 info.participants = data.participants;
122 return info;

125absl::Status AgentCollaborationCoordinator::LeaveSession() {
126 if (!active_) {
127 return absl::FailedPreconditionError("No collaborative session active");
128 }
129
130 const std::filesystem::path path = SessionFilePath(session_id_);
131 absl::Status status = absl::OkStatus();
132
133 if (hosting_) {
134 std::error_code ec;
135 std::filesystem::remove(path, ec);
136 if (ec) {
137 status = absl::InternalError(
138 absl::StrFormat("Failed to clean up session file: %s", ec.message()));
139 }
140 } else {
141 auto data_or = LoadSessionFile(path);
142 if (data_or.ok()) {
143 SessionFileData data = std::move(data_or.value());
144 auto end = std::remove(data.participants.begin(), data.participants.end(),
145 local_user_);
146 data.participants.erase(end, data.participants.end());
147
148 if (data.participants.empty()) {
149 std::error_code ec;
150 std::filesystem::remove(path, ec);
151 if (ec) {
152 status = absl::InternalError(absl::StrFormat(
153 "Failed to remove empty session file: %s", ec.message()));
154 }
155 } else {
156 status = WriteSessionFile(path, data);
157 }
158 } else {
159 // If the session file has already disappeared, treat it as success.
160 status = absl::OkStatus();
161 }
162 }
163
164 active_ = false;
165 hosting_ = false;
166 session_id_.clear();
167 session_name_.clear();
168
169 return status;

232std::string AgentCollaborationCoordinator::NormalizeSessionCode(
233 const std::string& input) const {
234 std::string normalized = Trimmed(input);
235 normalized.erase(std::remove_if(normalized.begin(), normalized.end(),
236 [](unsigned char c) {
237 return !std::isalnum(
238 static_cast<unsigned char>(c));
239 }),
240 normalized.end());
241 std::transform(normalized.begin(), normalized.end(), normalized.begin(),
242 [](unsigned char c) {
243 return static_cast<char>(
244 std::toupper(static_cast<unsigned char>(c)));
245 });
246 return normalized;
247}

249std::string AgentCollaborationCoordinator::GenerateSessionCode() const {
250 static constexpr char kAlphabet[] = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789";
251 thread_local std::mt19937 rng{std::random_device{}()};
252 std::uniform_int_distribution<size_t> dist(0, sizeof(kAlphabet) - 2);
253
254 std::string code(6, '0');
255 for (char& ch : code) {
256 ch = kAlphabet[dist(rng)];
257 }
258 return code;
259}

274absl::StatusOr<AgentCollaborationCoordinator::SessionFileData>
275AgentCollaborationCoordinator::LoadSessionFile(
276 const std::filesystem::path& path) const {
277 std::ifstream file(path);
278 if (!file.is_open()) {
279 return absl::NotFoundError(
280 absl::StrFormat("Session %s does not exist", path.string()));
281 }
282
283 SessionFileData data;
284 data.session_code = path.stem().string();
285
286 std::string line;
287 while (std::getline(file, line)) {
288 auto pos = line.find(':');
289 if (pos == std::string::npos) {
290 continue;
291 }
292 std::string key = line.substr(0, pos);
293 std::string value = Trimmed(line.substr(pos + 1));
294 if (key == "name") {
295 data.session_name = value;
296 } else if (key == "code") {
297 data.session_code = NormalizeSessionCode(value);
298 } else if (key == "host") {
299 data.host = value;
300 data.participants.push_back(value);
301 } else if (key == "participant") {
302 if (std::find(data.participants.begin(), data.participants.end(), value) ==
303 data.participants.end()) {
304 data.participants.push_back(value);
305 }
306 }
307 }
308
309 if (data.session_name.empty()) {
310 data.session_name = data.session_code;
311 }
312 if (!data.host.empty()) {
313 auto host_it = std::find(data.participants.begin(), data.participants.end(),
314 data.host);
315 if (host_it == data.participants.end()) {
316 data.participants.insert(data.participants.begin(), data.host);
317 } else if (host_it != data.participants.begin()) {
318 std::rotate(data.participants.begin(), host_it,
319 std::next(host_it));
320 }
321 }
322
323 return data;
324}

326absl::Status AgentCollaborationCoordinator::WriteSessionFile(
327 const std::filesystem::path& path, const SessionFileData& data) const {
328 std::ofstream file(path, std::ios::trunc);
329 if (!file.is_open()) {
330 return absl::InternalError(
331 absl::StrFormat("Failed to write session file: %s", path.string()));
332 }
333
334 file << "name:" << data.session_name << "\n";
335 file << "code:" << data.session_code << "\n";
336 file << "host:" << data.host << "\n";
337
338 std::set<std::string> seen;
339 seen.insert(data.host);
340 for (const auto& participant : data.participants) {
341 if (seen.insert(participant).second) {
342 file << "participant:" << participant << "\n";
343 }
344 }
345
346 file.flush();
347 if (!file.good()) {
348 return absl::InternalError(
349 absl::StrFormat("Failed to flush session file: %s", path.string()));
350 }
351 return absl::OkStatus();
352}
