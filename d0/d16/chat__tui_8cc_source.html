172void ChatTUI::OnSubmit(const std::string& message) {
173 if (message.empty()) {
174 return;
175 }
176
177 if (message == "/exit") {
178 screen_.Exit();
179 return;
180 }
181 if (message == "/clear") {
182 agent_service_.ResetConversation();
183 return;
184 }
185 if (message == "/rom_info") {
186 // Send ROM info as a user message to get a response
187 auto response = agent_service_.SendMessage("Show me information about the loaded ROM");
188 if (!response.ok()) {
189 last_error_ = response.status().message();
190 } else {
191 last_error_.reset();
192 }
193 return;
194 }
195 if (message == "/help") {
196 // Send help request as a user message
197 auto response = agent_service_.SendMessage("What commands can I use?");
198 if (!response.ok()) {
199 last_error_ = response.status().message();
200 } else {
201 last_error_.reset();
202 }
203 return;
204 }
205
206 auto response = agent_service_.SendMessage(message);
207 if (!response.ok()) {
208 last_error_ = response.status().message();
209 } else {
210 last_error_.reset();
211 }
212}
