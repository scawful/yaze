70std::string GetRelativeTimeString(const std::filesystem::file_time_type& ftime) {
71 auto sctp = std::chrono::time_point_cast<std::chrono::system_clock::duration>(
72 ftime - std::filesystem::file_time_type::clock::now() +
73 std::chrono::system_clock::now());
74 auto now = std::chrono::system_clock::now();
75 auto diff = std::chrono::duration_cast<std::chrono::hours>(now - sctp);
76
77 int hours = diff.count();
78 if (hours < 24) {
79 return "Today";
80 } else if (hours < 48) {
81 return "Yesterday";
82 } else if (hours < 168) {
83 int days = hours / 24;
84 return absl::StrFormat("%d days ago", days);
85 } else if (hours < 720) {
86 int weeks = hours / 168;
87 return absl::StrFormat("%d week%s ago", weeks, weeks > 1 ? "s" : "");
88 } else {
89 int months = hours / 720;
90 return absl::StrFormat("%d month%s ago", months, months > 1 ? "s" : "");
91 }
92}

94// Draw a pixelated triforce in the background (ALTTP style)
95void DrawTriforceBackground(ImDrawList* draw_list, ImVec2 pos, float size, float alpha, float glow) {
96 // Make it pixelated - round size to nearest 4 pixels
97 size = std::round(size / 4.0f) * 4.0f;
98
99 // Calculate triangle points with pixel-perfect positioning
100 auto triangle = [&](ImVec2 center, float s, ImU32 color) {
101 // Round to pixel boundaries for crisp edges
102 float half_s = s / 2.0f;
103 float tri_h = s * 0.866f; // Height of equilateral triangle
104
105 // Fixed: Proper equilateral triangle with apex at top
106 ImVec2 p1(std::round(center.x), std::round(center.y - tri_h / 2.0f)); // Top apex
107 ImVec2 p2(std::round(center.x - half_s), std::round(center.y + tri_h / 2.0f)); // Bottom left
108 ImVec2 p3(std::round(center.x + half_s), std::round(center.y + tri_h / 2.0f)); // Bottom right
109
110 draw_list->AddTriangleFilled(p1, p2, p3, color);
111 };
112
113 ImU32 gold = ImGui::GetColorU32(ImVec4(1.0f, 0.843f, 0.0f, alpha));
114
115 // Proper triforce layout with three triangles
116 float small_size = size / 2.0f;
117 float small_height = small_size * 0.866f;
118
119 // Top triangle (centered above)
120 triangle(ImVec2(pos.x, pos.y), small_size, gold);
121
122 // Bottom left triangle
123 triangle(ImVec2(pos.x - small_size / 2.0f, pos.y + small_height), small_size, gold);
124
125 // Bottom right triangle
126 triangle(ImVec2(pos.x + small_size / 2.0f, pos.y + small_height), small_size, gold);
127}

341void WelcomeScreen::RefreshRecentProjects() {
342 recent_projects_.clear();
343
344 // Use the ProjectManager singleton to get recent files
345 auto& recent_files = core::RecentFilesManager::GetInstance().GetRecentFiles();
346
347 for (const auto& filepath : recent_files) {
348 if (recent_projects_.size() >= kMaxRecentProjects) break;
349
350 RecentProject project;
351 project.filepath = filepath;
352
353 // Extract filename
354 std::filesystem::path path(filepath);
355 project.name = path.filename().string();
356
357 // Get file modification time if it exists
358 if (std::filesystem::exists(path)) {
359 auto ftime = std::filesystem::last_write_time(path);
360 project.last_modified = GetRelativeTimeString(ftime);
361 project.rom_title = "ALTTP ROM";
362 } else {
363 project.last_modified = "File not found";
364 project.rom_title = "Missing";
365 }
366
367 recent_projects_.push_back(project);
