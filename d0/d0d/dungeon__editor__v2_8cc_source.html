31absl::Status DungeonEditorV2::Load() {

493void DungeonEditorV2::DrawRoomMatrixCard() {
494 gui::EditorCard matrix_card(
495 MakeCardTitle("Room Matrix").c_str(),
496 ICON_MD_GRID_VIEW, &show_room_matrix_);
497
498 matrix_card.SetDefaultSize(440, 520);
499
500 if (matrix_card.Begin()) {
501 // 16 wide x 19 tall = 304 cells (296 rooms + 8 empty)
502 constexpr int kRoomsPerRow = 16;
503 constexpr int kRoomsPerCol = 19;
504 constexpr int kTotalRooms = 0x128; // 296 rooms (0x00-0x127)
505 constexpr float kRoomCellSize = 24.0f; // Smaller cells like ZScream
506 constexpr float kCellSpacing = 1.0f; // Tighter spacing
507
508 ImDrawList* draw_list = ImGui::GetWindowDrawList();
509 ImVec2 canvas_pos = ImGui::GetCursorScreenPos();
510
511 int room_index = 0;
512 for (int row = 0; row < kRoomsPerCol; row++) {
513 for (int col = 0; col < kRoomsPerRow; col++) {
514 int room_id = room_index;
515 bool is_valid_room = (room_id < kTotalRooms);
516
517 ImVec2 cell_min = ImVec2(
518 canvas_pos.x + col * (kRoomCellSize + kCellSpacing),
519 canvas_pos.y + row * (kRoomCellSize + kCellSpacing));
520 ImVec2 cell_max = ImVec2(
521 cell_min.x + kRoomCellSize,
522 cell_min.y + kRoomCellSize);
523
524 if (is_valid_room) {
525 // Use simple deterministic color based on room ID (no loading needed)
526 ImU32 bg_color;
527
528 // Generate color from room ID - much faster than loading
529 int hue = (room_id * 37) % 360; // Distribute colors across spectrum
530 int saturation = 40 + (room_id % 3) * 15;
531 int brightness = 50 + (room_id % 5) * 10;
532
533 // Convert HSV to RGB
534 float h = hue / 60.0f;
535 float s = saturation / 100.0f;
536 float v = brightness / 100.0f;
537
538 int i = static_cast<int>(h);
539 float f = h - i;
540 int p = static_cast<int>(v * (1 - s) * 255);
541 int q = static_cast<int>(v * (1 - s * f) * 255);
542 int t = static_cast<int>(v * (1 - s * (1 - f)) * 255);
543 int val = static_cast<int>(v * 255);
544
545 switch (i % 6) {
546 case 0: bg_color = IM_COL32(val, t, p, 255); break;
547 case 1: bg_color = IM_COL32(q, val, p, 255); break;
548 case 2: bg_color = IM_COL32(p, val, t, 255); break;
549 case 3: bg_color = IM_COL32(p, q, val, 255); break;
550 case 4: bg_color = IM_COL32(t, p, val, 255); break;
551 case 5: bg_color = IM_COL32(val, p, q, 255); break;
552 default: bg_color = IM_COL32(60, 60, 70, 255); break;
553 }
554
555 // Check if room is currently selected
556 bool is_current = (current_room_id_ == room_id);
557
558 // Check if room is open in a card
559 bool is_open = false;
560 for (int i = 0; i < active_rooms_.Size; i++) {
561 if (active_rooms_[i] == room_id) {
562 is_open = true;
563 break;
564 }
565 }
566
567 // Draw cell background with palette color
568 draw_list->AddRectFilled(cell_min, cell_max, bg_color);
569
570 // Draw outline ONLY for current/open rooms
571 if (is_current) {
572 // Light green for current room
573 draw_list->AddRect(cell_min, cell_max,
574 IM_COL32(144, 238, 144, 255), 0.0f, 0, 2.5f);
575 } else if (is_open) {
576 // Green for open rooms
577 draw_list->AddRect(cell_min, cell_max,
578 IM_COL32(0, 200, 0, 255), 0.0f, 0, 2.0f);
579 } else {
580 // Subtle gray border for all rooms
581 draw_list->AddRect(cell_min, cell_max,
582 IM_COL32(80, 80, 80, 200), 0.0f, 0, 1.0f);
583 }
584
585 // Draw room ID (small text)
586 std::string room_label = absl::StrFormat("%02X", room_id);
587 ImVec2 text_size = ImGui::CalcTextSize(room_label.c_str());
588 ImVec2 text_pos = ImVec2(
589 cell_min.x + (kRoomCellSize - text_size.x) * 0.5f,
590 cell_min.y + (kRoomCellSize - text_size.y) * 0.5f);
591
592 // Use smaller font if available
593 draw_list->AddText(text_pos, IM_COL32(220, 220, 220, 255),
594 room_label.c_str());
595
596 // Handle clicks
597 ImGui::SetCursorScreenPos(cell_min);
598 ImGui::InvisibleButton(
599 absl::StrFormat("##room%d", room_id).c_str(),
600 ImVec2(kRoomCellSize, kRoomCellSize));
601
602 if (ImGui::IsItemClicked()) {
603 OnRoomSelected(room_id);
604 }
605
606 // Hover tooltip with room name and status
607 if (ImGui::IsItemHovered()) {
608 ImGui::BeginTooltip();
609 if (room_id < static_cast<int>(std::size(zelda3::kRoomNames))) {
610 ImGui::Text("%s", zelda3::kRoomNames[room_id].data());
611 } else {
612 ImGui::Text("Room %03X", room_id);
613 }
614 ImGui::Text("Status: %s", is_open ? "Open" : is_current ? "Current" : "Closed");
615 ImGui::Text("Click to %s", is_open ? "focus" : "open");
616 ImGui::EndTooltip();
617 }
618 } else {
619 // Empty cell
620 draw_list->AddRectFilled(cell_min, cell_max,
621 IM_COL32(30, 30, 30, 255));
622 draw_list->AddRect(cell_min, cell_max,
623 IM_COL32(50, 50, 50, 255));
624 }
625
626 room_index++;
627 }
628 }
629
630 // Advance cursor past the grid
631 ImGui::Dummy(ImVec2(
632 kRoomsPerRow * (kRoomCellSize + kCellSpacing),
633 kRoomsPerCol * (kRoomCellSize + kCellSpacing)));
634 }
635 matrix_card.End();
636}

690 int col = current_block % max_blocks_per_row;

697 room_gfx_canvas.draw_list()->AddImage(
698 (ImTextureID)(intptr_t)gfx_sheet.texture(),
699 ImVec2(x, y),
700 ImVec2(x + block_width, y + block_height));
701 }
702 }
703 current_block++;
704 }
705
706 room_gfx_canvas.DrawGrid(32.0f);
707 room_gfx_canvas.DrawOverlay();
708 } else {
709 ImGui::TextDisabled("No room selected");
710 }
711 }
712 graphics_card.End();
