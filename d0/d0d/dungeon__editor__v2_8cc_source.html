104 // Column 2: Canvas (fully delegated)
105 TableNextColumn();
106 if (BeginTabBar("##RoomTabs")) {
107 for (int i = 0; i < active_rooms_.Size; i++) {
108 int room_id = active_rooms_[i];
109 bool open = true;
110
111 std::string tab_name_str;
112 const char* tab_name;
113 if (room_id >= 0 && static_cast<size_t>(room_id) < std::size(zelda3::kRoomNames)) {
114 tab_name = zelda3::kRoomNames[room_id].data();
115 } else {
116 tab_name_str = absl::StrFormat("Room %03X", room_id);
117 tab_name = tab_name_str.c_str();
118 }
119
120 if (BeginTabItem(tab_name, &open)) {
121 DrawRoomTab(room_id);
122 EndTabItem();
123 }
124
125 if (!open) {
126 active_rooms_.erase(active_rooms_.Data + i);
127 i--;
128 }
129 }
130
131 // Add tab button
132 if (ImGui::TabItemButton(ICON_MD_ADD, ImGuiTabItemFlags_Trailing | ImGuiTabItemFlags_NoTooltip)) {
133 OnRoomSelected(room_selector_.current_room_id());
134 }
135
136 EndTabBar();
137 }
138
139 // Column 3: Object Selector (fully delegated)
140 TableNextColumn();

147void DungeonEditorV2::DrawRoomTab(int room_id) {
148 if (room_id < 0 || room_id >= static_cast<int>(rooms_.size())) {
149 ImGui::Text("Invalid room ID: %d", room_id);
150 return;
151 }
152
153 // Lazy load room data

156 if (!status.ok()) {
157 ImGui::TextColored(ImVec4(1, 0, 0, 1), "Failed to load room: %s",
158 status.message().data());
159 return;
160 }
161 }
162
163 // Quick controls
164 ImGui::Text("Room %03X", room_id);
165 ImGui::SameLine();
166 if (ImGui::Button("Load Graphics")) {
167 (void)room_loader_.LoadAndRenderRoomGraphics(rooms_[room_id]);
168 }
169 ImGui::SameLine();
170 if (ImGui::Button("Save")) {
171 auto status = rooms_[room_id].SaveObjects();
172 if (!status.ok()) {
173 ImGui::TextColored(ImVec4(1, 0, 0, 1), "Save failed: %s",
174 status.message().data());
175 }
176 }
177
178 ImGui::Separator();
179
180 // Canvas - fully delegated to DungeonCanvasViewer
181 // DungeonCanvasViewer has DrawDungeonCanvas() method
182 canvas_viewer_.DrawDungeonCanvas(room_id);

185void DungeonEditorV2::OnRoomSelected(int room_id) {

187
188 // Check if already open

191 return; // Already open
192 }
193 }
194
195 // Add new tab
