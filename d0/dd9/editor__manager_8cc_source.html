84void EditorManager::LoadUserSettings() {
85 try {
86 auto data = util::LoadConfigFile(settings_filename_);
87 if (!data.empty()) {
88 std::istringstream ss(data);
89 std::string line;
90 while (std::getline(ss, line)) {
91 auto eq = line.find('=');
92 if (eq == std::string::npos)
93 continue;
94 auto key = line.substr(0, eq);
95 auto val = line.substr(eq + 1);
96 if (key == "font_global_scale")
97 font_global_scale_ = std::stof(val);
98 if (key == "autosave_enabled")
99 autosave_enabled_ = (val == "1");
100 if (key == "autosave_interval_secs")
101 autosave_interval_secs_ = std::stof(val);
102 }
103 ImGui::GetIO().FontGlobalScale = font_global_scale_;
104 }
105 } catch (...) {}
106}

116void EditorManager::RefreshWorkspacePresets() {
117 // Safe clearing with error handling
118 try {
119 // Create a new vector instead of clearing to avoid corruption
120 std::vector<std::string> new_presets;
121
122 // Try to read a simple index file of presets
123 try {
124 auto data = util::LoadConfigFile("workspace_presets.txt");
125 if (!data.empty()) {
126 std::istringstream ss(data);
127 std::string name;
128 while (std::getline(ss, name)) {
129 // Trim whitespace and validate
130 name.erase(0, name.find_first_not_of(" \t\r\n"));
131 name.erase(name.find_last_not_of(" \t\r\n") + 1);
132 if (!name.empty() &&
133 name.length() < 256) { // Reasonable length limit
134 new_presets.emplace_back(std::move(name));
135 }
136 }
137 }
138 } catch (const std::exception& e) {
139 LOG_WARN("EditorManager", "Failed to load workspace presets: %s", e.what());
140 }
141
142 // Safely replace the vector
143 workspace_presets_ = std::move(new_presets);
144 workspace_presets_loaded_ = true;
145
146 } catch (const std::exception& e) {
147 LOG_ERROR("EditorManager", "Error in RefreshWorkspacePresets: %s", e.what());
148 // Ensure we have a valid empty vector
149 workspace_presets_ = std::vector<std::string>();
150 workspace_presets_loaded_ =
151 true; // Mark as loaded even if empty to avoid retry
152 }
153}

155void EditorManager::SaveWorkspacePreset(const std::string& name) {
156 if (name.empty())
157 return;
158 std::string ini_name = absl::StrCat("yaze_workspace_", name, ".ini");
159 ImGui::SaveIniSettingsToDisk(ini_name.c_str());
160
161 // Ensure presets are loaded before updating
162 if (!workspace_presets_loaded_) {
163 RefreshWorkspacePresets();
164 }
165
166 // Update index
167 if (std::find(workspace_presets_.begin(), workspace_presets_.end(), name) ==
168 workspace_presets_.end()) {
169 workspace_presets_.emplace_back(name);
170 try {
171 std::ostringstream ss;
172 for (const auto& n : workspace_presets_)
173 ss << n << "\n";
174 util::SaveFile("workspace_presets.txt", ss.str());
175 } catch (const std::exception& e) {
176 LOG_WARN("EditorManager", "Failed to save workspace presets: %s", e.what());
177 }
178 }
179 last_workspace_preset_ = name;
180}

642absl::Status EditorManager::Update() {
643 // Update timing manager for accurate delta time across the application
644 // This fixes animation timing issues that occur when mouse isn't moving
645 core::TimingManager::Get().Update();
646
647 popup_manager_->DrawPopups();
648 ExecuteShortcuts(context_.shortcut_manager);
649 toast_manager_.Draw();
650
651 // Draw editor selection dialog
652 if (show_editor_selection_) {
653 editor_selection_dialog_.Show(&show_editor_selection_);
654 }
655
656#ifdef YAZE_WITH_GRPC
657 // Update agent editor dashboard
658 status_ = agent_editor_.Update();
659
660 // Draw chat widget separately (always visible when active)
661 if (agent_editor_.GetChatWidget()) {
662 agent_editor_.GetChatWidget()->Draw();
663 }
664#endif
665
666 // Draw background grid effects for the entire viewport
667 if (ImGui::GetCurrentContext()) {
668 ImDrawList* bg_draw_list = ImGui::GetBackgroundDrawList();
669 const ImGuiViewport* viewport = ImGui::GetMainViewport();
670
671 auto& theme_manager = gui::ThemeManager::Get();
672 auto current_theme = theme_manager.GetCurrentTheme();
673 auto& bg_renderer = gui::BackgroundRenderer::Get();
674
675 // Draw grid covering the entire main viewport
676 ImVec2 grid_pos = viewport->WorkPos;
677 ImVec2 grid_size = viewport->WorkSize;
678 bg_renderer.RenderDockingBackground(bg_draw_list, grid_pos, grid_size,
679 current_theme.primary);
680 }
681
682 // Ensure TestManager always has the current ROM
683 static Rom* last_test_rom = nullptr;
684 if (last_test_rom != current_rom_) {
685 LOG_DEBUG("EditorManager",
686 "EditorManager::Update - ROM changed, updating TestManager: %p -> "
687 "%p",
688 (void*)last_test_rom, (void*)current_rom_);
689 test::TestManager::Get().SetCurrentRom(current_rom_);
690 last_test_rom = current_rom_;
691 }
692
693 // Autosave timer
694 if (autosave_enabled_ && current_rom_ && current_rom_->dirty()) {
695 autosave_timer_ += ImGui::GetIO().DeltaTime;
696 if (autosave_timer_ >= autosave_interval_secs_) {
697 autosave_timer_ = 0.0f;
698 Rom::SaveSettings s;
699 s.backup = true;
700 s.save_new = false;
701 auto st = current_rom_->SaveToFile(s);
702 if (st.ok()) {
703 toast_manager_.Show("Autosave completed", editor::ToastType::kSuccess);
704 } else {
705 toast_manager_.Show(std::string(st.message()),
706 editor::ToastType::kError, 5.0f);
707 }
708 }
709 } else {
710 autosave_timer_ = 0.0f;
711 }
712
713 // Check if ROM is loaded before allowing editor updates
714 if (!current_editor_set_) {
715 // Show welcome screen when no session is active, but only if not manually closed
716 if (sessions_.empty() && !welcome_screen_manually_closed_) {
717 show_welcome_screen_ = true;
718 }
719 // Don't auto-show here, let the manual control handle it
720 return absl::OkStatus();
721 }
722
723 // Check if current ROM is valid
724 if (!current_rom_) {
725 // Only show welcome screen for truly empty state, not when ROM is loaded but current_rom_ is null
726 if (sessions_.empty() && !welcome_screen_manually_closed_) {
727 show_welcome_screen_ = true;
728 }
729 return absl::OkStatus();
730 }
731
732 // ROM is loaded and valid - don't auto-show welcome screen
733 // Welcome screen should only be shown manually at this point
734
735 // Iterate through ALL sessions to support multi-session docking
736 for (size_t session_idx = 0; session_idx < sessions_.size(); ++session_idx) {
737 auto& session = sessions_[session_idx];
738 if (!session.rom.is_loaded())
739 continue; // Skip sessions with invalid ROMs
740
741 for (auto editor : session.editors.active_editors_) {
742 if (*editor->active()) {
743 if (editor->type() == EditorType::kOverworld) {
744 auto& overworld_editor = static_cast<OverworldEditor&>(*editor);
745 if (overworld_editor.jump_to_tab() != -1) {
746 session.editors.dungeon_editor_.set_active(true);
747 // Set the dungeon editor to the jump to tab
748 session.editors.dungeon_editor_.add_room(
749 overworld_editor.jump_to_tab());
750 overworld_editor.jump_to_tab_ = -1;
751 }
752 }
753
754 // Generate unique window titles and IDs for multi-session support
755 std::string window_title =
756 GenerateUniqueEditorTitle(editor->type(), session_idx);
757
758 // Note: PushID removed - window title provides sufficient uniqueness
759 // and PushID was causing ID stack corruption issues
760
761 // Set window to maximize on first open (use FirstUseEver instead of IsWindowAppearing check)
762 ImGui::SetNextWindowSize(ImGui::GetMainViewport()->WorkSize, ImGuiCond_FirstUseEver);
763 ImGui::SetNextWindowPos(ImGui::GetMainViewport()->WorkPos, ImGuiCond_FirstUseEver);
764
765 if (ImGui::Begin(window_title.c_str(), editor->active(),
766 ImGuiWindowFlags_None)) { // Allow full docking
767 // Temporarily switch context for this editor's update
768 Rom* prev_rom = current_rom_;
769 EditorSet* prev_editor_set = current_editor_set_;
770 size_t prev_session_id = context_.session_id;
771
772 current_rom_ = &session.rom;
773 current_editor_set_ = &session.editors;
774 current_editor_ = editor;
775 context_.session_id = session_idx; // Set session ID for child panels
776
777 status_ = editor->Update();
778
779 // Route editor errors to toast manager
780 if (!status_.ok()) {
781 std::string editor_name =
782 "Editor"; // Get actual editor name if available
783 if (editor == &session.editors.overworld_editor_)
784 editor_name = "Overworld Editor";
785 else if (editor == &session.editors.dungeon_editor_)
786 editor_name = "Dungeon Editor";
787 else if (editor == &session.editors.sprite_editor_)
788 editor_name = "Sprite Editor";
789 else if (editor == &session.editors.graphics_editor_)
790 editor_name = "Graphics Editor";
791 else if (editor == &session.editors.music_editor_)
792 editor_name = "Music Editor";
793 else if (editor == &session.editors.palette_editor_)
794 editor_name = "Palette Editor";
795 else if (editor == &session.editors.screen_editor_)
796 editor_name = "Screen Editor";
797
798 toast_manager_.Show(
799 absl::StrFormat("%s Error: %s", editor_name, status_.message()),
800 editor::ToastType::kError, 8.0f);
801 }
802
803 // Restore context
804 current_rom_ = prev_rom;
805 current_editor_set_ = prev_editor_set;
806 context_.session_id = prev_session_id; // Restore previous session ID
807 }
808 ImGui::End();
809 // PopID removed to match PushID removal above
810 }
811 }
812 }
813
814 if (show_performance_dashboard_) {
815 gfx::PerformanceDashboard::Get().Render();
816 }
817
818 // Always draw proposal drawer (it manages its own visibility)
819 proposal_drawer_.Draw();
820
821#ifdef YAZE_WITH_GRPC
822 // Update ROM context for agent editor
823 if (current_rom_ && current_rom_->is_loaded()) {
824 agent_editor_.SetRomContext(current_rom_);
825 }
826#endif
827
828 return absl::OkStatus();
829}
