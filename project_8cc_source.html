21 // Helper functions for parsing key-value pairs
22 std::pair<std::string, std::string> ParseKeyValue(const std::string& line) {
23 size_t eq_pos = line.find('=');
24 if (eq_pos == std::string::npos) return {"", ""};
25
26 std::string key = line.substr(0, eq_pos);
27 std::string value = line.substr(eq_pos + 1);
28
29 // Trim whitespace
30 key.erase(0, key.find_first_not_of(" \t"));
31 key.erase(key.find_last_not_of(" \t") + 1);
32 value.erase(0, value.find_first_not_of(" \t"));
33 value.erase(value.find_last_not_of(" \t") + 1);
34
35 return {key, value};
36 }

38 bool ParseBool(const std::string& value) {
39 return value == "true" || value == "1" || value == "yes";
40 }
41
42 float ParseFloat(const std::string& value) {
43 try {
44 return std::stof(value);
45 } catch (...) {
46 return 0.0f;
47 }
48 }
49
50 std::vector<std::string> ParseStringList(const std::string& value) {
51 std::vector<std::string> result;
52 if (value.empty()) return result;
53
54 std::vector<std::string> parts = absl::StrSplit(value, ',');
55 for (const auto& part : parts) {
56 std::string trimmed = std::string(part);
57 trimmed.erase(0, trimmed.find_first_not_of(" \t"));
58 trimmed.erase(trimmed.find_last_not_of(" \t") + 1);
59 if (!trimmed.empty()) {
60 result.push_back(trimmed);
61 }
62 }
63 return result;
64 }

138absl::Status YazeProject::LoadFromYazeFormat(const std::string& project_path) {
139 std::ifstream file(project_path);
140 if (!file.is_open()) {
141 return absl::InvalidArgumentError(absl::StrFormat("Cannot open project file: %s", project_path));
142 }
143
144 std::string line;
145 std::string current_section = "";
146
147 while (std::getline(file, line)) {
148 // Skip empty lines and comments
149 if (line.empty() || line[0] == '#') continue;
150
151 // Check for section headers [section_name]
152 if (line[0] == '[' && line.back() == ']') {
153 current_section = line.substr(1, line.length() - 2);
154 continue;
155 }
156
157 auto [key, value] = ParseKeyValue(line);
158 if (key.empty()) continue;
159
160 // Parse based on current section
161 if (current_section == "project") {
162 if (key == "name") name = value;
163 else if (key == "description") metadata.description = value;
164 else if (key == "author") metadata.author = value;
165 else if (key == "license") metadata.license = value;
166 else if (key == "version") metadata.version = value;
167 else if (key == "created_date") metadata.created_date = value;
168 else if (key == "last_modified") metadata.last_modified = value;
169 else if (key == "yaze_version") metadata.yaze_version = value;
170 else if (key == "tags") metadata.tags = ParseStringList(value);
171 }
172 else if (current_section == "files") {
173 if (key == "rom_filename") rom_filename = value;
174 else if (key == "rom_backup_folder") rom_backup_folder = value;
175 else if (key == "code_folder") code_folder = value;
176 else if (key == "assets_folder") assets_folder = value;
177 else if (key == "patches_folder") patches_folder = value;
178 else if (key == "labels_filename") labels_filename = value;
179 else if (key == "symbols_filename") symbols_filename = value;
180 else if (key == "output_folder") output_folder = value;
181 else if (key == "additional_roms") additional_roms = ParseStringList(value);
182 }
183 else if (current_section == "feature_flags") {
184 if (key == "load_custom_overworld") feature_flags.overworld.kLoadCustomOverworld = ParseBool(value);
185 else if (key == "apply_zs_custom_overworld_asm") feature_flags.overworld.kApplyZSCustomOverworldASM = ParseBool(value);
186 else if (key == "save_dungeon_maps") feature_flags.kSaveDungeonMaps = ParseBool(value);
187 else if (key == "save_graphics_sheet") feature_flags.kSaveGraphicsSheet = ParseBool(value);
188 else if (key == "log_instructions") feature_flags.kLogInstructions = ParseBool(value);
189 }
190 else if (current_section == "workspace") {
191 if (key == "font_global_scale") workspace_settings.font_global_scale = ParseFloat(value);
192 else if (key == "dark_mode") workspace_settings.dark_mode = ParseBool(value);
193 else if (key == "ui_theme") workspace_settings.ui_theme = value;
194 else if (key == "autosave_enabled") workspace_settings.autosave_enabled = ParseBool(value);
195 else if (key == "autosave_interval_secs") workspace_settings.autosave_interval_secs = ParseFloat(value);
196 else if (key == "backup_on_save") workspace_settings.backup_on_save = ParseBool(value);
197 else if (key == "show_grid") workspace_settings.show_grid = ParseBool(value);
198 else if (key == "show_collision") workspace_settings.show_collision = ParseBool(value);
199 else if (key == "last_layout_preset") workspace_settings.last_layout_preset = value;
200 else if (key == "saved_layouts") workspace_settings.saved_layouts = ParseStringList(value);
201 else if (key == "recent_files") workspace_settings.recent_files = ParseStringList(value);
202 }
203 else if (current_section == "build") {
204 if (key == "build_script") build_script = value;
205 else if (key == "output_folder") output_folder = value;
206 else if (key == "git_repository") git_repository = value;
207 else if (key == "track_changes") track_changes = ParseBool(value);
208 else if (key == "build_configurations") build_configurations = ParseStringList(value);
209 }
210 else if (current_section.starts_with("labels_")) {
211 // Resource labels: [labels_type_name] followed by key=value pairs
212 std::string label_type = current_section.substr(7); // Remove "labels_" prefix
213 resource_labels[label_type][key] = value;
214 }
215 else if (current_section == "keybindings") {
216 workspace_settings.custom_keybindings[key] = value;
217 }
218 else if (current_section == "editor_visibility") {
219 workspace_settings.editor_visibility[key] = ParseBool(value);
220 }
221 else if (current_section == "zscream_compatibility") {
222 if (key == "original_project_file") zscream_project_file = value;
223 else zscream_mappings[key] = value;
224 }
225 }
226
227 file.close();
228 return absl::OkStatus();
229}

339 }
340 file << "\n";
341 }

343 file << "# End of YAZE Project File\n";
344 file.close();
345
346 return absl::OkStatus();

408absl::Status YazeProject::Validate() const {
409 std::vector<std::string> errors;
410
411 if (name.empty()) errors.push_back("Project name is required");
412 if (filepath.empty()) errors.push_back("Project file path is required");
413 if (rom_filename.empty()) errors.push_back("ROM file is required");
414
415 // Check if files exist
416 if (!rom_filename.empty() && !std::filesystem::exists(GetAbsolutePath(rom_filename))) {
417 errors.push_back("ROM file does not exist: " + rom_filename);
418 }
419
420 if (!code_folder.empty() && !std::filesystem::exists(GetAbsolutePath(code_folder))) {
421 errors.push_back("Code folder does not exist: " + code_folder);
422 }
423
424 if (!labels_filename.empty() && !std::filesystem::exists(GetAbsolutePath(labels_filename))) {
425 errors.push_back("Labels file does not exist: " + labels_filename);
426 }
427
428 if (!errors.empty()) {
429 return absl::InvalidArgumentError(absl::StrJoin(errors, "; "));
430 }
431
432 return absl::OkStatus();
433}

716// Compatibility implementations for ResourceLabelManager and related classes
717bool ResourceLabelManager::LoadLabels(const std::string& filename) {
718 filename_ = filename;
719 std::ifstream file(filename);
720 if (!file.is_open()) {
721 labels_loaded_ = false;
722 return false;
723 }
724
725 labels_.clear();
726 std::string line;
727 std::string current_type = "";
728
729 while (std::getline(file, line)) {
730 if (line.empty() || line[0] == '#') continue;
731
732 // Check for type headers [type_name]
733 if (line[0] == '[' && line.back() == ']') {
734 current_type = line.substr(1, line.length() - 2);
735 continue;
736 }
737
738 // Parse key=value pairs
739 size_t eq_pos = line.find('=');
740 if (eq_pos != std::string::npos && !current_type.empty()) {
741 std::string key = line.substr(0, eq_pos);
742 std::string value = line.substr(eq_pos + 1);
743 labels_[current_type][key] = value;
744 }
745 }
746
747 file.close();
748 labels_loaded_ = true;
749 return true;
750}

752bool ResourceLabelManager::SaveLabels() {
753 if (filename_.empty()) return false;
754
755 std::ofstream file(filename_);
756 if (!file.is_open()) return false;
757
758 file << "# YAZE Resource Labels\n";
759 file << "# Format: [type] followed by key=value pairs\n\n";
760
761 for (const auto& [type, type_labels] : labels_) {
762 if (!type_labels.empty()) {
763 file << "[" << type << "]\n";
764 for (const auto& [key, value] : type_labels) {
765 file << key << "=" << value << "\n";
766 }
767 file << "\n";
768 }
769 }
770
771 file.close();
772 return true;
773}
