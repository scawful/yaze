233absl::Status SetColorsPalette(Rom& rom, int index, gfx::SnesPalette& current,
234 gfx::SnesPalette main, gfx::SnesPalette animated,
235 gfx::SnesPalette aux1, gfx::SnesPalette aux2,
236 gfx::SnesPalette hud, gfx::SnesColor bgrcolor,
237 gfx::SnesPalette spr, gfx::SnesPalette spr2) {
238 // Palettes infos, color 0 of a palette is always transparent (the arrays
239 // contains 7 colors width wide) There is 16 color per line so 16*Y
240
241 // Left side of the palette - Main, Animated
242 std::vector<gfx::SnesColor> new_palette(256);
243
244 // Main Palette, Location 0,2 : 35 colors [7x5]
245 int k = 0;
246 for (int y = 2; y < 7; y++) {
247 for (int x = 1; x < 8; x++) {
248 new_palette[x + (16 * y)] = main[k];
249 k++;
250 }
251 }
252
253 // Animated Palette, Location 0,7 : 7colors
254 for (int x = 1; x < 8; x++) {
255 new_palette[(16 * 7) + (x)] = animated[(x - 1)];
256 }
257
258 // Right side of the palette - Aux1, Aux2
259
260 // Aux1 Palette, Location 8,2 : 21 colors [7x3]
261 k = 0;
262 for (int y = 2; y < 5; y++) {
263 for (int x = 9; x < 16; x++) {
264 new_palette[x + (16 * y)] = aux1[k];
265 k++;
266 }
267 }
268
269 // Aux2 Palette, Location 8,5 : 21 colors [7x3]
270 k = 0;
271 for (int y = 5; y < 8; y++) {
272 for (int x = 9; x < 16; x++) {
273 new_palette[x + (16 * y)] = aux2[k];
274 k++;
275 }
276 }
277
278 // Hud Palette, Location 0,0 : 32 colors [16x2]
279 for (int i = 0; i < 32; i++) {
280 new_palette[i] = hud[i];
281 }
282
283 // Hardcoded grass color (that might change to become invisible instead)
284 for (int i = 0; i < 8; i++) {
285 new_palette[(i * 16)] = bgrcolor;
286 new_palette[(i * 16) + 8] = bgrcolor;
287 }
288
289 // Sprite Palettes
290 k = 0;
291 for (int y = 8; y < 9; y++) {
292 for (int x = 1; x < 8; x++) {
293 auto pal_group = rom.palette_group().sprites_aux1;
294 new_palette[x + (16 * y)] = pal_group[1][k];
295 k++;
296 }
297 }
298
299 // Sprite Palettes
300 k = 0;
301 for (int y = 8; y < 9; y++) {
302 for (int x = 9; x < 16; x++) {
303 auto pal_group = rom.palette_group().sprites_aux3;
304 new_palette[x + (16 * y)] = pal_group[0][k];

306 }
307 }

370absl::Status OverworldMap::LoadPalette() {
371 int previousPalId = index_ > 0 ? rom_[overworldMapPalette + parent_ - 1] : 0;
372 int previousSprPalId =
373 index_ > 0 ? rom_[overworldSpritePalette + parent_ - 1] : 0;
374
375 area_palette_ = std::min((int)area_palette_, 0xA3);
376
377 uchar pal0 = 0;
378 uchar pal1 = rom_[rom_.version_constants().overworldMapPaletteGroup +
379 (area_palette_ * 4)];
380 uchar pal2 = rom_[rom_.version_constants().overworldMapPaletteGroup +
381 (area_palette_ * 4) + 1];
382 uchar pal3 = rom_[rom_.version_constants().overworldMapPaletteGroup +
383 (area_palette_ * 4) + 2];
384 uchar pal4 =
385 rom_[overworldSpritePaletteGroup + (sprite_palette_[game_state_] * 2)];
386 uchar pal5 = rom_[overworldSpritePaletteGroup +
387 (sprite_palette_[game_state_] * 2) + 1];
388
389 auto grass_pal_group = rom_.palette_group().grass;
390 ASSIGN_OR_RETURN(gfx::SnesColor bgr, grass_pal_group[0].GetColor(0));
391
392 auto ow_aux_pal_group = rom_.palette_group().overworld_aux;
393 ASSIGN_OR_RETURN(gfx::SnesPalette aux1,
394 GetPalette(ow_aux_pal_group, pal1, previousPalId, 20));
395 ASSIGN_OR_RETURN(gfx::SnesPalette aux2,
396 GetPalette(ow_aux_pal_group, pal2, previousPalId, 20));
397
398 // Additional handling of `pal3` and `parent_`
399 if (pal3 == 255) {
400 pal3 = rom_[rom_.version_constants().overworldMapPaletteGroup +
401 (previousPalId * 4) + 2];
402 }
403
404 if (parent_ < 0x40) {
405 pal0 = parent_ == 0x03 || parent_ == 0x05 || parent_ == 0x07 ? 2 : 0;
406 ASSIGN_OR_RETURN(bgr, grass_pal_group[0].GetColor(0));
407 } else if (parent_ >= 0x40 && parent_ < 0x80) {
408 pal0 = parent_ == 0x43 || parent_ == 0x45 || parent_ == 0x47 ? 3 : 1;
409 ASSIGN_OR_RETURN(bgr, grass_pal_group[0].GetColor(1));
410 } else if (parent_ >= 128 && parent_ < kNumOverworldMaps) {
411 pal0 = 0;
412 ASSIGN_OR_RETURN(bgr, grass_pal_group[0].GetColor(2));
413 }
414
415 if (parent_ == 0x88) {
416 pal0 = 4;
417 }
418
419 auto ow_main_pal_group = rom_.palette_group().overworld_main;
420 ASSIGN_OR_RETURN(gfx::SnesPalette main,
421 GetPalette(ow_main_pal_group, pal0, previousPalId, 255));
422 auto ow_animated_pal_group = rom_.palette_group().overworld_animated;
423 ASSIGN_OR_RETURN(gfx::SnesPalette animated,
424 GetPalette(ow_animated_pal_group, std::min((int)pal3, 13),
425 previousPalId, 14));
426
427 auto hud_pal_group = rom_.palette_group().hud;
428 gfx::SnesPalette hud = hud_pal_group[0];
429
430 ASSIGN_OR_RETURN(gfx::SnesPalette spr,
431 GetPalette(rom_.palette_group().sprites_aux3, pal4,
432 previousSprPalId, 24));
433 ASSIGN_OR_RETURN(gfx::SnesPalette spr2,
434 GetPalette(rom_.palette_group().sprites_aux3, pal5,
435 previousSprPalId, 24));
436
437 RETURN_IF_ERROR(palette_internal::SetColorsPalette(
438 rom_, parent_, current_palette_, main, animated, aux1, aux2, hud, bgr,
439 spr, spr2));
440
441 if (palettesets_.count(area_palette_) == 0) {
442 palettesets_[area_palette_] = gfx::Paletteset{
443 main, animated, aux1, aux2, bgr, hud, spr, spr2, current_palette_};
444 }
445
446 return absl::OkStatus();
447}

477absl::Status OverworldMap::BuildTiles16Gfx(int count) {
478 if (current_blockset_.size() == 0) current_blockset_.resize(0x100000, 0x00);
479
480 const int offsets[] = {0x00, 0x08, 0x400, 0x408};
481 auto yy = 0;
482 auto xx = 0;
483
484 for (auto i = 0; i < count; i++) {
485 for (auto tile = 0; tile < 0x04; tile++) {
486 gfx::TileInfo info = tiles16_[i].tiles_info[tile];
487 int offset = offsets[tile];
488 for (auto y = 0; y < 0x08; ++y) {
489 for (auto x = 0; x < 0x08; ++x) {
490 int mx = x;
491 int my = y;
492
493 if (info.horizontal_mirror_ != 0) {
494 mx = 0x07 - x;
495 }
496
497 if (info.vertical_mirror_ != 0) {
498 my = 0x07 - y;
499 }
500
501 int xpos = ((info.id_ % 0x10) * 0x08);
502 int ypos = (((info.id_ / 0x10)) * 0x400);
503 int source = ypos + xpos + (x + (y * 0x80));
504
505 auto destination = xx + yy + offset + (mx + (my * 0x80));
506 current_blockset_[destination] =
507 (current_gfx_[source] & 0x0F) + (info.palette_ * 0x10);
508 }
509 }
510 }
511
512 xx += 0x10;
513 if (xx >= 0x80) {
514 yy += 0x800;
515 xx = 0;
516 }
517 }
518
519 return absl::OkStatus();
520}

538absl::Status OverworldMap::BuildBitmap(OWBlockset& world_blockset) {
539 if (bitmap_data_.size() != 0) {
540 bitmap_data_.clear();
541 }
542 bitmap_data_.reserve(0x40000);
543 for (int i = 0; i < 0x40000; i++) {
544 bitmap_data_.push_back(0x00);
545 }
546
547 int superY = ((index_ - (world_ * 0x40)) / 0x08);
548 int superX = index_ - (world_ * 0x40) - (superY * 0x08);
549
550 for (int y = 0; y < 0x20; y++) {
551 for (int x = 0; x < 0x20; x++) {
552 auto xt = x + (superX * 0x20);
553 auto yt = y + (superY * 0x20);
554 CopyTile8bpp16((x * 0x10), (y * 0x10), world_blockset[xt][yt],
555 bitmap_data_, current_blockset_);
556 }
557 }
558 return absl::OkStatus();
559}
