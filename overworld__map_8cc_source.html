458absl::Status SetColorsPalette(Rom &rom, int index, gfx::SnesPalette &current,
459 gfx::SnesPalette main, gfx::SnesPalette animated,
460 gfx::SnesPalette aux1, gfx::SnesPalette aux2,
461 gfx::SnesPalette hud, gfx::SnesColor bgrcolor,
462 gfx::SnesPalette spr, gfx::SnesPalette spr2) {
463 // Palettes infos, color 0 of a palette is always transparent (the arrays
464 // contains 7 colors width wide) There is 16 color per line so 16*Y
465
466 // Left side of the palette - Main, Animated
467 std::array<gfx::SnesColor, 256> new_palette = {};
468
469 // Main Palette, Location 0,2 : 35 colors [7x5]
470 int k = 0;
471 for (int y = 2; y < 7; y++) {
472 for (int x = 1; x < 8; x++) {
473 new_palette[x + (16 * y)] = main[k];
474 k++;
475 }
476 }
477
478 // Animated Palette, Location 0,7 : 7colors
479 for (int x = 1; x < 8; x++) {
480 new_palette[(16 * 7) + (x)] = animated[(x - 1)];
481 }
482
483 // Right side of the palette - Aux1, Aux2
484
485 // Aux1 Palette, Location 8,2 : 21 colors [7x3]
486 k = 0;
487 for (int y = 2; y < 5; y++) {
488 for (int x = 9; x < 16; x++) {
489 new_palette[x + (16 * y)] = aux1[k];
490 k++;
491 }
492 }
493
494 // Aux2 Palette, Location 8,5 : 21 colors [7x3]
495 k = 0;
496 for (int y = 5; y < 8; y++) {
497 for (int x = 9; x < 16; x++) {
498 new_palette[x + (16 * y)] = aux2[k];
499 k++;
500 }
501 }
502
503 // Hud Palette, Location 0,0 : 32 colors [16x2]
504 for (int i = 0; i < 32; i++) {
505 new_palette[i] = hud[i];
506 }
507
508 // Hardcoded grass color (that might change to become invisible instead)
509 for (int i = 0; i < 8; i++) {
510 new_palette[(i * 16)] = bgrcolor;
511 new_palette[(i * 16) + 8] = bgrcolor;
512 }
513
514 // Sprite Palettes
515 k = 0;
516 for (int y = 8; y < 9; y++) {
517 for (int x = 1; x < 8; x++) {
518 new_palette[x + (16 * y)] = rom.palette_group().sprites_aux1[1][k];
519 k++;
520 }
521 }
522
523 // Sprite Palettes
524 k = 0;
525 for (int y = 8; y < 9; y++) {
526 for (int x = 9; x < 16; x++) {
527 new_palette[x + (16 * y)] = rom.palette_group().sprites_aux3[0][k];
528 k++;
529 }
530 }
531
532 // Sprite Palettes
533 k = 0;
534 for (int y = 9; y < 13; y++) {
535 for (int x = 1; x < 16; x++) {
536 new_palette[x + (16 * y)] = rom.palette_group().global_sprites[0][k];
537 k++;
538 }
539 }
540
541 // Sprite Palettes
542 k = 0;
543 for (int y = 13; y < 14; y++) {
544 for (int x = 1; x < 8; x++) {
545 new_palette[x + (16 * y)] = spr[k];
546 k++;
547 }
548 }
549
550 // Sprite Palettes
551 k = 0;
552 for (int y = 14; y < 15; y++) {
553 for (int x = 1; x < 8; x++) {
554 new_palette[x + (16 * y)] = spr2[k];
555 k++;
556 }
557 }

560 k = 0;

562 for (int x = 1; x < 16; x++) {
563 new_palette[x + (16 * y)] = rom.palette_group().armors[0][k];
564 k++;
565 }
566 }
567
568 current.Create(new_palette);
569 for (int i = 0; i < 256; i++) {
570 current[(i / 16) * 16].set_transparent(true);
571 }
572
573 return absl::OkStatus();

590absl::Status OverworldMap::LoadPalette() {
591 int previous_pal_id =
592 index_ > 0 ? rom_[kOverworldMapPaletteIds + parent_ - 1] : 0;
593 int previous_spr_pal_id =
594 index_ > 0 ? rom_[kOverworldSpritePaletteIds + parent_ - 1] : 0;
595
596 area_palette_ = std::min((int)area_palette_, 0xA3);
597
598 uchar pal0 = 0;
599 uchar pal1 = rom_[rom_.version_constants().kOverworldMapPaletteGroup +
600 (area_palette_ * 4)];
601 uchar pal2 = rom_[rom_.version_constants().kOverworldMapPaletteGroup +
602 (area_palette_ * 4) + 1];
603 uchar pal3 = rom_[rom_.version_constants().kOverworldMapPaletteGroup +
604 (area_palette_ * 4) + 2];
605 uchar pal4 =
606 rom_[kOverworldSpritePaletteGroup + (sprite_palette_[game_state_] * 2)];
607 uchar pal5 = rom_[kOverworldSpritePaletteGroup +
608 (sprite_palette_[game_state_] * 2) + 1];
609
610 auto grass_pal_group = rom_.palette_group().grass;
611 ASSIGN_OR_RETURN(gfx::SnesColor bgr, grass_pal_group[0].GetColor(0));
612
613 auto ow_aux_pal_group = rom_.palette_group().overworld_aux;
614 ASSIGN_OR_RETURN(gfx::SnesPalette aux1,
615 GetPalette(ow_aux_pal_group, pal1, previous_pal_id, 20));
616 ASSIGN_OR_RETURN(gfx::SnesPalette aux2,
617 GetPalette(ow_aux_pal_group, pal2, previous_pal_id, 20));
618
619 // Additional handling of `pal3` and `parent_`
620 if (pal3 == 255) {
621 pal3 = rom_[rom_.version_constants().kOverworldMapPaletteGroup +
622 (previous_pal_id * 4) + 2];
623 }
624
625 if (parent_ < kDarkWorldMapIdStart) {
626 pal0 = parent_ == 0x03 || parent_ == 0x05 || parent_ == 0x07 ? 2 : 0;
627 ASSIGN_OR_RETURN(bgr, grass_pal_group[0].GetColor(0));
628 } else if (parent_ >= kDarkWorldMapIdStart &&
629 parent_ < kSpecialWorldMapIdStart) {
630 pal0 = parent_ == 0x43 || parent_ == 0x45 || parent_ == 0x47 ? 3 : 1;
631 ASSIGN_OR_RETURN(bgr, grass_pal_group[0].GetColor(1));
632 } else if (parent_ >= 128 && parent_ < kNumOverworldMaps) {
633 pal0 = 0;
634 ASSIGN_OR_RETURN(bgr, grass_pal_group[0].GetColor(2));
635 }
636
637 if (parent_ == 0x88) {
638 pal0 = 4;
639 }
640

698absl::Status OverworldMap::BuildTiles16Gfx(std::vector<gfx::Tile16> &tiles16,
699 int count) {
700 if (current_blockset_.size() == 0) current_blockset_.resize(0x100000, 0x00);
701
702 const int offsets[] = {0x00, 0x08, 0x400, 0x408};
703 auto yy = 0;
704 auto xx = 0;
705
706 for (auto i = 0; i < count; i++) {
707 for (auto tile = 0; tile < 0x04; tile++) {
708 gfx::TileInfo info = tiles16[i].tiles_info[tile];
709 int offset = offsets[tile];
710 for (auto y = 0; y < 0x08; ++y) {
711 for (auto x = 0; x < 0x08; ++x) {
712 int mx = x;
713 int my = y;
714
715 if (info.horizontal_mirror_ != 0) {
716 mx = 0x07 - x;
717 }
718
719 if (info.vertical_mirror_ != 0) {
720 my = 0x07 - y;
721 }
722
723 int xpos = ((info.id_ % 0x10) * 0x08);
724 int ypos = (((info.id_ / 0x10)) * 0x400);
725 int source = ypos + xpos + (x + (y * 0x80));
726
727 auto destination = xx + yy + offset + (mx + (my * 0x80));
728 current_blockset_[destination] =
729 (current_gfx_[source] & 0x0F) + (info.palette_ * 0x10);
730 }
731 }
732 }
733
734 xx += 0x10;
735 if (xx >= 0x80) {
736 yy += 0x800;
737 xx = 0;
738 }
739 }
740
741 return absl::OkStatus();
742}

744absl::Status OverworldMap::BuildBitmap(OverworldBlockset &world_blockset) {
745 if (bitmap_data_.size() != 0) {
746 bitmap_data_.clear();
747 }
748 bitmap_data_.reserve(0x40000);
749 for (int i = 0; i < 0x40000; i++) {
750 bitmap_data_.push_back(0x00);
751 }
752
753 int superY = ((index_ - (world_ * 0x40)) / 0x08);
754 int superX = index_ - (world_ * 0x40) - (superY * 0x08);
755
756 for (int y = 0; y < 0x20; y++) {
757 for (int x = 0; x < 0x20; x++) {
758 auto xt = x + (superX * 0x20);
759 auto yt = y + (superY * 0x20);
760 gfx::CopyTile8bpp16((x * 0x10), (y * 0x10), world_blockset[xt][yt],
761 bitmap_data_, current_blockset_);
762 }
763 }
764 return absl::OkStatus();
