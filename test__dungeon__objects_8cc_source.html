67std::vector<uint8_t> TestDungeonObjects::CreateObjectSubtypeTable(int base_addr, int count) {
68 std::vector<uint8_t> table(count * 2, 0x00);
69
70 for (int i = 0; i < count; i++) {
71 int addr = i * 2;
72 // Point to tile data at 0x1B52 + (i * 8)
73 int tile_offset = (i * 8) & 0xFFFF;
74 table[addr] = tile_offset & 0xFF;
75 table[addr + 1] = (tile_offset >> 8) & 0xFF;
76 }
77
78 return table;
79}

81std::vector<uint8_t> TestDungeonObjects::CreateTileData(int base_addr, int tile_count) {
82 std::vector<uint8_t> data(tile_count * 8, 0x00);
83
84 for (int i = 0; i < tile_count; i++) {
85 int addr = i * 8;
86 // Create simple tile data
87 for (int j = 0; j < 8; j++) {
88 data[addr + j] = (i + j) & 0xFF;

90 }
