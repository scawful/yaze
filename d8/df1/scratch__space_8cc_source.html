217 // Handle drawing patterns from overworld selections
218 auto mouse_position = scratch_canvas_.drawn_tile_position();
219
220 // Use 32x32 grid size (same as scratch canvas grid)
221 int start_tile_x = static_cast<int>(mouse_position.x) / 32;

223
224 // Get the selected tiles from overworld via clipboard
225 if (!context_ || !context_->shared_clipboard.has_overworld_tile16) {
226 return;
227 }
228
229 const auto& tile_ids = context_->shared_clipboard.overworld_tile16_ids;

231 int pattern_height = context_->shared_clipboard.overworld_height;

234 return;

236 auto& current_slot = scratch_spaces_[current_scratch_slot_];

238 int max_height = current_slot.height > 0 ? current_slot.height : 30;
239
240 // Draw the pattern to scratch space
241 int idx = 0;
242 for (int py = 0; py < pattern_height && (start_tile_y + py) < max_height;

254 current_slot.tile_data[scratch_x][scratch_y] = tile_id;
255 UpdateScratchBitmapTile(scratch_x, scratch_y, tile_id);

257 idx++;
258 }
259 }
260 }
261
262 // Mark scratch space as modified
263 current_slot.in_use = true;
264 if (current_slot.name == "Empty") {

270void OverworldEditor::UpdateScratchBitmapTile(int tile_x, int tile_y,
271 int tile_id, int slot) {
272 gfx::ScopedTimer timer("overworld_update_scratch_tile");
273
274 // Use current slot if not specified
275 if (slot == -1)
276 slot = current_scratch_slot_;
277
278 // Get the tile data from the tile16 blockset
279 auto tile_data = gfx::GetTilemapData(tile16_blockset_, tile_id);
280 if (tile_data.empty())
281 return;
282
283 auto& scratch_slot = scratch_spaces_[slot];
284
285 // Use canvas grid size (32x32) for consistent scaling
286 const int grid_size = 32;
287 int scratch_bitmap_width = scratch_slot.scratch_bitmap.width();
288 int scratch_bitmap_height = scratch_slot.scratch_bitmap.height();
289
290 // Calculate pixel position in scratch bitmap
291 for (int y = 0; y < 16; ++y) {
292 for (int x = 0; x < 16; ++x) {
293 int src_index = y * 16 + x;
294
295 // Scale to grid size - each tile takes up grid_size x grid_size pixels
296 int dst_x = tile_x * grid_size + x + x; // Double scaling for 32x32 grid
297 int dst_y = tile_y * grid_size + y + y;
298
299 // Bounds check for scratch bitmap
300 if (dst_x >= 0 && dst_x < scratch_bitmap_width && dst_y >= 0 &&
301 dst_y < scratch_bitmap_height &&
302 src_index < static_cast<int>(tile_data.size())) {
303
304 // Write 2x2 pixel blocks to fill the 32x32 grid space
305 for (int py = 0; py < 2 && (dst_y + py) < scratch_bitmap_height; ++py) {
306 for (int px = 0; px < 2 && (dst_x + px) < scratch_bitmap_width;
307 ++px) {
308 int dst_index = (dst_y + py) * scratch_bitmap_width + (dst_x + px);
309 scratch_slot.scratch_bitmap.WriteToPixel(dst_index,
310 tile_data[src_index]);
311 }
312 }
313 }
314 }
315 }
316
317 scratch_slot.scratch_bitmap.set_modified(true);
318 // Use batch operations for texture updates
319 scratch_slot.scratch_bitmap.QueueTextureUpdate(
320 core::Renderer::Get().renderer());
321 scratch_slot.in_use = true;
322}

324absl::Status OverworldEditor::SaveCurrentSelectionToScratch(int slot) {
325 gfx::ScopedTimer timer("overworld_save_selection_to_scratch");
326
327 if (slot < 0 || slot >= 4) {
328 return absl::InvalidArgumentError("Invalid scratch slot");
329 }
330
331 if (ow_map_canvas_.select_rect_active() &&
332 !ow_map_canvas_.selected_tiles().empty()) {
333 // Calculate actual selection dimensions from overworld rectangle
334 const auto& selected_points = ow_map_canvas_.selected_points();
335 if (selected_points.size() >= 2) {
336 const auto start = selected_points[0];
337 const auto end = selected_points[1];
338
339 // Calculate width and height in tiles
340 int selection_width =
341 std::abs(static_cast<int>((end.x - start.x) / 16)) + 1;
342 int selection_height =
343 std::abs(static_cast<int>((end.y - start.y) / 16)) + 1;
344
345 // Update scratch space dimensions to match selection
346 scratch_spaces_[slot].width = std::max(1, std::min(selection_width, 32));
347 scratch_spaces_[slot].height =
348 std::max(1, std::min(selection_height, 32));
349 scratch_spaces_[slot].in_use = true;
350 scratch_spaces_[slot].name =
351 absl::StrFormat("Selection %dx%d", scratch_spaces_[slot].width,
352 scratch_spaces_[slot].height);
353
354 // Recreate bitmap with new dimensions
355 int bitmap_width = scratch_spaces_[slot].width * 16;
356 int bitmap_height = scratch_spaces_[slot].height * 16;
357 std::vector<uint8_t> empty_data(bitmap_width * bitmap_height, 0);
358 scratch_spaces_[slot].scratch_bitmap.Create(bitmap_width, bitmap_height,
359 8, empty_data);
360 if (all_gfx_loaded_) {
361 palette_ = overworld_.current_area_palette();
362 scratch_spaces_[slot].scratch_bitmap.SetPalette(palette_);
363 core::Renderer::Get().RenderBitmap(
364 &scratch_spaces_[slot].scratch_bitmap);
365 }
366
367 // Save selected tiles to scratch data with proper layout
368 overworld_.set_current_world(current_world_);
369 overworld_.set_current_map(current_map_);
370
371 int idx = 0;
372 for (int y = 0;
373 y < scratch_spaces_[slot].height &&
374 idx < static_cast<int>(ow_map_canvas_.selected_tiles().size());
375 ++y) {
376 for (int x = 0;
377 x < scratch_spaces_[slot].width &&
378 idx < static_cast<int>(ow_map_canvas_.selected_tiles().size());
379 ++x) {
380 if (idx < static_cast<int>(ow_map_canvas_.selected_tiles().size())) {
381 int tile_id = overworld_.GetTileFromPosition(
382 ow_map_canvas_.selected_tiles()[idx]);
383 if (x < 32 && y < 32) {
384 scratch_spaces_[slot].tile_data[x][y] = tile_id;
385 }
386 // Update the bitmap immediately
387 UpdateScratchBitmapTile(x, y, tile_id, slot);
388 idx++;
389 }
390 }
391 }
392 }
393 } else {
394 // Default single-tile scratch space
395 scratch_spaces_[slot].width = 16; // Default size
396 scratch_spaces_[slot].height = 16;
397 scratch_spaces_[slot].name = absl::StrFormat("Map %d Area", current_map_);
398 scratch_spaces_[slot].in_use = true;
399 }
400
401 // Process all queued texture updates at once
402 gfx::Arena::Get().ProcessBatchTextureUpdates();
403
404 return absl::OkStatus();
405}
