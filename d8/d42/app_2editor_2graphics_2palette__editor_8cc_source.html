61int CustomFormatString(char* buf, size_t buf_size, const char* fmt, ...) {
62 va_list args;
63 va_start(args, fmt);
64#ifdef IMGUI_USE_STB_SPRINTF
65 int w = stbsp_vsnprintf(buf, (int)buf_size, fmt, args);
66#else
67 int w = vsnprintf(buf, buf_size, fmt, args);
68#endif
69 va_end(args);
70 if (buf == nullptr) return w;
71 if (w == -1 || w >= (int)buf_size) w = (int)buf_size - 1;
72 buf[w] = 0;
73 return w;
74}

455void PaletteEditor::AddRecentlyUsedColor(const SnesColor& color) {
456 // Check if color already exists in recently used
457 auto it = std::find_if(
458 recently_used_colors_.begin(), recently_used_colors_.end(),
459 [&color](const SnesColor& c) { return c.snes() == color.snes(); });
460
461 // If found, remove it to re-add at front
462 if (it != recently_used_colors_.end()) {
463 recently_used_colors_.erase(it);
464 }
465
466 // Add at front
467 recently_used_colors_.insert(recently_used_colors_.begin(), color);
468
469 // Limit size
470 if (recently_used_colors_.size() > 16) {
471 recently_used_colors_.pop_back();
472 }

475absl::Status PaletteEditor::HandleColorPopup(gfx::SnesPalette& palette, int i,
476 int j, int n) {
477 auto col = gfx::ToFloatArray(palette[n]);
478 auto original_color = palette[n];
479
480 if (gui::SnesColorEdit4("Edit Color", &palette[n], kColorPopupFlags)) {
481 history_.RecordChange(/*group_name=*/std::string(kPaletteGroupNames[i]),
482 /*palette_index=*/j, /*color_index=*/n,
483 original_color, palette[n]);
484 palette[n].set_modified(true);
485
486 // Add to recently used colors
487 AddRecentlyUsedColor(palette[n]);
488 }
489
490 // Color information display
491 char buf[64];
492 int cr = F32_TO_INT8_SAT(col[0]);
493 int cg = F32_TO_INT8_SAT(col[1]);
494 int cb = F32_TO_INT8_SAT(col[2]);
495
496 Text("RGB: %d, %d, %d", cr, cg, cb);
497 Text("SNES: $%04X", palette[n].snes());
498
499 Separator();
500
501 if (Button("Copy as..", ImVec2(-1, 0))) OpenPopup("Copy");
502 if (BeginPopup("Copy")) {
503 CustomFormatString(buf, IM_ARRAYSIZE(buf), "(%.3ff, %.3ff, %.3ff)", col[0],
504 col[1], col[2]);
505 if (Selectable(buf)) SetClipboardText(buf);
506
507 CustomFormatString(buf, IM_ARRAYSIZE(buf), "(%d,%d,%d)", cr, cg, cb);
508 if (Selectable(buf)) SetClipboardText(buf);
509
510 CustomFormatString(buf, IM_ARRAYSIZE(buf), "#%02X%02X%02X", cr, cg, cb);
511 if (Selectable(buf)) SetClipboardText(buf);
512
513 // SNES Format
514 CustomFormatString(buf, IM_ARRAYSIZE(buf), "$%04X",
515 ConvertRgbToSnes(ImVec4(col[0], col[1], col[2], 1.0f)));
516 if (Selectable(buf)) SetClipboardText(buf);
517
518 EndPopup();
519 }
520
521 // Add a button to add this color to custom palette
522 if (Button("Add to Custom Palette", ImVec2(-1, 0))) {
523 custom_palette_.push_back(palette[n]);
524 }
525
526 EndPopup();
527 return absl::OkStatus();
