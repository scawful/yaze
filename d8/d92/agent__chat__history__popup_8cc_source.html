78void AgentChatHistoryPopup::DrawMessageList() {
79 if (messages_.empty()) {
80 ImGui::TextDisabled("No messages yet. Start a conversation in the chat window.");
81 return;
82 }
83
84 // Calculate starting index for display limit
85 int start_index = messages_.size() > display_limit_ ?
86 messages_.size() - display_limit_ : 0;
87
88 for (int i = start_index; i < messages_.size(); ++i) {
89 const auto& msg = messages_[i];
90
91 // Skip internal messages
92 if (msg.is_internal) continue;
93
94 // Apply filter
95 if (message_filter_ == MessageFilter::kUserOnly &&
96 msg.sender != cli::agent::ChatMessage::Sender::kUser) continue;
97 if (message_filter_ == MessageFilter::kAgentOnly &&
98 msg.sender != cli::agent::ChatMessage::Sender::kAgent) continue;
99
100 DrawMessage(msg, i);
101 }

104void AgentChatHistoryPopup::DrawMessage(const cli::agent::ChatMessage& msg, int index) {
105 ImGui::PushID(index);

114
115 const char* sender_label = from_user ? ICON_MD_PERSON " You" : ICON_MD_SMART_TOY " Agent";
116
117 // Message header
118 ImGui::TextColored(header_color, "%s", sender_label);
119
120 ImGui::SameLine();
121 ImGui::TextDisabled("%s", absl::FormatTime("%H:%M:%S", msg.timestamp, absl::LocalTimeZone()).c_str());
122
123 // Message content
124 ImGui::Indent(10.0f);
125
126 if (msg.table_data.has_value()) {
127 ImGui::TextColored(ImVec4(0.7f, 0.7f, 0.9f, 1.0f), ICON_MD_TABLE_CHART " [Table Data]");

129 ImGui::TextColored(ImVec4(0.7f, 0.7f, 0.9f, 1.0f), ICON_MD_DATA_OBJECT " [Structured Response]");

131 // Truncate long messages
132 std::string content = msg.message;
133 if (content.length() > 200) {
134 content = content.substr(0, 197) + "...";
135 }
136 ImGui::TextWrapped("%s", content.c_str());
137 }
138
139 // Show proposal indicator if present
140 if (msg.proposal.has_value()) {
141 ImGui::TextColored(ImVec4(0.2f, 0.8f, 0.4f, 1.0f),
142 ICON_MD_PREVIEW " Proposal: %s", msg.proposal->id.c_str());
143 }
144
145 ImGui::Unindent(10.0f);
146 ImGui::Spacing();
147 ImGui::Separator();
148
149 ImGui::PopID();
