33 // --- Configure Logging System ---
34 auto string_to_log_level = [](const std::string& s) {
35 std::string upper_s;
36 std::transform(s.begin(), s.end(), std::back_inserter(upper_s),
37 ::toupper);
38 if (upper_s == "YAZE_DEBUG") return yaze::util::LogLevel::YAZE_DEBUG;
39 if (upper_s == "INFO") return yaze::util::LogLevel::INFO;
40 if (upper_s == "WARN" || upper_s == "WARNING")
41 return yaze::util::LogLevel::WARNING;
42 if (upper_s == "ERROR") return yaze::util::LogLevel::ERROR;
43 if (upper_s == "FATAL") return yaze::util::LogLevel::FATAL;
44 return yaze::util::LogLevel::INFO; // Default
45 };

47 auto split_categories = [](const std::string& s) {
