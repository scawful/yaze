38void MaybeStripSmcHeader(std::vector<uint8_t> &rom_data, unsigned long &size) {
39 if (size % kBaseRomSize == kHeaderSize && size >= kHeaderSize) {
40 rom_data.erase(rom_data.begin(), rom_data.begin() + kHeaderSize);

76absl::StatusOr<std::vector<uint8_t>> Load2BppGraphics(const Rom &rom) {
77 std::vector<uint8_t> sheet;
78 const uint8_t sheets[] = {0x71, 0x72, 0xDA, 0xDB, 0xDC, 0xDD, 0xDE};
79 for (const auto &sheet_id : sheets) {
80 auto offset = GetGraphicsAddress(rom.data(), sheet_id,
81 rom.version_constants().kOverworldGfxPtr1,
82 rom.version_constants().kOverworldGfxPtr2,
83 rom.version_constants().kOverworldGfxPtr3);
84 ASSIGN_OR_RETURN(auto decomp_sheet,
85 gfx::lc_lz2::DecompressV2(rom.data(), offset));
86 auto converted_sheet = gfx::SnesTo8bppSheet(decomp_sheet, 2);
87 for (const auto &each_pixel : converted_sheet) {
88 sheet.push_back(each_pixel);
89 }
90 }

115absl::StatusOr<gfx::Bitmap> LoadFontGraphics(const Rom &rom) {
116 std::vector<uint8_t> data(0x2000);
117 for (int i = 0; i < 0x2000; i++) {
118 data[i] = rom.data()[0x70000 + i];
119 }
120
121 std::vector<uint8_t> new_data(0x4000);
122 std::vector<uint8_t> mask = {0x80, 0x40, 0x20, 0x10, 0x08, 0x04, 0x02, 0x01};
123 int sheet_position = 0;
124
125 // 8x8 tile
126 for (int s = 0; s < 4; s++) { // Per Sheet
127 for (int j = 0; j < 4; j++) { // Per Tile Line Y
128 for (int i = 0; i < 16; i++) { // Per Tile Line X
129 for (int y = 0; y < 8; y++) { // Per Pixel Line
130 uint8_t line_bits0 =
131 data[(y * 2) + (i * 16) + (j * 256) + sheet_position];
132 uint8_t line_bits1 =
133 data[(y * 2) + (i * 16) + (j * 256) + 1 + sheet_position];
134
135 for (int x = 0; x < 4; x++) { // Per Pixel X
136 uint8_t pixdata = 0;
137 uint8_t pixdata2 = 0;
138
139 if ((line_bits0 & mask[(x * 2)]) == mask[(x * 2)]) {
140 pixdata += 1;
141 }
142 if ((line_bits1 & mask[(x * 2)]) == mask[(x * 2)]) {
143 pixdata += 2;
144 }
145
146 if ((line_bits0 & mask[(x * 2) + 1]) == mask[(x * 2) + 1]) {
147 pixdata2 += 1;
148 }
149 if ((line_bits1 & mask[(x * 2) + 1]) == mask[(x * 2) + 1]) {
150 pixdata2 += 2;
151 }
152
153 new_data[(y * 64) + (x) + (i * 4) + (j * 512) + (s * 2048)] =
154 (uint8_t)((pixdata << 4) | pixdata2);
155 }
156 }
157 }
158 }
159
160 sheet_position += 0x400;
161 }
162
163 std::vector<uint8_t> fontgfx16_data(0x4000);
164 for (int i = 0; i < 0x4000; i++) {
165 fontgfx16_data[i] = new_data[i];
166 }
167
168 gfx::Bitmap font_gfx;
169 font_gfx.Create(128, 128, 64, fontgfx16_data);
170 return font_gfx;
171}

180
181 for (uint32_t i = 0; i < kNumGfxSheets; i++) {

183 sheet.resize(Uncompressed3BPPSize);

186 rom.version_constants().kOverworldGfxPtr2,

189 sheet.begin());
190 bpp3 = true;

192 bpp3 = false;
193 } else {
194 auto offset = GetGraphicsAddress(
195 rom.data(), i, rom.version_constants().kOverworldGfxPtr1,
196 rom.version_constants().kOverworldGfxPtr2,
197 rom.version_constants().kOverworldGfxPtr3);

199 bpp3 = true;
200 }
201

218 graphics_sheets[i].CreateTexture(renderer);
219 }
220
221 for (int j = 0; j < graphics_sheets[i].size(); ++j) {
222 rom.mutable_graphics_buffer()->push_back(graphics_sheets[i].at(j));
223 }
224
225 } else {
226 for (int j = 0; j < graphics_sheets[0].size(); ++j) {
227 rom.mutable_graphics_buffer()->push_back(0xFF);
228 }
229 }

276absl::Status Rom::LoadFromFile(const std::string &filename,
277 const RomLoadOptions &options) {
278 if (filename.empty()) {
279 return absl::InvalidArgumentError(
280 "Could not load ROM: parameter `filename` is empty.");
281 }
282
283 // Validate file exists before proceeding
284 if (!std::filesystem::exists(filename)) {
285 return absl::NotFoundError(
286 absl::StrCat("ROM file does not exist: ", filename));
287 }
288
289 filename_ = std::filesystem::absolute(filename).string();
290 short_name_ = filename_.substr(filename_.find_last_of("/\\") + 1);
291
292 std::ifstream file(filename_, std::ios::binary);
293 if (!file.is_open()) {
294 return absl::NotFoundError(
295 absl::StrCat("Could not open ROM file: ", filename_));
296 }
297
298 // Get file size and validate
299 try {
300 size_ = std::filesystem::file_size(filename_);
301
302 // Validate ROM size (minimum 32KB, maximum 8MB for expanded ROMs)
303 if (size_ < 32768) {
304 return absl::InvalidArgumentError(
305 absl::StrFormat("ROM file too small (%zu bytes), minimum is 32KB", size_));
306 }
307 if (size_ > 8 * 1024 * 1024) {
308 return absl::InvalidArgumentError(
309 absl::StrFormat("ROM file too large (%zu bytes), maximum is 8MB", size_));
310 }
311 } catch (const std::filesystem::filesystem_error &e) {
312 // Try to get the file size from the open file stream
313 file.seekg(0, std::ios::end);
314 if (!file) {
315 return absl::InternalError(absl::StrCat(
316 "Could not get file size: ", filename_, " - ", e.what()));
317 }
318 size_ = file.tellg();
319
320 // Validate size from stream
321 if (size_ < 32768 || size_ > 8 * 1024 * 1024) {
322 return absl::InvalidArgumentError(
323 absl::StrFormat("Invalid ROM size: %zu bytes", size_));
324 }
325 }
326
327 // Allocate and read ROM data
328 try {
329 rom_data_.resize(size_);
330 file.seekg(0, std::ios::beg);
331 file.read(reinterpret_cast<char *>(rom_data_.data()), size_);
332
333 if (!file) {
334 return absl::InternalError(
335 absl::StrFormat("Failed to read ROM data, read %zu of %zu bytes",
336 file.gcount(), size_));
337 }
338 } catch (const std::bad_alloc& e) {
339 return absl::ResourceExhaustedError(
340 absl::StrFormat("Failed to allocate memory for ROM (%zu bytes)", size_));
341 }
342
343 file.close();
344
345 if (!options.load_zelda3_content) {
346 if (options.strip_header) {
347 MaybeStripSmcHeader(rom_data_, size_);
348 }
349 size_ = rom_data_.size();
350 } else {
351 RETURN_IF_ERROR(LoadZelda3(options));
352 }
353
354 if (options.load_resource_labels) {
355 resource_label_manager_.LoadLabels(
356 absl::StrFormat("%s.labels", filename));
357 }
358
359 return absl::OkStatus();
360}

393absl::Status Rom::LoadZelda3(const RomLoadOptions &options) {
394 if (rom_data_.empty()) {
395 return absl::FailedPreconditionError("ROM data is empty");
396 }
397
398 if (options.strip_header) {
399 MaybeStripSmcHeader(rom_data_, size_);
400 }
401
402 size_ = rom_data_.size();
403
404 constexpr uint32_t kTitleStringOffset = 0x7FC0;
405 constexpr uint32_t kTitleStringLength = 20;
406 constexpr uint32_t kTitleStringOffsetWithHeader = 0x81C0;
407
408 if (options.populate_metadata) {
409 uint32_t offset = options.strip_header ? kTitleStringOffset
410 : kTitleStringOffsetWithHeader;
411 if (offset + kTitleStringLength > rom_data_.size()) {
412 return absl::OutOfRangeError(
413 "ROM image is too small to contain title metadata.");
414 }
415 title_.assign(rom_data_.begin() + offset,
416 rom_data_.begin() + offset + kTitleStringLength);
417 if (rom_data_[offset + 0x19] == 0) {
418 version_ = zelda3_version::JP;
419 } else {
420 version_ = zelda3_version::US;
421 }
422 }
423
424 if (options.populate_palettes) {
425 palette_groups_.clear();
426 RETURN_IF_ERROR(gfx::LoadAllPalettes(rom_data_, palette_groups_));
427 } else {
428 palette_groups_.clear();
429 }
430
431 if (options.populate_gfx_groups) {
432 RETURN_IF_ERROR(LoadGfxGroups());
433 } else {
434 main_blockset_ids = {};
435 room_blockset_ids = {};
436 spriteset_ids = {};
437 paletteset_ids = {};
438 }
439
440 if (options.expand_to_full_image) {
441 if (rom_data_.size() < kBaseRomSize * 2) {
442 rom_data_.resize(kBaseRomSize * 2);
443 }
444 }
445
446 size_ = rom_data_.size();
447
448 return absl::OkStatus();
449}

517absl::Status Rom::SaveToFile(const SaveSettings &settings) {
518 absl::Status non_firing_status;
519 if (rom_data_.empty()) {
520 return absl::InternalError("ROM data is empty.");
521 }
522
523 std::string filename = settings.filename;
524 auto backup = settings.backup;
525 auto save_new = settings.save_new;
526
527 // Check if filename is empty
528 if (filename == "") {
529 filename = filename_;
530 }
531
532 // Check if backup is enabled
533 if (backup) {
534 // Create a backup file with timestamp in its name
535 auto now = std::chrono::system_clock::now();
536 auto now_c = std::chrono::system_clock::to_time_t(now);
537 std::string backup_filename =
538 absl::StrCat(filename, "_backup_", std::ctime(&now_c));
539
540 // Remove newline character from ctime()
541 backup_filename.erase(
542 std::remove(backup_filename.begin(), backup_filename.end(), '\n'),
543 backup_filename.end());
544
545 // Replace spaces with underscores
546 std::replace(backup_filename.begin(), backup_filename.end(), ' ', '_');
547
548 // Now, copy the original file to the backup file
549 try {
550 std::filesystem::copy(filename_, backup_filename,
551 std::filesystem::copy_options::overwrite_existing);
552 } catch (const std::filesystem::filesystem_error &e) {
553 non_firing_status = absl::InternalError(absl::StrCat(
554 "Could not create backup file: ", backup_filename, " - ", e.what()));
555 }
556 }
557
558 // Run the other save functions
559 if (settings.z3_save) {
560 if (core::FeatureFlags::get().kSaveAllPalettes)
561 RETURN_IF_ERROR(SaveAllPalettes());
562 if (core::FeatureFlags::get().kSaveGfxGroups)
563 RETURN_IF_ERROR(SaveGfxGroups());
564 }
565
566 if (save_new) {
567 // Create a file of the same name and append the date between the filename
568 // and file extension
569 auto now = std::chrono::system_clock::now();
570 auto now_c = std::chrono::system_clock::to_time_t(now);
571 auto filename_no_ext = filename.substr(0, filename.find_last_of("."));
572 std::cout << filename_no_ext << std::endl;
573 filename = absl::StrCat(filename_no_ext, "_", std::ctime(&now_c));
574 // Remove spaces from new_filename and replace with _
575 filename.erase(std::remove(filename.begin(), filename.end(), ' '),
576 filename.end());
577 // Remove newline character from ctime()
578 filename.erase(std::remove(filename.begin(), filename.end(), '\n'),
579 filename.end());
580 // Add the file extension back to the new_filename
581 filename = filename + ".sfc";
582 std::cout << filename << std::endl;
583 }
584
585 // Open the file for writing and truncate existing content
586 std::ofstream file(filename.data(), std::ios::binary | std::ios::trunc);
587 if (!file) {
588 return absl::InternalError(
589 absl::StrCat("Could not open ROM file for writing: ", filename));
590 }
591
592 // Save the data to the file
593 try {
594 file.write(
595 static_cast<const char *>(static_cast<const void *>(rom_data_.data())),
596 rom_data_.size());
597 } catch (const std::ofstream::failure &e) {
598 return absl::InternalError(absl::StrCat(
599 "Error while writing to ROM file: ", filename, " - ", e.what()));
600 }
601
602 // Check for write errors
603 if (!file) {
604 return absl::InternalError(
605 absl::StrCat("Error while writing to ROM file: ", filename));
606 }
607
608 if (non_firing_status.ok()) dirty_ = false;
609 return non_firing_status.ok() ? absl::OkStatus() : non_firing_status;
610}
