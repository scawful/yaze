29bool HandleInput(ftxui::ScreenInteractive &screen, ftxui::Event &event,
30 int &selected) {
31 if (event == Event::ArrowDown || event == Event::Character('j')) {
32 selected++;
33 return true;
34 }
35 if (event == Event::ArrowUp || event == Event::Character('k')) {
36 if (selected != 0) selected--;
37 return true;
38 }
39 if (event == Event::Character('q')) {
40 SwitchComponents(screen, LayoutID::kExit);
41 return true;
42 }
43 return false;
44}

684void HexViewerComponent(ftxui::ScreenInteractive &screen) {
685 ReturnIfRomNotLoaded(screen);
686
687 auto back_button =
688 Button("Back", [&] { SwitchComponents(screen, LayoutID::kMainMenu); });
689
690 static int offset = 0;
691 const int lines_to_show = 20;
692
693 auto renderer = Renderer(back_button, [&] {
694 std::vector<Element> rows;
695 for (int i = 0; i < lines_to_show; ++i) {
696 int current_offset = offset + (i * 16);
697 if (current_offset >= static_cast<int>(app_context.rom.size())) {
698 break;
699 }
700
701 Elements row;
702 row.push_back(text(absl::StrFormat("0x%08X: ", current_offset)) | color(Color::Yellow));
703
704 for (int j = 0; j < 16; ++j) {
705 if (current_offset + j < static_cast<int>(app_context.rom.size())) {
706 row.push_back(text(absl::StrFormat("%02X ", app_context.rom.vector()[current_offset + j])));
707 } else {
708 row.push_back(text(" "));
709 }
710 }
711
712 row.push_back(separator());
713
714 for (int j = 0; j < 16; ++j) {
715 if (current_offset + j < static_cast<int>(app_context.rom.size())) {
716 char c = app_context.rom.vector()[current_offset + j];
717 row.push_back(text(std::isprint(c) ? std::string(1, c) : "."));
718 } else {
719 row.push_back(text(" "));
720 }
721 }
722
723 rows.push_back(hbox(row));
724 }
725
726 return vbox({
727 text("Hex Viewer") | center | bold,
728 separator(),
729 vbox(rows) | frame | flex,
730 separator(),
731 text(absl::StrFormat("Offset: 0x%08X", offset)),
732 separator(),
733 back_button->Render() | center,
734 }) | border;
735 });
736
737 auto event_handler = CatchEvent(renderer, [&](Event event) {
738 if (event == Event::ArrowUp) offset = std::max(0, offset - 16);
739 if (event == Event::ArrowDown) offset = std::min(static_cast<int>(app_context.rom.size()) - (lines_to_show * 16), offset + 16);
740 if (event == Event::PageUp) offset = std::max(0, offset - (lines_to_show * 16));
741 if (event == Event::PageDown) offset = std::min(static_cast<int>(app_context.rom.size()) - (lines_to_show * 16), offset + (lines_to_show * 16));
742 if (event == Event::Character('q')) SwitchComponents(screen, LayoutID::kExit);
743 if (event == Event::Return) SwitchComponents(screen, LayoutID::kMainMenu);
744 return false;
745 });
746
747 screen.Loop(event_handler);
748}
