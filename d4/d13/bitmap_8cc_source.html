290void Bitmap::SetPaletteWithTransparent(const SnesPalette &palette, size_t index,
291 int length) {
292 // Store palette even if surface isn't ready yet
293 palette_ = palette;
294
295 // If surface isn't created yet, just store the palette for later
296 if (surface_ == nullptr) {
297 return; // Palette will be applied when surface is created
298 }
299
300 // CRITICAL FIX: Use index directly as palette slot, not index * 7
301 // For 8-color palettes, index should be 0-7, not 0-49
302 if (index >= palette.size()) {
303 throw std::invalid_argument("Invalid palette index");
304 }
305
306 if (length < 0 || length > 8) {
307 throw std::invalid_argument("Invalid palette length (must be 0-8 for 8-color palettes)");
308 }
309
310 if (index + length > palette.size()) {
311 throw std::invalid_argument("Palette index + length exceeds size");
312 }
313
314 // Extract 8-color sub-palette starting at the specified index
315 // This correctly handles both 256-color overworld palettes and smaller palettes
316 std::vector<ImVec4> colors;
317
318 // Always start with transparent color (index 0)
319 colors.push_back(ImVec4(0, 0, 0, 0));
320
321 // Extract up to 7 colors from the palette starting at index
322 for (size_t i = 0; i < 7 && (index + i) < palette.size(); ++i) {
323 auto &pal_color = palette[index + i];
324 colors.push_back(pal_color.rgb());
325 }
326
327 // Ensure we have exactly 8 colors (transparent + 7 data colors)
328 while (colors.size() < 8) {
329 colors.push_back(ImVec4(0, 0, 0, 1.0f)); // Fill with black if needed
330 }
331
332 // CRITICAL FIX: Keep the original complete palette in palette_ member
333 // Only update the SDL surface palette for display purposes
334 // This prevents breaking other editors that expect the complete palette
335 if (palette_.size() != palette.size()) {
336 palette_ = palette; // Store complete palette
337 InvalidatePaletteCache(); // Update cache with complete palette
338 }
339
340 // Apply the 8-color sub-palette to SDL surface for display
341 SDL_UnlockSurface(surface_);
342 for (int color_index = 0; color_index < 8 && color_index < static_cast<int>(colors.size()); ++color_index) {
343 if (color_index < surface_->format->palette->ncolors) {
344 surface_->format->palette->colors[color_index].r = static_cast<Uint8>(colors[color_index].x * 255);
345 surface_->format->palette->colors[color_index].g = static_cast<Uint8>(colors[color_index].y * 255);
346 surface_->format->palette->colors[color_index].b = static_cast<Uint8>(colors[color_index].z * 255);
347 surface_->format->palette->colors[color_index].a = static_cast<Uint8>(colors[color_index].w * 255);
348 }
349 }
350 SDL_LockSurface(surface_);
351}

584uint32_t Bitmap::HashColor(const ImVec4& color) {
585 // Convert float values to integers for consistent hashing
586 uint32_t r = static_cast<uint32_t>(color.x * 255.0F) & 0xFF;
587 uint32_t g = static_cast<uint32_t>(color.y * 255.0F) & 0xFF;
588 uint32_t b = static_cast<uint32_t>(color.z * 255.0F) & 0xFF;
589 uint32_t a = static_cast<uint32_t>(color.w * 255.0F) & 0xFF;
590
591 // Simple hash combining all components
592 return (r << 24) | (g << 16) | (b << 8) | a;

652bool Bitmap::ValidateDataSurfaceSync() {
653 if (!surface_ || !surface_->pixels || data_.empty()) {
654 SDL_Log("ValidateDataSurfaceSync: surface or data is null/empty");
655 return false;
656 }
657
658 // Check if data and surface are synchronized
659 size_t surface_size = static_cast<size_t>(surface_->h * surface_->pitch);
660 size_t data_size = data_.size();
661 size_t compare_size = std::min(data_size, surface_size);
662
663 if (compare_size == 0) {
664 SDL_Log("ValidateDataSurfaceSync: invalid sizes - surface: %zu, data: %zu",
665 surface_size, data_size);
666 return false;
667 }
668
669 // Compare first few bytes to check synchronization
670 if (memcmp(surface_->pixels, data_.data(), compare_size) != 0) {
671 SDL_Log("ValidateDataSurfaceSync: data and surface are not synchronized");
672 return false;
673 }
674
675 return true;
676}
