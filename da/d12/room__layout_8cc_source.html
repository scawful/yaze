11 // Map layout code to actual VRAM tile ID

16 // a tile in the room's graphics sheets (current_gfx16_)

186RoomLayoutObject RoomLayout::CreateLayoutObject(int16_t tile_id, uint8_t x,
187 uint8_t y, uint8_t layer) {
188 // Determine type based on tile ID
189 RoomLayoutObject::Type type = RoomLayoutObject::Type::kUnknown;
190 if (tile_id >= 0x01 && tile_id <= 0x20) {
191 type = RoomLayoutObject::Type::kWall;
192 } else if (tile_id >= 0x21 && tile_id <= 0x40) {
193 type = RoomLayoutObject::Type::kFloor;
194 } else if (tile_id >= 0x41 && tile_id <= 0x60) {
195 type = RoomLayoutObject::Type::kCeiling;
196 } else if (tile_id >= 0x61 && tile_id <= 0x80) {
197 type = RoomLayoutObject::Type::kWater;
198 } else if (tile_id >= 0x81 && tile_id <= 0xA0) {
199 type = RoomLayoutObject::Type::kStairs;
200 } else if (tile_id >= 0xA1 && tile_id <= 0xC0) {
201 type = RoomLayoutObject::Type::kDoor;
202 }
203
204 return RoomLayoutObject(tile_id, x, y, type, layer);
205}
