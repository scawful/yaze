60 // Load room layout from room_object_layout_pointer

63 (rom_data[room_object_layout_pointer + 1] << 8) +

65 layout_pointer = SnesToPc(layout_pointer);
66
67 // Enhanced bounds checking for layout pointer
68 if (layout_pointer < 0 || layout_pointer >= (int)rom_->size()) {
69 return absl::OutOfRangeError(

71 }
72
73 // Get the layout address for this room
74 int layout_address = layout_pointer + (room_id * 3);
75
76 // Enhanced bounds checking for layout address
77 if (layout_address < 0 || layout_address + 2 >= (int)rom_->size()) {
78 return absl::OutOfRangeError(
79 absl::StrFormat("Layout address out of range: %#06x", layout_address));
80 }
81
82 // Read the layout data (3 bytes: bank, high, low)
83 uint8_t bank = rom_data[layout_address + 2];
84 uint8_t high = rom_data[layout_address + 1];
85 uint8_t low = rom_data[layout_address];

91 return absl::OutOfRangeError(absl::StrFormat(

120absl::Status RoomLayout::ParseLayoutData(const std::vector<uint8_t>& data) {
121 objects_.clear();
122 objects_.reserve(width_ * height_);
123
124 // Parse the layout data to create layout objects
125 // This is a simplified implementation - in reality, the format is more
126 // complex
127 for (int y = 0; y < height_; ++y) {
128 for (int x = 0; x < width_; ++x) {
129 int index = y * width_ + x;
130 if (index >= (int)data.size()) continue;
131
132 uint8_t tile_id = data[index];
133
134 // Determine object type based on tile ID
135 RoomLayoutObject::Type type = RoomLayoutObject::Type::kUnknown;
136 if (tile_id == 0) {
137 // Empty space - skip
138 continue;
139 } else if (tile_id >= 0x01 && tile_id <= 0x20) {
140 // Wall tiles
141 type = RoomLayoutObject::Type::kWall;
142 } else if (tile_id >= 0x21 && tile_id <= 0x40) {
143 // Floor tiles
144 type = RoomLayoutObject::Type::kFloor;
145 } else if (tile_id >= 0x41 && tile_id <= 0x60) {
146 // Ceiling tiles
147 type = RoomLayoutObject::Type::kCeiling;
148 } else if (tile_id >= 0x61 && tile_id <= 0x80) {
149 // Water tiles
150 type = RoomLayoutObject::Type::kWater;
151 } else if (tile_id >= 0x81 && tile_id <= 0xA0) {
152 // Stairs
153 type = RoomLayoutObject::Type::kStairs;
154 } else if (tile_id >= 0xA1 && tile_id <= 0xC0) {
155 // Doors
156 type = RoomLayoutObject::Type::kDoor;
157 }
158
159 // Create layout object
160 objects_.emplace_back(tile_id, x, y, type, 0);
161 }
162 }
163
164 return absl::OkStatus();
165}
