251void DungeonEditor::OnRoomSelected(int room_id) {
252 // Update current room ID
253 current_room_id_ = room_id;
254
255 // Check if room is already open in a tab
256 int existing_tab_index = -1;
257 for (int i = 0; i < active_rooms_.Size; i++) {
258 if (active_rooms_[i] == room_id) {
259 existing_tab_index = i;
260 break;
261 }
262 }
263
264 if (existing_tab_index >= 0) {
265 // Room is already open, switch to that tab
266 current_active_room_tab_ = existing_tab_index;
267 } else {
268 // Room is not open, add it as a new tab
269 active_rooms_.push_back(room_id);
270 current_active_room_tab_ = active_rooms_.Size - 1;
271 }
272
273 // Update the room selector's active rooms list
274 room_selector_.set_active_rooms(active_rooms_);
275}

483void DungeonEditor::DrawDungeonTabView() {
484 static int next_tab_id = 0;
485
486 if (BeginTabBar("MyTabBar", kDungeonTabBarFlags)) {
487 if (ImGui::TabItemButton(ICON_MD_ADD, kDungeonTabFlags)) {
488 if (std::find(active_rooms_.begin(), active_rooms_.end(),
489 current_room_id_) != active_rooms_.end()) {
490 // Room is already open
491 next_tab_id++;
492 }
493 active_rooms_.push_back(next_tab_id++); // Add new tab
494 }
495
496 // Submit our regular tabs
497 for (int n = 0; n < active_rooms_.Size;) {
498 bool open = true;
499
500 if (active_rooms_[n] > sizeof(zelda3::kRoomNames) / 4) {
501 active_rooms_.erase(active_rooms_.Data + n);
502 continue;
503 }
504
505 if (BeginTabItem(zelda3::kRoomNames[active_rooms_[n]].data(), &open,
506 ImGuiTabItemFlags_None)) {
507 current_active_room_tab_ = n; // Track which tab is currently active
508 DrawDungeonCanvas(active_rooms_[n]);
509 EndTabItem();
510 }
511
512 if (!open)
513 active_rooms_.erase(active_rooms_.Data + n);
514 else
515 n++;
516 }
517
518 EndTabBar();
519 }
520 ImGui::Separator();

704 (void)LoadAndRenderRoomGraphics(rooms_[room_id]);
705 }
706
707 // Load room objects if not already loaded
708 if (rooms_[room_id].GetTileObjects().empty()) {
709 rooms_[room_id].LoadObjects();
710 }
711
712 // Render background layers with proper positioning
713 renderer_.RenderRoomBackgroundLayers(room_id);
714
715 // Render room objects and sprites with improved graphics
716 if (current_palette_id_ < current_palette_group_.size()) {
717 auto room_palette = current_palette_group_[current_palette_id_];
718
719 // Render regular objects with improved fallback
720 for (const auto& object : rooms_[room_id].GetTileObjects()) {
721 renderer_.RenderObjectInCanvas(object, room_palette);
722 }
723
724 // Render sprites as simple 16x16 squares with labels
725 renderer_.RenderSprites(rooms_[room_id]);
726 }
727 }
728
729 // Phase 5: Render with integrated object editor
730 RenderRoomWithObjects(room_id);
731
732 // Draw selection box and drag preview using component
733 object_interaction_.DrawSelectBox();
734 object_interaction_.DrawDragPreview();
735
736 canvas_.DrawGrid();
737 canvas_.DrawOverlay();
738}

744void DungeonEditor::UpdateObjectEditor() {
745 if (!object_editor_ || !rom_ || !rom_->is_loaded()) {
746 return;
747 }
748
749 // Get current room ID
750 int room_id = current_room_id_;
751 if (!active_rooms_.empty() && current_active_room_tab_ < active_rooms_.Size) {
752 room_id = active_rooms_[current_active_room_tab_];
753 }
754
755 if (room_id < 0 || room_id >= rooms_.size()) {
756 return;
757 }
758
759 // Ensure room graphics and objects are loaded
760 auto& room = rooms_[room_id];
761
762 // Load room graphics if not already loaded (this populates arena buffers)
763 if (room.blocks().empty()) {
764 auto status = LoadAndRenderRoomGraphics(room);
765 if (!status.ok()) {
766 // Log error but continue
767 return;
768 }
769 }
770
771 // Load room objects if not already loaded
772 if (room.GetTileObjects().empty()) {
773 room.LoadObjects();
774 }
775

777 // The object editor should work with the room's tile_objects_ directly
778 // rather than maintaining its own copy

781void DungeonEditor::RenderRoomWithObjects(int room_id) {
782 if (!object_editor_ || room_id < 0 || room_id >= rooms_.size()) {
783 return;
784 }
785
786 // Ensure room graphics are loaded and rendered to arena buffers first
787 if (rooms_[room_id].blocks().empty()) {
788 // Room graphics not loaded yet, will be loaded by LoadAndRenderRoomGraphics
789 return;
790 }
791
792 // Get the arena buffers for rendering - these should already be populated
793 // by Room::RenderRoomGraphics() which was called in LoadAndRenderRoomGraphics
794 auto& bg1_bitmap = gfx::Arena::Get().bg1().bitmap();
795 auto& bg2_bitmap = gfx::Arena::Get().bg2().bitmap();
796
797 if (!bg1_bitmap.is_active() || !bg2_bitmap.is_active()) {
798 // Arena bitmaps not initialized, this means RenderRoomGraphics wasn't called
799 return;
800 }
801
802 // Render layer visualization if enabled (draws on top of existing bitmap)
803 if (enable_layer_visualization_) {
804 object_editor_->RenderLayerVisualization(bg1_bitmap);
805 }
806
807 // Render selection highlights if enabled (draws on top of existing bitmap)
808 if (enable_selection_highlight_) {
809 object_editor_->RenderSelectionHighlight(bg1_bitmap);
810 }
