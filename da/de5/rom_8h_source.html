151 virtual absl::Status WriteHelper(const WriteAction& action) {
152 if (std::holds_alternative<uint8_t>(action.value)) {
153 return WriteByte(action.address, std::get<uint8_t>(action.value));
154 } else if (std::holds_alternative<uint16_t>(action.value) ||
155 std::holds_alternative<short>(action.value)) {
156 return WriteShort(action.address, std::get<uint16_t>(action.value));
157 } else if (std::holds_alternative<std::vector<uint8_t>>(action.value)) {
158 return WriteVector(action.address,
159 std::get<std::vector<uint8_t>>(action.value));
160 } else if (std::holds_alternative<gfx::SnesColor>(action.value)) {
161 return WriteColor(action.address, std::get<gfx::SnesColor>(action.value));
162 } else if (std::holds_alternative<std::vector<gfx::SnesColor>>(
163 action.value)) {
164 return absl::UnimplementedError(
165 "WriteHelper: std::vector<gfx::SnesColor>");
166 }
167 auto error_message = absl::StrFormat("Invalid write argument type: %s",
168 typeid(action.value).name());
169 return absl::InvalidArgumentError(error_message);
170 }

172 template <typename T>
173 absl::Status ReadHelper(T& var, int address) {
174 if constexpr (std::is_same_v<T, uint8_t>) {
175 ASSIGN_OR_RETURN(auto result, ReadByte(address));
176 var = result;
177 } else if constexpr (std::is_same_v<T, uint16_t>) {
178 ASSIGN_OR_RETURN(auto result, ReadWord(address));
179 var = result;
180 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
181 ASSIGN_OR_RETURN(auto result, ReadByteVector(address, var.size()));
182 var = result;
183 }
184 return absl::OkStatus();
185 }
