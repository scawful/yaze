997void Canvas::DrawSelectRect(int current_map, int tile_size, float scale) {
998 gfx::ScopedTimer timer("canvas_select_rect");
999
1000 const ImGuiIO& io = GetIO();
1001 const ImVec2 origin(canvas_p0_.x + scrolling_.x, canvas_p0_.y + scrolling_.y);
1002 const ImVec2 mouse_pos(io.MousePos.x - origin.x, io.MousePos.y - origin.y);
1003 static ImVec2 drag_start_pos;
1004 const float scaled_size = tile_size * scale;
1005 static bool dragging = false;
1006 constexpr int small_map_size = 0x200;
1007
1008 // Only handle mouse events if the canvas is hovered
1009 const bool is_hovered = IsItemHovered();
1010 if (!is_hovered) {
1011 return;
1012 }
1013
1014 // Calculate superX and superY accounting for world offset
1015 int superY, superX;
1016 if (current_map < 0x40) {
1017 // Light World
1018 superY = current_map / 8;
1019 superX = current_map % 8;
1020 } else if (current_map < 0x80) {
1021 // Dark World
1022 superY = (current_map - 0x40) / 8;
1023 superX = (current_map - 0x40) % 8;
1024 } else {
1025 // Special World
1026 superY = (current_map - 0x80) / 8;
1027 superX = (current_map - 0x80) % 8;
1028 }
1029
1030 // Handle right click for single tile selection
1031 if (IsMouseClicked(ImGuiMouseButton_Right)) {
1032 ImVec2 painter_pos = AlignPosToGrid(mouse_pos, scaled_size);
1033 int painter_x = painter_pos.x;
1034 int painter_y = painter_pos.y;
1035
1036 auto tile16_x = (painter_x % small_map_size) / (small_map_size / 0x20);
1037 auto tile16_y = (painter_y % small_map_size) / (small_map_size / 0x20);
1038
1039 int index_x = superX * 0x20 + tile16_x;
1040 int index_y = superY * 0x20 + tile16_y;
1041 selected_tile_pos_ = ImVec2(index_x, index_y);
1042 selected_points_.clear();
1043 select_rect_active_ = false;
1044
1045 // Start drag position for rectangle selection
1046 drag_start_pos = AlignPosToGrid(mouse_pos, scaled_size);
1047 }
1048
1049 // Calculate the rectangle's top-left and bottom-right corners
1050 ImVec2 drag_end_pos = AlignPosToGrid(mouse_pos, scaled_size);
1051 if (ImGui::IsMouseDragging(ImGuiMouseButton_Right)) {
1052 auto start = ImVec2(canvas_p0_.x + drag_start_pos.x,
1053 canvas_p0_.y + drag_start_pos.y);
1054 auto end = ImVec2(canvas_p0_.x + drag_end_pos.x + tile_size,
1055 canvas_p0_.y + drag_end_pos.y + tile_size);
1056 draw_list_->AddRect(start, end, kWhiteColor);
1057 dragging = true;
1058 }
1059
1060 if (dragging && !ImGui::IsMouseDown(ImGuiMouseButton_Right)) {
1061 // Release dragging mode
1062 dragging = false;
1063
1064 // Calculate the bounds of the rectangle in terms of 16x16 tile indices
1065 constexpr int tile16_size = 16;
1066 int start_x = std::floor(drag_start_pos.x / scaled_size) * tile16_size;
1067 int start_y = std::floor(drag_start_pos.y / scaled_size) * tile16_size;
1068 int end_x = std::floor(drag_end_pos.x / scaled_size) * tile16_size;
1069 int end_y = std::floor(drag_end_pos.y / scaled_size) * tile16_size;
1070
1071 // Swap the start and end positions if they are in the wrong order
1072 if (start_x > end_x)
1073 std::swap(start_x, end_x);
1074 if (start_y > end_y)
1075 std::swap(start_y, end_y);
1076
1077 selected_tiles_.clear();
1078 selected_tiles_.reserve(((end_x - start_x) / tile16_size + 1) *
1079 ((end_y - start_y) / tile16_size + 1));
1080
1081 // Number of tiles per local map (since each tile is 16x16)
1082 constexpr int tiles_per_local_map = small_map_size / 16;
1083
1084 // Loop through the tiles in the rectangle and store their positions
1085 for (int y = start_y; y <= end_y; y += tile16_size) {
1086 for (int x = start_x; x <= end_x; x += tile16_size) {
1087 // Determine which local map (512x512) the tile is in
1088 int local_map_x = (x / small_map_size) % 8;
1089 int local_map_y = (y / small_map_size) % 8;
1090
1091 // Calculate the tile's position within its local map
1092 int tile16_x = (x % small_map_size) / tile16_size;
1093 int tile16_y = (y % small_map_size) / tile16_size;
1094
1095 // Calculate the index within the overall map structure
1096 int index_x = local_map_x * tiles_per_local_map + tile16_x;
1097 int index_y = local_map_y * tiles_per_local_map + tile16_y;
1098
1099 selected_tiles_.emplace_back(index_x, index_y);
1100 }
1101 }
1102 // Clear and add the calculated rectangle points
1103 selected_points_.clear();
1104 selected_points_.push_back(drag_start_pos);
1105 selected_points_.push_back(drag_end_pos);
1106 select_rect_active_ = true;
1107 }
1108}

1200void Canvas::DrawBitmapGroup(std::vector<int>& group, gfx::Tilemap& tilemap,
1201 int tile_size, float scale, int local_map_size,
1202 ImVec2 total_map_size) {

1204 // points_ should contain exactly two points
1205 return;

1207 if (group.empty()) {
1208 // group should not be empty
1209 return;
1210 }

1213 bool use_optimized_rendering =
1214 group.size() > 128; // Optimize for large selections
1215
1216 // Use provided map sizes for proper boundary handling
1217 const int small_map = local_map_size;
1218 const float large_map_width = total_map_size.x;
1219 const float large_map_height = total_map_size.y;
1220
1221 // Pre-calculate common values to avoid repeated computation
1222 const float tile_scale = tile_size * scale;
1223 const int atlas_tiles_per_row = tilemap.atlas.width() / tilemap.tile_size.x;
1224
1225 // Top-left and bottom-right corners of the rectangle
1226 ImVec2 rect_top_left = selected_points_[0];
1227 ImVec2 rect_bottom_right = selected_points_[1];
1228
1229 // Calculate the start and end tiles in the grid
1230 int start_tile_x =
1231 static_cast<int>(std::floor(rect_top_left.x / (tile_size * scale)));
1232 int start_tile_y =
1233 static_cast<int>(std::floor(rect_top_left.y / (tile_size * scale)));
1234 int end_tile_x =
1235 static_cast<int>(std::floor(rect_bottom_right.x / (tile_size * scale)));
1236 int end_tile_y =
1237 static_cast<int>(std::floor(rect_bottom_right.y / (tile_size * scale)));
1238
1239 if (start_tile_x > end_tile_x)
1240 std::swap(start_tile_x, end_tile_x);
1241 if (start_tile_y > end_tile_y)
1242 std::swap(start_tile_y, end_tile_y);
1243
1244 // Calculate the size of the rectangle in 16x16 grid form
1245 int rect_width = (end_tile_x - start_tile_x) * tile_size;
1246 int rect_height = (end_tile_y - start_tile_y) * tile_size;
1247
1248 int tiles_per_row = rect_width / tile_size;
1249 int tiles_per_col = rect_height / tile_size;
1250
1251 int i = 0;
1252 for (int y = 0; y < tiles_per_col + 1; ++y) {
1253 for (int x = 0; x < tiles_per_row + 1; ++x) {
1254 // Check bounds to prevent access violations
1255 if (i >= static_cast<int>(group.size())) {
1256 break;
1257 }

1260

1262 auto tilemap_size = tilemap.map_size.x;
1263 if (tile_id >= 0 && tile_id < tilemap_size) {
1264 // Calculate the position of the tile within the rectangle

1267
1268 // OPTIMIZATION: Use pre-calculated values for better performance with large selections
1269 if (tilemap.atlas.is_active() && tilemap.atlas.texture() &&
1270 atlas_tiles_per_row > 0) {
1271 int atlas_tile_x =
1272 (tile_id % atlas_tiles_per_row) * tilemap.tile_size.x;
1273 int atlas_tile_y =
1274 (tile_id / atlas_tiles_per_row) * tilemap.tile_size.y;
1275
1276 // Simple bounds check
1277 if (atlas_tile_x >= 0 && atlas_tile_x < tilemap.atlas.width() &&
1278 atlas_tile_y >= 0 && atlas_tile_y < tilemap.atlas.height()) {
1279
1280 // Calculate UV coordinates once for efficiency
1281 const float atlas_width = static_cast<float>(tilemap.atlas.width());
1282 const float atlas_height =
1283 static_cast<float>(tilemap.atlas.height());
1284 ImVec2 uv0 =
1285 ImVec2(atlas_tile_x / atlas_width, atlas_tile_y / atlas_height);
1286 ImVec2 uv1 =
1287 ImVec2((atlas_tile_x + tilemap.tile_size.x) / atlas_width,
1288 (atlas_tile_y + tilemap.tile_size.y) / atlas_height);
1289
1290 // Calculate screen positions
1291 float screen_x = canvas_p0_.x + scrolling_.x + tile_pos_x;
1292 float screen_y = canvas_p0_.y + scrolling_.y + tile_pos_y;
1293 float screen_w = tilemap.tile_size.x * scale;

1298 ? IM_COL32(255, 255, 255, 200)
1299 : IM_COL32(255, 255, 255, 150);
1300
1301 // Draw from atlas texture with optimized parameters
1302 draw_list_->AddImage(
1303 (ImTextureID)(intptr_t)tilemap.atlas.texture(),
1304 ImVec2(screen_x, screen_y),
1305 ImVec2(screen_x + screen_w, screen_y + screen_h), uv0, uv1,
1306 alpha_color);
1307 }
1308 }
1309 }
1310 i++;
1311 }
1312 // Break outer loop if we've run out of tiles
1313 if (i >= static_cast<int>(group.size())) {
1314 break;
1315 }
1316 }
1317
1318 // Performance optimization completed - tiles are now rendered with pre-calculated values
1319
1320 // Reposition rectangle to follow mouse, but clamp to prevent wrapping across map boundaries
1321 const ImGuiIO& io = GetIO();
1322 const ImVec2 origin(canvas_p0_.x + scrolling_.x, canvas_p0_.y + scrolling_.y);
1323 const ImVec2 mouse_pos(io.MousePos.x - origin.x, io.MousePos.y - origin.y);
1324
1325 // CRITICAL FIX: Clamp BEFORE grid alignment for smoother dragging behavior
1326 // This prevents the rectangle from even attempting to cross boundaries during drag
1327 ImVec2 clamped_mouse_pos = mouse_pos;
1328
1329 if (config_.clamp_rect_to_local_maps) {
1330 // Calculate which local map the mouse is in
1331 int mouse_local_map_x = static_cast<int>(mouse_pos.x) / small_map;
1332 int mouse_local_map_y = static_cast<int>(mouse_pos.y) / small_map;
1333
1334 // Calculate where the rectangle END would be if we place it at mouse position
1335 float potential_end_x = mouse_pos.x + rect_width;
1336 float potential_end_y = mouse_pos.y + rect_height;
1337
1338 // Check if this would cross local map boundary (512x512 blocks)
1339 int potential_end_map_x = static_cast<int>(potential_end_x) / small_map;
1340 int potential_end_map_y = static_cast<int>(potential_end_y) / small_map;
1341
1342 // Clamp mouse position to prevent crossing during drag
1343 if (potential_end_map_x != mouse_local_map_x) {
1344 // Would cross horizontal boundary - clamp mouse to safe zone
1345 float max_mouse_x = (mouse_local_map_x + 1) * small_map - rect_width;
1346 clamped_mouse_pos.x = std::min(mouse_pos.x, max_mouse_x);
1347 }
1348
1349 if (potential_end_map_y != mouse_local_map_y) {
1350 // Would cross vertical boundary - clamp mouse to safe zone
1351 float max_mouse_y = (mouse_local_map_y + 1) * small_map - rect_height;
1352 clamped_mouse_pos.y = std::min(mouse_pos.y, max_mouse_y);
1353 }
1354 }
1355
1356 // Now grid-align the clamped position
1357 auto new_start_pos = AlignPosToGrid(clamped_mouse_pos, tile_size * scale);
1358
1359 // Additional safety: clamp to overall map bounds
1360 new_start_pos.x =
1361 std::clamp(new_start_pos.x, 0.0f, large_map_width - rect_width);
1362 new_start_pos.y =
1363 std::clamp(new_start_pos.y, 0.0f, large_map_height - rect_height);
1364
1365 selected_points_.clear();
1366 selected_points_.push_back(new_start_pos);
1367 selected_points_.push_back(
1368 ImVec2(new_start_pos.x + rect_width, new_start_pos.y + rect_height));
1369 select_rect_active_ = true;
1370}
