66void Apu::RunCycles(uint64_t master_cycles) {
67 // Convert CPU master cycles to APU cycles target and step SPC/DSP accordingly.
68 const double ratio = memory_.pal_timing() ? apuCyclesPerMasterPal : apuCyclesPerMaster;
69
70 // Track last master cycles to only advance by the delta
71 uint64_t master_delta = master_cycles - g_last_master_cycles;
72 g_last_master_cycles = master_cycles;
73
74 const uint64_t target_apu_cycles = cycles_ + static_cast<uint64_t>(master_delta * ratio);
75
76 // Watchdog to detect infinite loops
77 static uint64_t last_log_cycle = 0;
78 static uint16_t last_pc = 0;
79 static int stuck_counter = 0;
80 static bool logged_transfer_state = false;
81
82 while (cycles_ < target_apu_cycles) {
83 // Execute one SPC700 opcode (variable cycles) then advance APU cycles accordingly.
84 uint16_t current_pc = spc700_.PC;
85
86 // IPL ROM protocol analysis - let it run to see what happens
87 // Log IPL ROM transfer loop activity (every 1000 cycles when in critical range)
88 static uint64_t last_ipl_log = 0;
89 if (rom_readable_ && current_pc >= 0xFFD6 && current_pc <= 0xFFED) {
90 if (cycles_ - last_ipl_log > 10000) {
91 LOG_INFO("APU", "IPL ROM loop: PC=$%04X Y=$%02X Ports: F4=$%02X F5=$%02X F6=$%02X F7=$%02X",
92 current_pc, spc700_.Y, in_ports_[0], in_ports_[1], in_ports_[2], in_ports_[3]);
93 LOG_INFO("APU", " Out ports: F4=$%02X F5=$%02X F6=$%02X F7=$%02X ZP: $00=$%02X $01=$%02X",
94 out_ports_[0], out_ports_[1], out_ports_[2], out_ports_[3], ram[0x00], ram[0x01]);
95 last_ipl_log = cycles_;
96 }
97 }
98
99 // Detect if SPC is stuck in tight loop
100 if (current_pc == last_pc) {
101 stuck_counter++;
102 if (stuck_counter > 10000 && cycles_ - last_log_cycle > 10000) {
103 LOG_WARN("APU", "SPC700 stuck at PC=$%04X for %d iterations",
104 current_pc, stuck_counter);
105 LOG_WARN("APU", "Port Status: F4=$%02X F5=$%02X F6=$%02X F7=$%02X",
106 in_ports_[0], in_ports_[1], in_ports_[2], in_ports_[3]);
107 LOG_WARN("APU", "Out Ports: F4=$%02X F5=$%02X F6=$%02X F7=$%02X",
108 out_ports_[0], out_ports_[1], out_ports_[2], out_ports_[3]);
109 LOG_WARN("APU", "IPL ROM enabled: %s", rom_readable_ ? "YES" : "NO");
110 LOG_WARN("APU", "SPC700 Y=$%02X, ZP $00=$%02X $01=$%02X",
111 spc700_.Y, ram[0x00], ram[0x01]);
112 if (!logged_transfer_state && ram[0x00] == 0x19 && ram[0x01] == 0x00) {
113 LOG_WARN("APU", "Uploaded byte at $0019 = $%02X", ram[0x0019]);
114 logged_transfer_state = true;
115 }
116 last_log_cycle = cycles_;
117 stuck_counter = 0;
118 }
119 } else {
120 stuck_counter = 0;
121 }
122 last_pc = current_pc;
123
124 spc700_.RunOpcode();
125
126 // Get the actual cycle count from the last opcode execution
127 // This is critical for proper IPL ROM handshake timing
128 int spc_cycles = spc700_.GetLastOpcodeCycles();
129
130 // Advance APU cycles based on actual SPC700 opcode timing
131 // The SPC700 runs at 1.024 MHz, and we need to synchronize with the DSP/timers
132 for (int i = 0; i < spc_cycles; ++i) {
133 Cycle();
134 }
135 }
136}

138void Apu::Cycle() {
139 if ((cycles_ & 0x1f) == 0) {
140 // every 32 cycles
141 dsp_.Cycle();
142 }
143
144 // handle timers
145 for (int i = 0; i < 3; i++) {
146 if (timer_[i].cycles == 0) {
147 timer_[i].cycles = i == 2 ? 16 : 128;
148 if (timer_[i].enabled) {
149 timer_[i].divider++;
150 if (timer_[i].divider == timer_[i].target) {
151 timer_[i].divider = 0;
152 timer_[i].counter++;
153 timer_[i].counter &= 0xf;
154 }
155 }
156 }
157 timer_[i].cycles--;
158 }
159
160 cycles_++;
161}

163uint8_t Apu::Read(uint16_t adr) {
164 static int port_read_count = 0;
165 switch (adr) {
166 case 0xf0:
167 case 0xf1:
168 case 0xfa:
169 case 0xfb:
170 case 0xfc: {
171 return 0;
172 }
173 case 0xf2: {
174 return dsp_adr_;
175 }
176 case 0xf3: {
177 return dsp_.Read(dsp_adr_ & 0x7f);
178 }
179 case 0xf4:
180 case 0xf5:
181 case 0xf6:
182 case 0xf7: {
183 uint8_t val = in_ports_[adr - 0xf4];
184 port_read_count++;
185 if (port_read_count < 10) { // Reduced to prevent logging overflow crash
186 LOG_INFO("APU", "SPC read port $%04X (F%d) = $%02X at PC=$%04X",
187 adr, adr - 0xf4 + 4, val, spc700_.PC);
188 }
189 return val;
190 }
191 case 0xf8:
192 case 0xf9: {
193 // Not I/O ports on real hardware; treat as general RAM region.
194 return ram[adr];
195 }
196 case 0xfd:
197 case 0xfe:
198 case 0xff: {
199 uint8_t ret = timer_[adr - 0xfd].counter;
200 timer_[adr - 0xfd].counter = 0;
201 return ret;
202 }
203 }
204 if (rom_readable_ && adr >= 0xffc0) {
205 return bootRom[adr - 0xffc0];
206 }
207 return ram[adr];
208}

210void Apu::Write(uint16_t adr, uint8_t val) {
211 static int port_write_count = 0;
212
213 switch (adr) {
214 case 0xf0: {
215 break; // test register
216 }
217 case 0xf1: {
218 bool old_rom_readable = rom_readable_;
219 for (int i = 0; i < 3; i++) {
220 if (!timer_[i].enabled && (val & (1 << i))) {
221 timer_[i].divider = 0;
222 timer_[i].counter = 0;
223 }
224 timer_[i].enabled = val & (1 << i);
225 }
226 if (val & 0x10) {
227 in_ports_[0] = 0;
228 in_ports_[1] = 0;
229 }
230 if (val & 0x20) {
231 in_ports_[2] = 0;
232 in_ports_[3] = 0;
233 }
234 // IPL ROM mapping: initially enabled; writing 1 to bit7 disables IPL ROM.
235 rom_readable_ = (val & 0x80) == 0;
236 if (old_rom_readable != rom_readable_) {
237 LOG_INFO("APU", "Control register $F1 = $%02X - IPL ROM %s at PC=$%04X",
238 val, rom_readable_ ? "ENABLED" : "DISABLED", spc700_.PC);
239 // When IPL ROM is disabled, reset transfer tracking
240 if (!rom_readable_) {
241 in_transfer_ = false;
242 transfer_size_ = 0;
243 }
244 }
245 break;
246 }
247 case 0xf2: {
248 dsp_adr_ = val;
249 break;
250 }
251 case 0xf3: {
252 if (dsp_adr_ < 0x80) dsp_.Write(dsp_adr_, val);
253 break;
254 }
255 case 0xf4:
256 case 0xf5:
257 case 0xf6:
258 case 0xf7: {
259 out_ports_[adr - 0xf4] = val;
260 port_write_count++;
261 if (port_write_count < 10) { // Reduced to prevent logging overflow crash
262 LOG_INFO("APU", "SPC wrote port $%04X (F%d) = $%02X at PC=$%04X [APU_cycles=%llu]",
263 adr, adr - 0xf4 + 4, val, spc700_.PC, cycles_);
264 }
265
266 // Track when SPC enters transfer loop (echoes counter back)
267 // PC is at $FFE2 when the MOVSY write completes (CB F4 is 2 bytes at $FFE0)
268 if (adr == 0xf4 && spc700_.PC == 0xFFE2 && rom_readable_) {
269 // SPC is echoing counter back - we're in data transfer phase
270 if (!in_transfer_ && ram[0x00] != 0 && ram[0x01] == 0) {
271 // Small destination address (< $0100) suggests small transfer
272 // ALTTP uses $0019 for bootstrap
273 if (ram[0x00] < 0x80) {
274 transfer_size_ = 1; // Assume 1-byte bootstrap transfer
275 in_transfer_ = true;
276 LOG_INFO("APU", "Detected small transfer start: dest=$%02X%02X, size=%d",
277 ram[0x01], ram[0x00], transfer_size_);
278 }
279 }
280 }
281 break;
282 }
283 case 0xf8:
284 case 0xf9: {
285 // General RAM
286 break;
287 }
288 case 0xfa:
289 case 0xfb:
290 case 0xfc: {
291 timer_[adr - 0xfa].target = val;
292 break;
293 }
294 }
295 ram[adr] = val;
296}
