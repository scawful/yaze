198 int GetTileIdFromMousePos() {
199 float x = mouse_pos_in_canvas_.x;
200 float y = mouse_pos_in_canvas_.y;
201 int num_columns = (canvas_sz_.x / global_scale_) / custom_step_;
202 int num_rows = (canvas_sz_.y / global_scale_) / custom_step_;
203 int tile_id = (x / custom_step_) + (y / custom_step_) * num_columns;
204 tile_id = tile_id / global_scale_;
205 if (tile_id >= num_columns * num_rows) {
206 tile_id = -1; // Invalid tile ID
207 }
