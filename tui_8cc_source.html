25bool HandleInput(ftxui::ScreenInteractive &screen, ftxui::Event &event,
26 int &selected) {
27 if (event == Event::ArrowDown || event == Event::Character('j')) {
28 selected++;
29 return true;
30 }
31 if (event == Event::ArrowUp || event == Event::Character('k')) {
32 if (selected != 0) selected--;
33 return true;
34 }
35 if (event == Event::Character('q')) {
36 SwitchComponents(screen, LayoutID::kExit);
37 return true;
38 }
39 return false;
40}
