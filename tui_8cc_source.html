30bool HandleInput(ftxui::ScreenInteractive &screen, ftxui::Event &event,
31 int &selected) {
32 if (event == Event::ArrowDown || event == Event::Character('j')) {
33 selected++;
34 return true;
35 }
36 if (event == Event::ArrowUp || event == Event::Character('k')) {
37 if (selected != 0) selected--;
38 return true;
39 }
40 if (event == Event::Character('q')) {
41 SwitchComponents(screen, LayoutID::kExit);
42 return true;
43 }
44 return false;
45}

853 // Tracks which menu item is selected.
854 static int selected = 0;
855 MenuOption option;
856 option.focused_entry = &selected;
857 auto menu = Menu(&kMainMenuEntries, &selected, option);

875 return vbox({
876 separator(),
877 title | center,
878 separator(),
879 menu->Render() | center,

904 return true;
