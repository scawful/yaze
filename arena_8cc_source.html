130void Arena::UpdateTexture(SDL_Texture* texture, SDL_Surface* surface) {
131 if (!texture || !surface) {

141 // Convert surface to RGBA8888 format for texture compatibility

143 std::unique_ptr<SDL_Surface, core::SDL_Surface_Deleter>(
144 SDL_ConvertSurfaceFormat(surface, SDL_PIXELFORMAT_RGBA8888, 0),
145 core::SDL_Surface_Deleter());
146
147 if (!converted_surface) {
148 SDL_Log("SDL_ConvertSurfaceFormat failed: %s", SDL_GetError());

150 }
151
152 // Lock texture for direct pixel access
153 void* pixels;

156 SDL_Log("SDL_LockTexture failed: %s", SDL_GetError());
157 return;
158 }
159
160 // Copy pixel data efficiently
161 memcpy(pixels, converted_surface->pixels,
162 converted_surface->h * converted_surface->pitch);
163
164 SDL_UnlockTexture(texture);

268void Arena::UpdateTextureRegion(SDL_Texture* texture, SDL_Surface* surface, SDL_Rect* rect) {
269 if (!texture || !surface) {
270 SDL_Log("Invalid texture or surface passed to UpdateTextureRegion");
271 return;
272 }
273
274 if (surface->pixels == nullptr) {
275 SDL_Log("Surface pixels are nullptr");
276 return;
277 }
278
279 // Convert surface to RGBA8888 format for texture compatibility
280 auto converted_surface =
281 std::unique_ptr<SDL_Surface, core::SDL_Surface_Deleter>(
282 SDL_ConvertSurfaceFormat(surface, SDL_PIXELFORMAT_RGBA8888, 0),
283 core::SDL_Surface_Deleter());
284
285 if (!converted_surface) {
286 SDL_Log("SDL_ConvertSurfaceFormat failed: %s", SDL_GetError());
287 return;
288 }
289
290 // Lock texture for direct pixel access
291 void* pixels;
292 int pitch;
293 if (SDL_LockTexture(texture, rect, &pixels, &pitch) != 0) {
294 SDL_Log("SDL_LockTexture failed: %s", SDL_GetError());
295 return;
296 }
297
298 // Copy pixel data efficiently
299 if (rect) {
300 // Copy only the specified region
301 int src_offset = rect->y * converted_surface->pitch + rect->x * 4; // 4 bytes per RGBA pixel
302 int dst_offset = 0;
303 for (int y = 0; y < rect->h; y++) {
304 memcpy(static_cast<char*>(pixels) + dst_offset,
305 static_cast<char*>(converted_surface->pixels) + src_offset,
306 rect->w * 4);
307 src_offset += converted_surface->pitch;
308 dst_offset += pitch;
309 }
310 } else {
311 // Copy entire surface
312 memcpy(pixels, converted_surface->pixels,
313 converted_surface->h * converted_surface->pitch);
314 }
315
316 SDL_UnlockTexture(texture);
317}
