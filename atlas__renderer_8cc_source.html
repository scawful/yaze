26int AtlasRenderer::AddBitmap(const Bitmap& bitmap) {
27 if (!bitmap.is_active() || !bitmap.texture()) {
28 return -1; // Invalid bitmap
29 }
30
31 ScopedTimer timer("atlas_add_bitmap");
32
33 // Try to pack into current atlas
34 SDL_Rect uv_rect;
35 if (PackBitmap(*atlases_[current_atlas_], bitmap, uv_rect)) {
36 int atlas_id = next_atlas_id_++;
37 auto& atlas = *atlases_[current_atlas_];
38
39 // Create atlas entry
40 atlas.entries.emplace_back(atlas_id, uv_rect, bitmap.texture());
41 atlas_lookup_[atlas_id] = &atlas.entries.back();

43 // Copy bitmap data to atlas texture
44 SDL_SetRenderTarget(renderer_, atlas.texture);
45 SDL_RenderCopy(renderer_, bitmap.texture(), nullptr, &uv_rect);
46 SDL_SetRenderTarget(renderer_, nullptr);
47
48 return atlas_id;
49 }
50
51 // Current atlas is full, create new one
52 CreateNewAtlas();
53 if (PackBitmap(*atlases_[current_atlas_], bitmap, uv_rect)) {
54 int atlas_id = next_atlas_id_++;
55 auto& atlas = *atlases_[current_atlas_];
56

110void AtlasRenderer::RenderBatch(const std::vector<RenderCommand>& render_commands) {
111 if (render_commands.empty()) {
112 return;
113 }
114
115 ScopedTimer timer("atlas_batch_render");
116
117 // Group commands by atlas for efficient rendering
118 std::unordered_map<int, std::vector<const RenderCommand*>> atlas_groups;
119
120 for (const auto& cmd : render_commands) {
121 auto it = atlas_lookup_.find(cmd.atlas_id);
122 if (it != atlas_lookup_.end() && it->second->in_use) {
123 // Find which atlas contains this entry
124 for (size_t i = 0; i < atlases_.size(); ++i) {
125 for (const auto& entry : atlases_[i]->entries) {
126 if (entry.atlas_id == cmd.atlas_id) {
127 atlas_groups[i].push_back(&cmd);
128 break;
129 }
130 }
131 }
132 }
133 }
134
135 // Render each atlas group
136 for (const auto& [atlas_index, commands] : atlas_groups) {
137 if (commands.empty()) continue;
138
139 auto& atlas = *atlases_[atlas_index];
140
141 // Set atlas texture
142 SDL_SetTextureBlendMode(atlas.texture, SDL_BLENDMODE_BLEND);
143
144 // Render all commands for this atlas
145 for (const auto* cmd : commands) {
146 auto it = atlas_lookup_.find(cmd->atlas_id);
147 if (it == atlas_lookup_.end()) continue;
148
149 AtlasEntry* entry = it->second;
150
151 // Calculate destination rectangle
152 SDL_Rect dest_rect = {
153 static_cast<int>(cmd->x),
154 static_cast<int>(cmd->y),
155 static_cast<int>(entry->uv_rect.w * cmd->scale_x),
156 static_cast<int>(entry->uv_rect.h * cmd->scale_y)
157 };
158
159 // Apply rotation if needed
160 if (std::abs(cmd->rotation) > 0.001F) {
161 // For rotation, we'd need to use SDL_RenderCopyEx
162 // This is a simplified version
163 SDL_RenderCopy(renderer_, atlas.texture, &entry->uv_rect, &dest_rect);
164 } else {
165 SDL_RenderCopy(renderer_, atlas.texture, &entry->uv_rect, &dest_rect);
166 }
167 }
168 }
169}

171AtlasStats AtlasRenderer::GetStats() const {
172 AtlasStats stats;
173
174 stats.total_atlases = atlases_.size();
175
176 for (const auto& atlas : atlases_) {
177 stats.total_entries += atlas->entries.size();
178 stats.used_entries += std::count_if(atlas->entries.begin(), atlas->entries.end(),
179 [](const AtlasEntry& entry) { return entry.in_use; });
180
181 // Calculate memory usage (simplified)
182 stats.total_memory += atlas->size * atlas->size * 4; // RGBA8888
183 }
184
185 if (stats.total_entries > 0) {
186 stats.utilization_percent = (static_cast<float>(stats.used_entries) / stats.total_entries) * 100.0F;
187 }
188
189 return stats;
190}

306void AtlasRenderer::RebuildAtlas(Atlas& atlas) {
307 // Clear used regions
308 std::fill(atlas.used_regions.begin(), atlas.used_regions.end(), false);
309
310 // Rebuild atlas texture by copying from source textures
311 SDL_SetRenderTarget(renderer_, atlas.texture);
312 SDL_SetRenderDrawColor(renderer_, 0, 0, 0, 0);
313 SDL_RenderClear(renderer_);
314
315 for (auto& entry : atlas.entries) {
316 if (entry.in_use && entry.texture) {
317 SDL_RenderCopy(renderer_, entry.texture, nullptr, &entry.uv_rect);
318 MarkRegionUsed(atlas, entry.uv_rect, true);
319 }
320 }
321
322 SDL_SetRenderTarget(renderer_, nullptr);
323}

325SDL_Rect AtlasRenderer::FindFreeRegion(Atlas& atlas, int width, int height) {
326 // Simple first-fit algorithm
327 for (int y = 0; y <= atlas.size - height; ++y) {
328 for (int x = 0; x <= atlas.size - width; ++x) {
329 bool can_fit = true;
330
331 // Check if region is free
332 for (int dy = 0; dy < height && can_fit; ++dy) {
333 for (int dx = 0; dx < width && can_fit; ++dx) {
334 int index = (y + dy) * atlas.size + (x + dx);
335 if (index >= static_cast<int>(atlas.used_regions.size()) || atlas.used_regions[index]) {
336 can_fit = false;
337 }
338 }
339 }
340
341 if (can_fit) {
342 return {x, y, width, height};
343 }
344 }
345 }
346
347 return {0, 0, 0, 0}; // No space found
348}

350void AtlasRenderer::MarkRegionUsed(Atlas& atlas, const SDL_Rect& rect, bool used) {
351 for (int y = rect.y; y < rect.y + rect.h; ++y) {
352 for (int x = rect.x; x < rect.x + rect.w; ++x) {
353 int index = y * atlas.size + x;
354 if (index >= 0 && index < static_cast<int>(atlas.used_regions.size())) {
355 atlas.used_regions[index] = used;
356 }
357 }
358 }
