386void OverworldEditor::DrawOverworldEdits() {
387 // Determine which overworld map the user is currently editing.
388 auto mouse_position = ow_map_canvas_.drawn_tile_position();
389 int map_x = mouse_position.x / kOverworldMapSize;
390 int map_y = mouse_position.y / kOverworldMapSize;
391 current_map_ = map_x + map_y * 8;
392 if (current_world_ == 1) {
393 current_map_ += 0x40;
394 } else if (current_world_ == 2) {
395 current_map_ += 0x80;
396 }
397
398 // Render the updated map bitmap.
399 RenderUpdatedMapBitmap(mouse_position,
400 tile16_individual_data_[current_tile16_]);
401
402 // Calculate the correct superX and superY values
403 int superY = current_map_ / 8;
404 int superX = current_map_ % 8;
405 int mouse_x = mouse_position.x;
406 int mouse_y = mouse_position.y;
407 // Calculate the correct tile16_x and tile16_y positions
408 int tile16_x = (mouse_x % kOverworldMapSize) / (kOverworldMapSize / 32);
409 int tile16_y = (mouse_y % kOverworldMapSize) / (kOverworldMapSize / 32);
410
411 // Update the overworld_.map_tiles() based on tile16 ID and current world
412 auto &selected_world =
413 (current_world_ == 0) ? overworld_.mutable_map_tiles()->light_world
414 : (current_world_ == 1) ? overworld_.mutable_map_tiles()->dark_world
415 : overworld_.mutable_map_tiles()->special_world;
416
417 int index_x = superX * 32 + tile16_x;
418 int index_y = superY * 32 + tile16_y;
419
420 selected_world[index_x][index_y] = current_tile16_;
421}

423void OverworldEditor::RenderUpdatedMapBitmap(
424 const ImVec2 &click_position, const std::vector<uint8_t> &tile_data) {
425 // Calculate the tile position relative to the current active map
426 constexpr int tile_size = 16; // Tile size is 16x16 pixels
427
428 // Calculate the tile index for x and y based on the click_position
429 int tile_index_x =
430 (static_cast<int>(click_position.x) % kOverworldMapSize) / tile_size;
431 int tile_index_y =
432 (static_cast<int>(click_position.y) % kOverworldMapSize) / tile_size;
433
434 // Calculate the pixel start position based on tile index and tile size
435 ImVec2 start_position;
436 start_position.x = tile_index_x * tile_size;
437 start_position.y = tile_index_y * tile_size;
438
439 // Get the current map's bitmap from the BitmapTable
440 gfx::Bitmap &current_bitmap = maps_bmp_[current_map_];
441
442 // Update the bitmap's pixel data based on the start_position and tile_data
443 for (int y = 0; y < tile_size; ++y) {
444 for (int x = 0; x < tile_size; ++x) {
445 int pixel_index =
446 (start_position.y + y) * kOverworldMapSize + (start_position.x + x);
447 current_bitmap.WriteToPixel(pixel_index, tile_data[y * tile_size + x]);
448 }
449 }
450
451 current_bitmap.set_modified(true);
452}

454void OverworldEditor::CheckForOverworldEdits() {
455 if (current_mode == EditingMode::DRAW_TILE) {
456 CheckForSelectRectangle();
457
458 // User has selected a tile they want to draw from the blockset.
459 if (!blockset_canvas_.points().empty() &&
460 !ow_map_canvas_.select_rect_active()) {
461 // Left click is pressed
462 if (ow_map_canvas_.DrawTilePainter(tile16_individual_[current_tile16_],
463 16)) {
464 DrawOverworldEdits();
465 }
466 }
467
468 if (ow_map_canvas_.select_rect_active()) {
469 if (ImGui::IsMouseClicked(ImGuiMouseButton_Left) ||
470 ImGui::IsMouseDragging(ImGuiMouseButton_Left)) {
471 auto &selected_world =
472 (current_world_ == 0) ? overworld_.mutable_map_tiles()->light_world
473 : (current_world_ == 1)
474 ? overworld_.mutable_map_tiles()->dark_world
475 : overworld_.mutable_map_tiles()->special_world;
476 // new_start_pos and new_end_pos
477 auto start = ow_map_canvas_.selected_points()[0];
478 auto end = ow_map_canvas_.selected_points()[1];
479
480 // Calculate the bounds of the rectangle in terms of 16x16 tile indices
481 constexpr int tile16_size = 16;
482 int start_x = std::floor(start.x / tile16_size) * tile16_size;
483 int start_y = std::floor(start.y / tile16_size) * tile16_size;
484 int end_x = std::floor(end.x / tile16_size) * tile16_size;
485 int end_y = std::floor(end.y / tile16_size) * tile16_size;
486
487 if (start_x > end_x) std::swap(start_x, end_x);
488 if (start_y > end_y) std::swap(start_y, end_y);
489
490 constexpr int local_map_size = 512; // Size of each local map
491 // Number of tiles per local map (since each tile is 16x16)
492 constexpr int tiles_per_local_map = local_map_size / 16;
493
494 for (int y = start_y, i = 0; y <= end_y; y += tile16_size) {
495 for (int x = start_x; x <= end_x; x += tile16_size, ++i) {
496 // Determine which local map (512x512) the tile is in
497 int local_map_x = x / local_map_size;
498 int local_map_y = y / local_map_size;
499
500 // Calculate the tile's position within its local map
501 int tile16_x = (x % local_map_size) / tile16_size;
502 int tile16_y = (y % local_map_size) / tile16_size;
503
504 // Calculate the index within the overall map structure
505 int index_x = local_map_x * tiles_per_local_map + tile16_x;
506 int index_y = local_map_y * tiles_per_local_map + tile16_y;
507 int tile16_id = overworld_.GetTileFromPosition(
508 ow_map_canvas_.selected_tiles()[i]);
509 selected_world[index_x][index_y] = tile16_id;
510 }
511 }
512
513 RefreshOverworldMap();
514 }
515 }
516 }

547absl::Status OverworldEditor::CheckForCurrentMap() {
548 // 4096x4096, 512x512 maps and some are larges maps 1024x1024
549 auto mouse_position = ImGui::GetIO().MousePos;
550 const auto large_map_size = 1024;
551 const auto canvas_zero_point = ow_map_canvas_.zero_point();
552
553 // Calculate which small map the mouse is currently over
554 int map_x = (mouse_position.x - canvas_zero_point.x) / kOverworldMapSize;
555 int map_y = (mouse_position.y - canvas_zero_point.y) / kOverworldMapSize;
556
557 // Calculate the index of the map in the `maps_bmp_` vector
558 current_map_ = map_x + map_y * 8;
559 int current_highlighted_map = current_map_;
560 if (current_world_ == 1) {
561 current_map_ += 0x40;
562 } else if (current_world_ == 2) {
563 current_map_ += 0x80;
564 }
565
566 current_parent_ = overworld_.overworld_map(current_map_)->parent();
567
568 auto current_map_x = current_highlighted_map % 8;
569 auto current_map_y = current_highlighted_map / 8;
570
571 if (overworld_.overworld_map(current_map_)->is_large_map() ||
572 overworld_.overworld_map(current_map_)->large_index() != 0) {
573 auto highlight_parent =
574 overworld_.overworld_map(current_highlighted_map)->parent();
575 auto parent_map_x = highlight_parent % 8;
576 auto parent_map_y = highlight_parent / 8;
577 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
578 parent_map_y * kOverworldMapSize, large_map_size,
579 large_map_size);
580 } else {
581 ow_map_canvas_.DrawOutline(current_map_x * kOverworldMapSize,
582 current_map_y * kOverworldMapSize,
583 kOverworldMapSize, kOverworldMapSize);
584 }
585
586 if (maps_bmp_[current_map_].modified() ||
587 ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
588 RefreshOverworldMap();
589 RETURN_IF_ERROR(RefreshTile16Blockset());
590 Renderer::GetInstance().UpdateBitmap(&maps_bmp_[current_map_]);
591 maps_bmp_[current_map_].set_modified(false);
592 }
593
594 return absl::OkStatus();
595}

992absl::Status OverworldEditor::LoadGraphics() {
993 // Load the Link to the Past overworld.
994 RETURN_IF_ERROR(overworld_.Load(*rom()))
995 palette_ = overworld_.AreaPalette();
996
997 // Create the area graphics image
998 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
999 0x80, kOverworldMapSize, 0x40, overworld_.current_graphics(),
1000 current_gfx_bmp_, palette_));
1001
1002 // Create the tile16 blockset image
1003 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1004 0x80, 0x2000, 0x08, overworld_.Tile16Blockset(), tile16_blockset_bmp_,
1005 palette_));
1006 map_blockset_loaded_ = true;
1007
1008 // Copy the tile16 data into individual tiles.
1009 auto tile16_data = overworld_.Tile16Blockset();
1010
1011 // Loop through the tiles and copy their pixel data into separate vectors
1012 for (int i = 0; i < 4096; i++) {
1013 // Create a new vector for the pixel data of the current tile
1014 std::vector<uint8_t> tile_data(16 * 16,
1015 0x00); // More efficient initialization
1016
1017 // Copy the pixel data for the current tile into the vector
1018 for (int ty = 0; ty < 16; ty++) {
1019 for (int tx = 0; tx < 16; tx++) {
1020 int position = tx + (ty * 0x10);
1021 uint8_t value =
1022 tile16_data[(i % 8 * 16) + (i / 8 * 16 * 0x80) + (ty * 0x80) + tx];
1023 tile_data[position] = value;
1024 }
1025 }
1026
1027 // Add the vector for the current tile to the vector of tile pixel data
1028 tile16_individual_data_.push_back(tile_data);
1029 tile16_individual_.emplace_back();
1030 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1031 0x10, 0x10, 0x80, tile16_individual_data_[i], tile16_individual_[i],
1032 palette_));
1033 }
1034
1035 // Render the overworld maps loaded from the ROM.
1036 for (int i = 0; i < zelda3::overworld::kNumOverworldMaps; ++i) {
1037 overworld_.set_current_map(i);
1038 auto palette = overworld_.AreaPalette();
1039 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1040 kOverworldMapSize, kOverworldMapSize, 0x200, overworld_.BitmapData(),
1041 maps_bmp_[i], palette));
1042 }
1043
1044 if (flags()->overworld.kDrawOverworldSprites) {
1045 RETURN_IF_ERROR(LoadSpriteGraphics());
1046 }
1047
1048 return absl::OkStatus();
1049}

1051absl::Status OverworldEditor::LoadSpriteGraphics() {
1052 // Render the sprites for each Overworld map
1053 for (int i = 0; i < 3; i++)
1054 for (auto const &sprite : overworld_.Sprites(i)) {
1055 int width = sprite.Width();
1056 int height = sprite.Height();
1057 int depth = 0x40;
1058 auto spr_gfx = sprite.PreviewGraphics();
1059 sprite_previews_[sprite.id()].Create(width, height, depth, spr_gfx);
1060 RETURN_IF_ERROR(sprite_previews_[sprite.id()].ApplyPalette(palette_));
1061 Renderer::GetInstance().RenderBitmap(&(sprite_previews_[sprite.id()]));
1062 }
1063 return absl::OkStatus();
1064}
