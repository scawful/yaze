351void OverworldEditor::DrawOverworldEdits() {
352 // Determine which overworld map the user is currently editing.
353 auto mouse_position = ow_map_canvas_.drawn_tile_position();
354 int map_x = mouse_position.x / kOverworldMapSize;
355 int map_y = mouse_position.y / kOverworldMapSize;
356 current_map_ = map_x + map_y * 8;
357 if (current_world_ == 1) {
358 current_map_ += 0x40;
359 } else if (current_world_ == 2) {
360 current_map_ += 0x80;
361 }
362
363 // Render the updated map bitmap.
364 RenderUpdatedMapBitmap(mouse_position,
365 tile16_individual_data_[current_tile16_]);
366
367 // Calculate the correct superX and superY values
368 int superY = current_map_ / 8;
369 int superX = current_map_ % 8;
370 int mouse_x = mouse_position.x;
371 int mouse_y = mouse_position.y;
372 // Calculate the correct tile16_x and tile16_y positions
373 int tile16_x = (mouse_x % kOverworldMapSize) / (kOverworldMapSize / 32);
374 int tile16_y = (mouse_y % kOverworldMapSize) / (kOverworldMapSize / 32);
375
376 // Update the overworld_.map_tiles() based on tile16 ID and current world
377 auto &selected_world =
378 (current_world_ == 0) ? overworld_.mutable_map_tiles()->light_world
379 : (current_world_ == 1) ? overworld_.mutable_map_tiles()->dark_world
380 : overworld_.mutable_map_tiles()->special_world;
381
382 int index_x = superX * 32 + tile16_x;
383 int index_y = superY * 32 + tile16_y;
384
385 selected_world[index_x][index_y] = current_tile16_;
386}

388void OverworldEditor::RenderUpdatedMapBitmap(
389 const ImVec2 &click_position, const std::vector<uint8_t> &tile_data) {
390 // Calculate the tile index for x and y based on the click_position
391 int tile_index_x =
392 (static_cast<int>(click_position.x) % kOverworldMapSize) / kTile16Size;
393 int tile_index_y =
394 (static_cast<int>(click_position.y) % kOverworldMapSize) / kTile16Size;
395
396 // Calculate the pixel start position based on tile index and tile size
397 ImVec2 start_position;
398 start_position.x = tile_index_x * kTile16Size;
399 start_position.y = tile_index_y * kTile16Size;
400
401 // Update the bitmap's pixel data based on the start_position and tile_data
402 gfx::Bitmap &current_bitmap = maps_bmp_[current_map_];
403 for (int y = 0; y < kTile16Size; ++y) {
404 for (int x = 0; x < kTile16Size; ++x) {
405 int pixel_index =
406 (start_position.y + y) * kOverworldMapSize + (start_position.x + x);
407 current_bitmap.WriteToPixel(pixel_index, tile_data[y * kTile16Size + x]);
408 }
409 }
410
411 current_bitmap.set_modified(true);
412}

414void OverworldEditor::CheckForOverworldEdits() {
415 CheckForSelectRectangle();
416
417 // User has selected a tile they want to draw from the blockset.
418 if (!blockset_canvas_.points().empty() &&
419 !ow_map_canvas_.select_rect_active()) {
420 // Left click is pressed
421 if (ow_map_canvas_.DrawTilePainter(tile16_individual_[current_tile16_],
422 kTile16Size)) {
423 DrawOverworldEdits();
424 }
425 }
426
427 if (ow_map_canvas_.select_rect_active()) {
428 if (ImGui::IsMouseClicked(ImGuiMouseButton_Left) ||
429 ImGui::IsMouseDragging(ImGuiMouseButton_Left)) {
430 auto &selected_world =
431 (current_world_ == 0) ? overworld_.mutable_map_tiles()->light_world
432 : (current_world_ == 1)
433 ? overworld_.mutable_map_tiles()->dark_world
434 : overworld_.mutable_map_tiles()->special_world;
435 // new_start_pos and new_end_pos
436 auto start = ow_map_canvas_.selected_points()[0];
437 auto end = ow_map_canvas_.selected_points()[1];
438
439 // Calculate the bounds of the rectangle in terms of 16x16 tile indices
440 int start_x = std::floor(start.x / kTile16Size) * kTile16Size;
441 int start_y = std::floor(start.y / kTile16Size) * kTile16Size;
442 int end_x = std::floor(end.x / kTile16Size) * kTile16Size;
443 int end_y = std::floor(end.y / kTile16Size) * kTile16Size;
444
445 if (start_x > end_x) std::swap(start_x, end_x);
446 if (start_y > end_y) std::swap(start_y, end_y);
447
448 constexpr int local_map_size = 512; // Size of each local map
449 // Number of tiles per local map (since each tile is 16x16)
450 constexpr int tiles_per_local_map = local_map_size / kTile16Size;
451
452 for (int y = start_y, i = 0; y <= end_y; y += kTile16Size) {
453 for (int x = start_x; x <= end_x; x += kTile16Size, ++i) {
454 // Determine which local map (512x512) the tile is in
455 int local_map_x = x / local_map_size;
456 int local_map_y = y / local_map_size;
457
458 // Calculate the tile's position within its local map
459 int tile16_x = (x % local_map_size) / kTile16Size;
460 int tile16_y = (y % local_map_size) / kTile16Size;
461
462 // Calculate the index within the overall map structure
463 int index_x = local_map_x * tiles_per_local_map + tile16_x;
464 int index_y = local_map_y * tiles_per_local_map + tile16_y;
465 int tile16_id = overworld_.GetTileFromPosition(
466 ow_map_canvas_.selected_tiles()[i]);
467 selected_world[index_x][index_y] = tile16_id;
468 }
469 }
470
471 RefreshOverworldMap();
472 }
473 }

503absl::Status OverworldEditor::CheckForCurrentMap() {
504 // 4096x4096, 512x512 maps and some are larges maps 1024x1024
505 const auto mouse_position = ImGui::GetIO().MousePos;
506 const int large_map_size = 1024;
507 const auto canvas_zero_point = ow_map_canvas_.zero_point();
508
509 // Calculate which small map the mouse is currently over
510 int map_x = (mouse_position.x - canvas_zero_point.x) / kOverworldMapSize;
511 int map_y = (mouse_position.y - canvas_zero_point.y) / kOverworldMapSize;
512
513 // Calculate the index of the map in the `maps_bmp_` vector
514 current_map_ = map_x + map_y * 8;
515 const int current_highlighted_map = current_map_;
516 if (current_world_ == 1) {
517 current_map_ += 0x40;
518 } else if (current_world_ == 2) {
519 current_map_ += 0x80;
520 }
521
522 current_parent_ = overworld_.overworld_map(current_map_)->parent();
523
524 if (overworld_.overworld_map(current_map_)->is_large_map() ||
525 overworld_.overworld_map(current_map_)->large_index() != 0) {
526 const int highlight_parent =
527 overworld_.overworld_map(current_highlighted_map)->parent();
528 const int parent_map_x = highlight_parent % 8;
529 const int parent_map_y = highlight_parent / 8;
530 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
531 parent_map_y * kOverworldMapSize, large_map_size,
532 large_map_size);
533 } else {
534 const int current_map_x = current_highlighted_map % 8;
535 const int current_map_y = current_highlighted_map / 8;
536 ow_map_canvas_.DrawOutline(current_map_x * kOverworldMapSize,
537 current_map_y * kOverworldMapSize,
538 kOverworldMapSize, kOverworldMapSize);
539 }
540
541 if (maps_bmp_[current_map_].modified() ||
542 ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
543 RefreshOverworldMap();
544 RETURN_IF_ERROR(RefreshTile16Blockset());
545 Renderer::GetInstance().UpdateBitmap(&maps_bmp_[current_map_]);
546 maps_bmp_[current_map_].set_modified(false);
547 }
548
549 return absl::OkStatus();
550}

970absl::Status OverworldEditor::LoadGraphics() {
971 // Load the Link to the Past overworld.
972 RETURN_IF_ERROR(overworld_.Load(*rom()))
973 palette_ = overworld_.AreaPalette();
974
975 // Create the area graphics image
976 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
977 0x80, kOverworldMapSize, 0x40, overworld_.current_graphics(),
978 current_gfx_bmp_, palette_));
979
980 // Create the tile16 blockset image
981 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
982 0x80, 0x2000, 0x08, overworld_.tile16_blockset_data(),
983 tile16_blockset_bmp_, palette_));
984 map_blockset_loaded_ = true;
985
986 // Copy the tile16 data into individual tiles.
987 auto tile16_data = overworld_.tile16_blockset_data();
988 tile16_individual_.reserve(kNumTile16Individual);
989
990 // Loop through the tiles and copy their pixel data into separate vectors
991 for (uint i = 0; i < kNumTile16Individual; i++) {
992 std::vector<uint8_t> tile_data(kTile16Size * kTile16Size, 0x00);
993
994 // Copy the pixel data for the current tile into the vector
995 for (int ty = 0; ty < kTile16Size; ty++) {
996 for (int tx = 0; tx < kTile16Size; tx++) {
997 int position = tx + (ty * kTile16Size);
998 uint8_t value =
999 tile16_data[(i % 8 * kTile16Size) + (i / 8 * kTile16Size * 0x80) +
1000 (ty * 0x80) + tx];
1001 tile_data[position] = value;
1002 }
1003 }
1004
1005 // Add the vector for the current tile to the vector of tile pixel data
1006 tile16_individual_data_.push_back(tile_data);
1007 tile16_individual_.emplace_back();
1008 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1009 kTile16Size, kTile16Size, 0x80, tile16_individual_data_[i],
1010 tile16_individual_[i], palette_));
1011 }
1012
1013 // Render the overworld maps loaded from the ROM.
1014 for (int i = 0; i < zelda3::overworld::kNumOverworldMaps; ++i) {
1015 overworld_.set_current_map(i);
1016 auto palette = overworld_.AreaPalette();
1017 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1018 kOverworldMapSize, kOverworldMapSize, 0x200, overworld_.BitmapData(),
1019 maps_bmp_[i], palette));
1020 }
1021
1022 if (flags()->overworld.kDrawOverworldSprites) {
1023 RETURN_IF_ERROR(LoadSpriteGraphics());
1024 }
1025
1026 return absl::OkStatus();
1027}
