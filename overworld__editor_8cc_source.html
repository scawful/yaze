405void OverworldEditor::DrawOverworldEdits() {
406 // Determine which overworld map the user is currently editing.
407 auto mouse_position = ow_map_canvas_.drawn_tile_position();
408 int map_x = mouse_position.x / kOverworldMapSize;
409 int map_y = mouse_position.y / kOverworldMapSize;
410 current_map_ = map_x + map_y * 8;
411 if (current_world_ == 1) {
412 current_map_ += 0x40;
413 } else if (current_world_ == 2) {
414 current_map_ += 0x80;
415 }
416
417 // Render the updated map bitmap.
418 RenderUpdatedMapBitmap(mouse_position,
419 tile16_individual_data_[current_tile16_]);
420
421 // Calculate the correct superX and superY values
422 int superY = current_map_ / 8;
423 int superX = current_map_ % 8;
424 int mouse_x = mouse_position.x;
425 int mouse_y = mouse_position.y;
426 // Calculate the correct tile16_x and tile16_y positions
427 int tile16_x = (mouse_x % kOverworldMapSize) / (kOverworldMapSize / 32);
428 int tile16_y = (mouse_y % kOverworldMapSize) / (kOverworldMapSize / 32);
429
430 // Update the overworld_.map_tiles() based on tile16 ID and current world
431 auto &selected_world =
432 (current_world_ == 0) ? overworld_.mutable_map_tiles()->light_world
433 : (current_world_ == 1) ? overworld_.mutable_map_tiles()->dark_world
434 : overworld_.mutable_map_tiles()->special_world;
435
436 int index_x = superX * 32 + tile16_x;
437 int index_y = superY * 32 + tile16_y;
438
439 selected_world[index_x][index_y] = current_tile16_;
440}

442void OverworldEditor::RenderUpdatedMapBitmap(
443 const ImVec2 &click_position, const std::vector<uint8_t> &tile_data) {
444 // Calculate the tile index for x and y based on the click_position
445 int tile_index_x =
446 (static_cast<int>(click_position.x) % kOverworldMapSize) / kTile16Size;
447 int tile_index_y =
448 (static_cast<int>(click_position.y) % kOverworldMapSize) / kTile16Size;
449
450 // Calculate the pixel start position based on tile index and tile size
451 ImVec2 start_position;
452 start_position.x = tile_index_x * kTile16Size;
453 start_position.y = tile_index_y * kTile16Size;
454
455 // Update the bitmap's pixel data based on the start_position and tile_data
456 gfx::Bitmap &current_bitmap = maps_bmp_[current_map_];
457 for (int y = 0; y < kTile16Size; ++y) {
458 for (int x = 0; x < kTile16Size; ++x) {
459 int pixel_index =
460 (start_position.y + y) * kOverworldMapSize + (start_position.x + x);
461 current_bitmap.WriteToPixel(pixel_index, tile_data[y * kTile16Size + x]);
462 }
463 }
464
465 current_bitmap.set_modified(true);
466}

468void OverworldEditor::CheckForOverworldEdits() {
469 CheckForSelectRectangle();
470
471 // User has selected a tile they want to draw from the blockset.
472 if (!blockset_canvas_.points().empty() &&
473 !ow_map_canvas_.select_rect_active()) {
474 // Left click is pressed
475 if (ow_map_canvas_.DrawTilePainter(tile16_individual_[current_tile16_],
476 kTile16Size)) {
477 DrawOverworldEdits();
478 }
479 }
480
481 if (ow_map_canvas_.select_rect_active()) {
482 if (ImGui::IsMouseClicked(ImGuiMouseButton_Left) ||
483 ImGui::IsMouseDragging(ImGuiMouseButton_Left)) {
484 auto &selected_world =
485 (current_world_ == 0) ? overworld_.mutable_map_tiles()->light_world
486 : (current_world_ == 1)
487 ? overworld_.mutable_map_tiles()->dark_world
488 : overworld_.mutable_map_tiles()->special_world;
489 // new_start_pos and new_end_pos
490 auto start = ow_map_canvas_.selected_points()[0];
491 auto end = ow_map_canvas_.selected_points()[1];
492
493 // Calculate the bounds of the rectangle in terms of 16x16 tile indices
494 int start_x = std::floor(start.x / kTile16Size) * kTile16Size;
495 int start_y = std::floor(start.y / kTile16Size) * kTile16Size;
496 int end_x = std::floor(end.x / kTile16Size) * kTile16Size;
497 int end_y = std::floor(end.y / kTile16Size) * kTile16Size;
498
499 if (start_x > end_x) std::swap(start_x, end_x);
500 if (start_y > end_y) std::swap(start_y, end_y);
501
502 constexpr int local_map_size = 512; // Size of each local map
503 // Number of tiles per local map (since each tile is 16x16)
504 constexpr int tiles_per_local_map = local_map_size / kTile16Size;
505
506 for (int y = start_y, i = 0; y <= end_y; y += kTile16Size) {
507 for (int x = start_x; x <= end_x; x += kTile16Size, ++i) {
508 // Determine which local map (512x512) the tile is in
509 int local_map_x = x / local_map_size;
510 int local_map_y = y / local_map_size;
511
512 // Calculate the tile's position within its local map
513 int tile16_x = (x % local_map_size) / kTile16Size;
514 int tile16_y = (y % local_map_size) / kTile16Size;
515
516 // Calculate the index within the overall map structure
517 int index_x = local_map_x * tiles_per_local_map + tile16_x;
518 int index_y = local_map_y * tiles_per_local_map + tile16_y;
519 int tile16_id = overworld_.GetTileFromPosition(
520 ow_map_canvas_.selected_tiles()[i]);
521 selected_world[index_x][index_y] = tile16_id;
522 }
523 }
524
525 RefreshOverworldMap();
526 }
527 }

557absl::Status OverworldEditor::CheckForCurrentMap() {
558 // 4096x4096, 512x512 maps and some are larges maps 1024x1024
559 const auto mouse_position = ImGui::GetIO().MousePos;
560 const int large_map_size = 1024;
561 const auto canvas_zero_point = ow_map_canvas_.zero_point();
562
563 // Calculate which small map the mouse is currently over
564 int map_x = (mouse_position.x - canvas_zero_point.x) / kOverworldMapSize;
565 int map_y = (mouse_position.y - canvas_zero_point.y) / kOverworldMapSize;
566
567 // Calculate the index of the map in the `maps_bmp_` vector
568 current_map_ = map_x + map_y * 8;
569 const int current_highlighted_map = current_map_;
570 if (current_world_ == 1) {
571 current_map_ += 0x40;
572 } else if (current_world_ == 2) {
573 current_map_ += 0x80;
574 }
575
576 current_parent_ = overworld_.overworld_map(current_map_)->parent();
577
578 if (overworld_.overworld_map(current_map_)->is_large_map() ||
579 overworld_.overworld_map(current_map_)->large_index() != 0) {
580 const int highlight_parent =
581 overworld_.overworld_map(current_highlighted_map)->parent();
582 const int parent_map_x = highlight_parent % 8;
583 const int parent_map_y = highlight_parent / 8;
584 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
585 parent_map_y * kOverworldMapSize, large_map_size,
586 large_map_size);
587 } else {
588 const int current_map_x = current_highlighted_map % 8;
589 const int current_map_y = current_highlighted_map / 8;
590 ow_map_canvas_.DrawOutline(current_map_x * kOverworldMapSize,
591 current_map_y * kOverworldMapSize,
592 kOverworldMapSize, kOverworldMapSize);
593 }
594
595 if (maps_bmp_[current_map_].modified() ||
596 ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
597 RefreshOverworldMap();
598 RETURN_IF_ERROR(RefreshTile16Blockset());
599 Renderer::GetInstance().UpdateBitmap(&maps_bmp_[current_map_]);
600 maps_bmp_[current_map_].set_modified(false);
601 }
602
603 return absl::OkStatus();
604}

1032absl::Status OverworldEditor::LoadGraphics() {
1033 // Load the Link to the Past overworld.
1034 RETURN_IF_ERROR(overworld_.Load(rom_))
1035 palette_ = overworld_.current_area_palette();
1036
1037 // Create the area graphics image
1038 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1039 0x80, kOverworldMapSize, 0x40, overworld_.current_graphics(),
1040 current_gfx_bmp_, palette_));
1041
1042 // Create the tile16 blockset image
1043 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1044 0x80, 0x2000, 0x08, overworld_.tile16_blockset_data(),
1045 tile16_blockset_bmp_, palette_));
1046 map_blockset_loaded_ = true;
1047
1048 // Copy the tile16 data into individual tiles.
1049 auto tile16_data = overworld_.tile16_blockset_data();
1050 tile16_individual_.reserve(zelda3::kNumTile16Individual);
1051
1052 // Loop through the tiles and copy their pixel data into separate vectors
1053 for (uint i = 0; i < zelda3::kNumTile16Individual; i++) {
1054 std::vector<uint8_t> tile_data(kTile16Size * kTile16Size, 0x00);
1055
1056 // Copy the pixel data for the current tile into the vector
1057 for (int ty = 0; ty < kTile16Size; ty++) {
1058 for (int tx = 0; tx < kTile16Size; tx++) {
1059 int position = tx + (ty * kTile16Size);
1060 uint8_t value =
1061 tile16_data[(i % 8 * kTile16Size) + (i / 8 * kTile16Size * 0x80) +
1062 (ty * 0x80) + tx];
1063 tile_data[position] = value;
1064 }
1065 }
1066
1067 // Add the vector for the current tile to the vector of tile pixel data
1068 tile16_individual_data_.push_back(tile_data);
1069 tile16_individual_.emplace_back();
1070 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1071 kTile16Size, kTile16Size, 0x80, tile16_individual_data_[i],
1072 tile16_individual_[i], palette_));
1073 }
1074
1075 // Render the overworld maps loaded from the ROM.
1076 for (int i = 0; i < zelda3::kNumOverworldMaps; ++i) {
1077 overworld_.set_current_map(i);
1078 auto palette = overworld_.current_area_palette();
1079 RETURN_IF_ERROR(Renderer::GetInstance().CreateAndRenderBitmap(
1080 kOverworldMapSize, kOverworldMapSize, 0x200,
1081 overworld_.current_map_bitmap_data(), maps_bmp_[i], palette));
1082 }
1083
1084 if (flags()->overworld.kDrawOverworldSprites) {
1085 RETURN_IF_ERROR(LoadSpriteGraphics());
1086 }
1087
1088 return absl::OkStatus();
1089}
