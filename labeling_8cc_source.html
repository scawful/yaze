55bool ResourceLabelManager::LoadLabels(const std::string& filename) {
56 std::ifstream file(filename);
57 if (!file.is_open()) {

62 }

64 file.open(filename);
65 if (!file.is_open()) {
66 return false;
67 }

70
71 std::string line;
72 while (std::getline(file, line)) {
73 std::istringstream iss(line);
74 std::string type, key, value;
75 if (std::getline(iss, type, ',') && std::getline(iss, key, ',') &&
76 std::getline(iss, value)) {
77 labels_[type][key] = value;
78 }
79 }
80 labels_loaded_ = true;
81 return true;
82}
