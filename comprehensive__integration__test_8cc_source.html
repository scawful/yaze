86 void ApplyV3Patch(Rom& rom) {
87 // Set ASM version to v3
88 ASSERT_TRUE(rom.WriteByte(OverworldCustomASMHasBeenApplied, 0x03).ok());
89
90 // Enable v3 features
91 ASSERT_TRUE(rom.WriteByte(OverworldCustomAreaSpecificBGEnabled, 0x01).ok());
92 ASSERT_TRUE(rom.WriteByte(OverworldCustomSubscreenOverlayEnabled, 0x01).ok());
93 ASSERT_TRUE(rom.WriteByte(OverworldCustomAnimatedGFXEnabled, 0x01).ok());
94 ASSERT_TRUE(rom.WriteByte(OverworldCustomTileGFXGroupEnabled, 0x01).ok());
95 ASSERT_TRUE(rom.WriteByte(OverworldCustomMosaicEnabled, 0x01).ok());
96 ASSERT_TRUE(rom.WriteByte(OverworldCustomMainPaletteEnabled, 0x01).ok());
97
98 // Apply v3 settings to first 10 maps for testing
99 for (int i = 0; i < 10; i++) {
100 // Set area sizes (mix of different sizes)
101 AreaSizeEnum area_size = static_cast<AreaSizeEnum>(i % 4);
102 ASSERT_TRUE(rom.WriteByte(kOverworldScreenSize + i, static_cast<uint8_t>(area_size)).ok());
103
104 // Set main palettes
105 ASSERT_TRUE(rom.WriteByte(OverworldCustomMainPaletteArray + i, i % 8).ok());
106
107 // Set area-specific background colors
108 uint16_t bg_color = 0x0000 + (i * 0x1000);
109 ASSERT_TRUE(rom.WriteByte(OverworldCustomAreaSpecificBGPalette + (i * 2),
110 bg_color & 0xFF).ok());
111 ASSERT_TRUE(rom.WriteByte(OverworldCustomAreaSpecificBGPalette + (i * 2) + 1,
112 (bg_color >> 8) & 0xFF).ok());
113
114 // Set subscreen overlays
115 uint16_t overlay = 0x0090 + i;
116 ASSERT_TRUE(rom.WriteByte(OverworldCustomSubscreenOverlayArray + (i * 2),
117 overlay & 0xFF).ok());
118 ASSERT_TRUE(rom.WriteByte(OverworldCustomSubscreenOverlayArray + (i * 2) + 1,
119 (overlay >> 8) & 0xFF).ok());
120
121 // Set animated GFX
122 ASSERT_TRUE(rom.WriteByte(OverworldCustomAnimatedGFXArray + i, 0x50 + i).ok());
123
124 // Set custom tile GFX groups (8 bytes per map)
125 for (int j = 0; j < 8; j++) {
126 ASSERT_TRUE(rom.WriteByte(OverworldCustomTileGFXGroupArray + (i * 8) + j,
127 0x20 + j + i).ok());
128 }
129
130 // Set mosaic settings
131 ASSERT_TRUE(rom.WriteByte(OverworldCustomMosaicArray + i, i % 16).ok());
132
133 // Set expanded message IDs
134 uint16_t message_id = 0x1000 + i;
135 ASSERT_TRUE(rom.WriteByte(kOverworldMessagesExpanded + (i * 2), message_id & 0xFF).ok());
136 ASSERT_TRUE(rom.WriteByte(kOverworldMessagesExpanded + (i * 2) + 1,
137 (message_id >> 8) & 0xFF).ok());
138 }
139 }
