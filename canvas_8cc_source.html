331void Canvas::DrawTileOnBitmap(int tile_size, gfx::Bitmap *bitmap,
332 ImVec4 color) {
333 const ImVec2 position = drawn_tile_pos_;
334 int tile_index_x = static_cast<int>(position.x / global_scale_) / tile_size;
335 int tile_index_y = static_cast<int>(position.y / global_scale_) / tile_size;
336
337 ImVec2 start_position(tile_index_x * tile_size, tile_index_y * tile_size);
338
339 // Update the bitmap's pixel data based on the start_position and color
340 for (int y = 0; y < tile_size; ++y) {
341 for (int x = 0; x < tile_size; ++x) {
342 // Calculate the actual pixel index in the bitmap
343 int pixel_index =
344 (start_position.y + y) * bitmap->width() + (start_position.x + x);
345
346 // Write the color to the pixel
347 bitmap->WriteColor(pixel_index, color);
348 }
349 }

378void Canvas::DrawSelectRect(int current_map, int tile_size, float scale) {
379 const ImGuiIO &io = GetIO();
380 const ImVec2 origin(canvas_p0_.x + scrolling_.x, canvas_p0_.y + scrolling_.y);
381 const ImVec2 mouse_pos(io.MousePos.x - origin.x, io.MousePos.y - origin.y);
382 static ImVec2 drag_start_pos;
383 const float scaled_size = tile_size * scale;
384 static bool dragging = false;
385 constexpr int small_map_size = 0x200;
386 int superY = current_map / 8;
387 int superX = current_map % 8;
388
389 // Handle right click for single tile selection
390 if (IsMouseClicked(ImGuiMouseButton_Right)) {
391 ImVec2 painter_pos = AlignPosToGrid(mouse_pos, scaled_size);
392 int painter_x = painter_pos.x;
393 int painter_y = painter_pos.y;
394
395 auto tile16_x = (painter_x % small_map_size) / (small_map_size / 0x20);
396 auto tile16_y = (painter_y % small_map_size) / (small_map_size / 0x20);
397
398 int index_x = superX * 0x20 + tile16_x;
399 int index_y = superY * 0x20 + tile16_y;
400 selected_tile_pos_ = ImVec2(index_x, index_y);
401 selected_points_.clear();
402 select_rect_active_ = false;
403
404 // Start drag position for rectangle selection
405 drag_start_pos = AlignPosToGrid(mouse_pos, scaled_size);
406 }
407
408 // Calculate the rectangle's top-left and bottom-right corners
409 ImVec2 drag_end_pos = AlignPosToGrid(mouse_pos, scaled_size);
410 if (ImGui::IsMouseDragging(ImGuiMouseButton_Right)) {
411 auto start = ImVec2(canvas_p0_.x + drag_start_pos.x,
412 canvas_p0_.y + drag_start_pos.y);
413 auto end = ImVec2(canvas_p0_.x + drag_end_pos.x + tile_size,
414 canvas_p0_.y + drag_end_pos.y + tile_size);
415 draw_list_->AddRect(start, end, kWhiteColor);
416 dragging = true;
417 }
418
419 if (dragging) {
420 // Release dragging mode
421 if (!ImGui::IsMouseDown(ImGuiMouseButton_Right)) {
422 dragging = false;
423
424 // Calculate the bounds of the rectangle in terms of 16x16 tile indices
425 constexpr int tile16_size = 16;
426 int start_x = std::floor(drag_start_pos.x / scaled_size) * tile16_size;
427 int start_y = std::floor(drag_start_pos.y / scaled_size) * tile16_size;
428 int end_x = std::floor(drag_end_pos.x / scaled_size) * tile16_size;
429 int end_y = std::floor(drag_end_pos.y / scaled_size) * tile16_size;
430
431 // Swap the start and end positions if they are in the wrong order
432 if (start_x > end_x) std::swap(start_x, end_x);
433 if (start_y > end_y) std::swap(start_y, end_y);
434
435 selected_tiles_.clear();
436 // Number of tiles per local map (since each tile is 16x16)
437 constexpr int tiles_per_local_map = small_map_size / 16;
438
439 // Loop through the tiles in the rectangle and store their positions
440 for (int y = start_y; y <= end_y; y += tile16_size) {
441 for (int x = start_x; x <= end_x; x += tile16_size) {
442 // Determine which local map (512x512) the tile is in
443 int local_map_x = x / small_map_size;
444 int local_map_y = y / small_map_size;
445
446 // Calculate the tile's position within its local map
447 int tile16_x = (x % small_map_size) / tile16_size;
448 int tile16_y = (y % small_map_size) / tile16_size;
449
450 // Calculate the index within the overall map structure
451 int index_x = local_map_x * tiles_per_local_map + tile16_x;
452 int index_y = local_map_y * tiles_per_local_map + tile16_y;
453
454 selected_tiles_.push_back(ImVec2(index_x, index_y));
455 }
456 }
457 // Clear and add the calculated rectangle points
458 selected_points_.clear();
459 selected_points_.push_back(drag_start_pos);
460 selected_points_.push_back(drag_end_pos);
461 select_rect_active_ = true;
462 }
463 }

544void Canvas::DrawBitmapGroup(std::vector<int> &group,
545 std::array<gfx::Bitmap, 4096> &tile16_individual_,
546 int tile_size, float scale) {
547 if (selected_points_.size() != 2) {
548 // points_ should contain exactly two points
549 return;
550 }
551 if (group.empty()) {
552 // group should not be empty
553 return;
554 }
555
556 // Top-left and bottom-right corners of the rectangle
557 ImVec2 rect_top_left = selected_points_[0];
558 ImVec2 rect_bottom_right = selected_points_[1];
559
560 // Calculate the start and end tiles in the grid
561 int start_tile_x =
562 static_cast<int>(std::floor(rect_top_left.x / (tile_size * scale)));
563 int start_tile_y =
564 static_cast<int>(std::floor(rect_top_left.y / (tile_size * scale)));
565 int end_tile_x =
566 static_cast<int>(std::floor(rect_bottom_right.x / (tile_size * scale)));
567 int end_tile_y =
568 static_cast<int>(std::floor(rect_bottom_right.y / (tile_size * scale)));
569
570 if (start_tile_x > end_tile_x) std::swap(start_tile_x, end_tile_x);
571 if (start_tile_y > end_tile_y) std::swap(start_tile_y, end_tile_y);
572
573 // Calculate the size of the rectangle in 16x16 grid form
574 int rect_width = (end_tile_x - start_tile_x) * tile_size;
575 int rect_height = (end_tile_y - start_tile_y) * tile_size;
576
577 int tiles_per_row = rect_width / tile_size;
578 int tiles_per_col = rect_height / tile_size;
579
580 int i = 0;
581 for (int y = 0; y < tiles_per_col + 1; ++y) {
582 for (int x = 0; x < tiles_per_row + 1; ++x) {
583 int tile_id = group[i];
584
585 // Check if tile_id is within the range of tile16_individual_
586 if (tile_id >= 0 && tile_id < tile16_individual_.size()) {
587 // Calculate the position of the tile within the rectangle
588 int tile_pos_x = (x + start_tile_x) * tile_size * scale;
589 int tile_pos_y = (y + start_tile_y) * tile_size * scale;
590
591 // Draw the tile bitmap at the calculated position
592 DrawBitmap(tile16_individual_[tile_id], tile_pos_x, tile_pos_y, scale,
593 150.0f);
594 i++;

788 // Create 2 channels and draw a Blue shape THEN a Red shape.
789 // You can create any number of channels. Tables API use 1 channel per

791 draw_list->ChannelsSplit(2);
792 draw_list->ChannelsSetCurrent(1);
793 draw_list->AddRectFilled(ImVec2(p1.x, p1.y), ImVec2(p1.x + 50, p1.y + 50),
