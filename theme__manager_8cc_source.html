242absl::Status ThemeManager::LoadThemeFromFile(const std::string& filepath) {
243 // Try multiple possible paths where theme files might be located
244 std::vector<std::string> possible_paths = {
245 filepath, // Absolute path
246 "assets/themes/" + filepath, // Relative from build dir
247 "../assets/themes/" + filepath, // Relative from bin dir
248 core::GetResourcePath("assets/themes/" + filepath), // Platform-specific resource path
249 };
250
251 std::ifstream file;
252 std::string successful_path;
253
254 for (const auto& path : possible_paths) {
255 util::logf("Trying to open theme file: %s", path.c_str());
256 file.open(path);
257 if (file.is_open()) {
258 successful_path = path;
259 util::logf("✅ Successfully opened theme file: %s", path.c_str());
260 break;
261 } else {
262 util::logf("❌ Failed to open theme file: %s", path.c_str());
263 file.clear(); // Clear any error flags before trying next path
264 }
265 }
266
267 if (!file.is_open()) {
268 return absl::InvalidArgumentError(absl::StrFormat("Cannot open theme file: %s (tried %zu paths)",
269 filepath, possible_paths.size()));
270 }
271
272 std::string content((std::istreambuf_iterator<char>(file)),
273 std::istreambuf_iterator<char>());
274 file.close();
275
276 if (content.empty()) {
277 return absl::InvalidArgumentError(absl::StrFormat("Theme file is empty: %s", successful_path));
278 }
279
280 EnhancedTheme theme;
281 auto parse_status = ParseThemeFile(content, theme);
282 if (!parse_status.ok()) {
283 return absl::InvalidArgumentError(absl::StrFormat("Failed to parse theme file %s: %s",
284 successful_path, parse_status.message()));
285 }
286
287 if (theme.name.empty()) {
288 return absl::InvalidArgumentError(absl::StrFormat("Theme file missing name: %s", successful_path));
289 }
290
291 themes_[theme.name] = theme;
292 return absl::OkStatus();
293}

460absl::Status ThemeManager::ParseThemeFile(const std::string& content, EnhancedTheme& theme) {
461 std::istringstream stream(content);
462 std::string line;
463 std::string current_section = "";
464
465 while (std::getline(stream, line)) {
466 // Skip empty lines and comments
467 if (line.empty() || line[0] == '#') continue;
468
469 // Check for section headers [section_name]
470 if (line[0] == '[' && line.back() == ']') {
471 current_section = line.substr(1, line.length() - 2);
472 continue;
473 }
474
475 size_t eq_pos = line.find('=');
476 if (eq_pos == std::string::npos) continue;
477
478 std::string key = line.substr(0, eq_pos);
479 std::string value = line.substr(eq_pos + 1);
480
481 // Trim whitespace and comments
482 key.erase(0, key.find_first_not_of(" \t"));
483 key.erase(key.find_last_not_of(" \t") + 1);
484 value.erase(0, value.find_first_not_of(" \t"));
485
486 // Remove inline comments
487 size_t comment_pos = value.find('#');
488 if (comment_pos != std::string::npos) {
489 value = value.substr(0, comment_pos);
490 }
491 value.erase(value.find_last_not_of(" \t") + 1);
492
493 // Parse based on section
494 if (current_section == "colors") {
495 Color color = ParseColorFromString(value);
496
497 if (key == "primary") theme.primary = color;
498 else if (key == "secondary") theme.secondary = color;

566Color ThemeManager::ParseColorFromString(const std::string& color_str) const {
567 std::vector<std::string> components = absl::StrSplit(color_str, ',');
568 if (components.size() != 4) {
569 return RGBA(255, 255, 255, 255); // White fallback
570 }
571
572 try {
573 int r = std::stoi(components[0]);
574 int g = std::stoi(components[1]);
575 int b = std::stoi(components[2]);
576 int a = std::stoi(components[3]);
577 return RGBA(r, g, b, a);
578 } catch (...) {
579 return RGBA(255, 255, 255, 255); // White fallback
580 }

700 }
701
702 util::logf("✅ Successfully saved theme '%s' to file: %s", theme.name.c_str(), filepath.c_str());
703 return absl::OkStatus();
