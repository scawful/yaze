102 void Get8x8Tile(int tile_index, int x, int y, std::vector<uint8_t> &tile_data,
103 int &tile_data_offset) {
104 int tile_offset = tile_index * (width_ * height_);
105 int tile_x = (x * 8) % width_;
106 int tile_y = (y * 8) % height_;
107 for (int i = 0; i < 8; i++) {
108 int row_offset = tile_offset + ((tile_y + i) * width_);
109 for (int j = 0; j < 8; j++) {
110 int pixel_offset = row_offset + (tile_x + j);
111 int pixel_value = data_[pixel_offset];
112 tile_data[tile_data_offset] = pixel_value;
113 tile_data_offset++;
114 }
115 }

118 void Get16x16Tile(int tile_index, int x, int y,
119 std::vector<uint8_t> &tile_data, int &tile_data_offset) {
120 int tile_offset = tile_index * (width_ * height_);
121 int tile_x = x * 16;
122 int tile_y = y * 16;
123 for (int i = 0; i < 16; i++) {
124 int row_offset = tile_offset + ((i / 8) * (width_ * 8));
125 for (int j = 0; j < 16; j++) {
126 int pixel_offset =
127 row_offset + ((j / 8) * 8) + ((i % 8) * width_) + (j % 8);
128 int pixel_value = data_[pixel_offset];
129 tile_data[tile_data_offset] = pixel_value;
130 tile_data_offset++;

135 void Get16x16Tile(int tile_x, int tile_y, std::vector<uint8_t> &tile_data,
136 int &tile_data_offset) {
137 // Assuming 'width_' and 'height_' are the dimensions of the bitmap
138 // and 'data_' is the bitmap data.
139 for (int ty = 0; ty < 16; ty++) {
140 for (int tx = 0; tx < 16; tx++) {
141 // Calculate the pixel position in the bitmap
142 int pixel_x = tile_x + tx;
143 int pixel_y = tile_y + ty;
144 int pixel_offset = pixel_y * width_ + pixel_x;
145 int pixel_value = data_[pixel_offset];
146
147 // Store the pixel value in the tile data
148 tile_data[tile_data_offset++] = pixel_value;
149 }
150 }

153 void WriteColor(int position, const ImVec4 &color) {
154 // Convert ImVec4 (RGBA) to SDL_Color (RGBA)
155 SDL_Color sdl_color;
156 sdl_color.r = static_cast<Uint8>(color.x * 255);
157 sdl_color.g = static_cast<Uint8>(color.y * 255);
158 sdl_color.b = static_cast<Uint8>(color.z * 255);

161 // Map SDL_Color to the nearest color index in the surface's palette
162 Uint8 index =
163 SDL_MapRGB(surface_->format, sdl_color.r, sdl_color.g, sdl_color.b);
164
165 // Write the color index to the pixel data
166 pixel_data_[position] = index;
167 modified_ = true;
