116 int &tile_data_offset) {

118 int tile_x = (x * 8) % width_;

131 void Get16x16Tile(int tile_index, int x, int y,
132 std::vector<uint8_t> &tile_data, int &tile_data_offset) {
133 int tile_offset = tile_index * (width_ * height_);
134 int tile_x = x * 16;
135 int tile_y = y * 16;
136 for (int i = 0; i < 16; i++) {
137 int row_offset = tile_offset + ((i / 8) * (width_ * 8));
138 for (int j = 0; j < 16; j++) {

148 void Get16x16Tile(int tile_x, int tile_y, std::vector<uint8_t> &tile_data,
149 int &tile_data_offset) {
150 // Assuming 'width_' and 'height_' are the dimensions of the bitmap
151 // and 'data_' is the bitmap data.
152 for (int ty = 0; ty < 16; ty++) {
153 for (int tx = 0; tx < 16; tx++) {
154 // Calculate the pixel position in the bitmap
155 int pixel_x = tile_x + tx;
156 int pixel_y = tile_y + ty;
157 int pixel_offset = pixel_y * width_ + pixel_x;
158 int pixel_value = data_[pixel_offset];
159
160 // Store the pixel value in the tile data
161 tile_data[tile_data_offset++] = pixel_value;
162 }
163 }

166 void WriteColor(int position, const ImVec4 &color) {
167 // Convert ImVec4 (RGBA) to SDL_Color (RGBA)
168 SDL_Color sdl_color;
169 sdl_color.r = static_cast<Uint8>(color.x * 255);

174 // Map SDL_Color to the nearest color index in the surface's palette
175 Uint8 index =

177
178 // Write the color index to the pixel data
179 pixel_data_[position] = index;
180 modified_ = true;
