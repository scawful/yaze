127 void Get8x8Tile(int tile_index, int x, int y, std::vector<uint8_t> &tile_data,
128 int &tile_data_offset) {
129 int tile_offset = tile_index * (width_ * height_);
130 int tile_x = (x * 8) % width_;
131 int tile_y = (y * 8) % height_;
132 for (int i = 0; i < 8; i++) {
133 int row_offset = tile_offset + ((tile_y + i) * width_);
134 for (int j = 0; j < 8; j++) {
135 int pixel_offset = row_offset + (tile_x + j);
136 int pixel_value = data_[pixel_offset];
137 tile_data[tile_data_offset] = pixel_value;
138 tile_data_offset++;
139 }
140 }

143 void Get16x16Tile(int tile_index, int x, int y,
144 std::vector<uint8_t> &tile_data, int &tile_data_offset) {
145 int tile_offset = tile_index * (width_ * height_);
146 int tile_x = x * 16;
147 int tile_y = y * 16;
148 for (int i = 0; i < 16; i++) {
149 int row_offset = tile_offset + ((i / 8) * (width_ * 8));
150 for (int j = 0; j < 16; j++) {
151 int pixel_offset =
152 row_offset + ((j / 8) * 8) + ((i % 8) * width_) + (j % 8);
153 int pixel_value = data_[pixel_offset];
154 tile_data[tile_data_offset] = pixel_value;
155 tile_data_offset++;
156 }
157 }

160 void Get16x16Tile(int tile_x, int tile_y, std::vector<uint8_t> &tile_data,
161 int &tile_data_offset) {
162 // Assuming 'width_' and 'height_' are the dimensions of the bitmap
163 // and 'data_' is the bitmap data.
164 for (int ty = 0; ty < 16; ty++) {
165 for (int tx = 0; tx < 16; tx++) {
166 // Calculate the pixel position in the bitmap
167 int pixel_x = tile_x + tx;
168 int pixel_y = tile_y + ty;
169 int pixel_offset = pixel_y * width_ + pixel_x;
170 int pixel_value = data_[pixel_offset];
171
172 // Store the pixel value in the tile data
173 tile_data[tile_data_offset++] = pixel_value;
174 }
175 }

178 void WriteColor(int position, const ImVec4 &color) {
179 // Convert ImVec4 (RGBA) to SDL_Color (RGBA)
180 SDL_Color sdl_color;
181 sdl_color.r = static_cast<Uint8>(color.x * 255);
182 sdl_color.g = static_cast<Uint8>(color.y * 255);
183 sdl_color.b = static_cast<Uint8>(color.z * 255);
184 sdl_color.a = static_cast<Uint8>(color.w * 255);
185
186 // Map SDL_Color to the nearest color index in the surface's palette
187 Uint8 index =
188 SDL_MapRGB(surface_->format, sdl_color.r, sdl_color.g, sdl_color.b);
189
190 // Write the color index to the pixel data
191 pixel_data_[position] = index;
192 modified_ = true;
193 }
