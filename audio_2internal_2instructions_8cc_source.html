54void Spc700::ADC(uint16_t adr) {
55 uint8_t value = read(adr);
56 uint16_t result = A + value + PSW.C;
57 PSW.V = ((A ^ result) & (adr ^ result) & 0x80);
58 PSW.C = (result > 0xFF);
59 PSW.H = ((A ^ adr ^ result) & 0x10);
60 A = result & 0xFF;
61 PSW.Z = ((A & 0xFF) == 0);
62 PSW.N = (A & 0x80);
63}

65void Spc700::ADCM(uint16_t& dest, uint8_t operand) {
66 uint8_t applyOn = read(dest);
67 int result = applyOn + operand + PSW.C;
68 PSW.V = (applyOn & 0x80) == (operand & 0x80) &&
69 (operand & 0x80) != (result & 0x80);
70 PSW.H = ((applyOn & 0xf) + (operand & 0xf) + PSW.C) > 0xf;
71 PSW.C = result > 0xff;
72 write(dest, result);
73 PSW.Z = ((result & 0xFF) == 0);
74 PSW.N = (result & 0x80);
75}
