320void DungeonObjectSelector::DrawRoomGraphics() {
321 const auto height = 0x40;
322 room_gfx_canvas_.DrawBackground();
323 room_gfx_canvas_.DrawContextMenu();
324 room_gfx_canvas_.DrawTileSelector(32);
325
326 if (rom_ && rom_->is_loaded() && rooms_) {
327 int active_room_id = current_room_id_;
328 auto& room = (*rooms_)[active_room_id];
329 auto blocks = room.blocks();
330
331 // Load graphics for this room if not already loaded
332 if (blocks.empty()) {
333 room.LoadRoomGraphics(room.blockset);
334 blocks = room.blocks();
335 }
336
337 int current_block = 0;
338 const int max_blocks_per_row = 2; // 2 blocks per row for 300px column
339 const int block_width = 128; // Reduced size to fit column
340 const int block_height = 32; // Reduced height
341
342 for (int block : blocks) {
343 if (current_block >= 16) break; // Only show first 16 blocks
344
345 // Ensure the graphics sheet is loaded and has a valid texture
346 if (block < gfx::Arena::Get().gfx_sheets().size()) {
347 auto& gfx_sheet = gfx::Arena::Get().gfx_sheets()[block];
348
349 // Calculate position in a grid layout instead of horizontal concatenation
350 int row = current_block / max_blocks_per_row;
351 int col = current_block % max_blocks_per_row;
352
353 int x = room_gfx_canvas_.zero_point().x + 2 + (col * block_width);
354 int y = room_gfx_canvas_.zero_point().y + 2 + (row * block_height);
355
356 // Ensure we don't exceed canvas bounds
357 if (x + block_width <= room_gfx_canvas_.zero_point().x + room_gfx_canvas_.width() &&
358 y + block_height <= room_gfx_canvas_.zero_point().y + room_gfx_canvas_.height()) {
359
360 // Only draw if the texture is valid
361 if (gfx_sheet.texture() != 0) {
362 room_gfx_canvas_.draw_list()->AddImage(
363 (ImTextureID)(intptr_t)gfx_sheet.texture(),
364 ImVec2(x, y),
365 ImVec2(x + block_width, y + block_height));
366 }
367 }
368 }
369 current_block += 1;
370 }
371 }
372 room_gfx_canvas_.DrawGrid(32.0f);
373 room_gfx_canvas_.DrawOverlay();
374}

499ImU32 DungeonObjectSelector::GetObjectTypeColor(int object_id) {
500 // Color-code objects based on their type and function
501 if (object_id >= 0x10 && object_id <= 0x1F) {
502 return IM_COL32(128, 128, 128, 255); // Gray for walls
503 } else if (object_id >= 0x20 && object_id <= 0x2F) {
504 return IM_COL32(139, 69, 19, 255); // Brown for floors
505 } else if (object_id == 0xF9 || object_id == 0xFA) {
506 return IM_COL32(255, 215, 0, 255); // Gold for chests
507 } else if (object_id >= 0x17 && object_id <= 0x1E) {
508 return IM_COL32(139, 69, 19, 255); // Brown for doors
509 } else if (object_id == 0x2F || object_id == 0x2B) {
510 return IM_COL32(160, 82, 45, 255); // Saddle brown for pots
511 } else if (object_id >= 0x138 && object_id <= 0x13B) {
512 return IM_COL32(255, 255, 0, 255); // Yellow for stairs
513 } else if (object_id >= 0x30 && object_id <= 0x3F) {
514 return IM_COL32(105, 105, 105, 255); // Dim gray for decorations
515 } else {
516 return IM_COL32(96, 96, 96, 255); // Default gray
517 }

665bool DungeonObjectSelector::MatchesObjectFilter(int obj_id, int filter_type) {
666 switch (filter_type) {
667 case 1: // Walls
668 return obj_id >= 0x10 && obj_id <= 0x1F;
669 case 2: // Floors
670 return obj_id >= 0x20 && obj_id <= 0x2F;
671 case 3: // Chests
672 return obj_id == 0xF9 || obj_id == 0xFA;
673 case 4: // Doors
674 return obj_id >= 0x17 && obj_id <= 0x1E;
675 case 5: // Decorations
676 return obj_id >= 0x30 && obj_id <= 0x3F;
677 case 6: // Stairs
678 return obj_id >= 0x138 && obj_id <= 0x13B;
679 default: // All
680 return true;
681 }
682}

684void DungeonObjectSelector::CalculateObjectDimensions(const zelda3::RoomObject& object, int& width, int& height) {
685 // Default base size
686 width = 16;
687 height = 16;
688
689 // For walls, use the size field to determine length
690 if (object.id_ >= 0x10 && object.id_ <= 0x1F) {
691 // Wall objects: size determines length and orientation
692 uint8_t size_x = object.size_ & 0x0F;
693 uint8_t size_y = (object.size_ >> 4) & 0x0F;
694
695 // Walls can be horizontal or vertical based on size parameters
696 if (size_x > size_y) {
697 // Horizontal wall
698 width = 16 + size_x * 16; // Each unit adds 16 pixels
699 height = 16;
700 } else if (size_y > size_x) {
701 // Vertical wall
702 width = 16;
703 height = 16 + size_y * 16;
704 } else {
705 // Square wall or corner
706 width = 16 + size_x * 8;
707 height = 16 + size_y * 8;
708 }
709 } else {
710 // For other objects, use standard size calculation
711 width = 16 + (object.size_ & 0x0F) * 8;
712 height = 16 + ((object.size_ >> 4) & 0x0F) * 8;
713 }
714
715 // Clamp to reasonable limits
716 width = std::min(width, 256);
717 height = std::min(height, 256);
718}
