186void DungeonEditor::LoadDungeonRoomSize() {
187 std::map<int, std::vector<int>> rooms_by_bank;
188 for (const auto &room : room_size_addresses_) {
189 int bank = room.second >> 16;
190 rooms_by_bank[bank].push_back(room.second);
191 }
192
193 // Process and calculate room sizes within each bank
194 for (auto &bank_rooms : rooms_by_bank) {
195 // Sort the rooms within this bank
196 std::ranges::sort(bank_rooms.second);
197
198 for (size_t i = 0; i < bank_rooms.second.size(); ++i) {
199 int room_ptr = bank_rooms.second[i];
200
201 // Identify the room ID for the current room pointer
202 int room_id =
203 std::ranges::find_if(room_size_addresses_, [room_ptr](
204 const auto &entry) {
205 return entry.second == room_ptr;
206 })->first;
207
208 if (room_ptr != 0x0A8000) {
209 if (i < bank_rooms.second.size() - 1) {
210 // Calculate size as difference between current room and next room
211 // in the same bank
212 room_sizes_[room_id] = bank_rooms.second[i + 1] - room_ptr;
213 } else {
214 // Calculate size for the last room in this bank
215 int bank_end_address = (bank_rooms.first << 16) | 0xFFFF;
216 room_sizes_[room_id] = bank_end_address - room_ptr + 1;
217 }
218 total_room_size_ += room_sizes_[room_id];
219 } else {
220 // Room with address 0x0A8000
221 room_sizes_[room_id] = 0x00;
222 }
223 }
224 }

700void DungeonEditor::RenderObjectInCanvas(const zelda3::RoomObject &object,
701 const gfx::SnesPalette &palette) {
702 // Validate ROM is loaded
703 if (!rom_ || !rom_->is_loaded()) {
704 return;
705 }
706
707 // Create a mutable copy of the object to ensure tiles are loaded
708 auto mutable_object = object;
709 mutable_object.set_rom(rom_);
710 mutable_object.EnsureTilesLoaded();
711
712 // Check if tiles were loaded successfully
713 if (mutable_object.tiles().empty()) {
714 return; // Skip objects without tiles
715 }
716
717 // Calculate palette hash for caching
718 uint64_t palette_hash = 0;
719 for (size_t i = 0; i < palette.size() && i < 16; ++i) {
720 palette_hash ^= std::hash<uint16_t>{}(palette[i].snes()) + 0x9e3779b9 +
721 (palette_hash << 6) + (palette_hash >> 2);
722 }
723
724 // Convert room coordinates to canvas coordinates using helper function
725 auto [canvas_x, canvas_y] = RoomToCanvasCoordinates(object.x_, object.y_);
726
727 // Check if object is within canvas bounds (accounting for scrolling)
728 if (!IsWithinCanvasBounds(canvas_x, canvas_y, 32)) {
729 return; // Skip objects outside visible area
730 }
731
732 // Check cache first
733 for (auto &cached : object_render_cache_) {
734 if (cached.object_id == object.id_ && cached.object_x == object.x_ &&
735 cached.object_y == object.y_ && cached.object_size == object.size_ &&
736 cached.palette_hash == palette_hash && cached.is_valid) {
737 // Use cached bitmap - canvas handles scrolling internally
738 canvas_.DrawBitmap(cached.rendered_bitmap, canvas_x, canvas_y, 1.0f, 255);
739 return;
740 }
741 }
742
743 // Render the object to a bitmap
744 auto render_result = object_renderer_.RenderObject(mutable_object, palette);
745 if (!render_result.ok()) {
746 return; // Skip if rendering failed
747 }
748
749 auto object_bitmap = std::move(render_result.value());
750
751 // Set the palette for the bitmap
752 object_bitmap.SetPalette(palette);
753
754 // Render the bitmap to a texture so it can be drawn
755 core::Renderer::Get().RenderBitmap(&object_bitmap);
756
757 // Draw the object bitmap to the canvas
758 // Canvas will handle scrolling and coordinate transformation
759 canvas_.DrawBitmap(object_bitmap, canvas_x, canvas_y, 1.0f, 255);
760
761 // Cache the rendered bitmap (create a copy for caching)
762 ObjectRenderCache cache_entry;
763 cache_entry.object_id = object.id_;
764 cache_entry.object_x = object.x_;
765 cache_entry.object_y = object.y_;
766 cache_entry.object_size = object.size_;
767 cache_entry.palette_hash = palette_hash;
768 cache_entry.rendered_bitmap = object_bitmap; // Copy instead of move
769 cache_entry.is_valid = true;
770
771 // Add to cache (limit cache size)
772 if (object_render_cache_.size() >= 100) {
773 object_render_cache_.erase(object_render_cache_.begin());
774 }
775 object_render_cache_.push_back(std::move(cache_entry));
776}
