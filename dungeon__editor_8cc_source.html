214void DungeonEditor::OnRoomSelected(int room_id) {
215 // Update current room ID
216 current_room_id_ = room_id;
217
218 // Check if room is already open in a tab
219 int existing_tab_index = -1;
220 for (int i = 0; i < active_rooms_.Size; i++) {
221 if (active_rooms_[i] == room_id) {
222 existing_tab_index = i;
223 break;
224 }
225 }
226
227 if (existing_tab_index >= 0) {
228 // Room is already open, switch to that tab
229 current_active_room_tab_ = existing_tab_index;
230 } else {
231 // Room is not open, add it as a new tab
232 active_rooms_.push_back(room_id);
233 current_active_room_tab_ = active_rooms_.Size - 1;
234 }
235
236 // Update the room selector's active rooms list
237 room_selector_.set_active_rooms(active_rooms_);
238}

405void DungeonEditor::DrawDungeonTabView() {
406 static int next_tab_id = 0;
407
408 if (BeginTabBar("MyTabBar", kDungeonTabBarFlags)) {
409 if (ImGui::TabItemButton(ICON_MD_ADD, kDungeonTabFlags)) {
410 if (std::find(active_rooms_.begin(), active_rooms_.end(),
411 current_room_id_) != active_rooms_.end()) {
412 // Room is already open
413 next_tab_id++;
414 }
415 active_rooms_.push_back(next_tab_id++); // Add new tab
416 }
417
418 // Submit our regular tabs
419 for (int n = 0; n < active_rooms_.Size;) {
420 bool open = true;
421
422 if (active_rooms_[n] > sizeof(zelda3::kRoomNames) / 4) {
423 active_rooms_.erase(active_rooms_.Data + n);
424 continue;
425 }
426
427 if (BeginTabItem(zelda3::kRoomNames[active_rooms_[n]].data(), &open,
428 ImGuiTabItemFlags_None)) {
429 current_active_room_tab_ = n; // Track which tab is currently active
430 DrawDungeonCanvas(active_rooms_[n]);
431 EndTabItem();
432 }
433
434 if (!open)
435 active_rooms_.erase(active_rooms_.Data + n);
436 else
437 n++;
438 }
439
440 EndTabBar();
441 }
442 ImGui::Separator();
443}
