23 absl::StatusOr<std::shared_ptr<gfx::Bitmap>> GetGraphicsSheet(int sheet_index) {
24 std::lock_guard<std::mutex> lock(mutex_);
25
26 // Validate sheet index
27 if (sheet_index < 0 || sheet_index >= 223) {
28 return absl::InvalidArgumentError("Invalid graphics sheet index");
29 }
30
31 // Check cache first
32 auto it = cache_.find(sheet_index);
33 if (it != cache_.end() && it->second.is_loaded) {
34 it->second.last_accessed = std::chrono::steady_clock::now();
35 it->second.access_count++;
36 cache_hits_++;
37 return it->second.sheet;
38 }
39
40 // Load from Arena
41 auto& arena = gfx::Arena::Get();
42 auto sheet = arena.gfx_sheet(sheet_index);
43
44 if (!sheet.is_active()) {
45 cache_misses_++;
46 return absl::NotFoundError("Graphics sheet not available");
47 }
48
49 // Cache the sheet
50 GraphicsSheetInfo info;
51 info.sheet = std::make_shared<gfx::Bitmap>(sheet);
52 info.is_loaded = true;
53 info.last_accessed = std::chrono::steady_clock::now();
54 info.access_count = 1;
55
56 cache_[sheet_index] = info;
57 cache_misses_++;
58
59 // Evict if cache is full
60 if (cache_.size() > max_cache_size_) {
61 EvictLeastRecentlyUsed();
62 }
63
64 return info.sheet;
65 }

120 void EvictLeastRecentlyUsed() {
121 auto oldest = cache_.end();

123

125 if (it->second.last_accessed < oldest_time) {
126 oldest = it;

146 void* Allocate(size_t size) {
147 std::lock_guard<std::mutex> lock(mutex_);
148
149 // Align to 8-byte boundary for optimal performance
150 size = (size + 7) & ~7;
151
152 if (current_offset_ + size > pool_size_) {
153 // Allocate new pool
154 pools_.push_back(std::make_unique<uint8_t[]>(pool_size_));

300 absl::StatusOr<std::vector<gfx::Tile16>> ParseSubtype1(int16_t object_id) {
301 int index = object_id & 0xFF;
302 int tile_ptr = kRoomObjectSubtype1 + (index * 2);
303
304 // Enhanced bounds checking
305 if (!ValidateROMAddress(tile_ptr, 2)) {
306 return absl::OutOfRangeError("Tile pointer out of range");
307 }
308
309 // Read tile data pointer
310 uint8_t low = rom_->data()[tile_ptr];
311 uint8_t high = rom_->data()[tile_ptr + 1];
312 int tile_data_ptr = kRoomObjectTileAddress + ((high << 8) | low);
313
314 // Validate tile data address
315 if (!ValidateROMAddress(tile_data_ptr, 64)) {
316 return absl::OutOfRangeError("Tile data address out of range");
317 }
318
319 return ReadTileData(tile_data_ptr, 8); // 8 tiles for subtype 1
320 }
321
322 absl::StatusOr<std::vector<gfx::Tile16>> ParseSubtype2(int16_t object_id) {
323 int index = (object_id & 0xFF) - 0x100;
324 int tile_ptr = kRoomObjectSubtype2 + (index * 2);
325
326 if (!ValidateROMAddress(tile_ptr, 2)) {
327 return absl::OutOfRangeError("Tile pointer out of range");
328 }
329
330 uint8_t low = rom_->data()[tile_ptr];
331 uint8_t high = rom_->data()[tile_ptr + 1];
332 int tile_data_ptr = kRoomObjectTileAddress + ((high << 8) | low);
333
334 if (!ValidateROMAddress(tile_data_ptr, 128)) {
335 return absl::OutOfRangeError("Tile data address out of range");
336 }
337
338 return ReadTileData(tile_data_ptr, 16); // 16 tiles for subtype 2
339 }

341 absl::StatusOr<std::vector<gfx::Tile16>> ParseSubtype3(int16_t object_id) {
342 int index = (object_id & 0xFF) - 0x200;
343 int tile_ptr = kRoomObjectSubtype3 + (index * 2);
344
345 if (!ValidateROMAddress(tile_ptr, 2)) {
346 return absl::OutOfRangeError("Tile pointer out of range");
347 }
348

350 uint8_t high = rom_->data()[tile_ptr + 1];
351 int tile_data_ptr = kRoomObjectTileAddress + ((high << 8) | low);
352
353 if (!ValidateROMAddress(tile_data_ptr, 256)) {

355 }
356

360 absl::StatusOr<std::vector<gfx::Tile16>> ReadTileData(int address, int tile_count) {
361 std::vector<gfx::Tile16> tiles;
362 tiles.reserve(tile_count);
363
364 for (int i = 0; i < tile_count; i++) {
365 int tile_address = address + (i * 8);
366
367 if (!ValidateROMAddress(tile_address, 8)) {
368 // Create placeholder tile for invalid data
369 tiles.emplace_back(gfx::TileInfo{}, gfx::TileInfo{}, gfx::TileInfo{}, gfx::TileInfo{});
370 continue;
371 }
372
373 // Read 4 tile infos (8 bytes total)
374 uint16_t w0 = rom_->data()[tile_address] | (rom_->data()[tile_address + 1] << 8);
375 uint16_t w1 = rom_->data()[tile_address + 2] | (rom_->data()[tile_address + 3] << 8);
376 uint16_t w2 = rom_->data()[tile_address + 4] | (rom_->data()[tile_address + 5] << 8);
377 uint16_t w3 = rom_->data()[tile_address + 6] | (rom_->data()[tile_address + 7] << 8);
378
379 tiles.emplace_back(gfx::WordToTileInfo(w0), gfx::WordToTileInfo(w1),
380 gfx::WordToTileInfo(w2), gfx::WordToTileInfo(w3));
381 }
382
383 return tiles;
384 }

406absl::StatusOr<gfx::Bitmap> ObjectRenderer::RenderObject(const RoomObject& object, const gfx::SnesPalette& palette) {
407 auto start_time = std::chrono::high_resolution_clock::now();
408
409 // Validate inputs
410 auto status = ValidateInputs(object, palette);
411 if (!status.ok()) return status;
412
413 // Ensure object has tiles loaded
414 if (object.tiles().empty()) {
415 return absl::FailedPreconditionError("Object has no tiles loaded");
416 }
417
418 // Create bitmap
419 int bitmap_width = std::min(512, static_cast<int>(object.tiles().size()) * 16);
420 int bitmap_height = std::min(512, 32);
421
422 auto bitmap_result = CreateBitmap(bitmap_width, bitmap_height);
423 if (!bitmap_result.ok()) return bitmap_result;
424
425 auto bitmap = std::move(bitmap_result.value());
426
427 // Render tiles
428 for (size_t i = 0; i < object.tiles().size(); ++i) {
429 int tile_x = (i % 2) * 16;
430 int tile_y = (i / 2) * 16;
431
432 auto tile_status = RenderTileToBitmap(object.tiles()[i], bitmap, tile_x, tile_y, palette);
433 if (!tile_status.ok()) {
434 // Continue with other tiles
435 continue;
436 }
437 }
438
439 // Update performance stats
440 auto end_time = std::chrono::high_resolution_clock::now();
441 if (performance_monitoring_enabled_) {
442 performance_monitor_->RecordRenderTime(end_time - start_time);
443 performance_monitor_->IncrementObjectCount();
444 performance_monitor_->IncrementTileCount(object.tiles().size());
445 }
446
447 return bitmap;
448}

450absl::StatusOr<gfx::Bitmap> ObjectRenderer::RenderObjects(const std::vector<RoomObject>& objects, const gfx::SnesPalette& palette) {
451 if (objects.empty()) {
452 return absl::InvalidArgumentError("No objects to render");
453 }
454
455 // Validate inputs
456 auto status = ValidateInputs(objects, palette);
457 if (!status.ok()) return status;
458
459 // Calculate optimal bitmap size
460 auto [width, height] = CalculateOptimalBitmapSize(objects);
461
462 auto bitmap_result = CreateBitmap(width, height);
463 if (!bitmap_result.ok()) return bitmap_result;
464
465 auto bitmap = std::move(bitmap_result.value());
466
467 // Collect all tiles for batch rendering
468 std::vector<TileRenderInfo> tile_infos;
469 tile_infos.reserve(objects.size() * 8);
470
471 for (const auto& object : objects) {
472 if (object.tiles().empty()) continue;
473
474 int obj_x = object.x_ * 16;
475 int obj_y = object.y_ * 16;
476
477 for (size_t i = 0; i < object.tiles().size(); ++i) {
478 int tile_x = obj_x + (i % 2) * 16;
479 int tile_y = obj_y + (i / 2) * 16;
480
481 if (tile_x >= -16 && tile_x < width && tile_y >= -16 && tile_y < height) {
482 TileRenderInfo info;
483 info.tile = &object.tiles()[i];
484 info.x = tile_x;
485 info.y = tile_y;
486 info.sheet_index = -1;
487 tile_infos.push_back(info);
488 }
489 }
490 }
491
492 // Batch render tiles
493 auto batch_status = BatchRenderTiles(tile_infos, bitmap, palette);
494 if (!batch_status.ok()) return batch_status;
495
496 // Update performance stats
497 if (performance_monitoring_enabled_) {
498 performance_monitor_->IncrementObjectCount();
499 performance_monitor_->IncrementTileCount(tile_infos.size());

720absl::Status ObjectRenderer::RenderTileToBitmap(const gfx::Tile16& tile, gfx::Bitmap& bitmap, int x, int y, const gfx::SnesPalette& palette) {
721 // Render the 4 sub-tiles of the Tile16
722 std::array<gfx::TileInfo, 4> sub_tiles = {
723 tile.tile0_, tile.tile1_, tile.tile2_, tile.tile3_
724 };
725
726 for (int i = 0; i < 4; ++i) {
727 const auto& tile_info = sub_tiles[i];
728 int sub_x = x + (i % 2) * 8;
729 int sub_y = y + (i / 2) * 8;
730
731 // Bounds check
732 if (sub_x < 0 || sub_y < 0 || sub_x >= bitmap.width() || sub_y >= bitmap.height()) {
733 continue;
734 }
735
736 // Get graphics sheet
737 int sheet_index = tile_info.id_ / 256;
738 auto sheet_result = graphics_cache_->GetGraphicsSheet(sheet_index);
739 if (!sheet_result.ok()) {
740 // Use fallback pattern
741 RenderTilePattern(bitmap, sub_x, sub_y, tile_info, palette);
742 continue;
743 }
744
745 auto graphics_sheet = sheet_result.value();
746 if (!graphics_sheet || !graphics_sheet->is_active()) {
747 RenderTilePattern(bitmap, sub_x, sub_y, tile_info, palette);
748 continue;
749 }
750
751 // Render 8x8 tile from graphics sheet
752 Render8x8Tile(bitmap, graphics_sheet.get(), tile_info, sub_x, sub_y, palette);
753 }
754
755 return absl::OkStatus();

758void ObjectRenderer::Render8x8Tile(gfx::Bitmap& bitmap, gfx::Bitmap* graphics_sheet, const gfx::TileInfo& tile_info, int x, int y, const gfx::SnesPalette& palette) {
759 int tile_x = (tile_info.id_ % 16) * 8;
760 int tile_y = ((tile_info.id_ % 256) / 16) * 8;
761
762 for (int py = 0; py < 8; ++py) {
763 for (int px = 0; px < 8; ++px) {
764 int final_x = x + px;
765 int final_y = y + py;
766
767 if (final_x < 0 || final_y < 0 || final_x >= bitmap.width() || final_y >= bitmap.height()) {
768 continue;
769 }
770
771 int src_x = tile_x + px;
772 int src_y = tile_y + py;
773
774 if (src_x < 0 || src_y < 0 || src_x >= graphics_sheet->width() || src_y >= graphics_sheet->height()) {
775 continue;
776 }
777
778 int pixel_index = src_y * graphics_sheet->width() + src_x;
779 if (pixel_index < 0 || pixel_index >= static_cast<int>(graphics_sheet->size())) {
780 continue;
781 }
782
783 uint8_t color_index = graphics_sheet->at(pixel_index);
784
785 if (color_index >= palette.size()) {
786 continue;
787 }
788
789 // Apply mirroring
790 int render_x = final_x;
791 int render_y = final_y;
792
793 if (tile_info.horizontal_mirror_) {
794 render_x = x + (7 - px);
795 if (render_x < 0 || render_x >= bitmap.width()) continue;
796 }
797 if (tile_info.vertical_mirror_) {
798 render_y = y + (7 - py);
799 if (render_y < 0 || render_y >= bitmap.height()) continue;
800 }
801
802 if (render_x >= 0 && render_y >= 0 && render_x < bitmap.width() && render_y < bitmap.height()) {
803 bitmap.SetPixel(render_x, render_y, palette[color_index]);
804 }
805 }
806 }

809void ObjectRenderer::RenderTilePattern(gfx::Bitmap& bitmap, int x, int y, const gfx::TileInfo& tile_info, const gfx::SnesPalette& palette) {
810 // Render a simple pattern for missing tiles
811 uint8_t color = (tile_info.id_ % 16) + 1;
812 if (color >= palette.size()) color = 1;
813
814 for (int py = 0; py < 8; ++py) {
815 for (int px = 0; px < 8; ++px) {
816 int final_x = x + px;
817 int final_y = y + py;
818
819 if (final_x >= 0 && final_y >= 0 && final_x < bitmap.width() && final_y < bitmap.height()) {
820 bitmap.SetPixel(final_x, final_y, palette[color]);
821 }
822 }

826absl::Status ObjectRenderer::BatchRenderTiles(const std::vector<TileRenderInfo>& tiles, gfx::Bitmap& bitmap, const gfx::SnesPalette& palette) {
827 // Group tiles by graphics sheet for efficiency
828 std::unordered_map<int, std::vector<TileRenderInfo>> sheet_tiles;
829
830 for (const auto& tile_info : tiles) {
831 if (tile_info.tile == nullptr) continue;
832
833 for (int i = 0; i < 4; i++) {
834 const gfx::TileInfo* sub_tile = nullptr;
835 switch (i) {
836 case 0: sub_tile = &tile_info.tile->tile0_; break;
837 case 1: sub_tile = &tile_info.tile->tile1_; break;
838 case 2: sub_tile = &tile_info.tile->tile2_; break;
839 case 3: sub_tile = &tile_info.tile->tile3_; break;
840 }
841
842 if (sub_tile == nullptr) continue;
843
844 int sheet_index = sub_tile->id_ / 256;
845 if (sheet_index >= 0 && sheet_index < 223) {
846 TileRenderInfo sheet_tile_info = tile_info;
847 sheet_tile_info.sheet_index = sheet_index;
848 sheet_tiles[sheet_index].push_back(sheet_tile_info);
849 }
850 }
851 }
852
853 // Render tiles for each graphics sheet
854 for (auto& [sheet_index, sheet_tiles_list] : sheet_tiles) {
855 auto sheet_result = graphics_cache_->GetGraphicsSheet(sheet_index);
856 if (!sheet_result.ok()) {
857 continue;
858 }
859
860 auto graphics_sheet = sheet_result.value();
861 if (!graphics_sheet || !graphics_sheet->is_active()) {
862 continue;
863 }
864
865 // Render all tiles from this sheet
866 for (const auto& tile_info : sheet_tiles_list) {
867 auto status = RenderTileToBitmap(*tile_info.tile, bitmap, tile_info.x, tile_info.y, palette);
868 if (!status.ok()) {
869 continue;
870 }
871 }
872 }
873
874 return absl::OkStatus();
875}

877std::pair<int, int> ObjectRenderer::CalculateOptimalBitmapSize(const std::vector<RoomObject>& objects) {
878 if (objects.empty()) {
879 return {256, 256};
880 }
881
882 int max_x = 0, max_y = 0;
883
884 for (const auto& obj : objects) {
885 int obj_max_x = obj.x_ * 16 + 16;
886 int obj_max_y = obj.y_ * 16 + 16;
887
888 max_x = std::max(max_x, obj_max_x);
889 max_y = std::max(max_y, obj_max_y);
890 }
891
892 // Round up to nearest power of 2
893 int width = 1;
894 int height = 1;
895
896 while (width < max_x) width <<= 1;
897 while (height < max_y) height <<= 1;
898
899 // Cap at maximum size
900 width = std::min(width, 2048);
901 height = std::min(height, 2048);
902
903 return {width, height};
904}

914// Legacy compatibility methods
915absl::Status ObjectRenderer::RenderTile(const gfx::Tile16& tile,
916 gfx::Bitmap& bitmap,
917 int x, int y,
918 const gfx::SnesPalette& palette) {
919
920 // Check if bitmap is valid
921 if (!bitmap.is_active() || bitmap.surface() == nullptr) {
922 return absl::FailedPreconditionError("Bitmap is not properly initialized");
923 }
924
925 // Get the graphics sheet from Arena - this contains the actual pixel data
926 auto& arena = gfx::Arena::Get();
927
928 // Render the 4 sub-tiles of the Tile16
929 std::array<gfx::TileInfo, 4> sub_tiles = {
930 tile.tile0_, tile.tile1_, tile.tile2_, tile.tile3_
931 };
932
933 for (int i = 0; i < 4; ++i) {
934 const auto& tile_info = sub_tiles[i];
935 int sub_x = x + (i % 2) * 8;
936 int sub_y = y + (i / 2) * 8;
937
938 // Get the graphics sheet that contains this tile
939 // Tile IDs are typically organized in sheets of 256 tiles each
940 int sheet_index = tile_info.id_ / 256;
941 if (sheet_index >= 223) { // Arena has 223 graphics sheets
942 sheet_index = 0; // Fallback to first sheet
943 }
944
945 auto graphics_sheet = arena.gfx_sheet(sheet_index);
946 if (!graphics_sheet.is_active()) {
947 // If graphics sheet is not loaded, create a simple pattern
948 RenderTilePattern(bitmap, sub_x, sub_y, tile_info, palette);
949 continue;
950 }
951
952 // Calculate tile position within the graphics sheet
953 int tile_x = (tile_info.id_ % 16) * 8; // 16 tiles per row, 8 pixels per tile
954 int tile_y = ((tile_info.id_ % 256) / 16) * 8; // 16 rows per sheet
955
956 // Render the 8x8 tile from the graphics sheet
957 for (int py = 0; py < 8; ++py) {
958 for (int px = 0; px < 8; ++px) {
959 if (sub_x + px < bitmap.width() && sub_y + py < bitmap.height()) {
960 // Get pixel from graphics sheet
961 int src_x = tile_x + px;
962 int src_y = tile_y + py;
963
964 if (src_x < graphics_sheet.width() && src_y < graphics_sheet.height()) {
965 int pixel_index = src_y * graphics_sheet.width() + src_x;
966 if (pixel_index < (int)graphics_sheet.size()) {
967 uint8_t color_index = graphics_sheet.at(pixel_index);
968
969 // Apply palette
970 if (color_index < palette.size()) {
971 // Apply mirroring if needed
972 int final_x = sub_x + px;
973 int final_y = sub_y + py;
974
975 if (tile_info.horizontal_mirror_) {
976 final_x = sub_x + (7 - px);
977 }
978 if (tile_info.vertical_mirror_) {
979 final_y = sub_y + (7 - py);
980 }
981
982 if (final_x < bitmap.width() && final_y < bitmap.height()) {
983 bitmap.SetPixel(final_x, final_y, palette[color_index]);
984 }
985 }
986 }
987 }
988 }
989 }
990 }
991 }
992
993 return absl::OkStatus();
994}
