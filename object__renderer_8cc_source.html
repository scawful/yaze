141absl::Status ObjectRenderer::RenderTile(const gfx::Tile16& tile,
142 gfx::Bitmap& bitmap,
143 int x, int y,
144 const gfx::SnesPalette& palette) {
145
146 // Check if bitmap is valid
147 if (!bitmap.is_active() || bitmap.surface() == nullptr) {
148 return absl::FailedPreconditionError("Bitmap is not properly initialized");
149 }
150
151 // Get the graphics sheet from Arena - this contains the actual pixel data
152 auto& arena = gfx::Arena::Get();
153
154 // Render the 4 sub-tiles of the Tile16
155 std::array<gfx::TileInfo, 4> sub_tiles = {
156 tile.tile0_, tile.tile1_, tile.tile2_, tile.tile3_
157 };
158
159 for (int i = 0; i < 4; ++i) {
160 const auto& tile_info = sub_tiles[i];
161 int sub_x = x + (i % 2) * 8;
162 int sub_y = y + (i / 2) * 8;
163
164 // Get the graphics sheet that contains this tile
165 // Tile IDs are typically organized in sheets of 256 tiles each
166 int sheet_index = tile_info.id_ / 256;
167 if (sheet_index >= 223) { // Arena has 223 graphics sheets
168 sheet_index = 0; // Fallback to first sheet
169 }
170
171 auto graphics_sheet = arena.gfx_sheet(sheet_index);
172 if (!graphics_sheet.is_active()) {
173 // If graphics sheet is not loaded, create a simple pattern
174 RenderTilePattern(bitmap, sub_x, sub_y, tile_info, palette);
175 continue;
176 }
177
178 // Calculate tile position within the graphics sheet
179 int tile_x = (tile_info.id_ % 16) * 8; // 16 tiles per row, 8 pixels per tile
180 int tile_y = ((tile_info.id_ % 256) / 16) * 8; // 16 rows per sheet
181
182 // Render the 8x8 tile from the graphics sheet
183 for (int py = 0; py < 8; ++py) {
184 for (int px = 0; px < 8; ++px) {
185 if (sub_x + px < bitmap.width() && sub_y + py < bitmap.height()) {
186 // Get pixel from graphics sheet
187 int src_x = tile_x + px;
188 int src_y = tile_y + py;
189
190 if (src_x < graphics_sheet.width() && src_y < graphics_sheet.height()) {
191 int pixel_index = src_y * graphics_sheet.width() + src_x;
192 if (pixel_index < (int)graphics_sheet.size()) {
193 uint8_t color_index = graphics_sheet.at(pixel_index);
194
195 // Apply palette
196 if (color_index < palette.size()) {
197 // Apply mirroring if needed
198 int final_x = sub_x + px;
199 int final_y = sub_y + py;
200
201 if (tile_info.horizontal_mirror_) {
202 final_x = sub_x + (7 - px);
203 }
204 if (tile_info.vertical_mirror_) {
205 final_y = sub_y + (7 - py);
206 }
207
208 if (final_x < bitmap.width() && final_y < bitmap.height()) {
209 bitmap.SetPixel(final_x, final_y, palette[color_index]);
210 }
211 }
212 }
213 }
214 }
215 }
216 }
217 }
218
219 return absl::OkStatus();
220}

236void ObjectRenderer::RenderTilePattern(gfx::Bitmap& bitmap, int x, int y,
237 const gfx::TileInfo& tile_info,
238 const gfx::SnesPalette& palette) {
239 // Create a simple pattern based on tile ID and palette
240 // This is used when the graphics sheet is not available
241
242 for (int py = 0; py < 8; ++py) {
243 for (int px = 0; px < 8; ++px) {
244 if (x + px < bitmap.width() && y + py < bitmap.height()) {
245 // Create a simple pattern based on tile ID
246 int pattern_value = (tile_info.id_ + px + py) % 16;
247
248 // Use different colors based on the pattern
249 int color_index = pattern_value % palette.size();
250 if (color_index > 0) { // Skip transparent color (index 0)
251 bitmap.SetPixel(x + px, y + py, palette[color_index]);
252 }
253 }
254 }
255 }
