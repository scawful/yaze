26SubtypeInfo DungeonObjectRenderer::FetchSubtypeInfo(uint16_t object_id) {
27 SubtypeInfo info;
28
29 // Determine the subtype based on objectId
30 uint8_t subtype = 1;
31
32 // Based on the subtype, fetch the correct pointers
33 switch (subtype) {
34 case 1: // Subtype 1
35 info.subtype_ptr = core::subtype1_tiles + (object_id & 0xFF) * 2;
36 info.routine_ptr = core::subtype1_tiles + 0x200 + (object_id & 0xFF) * 2;
37 std::cout << "Subtype 1 " << std::hex << info.subtype_ptr << std::endl;
38 std::cout << "Subtype 1 " << std::hex << info.routine_ptr << std::endl;
39 break;
40 case 2: // Subtype 2
41 info.subtype_ptr = core::subtype2_tiles + (object_id & 0x7F) * 2;
42 info.routine_ptr = core::subtype2_tiles + 0x80 + (object_id & 0x7F) * 2;
43 break;
44 case 3: // Subtype 3
45 info.subtype_ptr = core::subtype3_tiles + (object_id & 0xFF) * 2;
46 info.routine_ptr = core::subtype3_tiles + 0x100 + (object_id & 0xFF) * 2;
47 break;
48 default:
49 // Handle unknown subtype
50 throw std::runtime_error("Unknown subtype for object ID: " +

97void DungeonObjectRenderer::RenderObject(const SubtypeInfo& info) {
98 cpu.PB = 0x01;
99
100 // Push an initial value to the stack we can read later to confirm we are
101 // done
102 cpu.PushLong(0x01 << 16 | info.routine_ptr);
103
104 int i = 0;
105 while (true) {
106 uint8_t opcode = cpu.ReadByte(cpu.PB << 16 | cpu.PC);
