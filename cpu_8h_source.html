688 void compare(uint16_t register_value, uint16_t memory_value) {
689 uint16_t result;
690 if (GetIndexSize()) {
691 // 8-bit mode
692 uint8_t result8 = static_cast<uint8_t>(register_value) -
693 static_cast<uint8_t>(memory_value);
694 result = result8;
695 SetNegativeFlag(result & 0x80); // Negative flag for 8-bit
696 } else {
697 // 16-bit mode
698 result = register_value - memory_value;
699 SetNegativeFlag(result & 0x8000); // Negative flag for 16-bit
700 }
701 SetZeroFlag(result == 0); // Zero flag
702 SetCarryFlag(register_value >= memory_value); // Carry flag
703 }
