756 void compare(uint16_t register_value, uint16_t memory_value) {
757 uint16_t result;
758 if (GetIndexSize()) {
759 // 8-bit mode
760 uint8_t result8 = static_cast<uint8_t>(register_value) -
761 static_cast<uint8_t>(memory_value);
762 result = result8;
763 SetNegativeFlag(result & 0x80); // Negative flag for 8-bit
764 } else {
765 // 16-bit mode
766 result = register_value - memory_value;
767 SetNegativeFlag(result & 0x8000); // Negative flag for 16-bit
768 }
769 SetZeroFlag(result == 0); // Zero flag
770 SetCarryFlag(register_value >= memory_value); // Carry flag
771 }
