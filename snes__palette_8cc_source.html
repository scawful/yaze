210uint32_t GetPaletteAddress(const std::string& group_name, size_t palette_index,
211 size_t color_index) {
212 // Retrieve the base address for the palette group
213 uint32_t base_address = kPaletteGroupAddressMap.at(group_name);
214
215 // Retrieve the number of colors for each palette in the group
216 uint32_t colors_per_palette = kPaletteGroupColorCounts.at(group_name);
217
218 // Calculate the address for thes specified color in the ROM
219 uint32_t address = base_address + (palette_index * colors_per_palette * 2) +
220 (color_index * 2);
221
222 return address;
223}

235SnesPalette::SnesPalette(char* data) : size_(sizeof(data) / 2) {
236 assert((sizeof(data) % 4 == 0) && (sizeof(data) <= 32));
237 for (unsigned i = 0; i < sizeof(data); i += 2) {
238 SnesColor col;
239 col.set_snes(static_cast<uchar>(data[i + 1]) << 8);
240 col.set_snes(col.snes() | static_cast<uchar>(data[i]));
241 snes_color mColor = ConvertSNEStoRGB(col.snes());
242 col.set_rgb(ImVec4(mColor.red, mColor.green, mColor.blue, 1.f));
243 colors.push_back(col);
244 }
245 size_ = sizeof(data) / 2;
246}

304SnesPalette ReadPaletteFromRom(int offset, int num_colors, const uchar* rom) {
305 int color_offset = 0;
306 std::vector<gfx::SnesColor> colors(num_colors);
307
308 while (color_offset < num_colors) {
309 short color = (ushort)((rom[offset + 1]) << 8) | rom[offset];
310 snes_color new_color;
311 new_color.red = (color & 0x1F) * 8;
312 new_color.green = ((color >> 5) & 0x1F) * 8;
313 new_color.blue = ((color >> 10) & 0x1F) * 8;
314 colors[color_offset].set_snes(ConvertRGBtoSNES(new_color));
315 if (color_offset == 0) {
316 colors[color_offset].set_transparent(true);
317 }
318 color_offset++;
319 offset += 2;
320 }
321
322 return gfx::SnesPalette(colors);
323}
