109 std::string OptimizeMessageForDictionary(std::string messageString) {
110 std::stringstream protons;
111 bool command = false;
112 for (const auto& c : messageString) {
113 if (c == '[') {
114 command = true;
115 } else if (c == ']') {
116 command = false;
117 }
118
119 protons << c;
120 if (command) {
121 protons << CHEESE;
122 }
123 }
124
125 std::string protonsString = protons.str();
126 std::string replacedString = ReplaceAllDictionaryWords(protonsString);
127 std::string finalString =
128 absl::StrReplaceAll(replacedString, {{CHEESE, ""}});
129
130 return finalString;
131 }
