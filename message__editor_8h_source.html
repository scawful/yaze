196 DictionaryEntry(uint8_t i, std::string s)
197 : Contents(s), ID(i), Length(s.length()) {
198 Token = absl::StrFormat("[%s:%00X]", DICTIONARYTOKEN, ID);
199 Data = ParseMessageToData(Contents);

201
202 bool ContainedInString(std::string s) {
203 return s.find(Contents) != std::string::npos;
204 }
205
206 std::string ReplaceInstancesOfIn(std::string s) {
207 std::string replacedString = s;
208 size_t pos = replacedString.find(Contents);
209 while (pos != std::string::npos) {
210 replacedString.replace(pos, Contents.length(), Token);
211 pos = replacedString.find(Contents, pos + Token.length());
212 }
213 return replacedString;
214 }
