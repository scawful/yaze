204 std::string ReplaceInstancesOfIn(std::string s) {
205 std::string replacedString = s;
206 size_t pos = replacedString.find(Contents);
207 while (pos != std::string::npos) {
208 replacedString.replace(pos, Contents.length(), Token);
209 pos = replacedString.find(Contents, pos + Token.length());
210 }
211 return replacedString;
212 }
