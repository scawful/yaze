20// Convert SNES color to RGB
21struct RGB {
22 uint8_t r, g, b;
23};

25RGB SnesColorToRGB(uint16_t snes_color) {
26 // SNES color format: 0bbbbbgggggrrrrr (5 bits per channel)
27 uint8_t r = (snes_color & 0x1F) << 3;
28 uint8_t g = ((snes_color >> 5) & 0x1F) << 3;
29 uint8_t b = ((snes_color >> 10) & 0x1F) << 3;
30 return {r, g, b};
31}

33// Convert RGB to SNES color
34uint16_t RGBToSnesColor(uint8_t r, uint8_t g, uint8_t b) {
35 return ((r >> 3) & 0x1F) | (((g >> 3) & 0x1F) << 5) | (((b >> 3) & 0x1F) << 10);
36}

38// Parse hex color (supports #RRGGBB or RRGGBB)
39absl::StatusOr<RGB> ParseHexColor(const std::string& str) {
40 std::string clean = str;
41 if (!clean.empty() && clean[0] == '#') {
42 clean = clean.substr(1);
43 }
44
45 if (clean.length() != 6) {
46 return absl::InvalidArgumentError(
47 absl::StrFormat("Invalid hex color format: %s (expected 6 hex digits)", str));
48 }
49
50 try {
51 unsigned long value = std::stoul(clean, nullptr, 16);
52 RGB rgb;
53 rgb.r = (value >> 16) & 0xFF;
54 rgb.g = (value >> 8) & 0xFF;
55 rgb.b = value & 0xFF;
56 return rgb;
57 } catch (const std::exception& e) {
58 return absl::InvalidArgumentError(
59 absl::StrFormat("Failed to parse hex color '%s': %s", str, e.what()));
60 }
61}

63// Format color based on requested format
64std::string FormatColor(uint16_t snes_color, const std::string& format) {
65 if (format == "snes") {
66 return absl::StrFormat("$%04X", snes_color);
67 }
68
69 RGB rgb = SnesColorToRGB(snes_color);
70
71 if (format == "rgb") {
72 return absl::StrFormat("rgb(%d, %d, %d)", rgb.r, rgb.g, rgb.b);
73 }
74
75 // Default to hex
76 return absl::StrFormat("#%02X%02X%02X", rgb.r, rgb.g, rgb.b);
77}

243absl::Status HandlePaletteAnalyze(const std::vector<std::string>& args,
244 Rom* rom_context) {
245 if (!rom_context || !rom_context->is_loaded()) {
246 return absl::FailedPreconditionError("ROM not loaded");
247 }
248
249 // Parse arguments
250 std::string target_type;
251 std::string target_id;
252
253 for (const auto& arg : args) {
254 if (arg.rfind("--type=", 0) == 0) {
255 target_type = arg.substr(7);
256 } else if (arg.rfind("--id=", 0) == 0) {
257 target_id = arg.substr(5);
258 }
259 }
260
261 if (target_type.empty() || target_id.empty()) {
262 return absl::InvalidArgumentError("--type and --id required");
263 }
264
265 if (target_type == "palette") {
266 // Parse palette ID (assume format "group/palette")
267 size_t slash_pos = target_id.find('/');
268 if (slash_pos == std::string::npos) {
269 return absl::InvalidArgumentError(
270 "Palette ID format should be 'group/palette' (e.g., '0/0')");
271 }
272
273 int group = std::stoi(target_id.substr(0, slash_pos));
274 int palette = std::stoi(target_id.substr(slash_pos + 1));
275
276 // Read palette
277 constexpr uint32_t kPaletteBase = 0xDE6C8;
278 constexpr uint32_t kColorsPerPalette = 16;
279 constexpr uint32_t kBytesPerColor = 2;
280
281 uint32_t palette_addr = kPaletteBase +
282 (group * 8 * kColorsPerPalette * kBytesPerColor) +
283 (palette * kColorsPerPalette * kBytesPerColor);
284
285 // Analyze colors
286 std::map<uint16_t, int> color_usage;
287 int transparent_count = 0;
288 int darkest = 0xFFFF, brightest = 0;
289
290 for (int i = 0; i < kColorsPerPalette; ++i) {
291 uint32_t color_addr = palette_addr + (i * kBytesPerColor);
292 auto snes_color_or = rom_context->ReadWord(color_addr);
293 if (!snes_color_or.ok()) {
294 return snes_color_or.status();
295 }
296 uint16_t snes_color = snes_color_or.value();
297
298 color_usage[snes_color]++;
299
300 if (snes_color == 0) {
301 transparent_count++;
302 }
303
304 // Calculate brightness (simple sum of RGB components)
305 RGB rgb = SnesColorToRGB(snes_color);
306 int brightness = rgb.r + rgb.g + rgb.b;
307 if (brightness < (((darkest & 0x1F) + ((darkest >> 5) & 0x1F) + ((darkest >> 10) & 0x1F)) * 8)) {
308 darkest = snes_color;
309 }
310 if (brightness > (((brightest & 0x1F) + ((brightest >> 5) & 0x1F) + ((brightest >> 10) & 0x1F)) * 8)) {
311 brightest = snes_color;
312 }
313 }
314
315 // Output analysis
316 std::cout << absl::StrFormat("Palette Analysis: Group %d, Palette %d\n", group, palette);
317 std::cout << absl::StrFormat("Address: 0x%06X\n\n", palette_addr);
318 std::cout << absl::StrFormat("Total colors: %d\n", kColorsPerPalette);
319 std::cout << absl::StrFormat("Unique colors: %zu\n", color_usage.size());
320 std::cout << absl::StrFormat("Transparent/black (0): %d\n", transparent_count);
321 std::cout << absl::StrFormat("Darkest color: %s\n", FormatColor(darkest, "hex"));
322 std::cout << absl::StrFormat("Brightest color: %s\n", FormatColor(brightest, "hex"));
323
324 if (color_usage.size() < kColorsPerPalette) {
325 std::cout << "\nDuplicate colors found:\n";
326 for (const auto& [color, count] : color_usage) {
327 if (count > 1) {
328 std::cout << absl::StrFormat(" %s appears %d times\n",
329 FormatColor(color, "hex"), count);
330 }
331 }
332 }
333
334 } else {
335 return absl::UnimplementedError(
336 absl::StrFormat("Analysis for type '%s' not yet implemented", target_type));
337 }
338
339 return absl::OkStatus();
340}
