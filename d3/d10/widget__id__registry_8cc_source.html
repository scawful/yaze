64std::vector<std::string> WidgetIdRegistry::FindWidgets(
65 const std::string& pattern) const {
66 std::vector<std::string> matches;
67
68 // Simple glob-style pattern matching
69 // Supports: "*" (any), "?" (single char), exact matches
70 for (const auto& [path, info] : widgets_) {
71 bool match = false;
72
73 if (pattern == "*") {
74 match = true;
75 } else if (pattern.find('*') != std::string::npos) {
76 // Wildcard pattern - convert to simple substring match for now
77 std::string search = pattern;
78 search.erase(std::remove(search.begin(), search.end(), '*'), search.end());
79 if (!search.empty() && path.find(search) != std::string::npos) {
80 match = true;
81 }
82 } else {
83 // Exact match
84 if (path == pattern) {
85 match = true;
86 }
87 }
88
89 if (match) {
90 matches.push_back(path);
91 }
92 }
93
94 // Sort for consistent ordering
95 std::sort(matches.begin(), matches.end());
96 return matches;
