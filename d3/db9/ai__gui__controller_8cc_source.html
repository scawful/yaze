56absl::StatusOr<ControlResult> AIGUIController::ExecuteActions(
57 const std::vector<AIAction>& actions) {
58
59 ControlResult result;
60 result.success = false;
61
62 for (const auto& action : actions) {
63 int retry_count = 0;
64 bool action_succeeded = false;
65 AIAction current_action = action;
66
67 while (retry_count < config_.max_retries_per_action && !action_succeeded) {
68 result.iterations_performed++;
69
70 if (result.iterations_performed > config_.max_iterations) {
71 result.error_message = "Max iterations reached";
72 return result;
73 }
74
75 // Execute the action with vision verification
76 auto execute_result = ExecuteSingleAction(
77 current_action,
78 config_.enable_vision_verification
79 );
80
81 if (!execute_result.ok()) {
82 result.error_message = std::string(execute_result.status().message());
83 return result;
84 }
85
86 result.vision_analyses.push_back(*execute_result);
87 result.actions_executed.push_back(current_action);
88
89 if (execute_result->action_successful) {
90 action_succeeded = true;
91 }
92 else if (config_.enable_iterative_refinement) {
93 // Refine action and retry
94 auto refinement = vision_refiner_->RefineAction(
95 current_action,
96 *execute_result
97 );
98
99 if (!refinement.ok()) {
100 result.error_message =
101 absl::StrCat("Failed to refine action: ",
102 refinement.status().message());
103 return result;
104 }
105
106 if (refinement->needs_different_approach) {
107 result.error_message =
108 absl::StrCat("Action requires different approach: ",
109 refinement->reasoning);
110 return result;
111 }
112
113 if (refinement->needs_retry) {
114 // Update action parameters
115 for (const auto& [key, value] : refinement->adjusted_parameters) {
116 current_action.parameters[key] = value;
117 }
118 }
119
120 retry_count++;
121 }
122 else {
123 // No refinement, just fail
124 result.error_message = execute_result->error_message;
125 return result;
126 }
127 }
128
129 if (!action_succeeded) {
130 result.error_message =
131 absl::StrFormat("Action failed after %d retries", retry_count);
132 return result;
133 }
134 }
135
136 result.success = true;
137
138 // Capture final state
139 auto final_screenshot = CaptureCurrentState("final_state");
140 if (final_screenshot.ok()) {
141 result.screenshots_taken.push_back(*final_screenshot);
142
143 // Analyze final state
144 auto final_analysis = vision_refiner_->AnalyzeScreenshot(
145 *final_screenshot,
146 "Verify all actions completed successfully"
147 );
148
149 if (final_analysis.ok()) {
150 result.final_state_description = final_analysis->description;
151 }
152 }
153
154 return result;
155}

157absl::StatusOr<VisionAnalysisResult> AIGUIController::ExecuteSingleAction(
158 const AIAction& action,
159 bool verify_with_vision) {
160
161 VisionAnalysisResult result;
162
163 // Capture before screenshot
164 std::filesystem::path before_screenshot;
165 if (verify_with_vision) {
166 auto before_result = CaptureCurrentState("before_action");
167 if (!before_result.ok()) {
168 return before_result.status();
169 }
170 before_screenshot = *before_result;
171 }
172
173 // Wait for UI to settle
174 if (config_.screenshot_delay_ms > 0) {
175 std::this_thread::sleep_for(
176 std::chrono::milliseconds(config_.screenshot_delay_ms));
177 }
178
179 // Execute the action via gRPC
180 auto execute_status = ExecuteGRPCAction(action);
181 if (!execute_status.ok()) {
182 result.action_successful = false;
183 result.error_message = std::string(execute_status.message());
184 return result;
185 }
186
187 // Wait for action to complete
188 std::this_thread::sleep_for(
189 std::chrono::milliseconds(config_.screenshot_delay_ms));
190
191 if (verify_with_vision) {
192 // Capture after screenshot
193 auto after_result = CaptureCurrentState("after_action");
194 if (!after_result.ok()) {
195 return after_result.status();
196 }
197
198 // Verify with vision
199 return VerifyActionSuccess(action, before_screenshot, *after_result);
200 }
201 else {
202 // Assume success without verification
203 result.action_successful = true;
204 result.description = "Action executed (no vision verification)";
205 return result;
206 }
207}
