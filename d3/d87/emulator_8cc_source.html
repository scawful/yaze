23bool ShouldDisplay(const InstructionEntry& entry, const char* filter) {
24 if (filter[0] == '\0') {
25 return true;
26 }
27
28 // Supported fields: address, opcode, operands
29 if (entry.operands.find(filter) != std::string::npos) {
30 return true;
31 }
32
33 if (absl::StrFormat("%06X", entry.address).find(filter) !=
34 std::string::npos) {
35 return true;
36 }
37
38 if (opcode_to_mnemonic.at(entry.opcode).find(filter) != std::string::npos) {
39 return true;
40 }
41
42 return false;
43}

52void Emulator::Run(Rom* rom) {
53 static bool loaded = false;
54 if (!snes_.running() && rom->is_loaded()) {
55 // Use ARGB8888 format to match PPU output (XBGR layout with format=1)
56 // Add better error handling and texture optimization
57 ppu_texture_ = SDL_CreateTexture(core::Renderer::Get().renderer(),
58 SDL_PIXELFORMAT_ARGB8888,
59 SDL_TEXTUREACCESS_STREAMING, 512, 480);
60 if (ppu_texture_ == NULL) {
61 printf("Failed to create texture: %s\n", SDL_GetError());
62 return;
63 }
64
65 // Optimize texture for better performance
66 SDL_SetTextureBlendMode(ppu_texture_, SDL_BLENDMODE_NONE);
67 rom_data_ = rom->vector();
68 snes_.Init(rom_data_);
69
70 // Note: PPU pixel format set to 1 (XBGR) in Init() which matches ARGB8888 texture
71
72 wanted_frames_ = 1.0 / (snes_.memory().pal_timing() ? 50.0 : 60.0);
73 wanted_samples_ = 48000 / (snes_.memory().pal_timing() ? 50 : 60);
74 loaded = true;
75
76 count_frequency = SDL_GetPerformanceFrequency();
77 last_count = SDL_GetPerformanceCounter();
78 time_adder = 0.0;
79 frame_count_ = 0;
80 fps_timer_ = 0.0;
81 current_fps_ = 0.0;
82 }
83
84 RenderNavBar();
85
86 if (running_) {
87 HandleEvents();
88
89 uint64_t current_count = SDL_GetPerformanceCounter();
90 uint64_t delta = current_count - last_count;
91 last_count = current_count;
92 double seconds = delta / (double)count_frequency;
93 time_adder += seconds;
94
95 // Cap time accumulation to prevent spiral of death and improve stability
96 if (time_adder > wanted_frames_ * 3.0) {
97 time_adder = wanted_frames_ * 3.0;
98 }
99
100 // Track frames to skip for performance
101 int frames_to_process = 0;
102 while (time_adder >= wanted_frames_ - 0.002) {
103 time_adder -= wanted_frames_;
104 frames_to_process++;
105 }
106
107 // Limit maximum frames to process (prevent spiral of death)
108 if (frames_to_process > 4) {
109 frames_to_process = 4;
110 }
111
112 if (loaded && frames_to_process > 0) {
113 // Process frames (skip rendering for all but last frame if falling behind)
114 for (int i = 0; i < frames_to_process; i++) {
115 bool should_render = (i == frames_to_process - 1);
116
117 // Run frame
118 if (turbo_mode_) {
119 snes_.RunFrame();
120 }
121 snes_.RunFrame();
122
123 // Track FPS
124 frame_count_++;
125 fps_timer_ += wanted_frames_;
126 if (fps_timer_ >= 1.0) {
127 current_fps_ = frame_count_ / fps_timer_;
128 frame_count_ = 0;
129 fps_timer_ = 0.0;
130 }
131
132 // Only render and handle audio on the last frame
133 if (should_render) {
134 // Generate and queue audio samples with improved buffering
135 snes_.SetSamples(audio_buffer_, wanted_samples_);
136 uint32_t queued = SDL_GetQueuedAudioSize(audio_device_);
137 uint32_t target_buffer = wanted_samples_ * 4 * 2; // Target 2 frames buffered
138 uint32_t max_buffer = wanted_samples_ * 4 * 6; // Max 6 frames
139
140 if (queued < target_buffer) {
141 // Buffer is low, queue more audio
142 SDL_QueueAudio(audio_device_, audio_buffer_, wanted_samples_ * 4);
143 } else if (queued > max_buffer) {
144 // Buffer is too full, clear it to prevent lag
145 SDL_ClearQueuedAudio(audio_device_);
146 SDL_QueueAudio(audio_device_, audio_buffer_, wanted_samples_ * 4);
147 }
148
149 // Update PPU texture only on rendered frames
150 void* ppu_pixels_;
151 int ppu_pitch_;
152 if (SDL_LockTexture(ppu_texture_, NULL, &ppu_pixels_, &ppu_pitch_) ==
153 0) {
154 snes_.SetPixels(static_cast<uint8_t*>(ppu_pixels_));
155 SDL_UnlockTexture(ppu_texture_);
156 }
157 }
158 }
159 }
160 }
