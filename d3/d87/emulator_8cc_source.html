132void Emulator::Run(Rom* rom) {
133 // Lazy initialization: set renderer from Controller if not set yet
134 if (!renderer_) {
135 ImGui::TextColored(ImVec4(1.0f, 0.3f, 0.3f, 1.0f),
136 "Emulator renderer not initialized");
137 return;
138 }
139
140 // Initialize audio backend if not already done (lazy initialization)
141 if (!audio_backend_) {
142 audio_backend_ = audio::AudioBackendFactory::Create(
143 audio::AudioBackendFactory::BackendType::SDL2);
144
145 audio::AudioConfig config;
146 config.sample_rate = 48000;
147 config.channels = 2;
148 config.buffer_frames = 1024;
149 config.format = audio::SampleFormat::INT16;
150
151 if (!audio_backend_->Initialize(config)) {
152 LOG_ERROR("Emulator", "Failed to initialize audio backend");
153 } else {
154 LOG_INFO("Emulator", "Audio backend initialized (lazy): %s",
155 audio_backend_->GetBackendName().c_str());
156 }
157 }
158
159 // Initialize input manager if not already done
160 if (!input_manager_.IsInitialized()) {
161 if (!input_manager_.Initialize(input::InputBackendFactory::BackendType::SDL2)) {
162 LOG_ERROR("Emulator", "Failed to initialize input manager");
163 } else {
164 LOG_INFO("Emulator", "Input manager initialized: %s",
165 input_manager_.backend()->GetBackendName().c_str());
166 }
167 }
168
169 // Initialize SNES and create PPU texture on first run
170 // This happens lazily when user opens the emulator window
171 if (!snes_initialized_ && rom->is_loaded()) {
172 // Create PPU texture with correct format for SNES emulator
173 // ARGB8888 matches the XBGR format used by the SNES PPU (pixel format 1)
174 if (!ppu_texture_) {
175 ppu_texture_ = renderer_->CreateTextureWithFormat(
176 512, 480, SDL_PIXELFORMAT_ARGB8888, SDL_TEXTUREACCESS_STREAMING);
177 if (ppu_texture_ == NULL) {
178 printf("Failed to create PPU texture: %s\n", SDL_GetError());
179 return;
180 }
181 }
182
183 // Initialize SNES with ROM data (either from Initialize() or from rom parameter)
184 if (rom_data_.empty()) {
185 rom_data_ = rom->vector();
186 }
187 snes_.Init(rom_data_);
188
189 // Note: DisassemblyViewer recording is always enabled via callback
190 // No explicit setup needed - callback is set in Initialize()
191
192 // Note: PPU pixel format set to 1 (XBGR) in Init() which matches ARGB8888 texture
193
194 wanted_frames_ = 1.0 / (snes_.memory().pal_timing() ? 50.0 : 60.0);
195 wanted_samples_ = 48000 / (snes_.memory().pal_timing() ? 50 : 60);
196 snes_initialized_ = true;
197
198 count_frequency = SDL_GetPerformanceFrequency();
199 last_count = SDL_GetPerformanceCounter();
200 time_adder = 0.0;
201 frame_count_ = 0;
202 fps_timer_ = 0.0;
203 current_fps_ = 0.0;
204 }
205
206 RenderNavBar();
207
208 // Auto-pause emulator during window operations to prevent macOS crashes
209 static bool was_running_before_pause = false;
210 bool window_has_focus = ImGui::IsWindowFocused(ImGuiFocusedFlags_RootWindow);
211
212 // Check if window is being resized (set in HandleEvents)
213 if (yaze::core::g_window_is_resizing && running_) {
214 was_running_before_pause = true;
215 running_ = false;
216 } else if (!yaze::core::g_window_is_resizing && !running_ && was_running_before_pause) {
217 // Auto-resume after resize completes
218 running_ = true;
219 was_running_before_pause = false;
220 }
221
222 // Also pause when window loses focus to save CPU/battery
223 if (!window_has_focus && running_ && !was_running_before_pause) {
224 was_running_before_pause = true;
225 running_ = false;
226 } else if (window_has_focus && !running_ && was_running_before_pause && !yaze::core::g_window_is_resizing) {
227 // Don't auto-resume - let user manually resume
228 was_running_before_pause = false;
229 }
230
231 if (running_) {
232 // Poll input and update SNES controller state
233 input_manager_.Poll(&snes_, 1); // Player 1
234
235 uint64_t current_count = SDL_GetPerformanceCounter();
236 uint64_t delta = current_count - last_count;
237 last_count = current_count;
238 double seconds = delta / (double)count_frequency;
239 time_adder += seconds;
240
241 // Cap time accumulation to prevent spiral of death and improve stability
242 if (time_adder > wanted_frames_ * 3.0) {
243 time_adder = wanted_frames_ * 3.0;
244 }
245
246 // Track frames to skip for performance
247 int frames_to_process = 0;
248 while (time_adder >= wanted_frames_ - 0.002) {
249 time_adder -= wanted_frames_;
250 frames_to_process++;
251 }
252
253 // Limit maximum frames to process (prevent spiral of death)
254 if (frames_to_process > 4) {
255 frames_to_process = 4;
256 }
257
258 if (snes_initialized_ && frames_to_process > 0) {
259 // Process frames (skip rendering for all but last frame if falling behind)
260 for (int i = 0; i < frames_to_process; i++) {
261 bool should_render = (i == frames_to_process - 1);
262
263 // Run frame
264 if (turbo_mode_) {
265 snes_.RunFrame();
266 }
267 snes_.RunFrame();
268
269 // Track FPS
270 frame_count_++;
271 fps_timer_ += wanted_frames_;
272 if (fps_timer_ >= 1.0) {
273 current_fps_ = frame_count_ / fps_timer_;
274 frame_count_ = 0;
275 fps_timer_ = 0.0;
276 }
277
278 // Only render and handle audio on the last frame
279 if (should_render) {
280 // Generate and queue audio samples using audio backend
281 snes_.SetSamples(audio_buffer_, wanted_samples_);
282
283 // AUDIO DEBUG: Comprehensive diagnostics at regular intervals
284 static int audio_debug_counter = 0;
285 audio_debug_counter++;
286
287 // Log at frames 60 (1sec), 300 (5sec), 600 (10sec), then every 600 frames
288 bool should_debug = (audio_debug_counter == 60 || audio_debug_counter == 300 ||
289 audio_debug_counter == 600 || (audio_debug_counter % 600 == 0));
290
291 if (should_debug) {
292 // Check if buffer exists
293 if (!audio_buffer_) {
294 printf("[AUDIO ERROR] audio_buffer_ is NULL!\n");
295 } else {
296 // Check for audio samples
297 bool has_audio = false;
298 int16_t max_sample = 0;
299 int non_zero_count = 0;
300 for (int i = 0; i < wanted_samples_ * 2 && i < 100; i++) {
301 if (audio_buffer_[i] != 0) {
302 has_audio = true;
303 non_zero_count++;
304 if (std::abs(audio_buffer_[i]) > std::abs(max_sample)) {
305 max_sample = audio_buffer_[i];
306 }
307 }
308 }
309
310 // Backend status
311 auto audio_status = audio_backend_ ? audio_backend_->GetStatus() : audio::AudioStatus{};
312 bool backend_playing = audio_status.is_playing;
313
314 printf("\n[AUDIO DEBUG] Frame=%d (~%.1f sec)\n", audio_debug_counter, audio_debug_counter / 60.0f);
315 printf(" Backend: %s (Playing: %s)\n",
316 audio_backend_ ? audio_backend_->GetBackendName().c_str() : "NULL",
317 backend_playing ? "YES" : "NO");
318 printf(" Queued: %u frames\n", audio_status.queued_frames);
319 printf(" Buffer: wanted_samples=%d, non_zero=%d/%d, max=%d\n",
320 wanted_samples_, non_zero_count, std::min(wanted_samples_ * 2, 100), max_sample);
321 printf(" Samples: %s\n", has_audio ? "YES" : "SILENCE");
322
323 // APU state
324 if (snes_.running()) {
325 uint64_t apu_cycles = snes_.apu().GetCycles();
326 uint16_t spc_pc = snes_.apu().spc700().PC;
327 bool ipl_rom_active = (spc_pc >= 0xFFC0 && spc_pc <= 0xFFFF);
328
329 printf(" APU: %llu cycles, PC=$%04X %s\n",
330 apu_cycles, spc_pc, ipl_rom_active ? "(IPL ROM)" : "(Game Code)");
331
332 // Handshake status
333 auto& tracker = snes_.apu_handshake_tracker();
334 printf(" Handshake: %s\n", tracker.GetPhaseString().c_str());
335
336 if (ipl_rom_active && audio_debug_counter > 300) {
337 printf(" ⚠️ SPC700 STUCK IN IPL ROM - Handshake not completing!\n");
338 }
339 } else {
340 printf(" ⚠️ SNES not running!\n");
341 }
342
343 printf("\n");
344 }
345 }
346
347 // Smart buffer management using audio backend
348 if (audio_backend_) {
349 auto status = audio_backend_->GetStatus();
350 int num_samples = wanted_samples_ * 2; // Stereo
351
352 if (status.queued_frames < 2) {
353 // Buffer is low, queue more audio
354 if (!audio_backend_->QueueSamples(audio_buffer_, num_samples)) {
355 if (frame_count_ % 300 == 0) {
356 LOG_WARN("Emulator", "Failed to queue audio samples");
357 }
358 }
359 } else if (status.queued_frames > 6) {
360 // Buffer is too full, clear it to prevent lag
361 audio_backend_->Clear();
362 audio_backend_->QueueSamples(audio_buffer_, num_samples);
363 } else {
364 // Normal operation - queue samples
365 audio_backend_->QueueSamples(audio_buffer_, num_samples);
366 }
367 }
368
369 // Update PPU texture only on rendered frames
370 void* ppu_pixels_;
371 int ppu_pitch_;
372 if (renderer_->LockTexture(ppu_texture_, NULL, &ppu_pixels_, &ppu_pitch_)) {
373 snes_.SetPixels(static_cast<uint8_t*>(ppu_pixels_));
374 renderer_->UnlockTexture(ppu_texture_);
375
376 // WORKAROUND: Tiny delay after texture unlock to prevent macOS Metal crash
377 // macOS CoreAnimation/Metal driver bug in layer_presented() callback
378 // Without this, rapid texture updates corrupt Metal's frame tracking
379 SDL_Delay(1);
380 }
381 }
382 }
383 }
384 }

387}
