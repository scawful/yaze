32std::string TrimWhitespace(const std::string& input) {
33 auto begin = std::find_if_not(input.begin(), input.end(),
34 [](unsigned char c) { return std::isspace(c); });
35 auto end = std::find_if_not(input.rbegin(), input.rend(),
36 [](unsigned char c) { return std::isspace(c); })
37 .base();
38 if (begin >= end) {
39 return "";
40 }
41 return std::string(begin, end);
42}

60std::set<std::string> CollectObjectKeys(const nlohmann::json& array) {
61 std::set<std::string> keys;
62 for (const auto& item : array) {
63 if (!item.is_object()) {
64 continue;
65 }
66 for (const auto& [key, _] : item.items()) {
67 keys.insert(key);

73std::optional<ChatMessage::TableData> BuildTableData(const nlohmann::json& data) {
74 using TableData = ChatMessage::TableData;
75
76 if (data.is_object()) {
77 TableData table;
78 table.headers = {"Key", "Value"};
79 table.rows.reserve(data.size());
80 for (const auto& [key, value] : data.items()) {
81 table.rows.push_back({key, JsonValueToString(value)});
82 }
83 return table;
84 }
85
86 if (data.is_array()) {
87 TableData table;
88 if (data.empty()) {
89 table.headers = {"Value"};
90 return table;
91 }
92
93 const bool all_objects = std::all_of(data.begin(), data.end(), [](const nlohmann::json& item) {
94 return item.is_object();
95 });
96
97 if (all_objects) {
98 auto keys = CollectObjectKeys(data);
99 if (keys.empty()) {
100 table.headers = {"Value"};
101 for (const auto& item : data) {
102 table.rows.push_back({JsonValueToString(item)});
103 }
104 return table;
105 }
106
107 table.headers.assign(keys.begin(), keys.end());
108 table.rows.reserve(data.size());
109 for (const auto& item : data) {
110 std::vector<std::string> row;
111 row.reserve(table.headers.size());
112 for (const auto& key : table.headers) {
113 if (item.contains(key)) {
114 row.push_back(JsonValueToString(item.at(key)));
115 } else {
116 row.emplace_back("-");
117 }
118 }
119 table.rows.push_back(std::move(row));
120 }
121 return table;
122 }
123
124 table.headers = {"Value"};
125 table.rows.reserve(data.size());
126 for (const auto& item : data) {
127 table.rows.push_back({JsonValueToString(item)});
128 }
129 return table;
130 }
131
132 return std::nullopt;
133}

135ChatMessage CreateMessage(ChatMessage::Sender sender, const std::string& content) {
136 ChatMessage message;
137 message.sender = sender;
138 message.message = content;
139 message.timestamp = absl::Now();
140
141 if (sender == ChatMessage::Sender::kAgent) {
142 const std::string trimmed = TrimWhitespace(content);
143 if (!trimmed.empty() && (trimmed.front() == '{' || trimmed.front() == '[')) {
144 try {
145 nlohmann::json parsed = nlohmann::json::parse(trimmed);
146 message.table_data = BuildTableData(parsed);
147 message.json_pretty = parsed.dump(2);
148 } catch (const nlohmann::json::parse_error&) {
149 // Ignore parse errors, fall back to raw text.
150 }
151 }
152 }
153
154 return message;
155}

229 << agent_response.tool_calls.size() << util::colors::kReset << std::endl;
230 std::cout << util::colors::kDim << " - Commands: "

362 if (!response_text.empty()) {
363 response_text.append("\n\n");

366 response_text.append(absl::StrJoin(agent_response.commands, "\n"));
367 }
368

371 if (!response_text.empty()) {
372 response_text.append("\n\n");

375 "✅ Proposal %s ready with %d change%s (%d command%s).\n"
376 "Review it in the Proposal drawer or run `z3ed agent diff --proposal-id %s`.\n"
377 "Sandbox ROM: %s\nProposal JSON: %s",
378 metadata.id, proposal_result->change_count,
379 proposal_result->change_count == 1 ? "" : "s",
380 proposal_result->executed_commands,
381 proposal_result->executed_commands == 1 ? "" : "s",
382 metadata.id, metadata.sandbox_rom_path.string(),
383 proposal_result->proposal_json_path.string()));
384 } else if (attempted_proposal && !proposal_status.ok()) {
385 if (!response_text.empty()) {
386 response_text.append("\n\n");
387 }
388 response_text.append(absl::StrCat(
389 "⚠️ Failed to prepare a proposal automatically: ",
390 proposal_status.message()));
391 }
392
393 ChatMessage chat_response =
394 CreateMessage(ChatMessage::Sender::kAgent, response_text);
395 history_.push_back(chat_response);
396 return chat_response;
397 }
398
399 return absl::InternalError(
400 "Agent did not produce a response after executing tools.");
