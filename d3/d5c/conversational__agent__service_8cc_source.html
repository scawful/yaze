22std::string TrimWhitespace(const std::string& input) {
23 auto begin = std::find_if_not(input.begin(), input.end(),
24 [](unsigned char c) { return std::isspace(c); });
25 auto end = std::find_if_not(input.rbegin(), input.rend(),
26 [](unsigned char c) { return std::isspace(c); })
27 .base();
28 if (begin >= end) {
29 return "";
30 }
31 return std::string(begin, end);
32}

64 using TableData = ChatMessage::TableData;
