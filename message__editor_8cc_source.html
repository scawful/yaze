442void MessageEditor::DrawTileToPreview(int x, int y, int srcx, int srcy, int pal,
443 int sizex, int sizey) {
444 const int num_x_tiles = 16;
445 const int img_width = 512; // (imgwidth/2)
446 int draw_id = srcx + (srcy * 32);
447 for (int yl = 0; yl < sizey * 8; yl++) {
448 for (int xl = 0; xl < 4; xl++) {
449 int mx = xl;
450 int my = yl;
451
452 // Formula information to get tile index position in the array.
453 // ((ID / nbrofXtiles) * (imgwidth/2) + (ID - ((ID/16)*16) ))
454 int tx = ((draw_id / num_x_tiles) * img_width) +
455 ((draw_id - ((draw_id / 16) * 16)) * 4);
456 uint8_t pixel = font_gfx16_data_[tx + (yl * 64) + xl];
457
458 // nx,ny = object position, xx,yy = tile position, xl,yl = pixel
459 // position
460 int index = x + (y * 172) + (mx * 2) + (my * 172);
461 if ((pixel & 0x0F) != 0) {
462 current_font_gfx16_data_[index + 1] =
463 (uint8_t)((pixel & 0x0F) + (0 * 4));
464 }
465
466 if (((pixel >> 4) & 0x0F) != 0) {
467 current_font_gfx16_data_[index + 0] =
468 (uint8_t)(((pixel >> 4) & 0x0F) + (0 * 4));
469 }
470 }
471 }

484void MessageEditor::DrawCharacterToPreview(const std::vector<uint8_t> &text) {
485 for (const uint8_t &value : text) {
486 if (skip_next) {
487 skip_next = false;
488 continue;
489 }
490
491 if (value < 100) {
492 int srcy = value / 16;
493 int srcx = value - (value & (~0xF));
494
495 if (text_position_ >= 170) {
496 text_position_ = 0;
497 text_line_++;
498 }
499
500 DrawTileToPreview(text_position_, text_line_ * 16, srcx, srcy, 0, 1, 2);
501 text_position_ += width_array[value];
502 } else if (value == kLine1) {
503 text_position_ = 0;
504 text_line_ = 0;
505 } else if (value == kScrollVertical) {
506 text_position_ = 0;
507 text_line_ += 1;
508 } else if (value == kLine2) {
509 text_position_ = 0;
510 text_line_ = 1;
511 } else if (value == kLine3) {
512 text_position_ = 0;
513 text_line_ = 2;
514 } else if (value == 0x6B || value == 0x6D || value == 0x6E ||
515 value == 0x77 || value == 0x78 || value == 0x79 ||
516 value == 0x7A) {
517 skip_next = true;
518
519 continue;
520 } else if (value == 0x6C) // BCD numbers.
521 {
522 DrawCharacterToPreview('0');
523 skip_next = true;
524
525 continue;
526 } else if (value == 0x6A) {
527 // Includes parentheses to be longer, since player names can be up to 6
528 // characters.
529 DrawStringToPreview("(NAME)");
530 } else if (value >= DICTOFF && value < (DICTOFF + 97)) {
531 auto dictionaryEntry = GetDictionaryFromID(value - DICTOFF);
532 DrawCharacterToPreview(dictionaryEntry.Data);
533 }
534 }

587absl::Status MessageEditor::Save() {
588 std::vector<uint8_t> backup = rom()->vector();
589
590 for (int i = 0; i < kWidthArraySize; i++) {
591 RETURN_IF_ERROR(rom()->WriteByte(kCharactersWidth + i, width_array[i]));
592 }
593
594 int pos = kTextData;
595 bool in_second_bank = false;
596
597 for (const auto &message : list_of_texts_) {
598 for (const auto value : message.Data) {
599 RETURN_IF_ERROR(rom()->WriteByte(pos, value));
600
601 if (value == kBlockTerminator) {
602 // Make sure we didn't go over the space available in the first block.
603 // 0x7FFF available.
604 if ((!in_second_bank & pos) > kTextDataEnd) {
605 return absl::InternalError(DisplayTextOverflowError(pos, true));
606 }
607
608 // Switch to the second block.
609 pos = kTextData2 - 1;
610 in_second_bank = true;
611 }
612
613 pos++;
614 }
615
616 RETURN_IF_ERROR(rom()->WriteByte(
617 pos++, kMessageTerminator)); // , true, "Terminator text"
618 }
619
620 // Verify that we didn't go over the space available for the second block.
621 // 0x14BF available.
622 if ((in_second_bank & pos) > kTextData2End) {
623 // rom()->data() = backup;
624 return absl::InternalError(DisplayTextOverflowError(pos, false));
625 }
626
627 RETURN_IF_ERROR(rom()->WriteByte(pos, 0xFF)); // , true, "End of text"
628
629 return absl::OkStatus();
630}

632std::string MessageEditor::DisplayTextOverflowError(int pos, bool bank) {
633 int space = bank ? kTextDataEnd - kTextData : kTextData2End - kTextData2;
634 std::string bankSTR = bank ? "1st" : "2nd";
635 std::string posSTR =
636 bank ? absl::StrFormat("%X4", pos & 0xFFFF)
637 : absl::StrFormat("%X4", (pos - kTextData2) & 0xFFFF);
638 std::string message = absl::StrFormat(
639 "There is too much text data in the %s block to save.\n"
640 "Available: %X4 | Used: %s",
641 bankSTR, space, posSTR);
642 return message;
