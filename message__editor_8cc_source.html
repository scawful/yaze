256 std::vector<uint8_t> bytes;
257 std::stringstream stringBuilder;
258
259 int address = core::SnesToPc(
260 kTextData + (rom()->data()[kPointersDictionaries + (i * 2) + 1] << 8) +

281void MessageEditor::ReadAllTextData() {
282 int pos = kTextData;
283 int message_id = 0;
284 uint8_t current_byte;
285 std::vector<uint8_t> temp_bytes_raw;
286 std::vector<uint8_t> temp_bytes_parsed;
287
288 std::string current_message_raw;
289 std::string current_message_parsed;
290 TextElement text_element;
291
292 while (true) {
293 current_byte = rom()->data()[pos++];
294
295 if (current_byte == MESSAGETERMINATOR) {
296 auto message =
297 MessageData(message_id++, pos, current_message_raw, temp_bytes_raw,
298 current_message_parsed, temp_bytes_parsed);
299
300 list_of_texts_.push_back(message);
301
302 temp_bytes_raw.clear();
303 temp_bytes_parsed.clear();
304 current_message_raw.clear();
305 current_message_parsed.clear();
306
307 continue;
308 } else if (current_byte == 0xFF) {
309 break;
310 }
311
312 temp_bytes_raw.push_back(current_byte);
313
314 // Check for command.
315 text_element = FindMatchingCommand(current_byte);
316
317 if (!text_element.Empty()) {
318 temp_bytes_parsed.push_back(current_byte);
319 if (text_element.HasArgument) {
320 current_byte = rom()->data()[pos++];
321 temp_bytes_raw.push_back(current_byte);
322 temp_bytes_parsed.push_back(current_byte);
323 }
324
325 current_message_raw.append(
326 text_element.GetParameterizedToken(current_byte));
327 current_message_parsed.append(
328 text_element.GetParameterizedToken(current_byte));
329
330 if (text_element.Token == BANKToken) {
331 pos = kTextData2;
332 }
333
334 continue;
335 }
336
337 // Check for special characters.
338 text_element = FindMatchingSpecial(current_byte);
339 if (!text_element.Empty()) {
340 current_message_raw.append(text_element.GetParameterizedToken());
341 current_message_parsed.append(text_element.GetParameterizedToken());
342 temp_bytes_parsed.push_back(current_byte);
343 continue;
344 }
345
346 // Check for dictionary.
347 int dictionary = FindDictionaryEntry(current_byte);
348
349 if (dictionary >= 0) {
350 current_message_raw.append("[");
351 current_message_raw.append(DICTIONARYTOKEN);
352 current_message_raw.append(":");
353 current_message_raw.append(core::UppercaseHexWord(dictionary));
354 current_message_raw.append("]");
355
356 uint32_t address = core::Get24LocalFromPC(
357 rom()->data(), kPointersDictionaries + (dictionary * 2));
358 uint32_t address_end = core::Get24LocalFromPC(
359 rom()->data(), kPointersDictionaries + ((dictionary + 1) * 2));
360
361 for (uint32_t i = address; i < address_end; i++) {
362 temp_bytes_parsed.push_back(rom()->data()[i]);
363 current_message_parsed.append(ParseTextDataByte(rom()->data()[i]));
364 }
365
366 continue;
367 }
368
369 // Everything else.
370 if (CharEncoder.contains(current_byte)) {
371 std::string str = "";
372 str.push_back(CharEncoder.at(current_byte));
373 current_message_raw.append(str);
374 current_message_parsed.append(str);
375 temp_bytes_parsed.push_back(current_byte);
376 }
377 }
378}

402ParsedElement MessageEditor::FindMatchingElement(std::string str) {
403 std::smatch match;
404 for (auto& textElement : TextCommands) {
405 match = textElement.MatchMe(str);
406 if (match.size() > 0) {
407 if (textElement.HasArgument) {
408 return ParsedElement(textElement,
409 std::stoi(match[1].str(), nullptr, 16));
410 } else {
411 return ParsedElement(textElement, 0);
412 }
413 }
414 }
415
416 const auto dictionary_element =
417 TextElement(0x80, DICTIONARYTOKEN, true, "Dictionary");
418
419 match = dictionary_element.MatchMe(str);
420 if (match.size() > 0) {
421 return ParsedElement(dictionary_element,
422 DICTOFF + std::stoi(match[1].str(), nullptr, 16));
423 }
424 return ParsedElement();
425}

438std::vector<uint8_t> MessageEditor::ParseMessageToData(std::string str) {
439 std::vector<uint8_t> bytes;
440 std::string temp_string = str;
441 int pos = 0;
442
443 while (pos < temp_string.size()) {
444 // Get next text fragment.
445 if (temp_string[pos] == '[') {
446 int next = temp_string.find(']', pos);
447 if (next == -1) {
448 break;
449 }
450
451 ParsedElement parsedElement =
452 FindMatchingElement(temp_string.substr(pos, next - pos + 1));
453
454 const auto dictionary_element =
455 TextElement(0x80, DICTIONARYTOKEN, true, "Dictionary");
456
457 if (!parsedElement.Active) {
458 core::logf("Error parsing message: %s", temp_string);
459 break;
460 } else if (parsedElement.Parent == dictionary_element) {
461 bytes.push_back(parsedElement.Value);
462 } else {
463 bytes.push_back(parsedElement.Parent.ID);
464
465 if (parsedElement.Parent.HasArgument) {
466 bytes.push_back(parsedElement.Value);
467 }
468 }
469
470 pos = next + 1;
471 continue;
472 } else {
473 uint8_t bb = FindMatchingCharacter(temp_string[pos++]);
474
475 if (bb != 0xFF) {
476 core::logf("Error parsing message: %s", temp_string);
477 bytes.push_back(bb);
478 }
479 }
480 }
481
482 return bytes;
483}

510std::string MessageEditor::ParseTextDataByte(uint8_t value) {
511 if (CharEncoder.contains(value)) {
512 char c = CharEncoder.at(value);
513 std::string str = "";
514 str.push_back(c);
515 return str;
516 }
517
518 // Check for command.
519 TextElement textElement = FindMatchingCommand(value);
520 if (!textElement.Empty()) {
521 return textElement.GenericToken;
522 }
523
524 // Check for special characters.
525 textElement = FindMatchingSpecial(value);
526 if (!textElement.Empty()) {
527 return textElement.GenericToken;
528 }
529
530 // Check for dictionary.
531 int dictionary = FindDictionaryEntry(value);
532 if (dictionary >= 0) {
533 return absl::StrFormat("[%s:%X]", DICTIONARYTOKEN, dictionary);
534 }
535
536 return "";
537}

539void MessageEditor::DrawTileToPreview(int x, int y, int srcx, int srcy, int pal,
540 int sizex, int sizey) {
541 int drawid = srcx + (srcy * 32);
542 for (int yl = 0; yl < sizey * 8; yl++) {
543 for (int xl = 0; xl < 4; xl++) {
544 int mx = xl;
545 int my = yl;
546
547 // Formula information to get tile index position in the array.
548 // ((ID / nbrofXtiles) * (imgwidth/2) + (ID - ((ID/16)*16) ))
549 int tx = ((drawid / 16) * 512) + ((drawid - ((drawid / 16) * 16)) * 4);
550 uint8_t pixel = font_gfx16_data_[tx + (yl * 64) + xl];
551
552 // nx,ny = object position, xx,yy = tile position, xl,yl = pixel
553 // position
554 int index = x + (y * 172) + (mx * 2) + (my * 172);
555 if ((pixel & 0x0F) != 0) {
556 current_font_gfx16_data_[index + 1] =
557 (uint8_t)((pixel & 0x0F) + (0 * 4));
558 }
559
560 if (((pixel >> 4) & 0x0F) != 0) {
561 current_font_gfx16_data_[index + 0] =
562 (uint8_t)(((pixel >> 4) & 0x0F) + (0 * 4));
563 }
564 }
565 }

578void MessageEditor::DrawCharacterToPreview(const std::vector<uint8_t>& text) {
579 for (const uint8_t& value : text) {
580 if (skip_next) {
581 skip_next = false;
582 continue;
583 }
584
585 if (value < 100) {
586 int srcy = value / 16;
587 int srcx = value - (value & (~0xF));
588
589 if (text_position_ >= 170) {
590 text_position_ = 0;
591 text_line_++;
592 }
593
594 DrawTileToPreview(text_position_, text_line_ * 16, srcx, srcy, 0, 1, 2);
595 text_position_ += width_array[value];
596 } else if (value == kLine1) {
597 text_position_ = 0;
598 text_line_ = 0;
599 } else if (value == kScrollVertical) {
600 text_position_ = 0;
601 text_line_ += 1;
602 } else if (value == kLine2) {
603 text_position_ = 0;
604 text_line_ = 1;
605 } else if (value == kLine3) {
606 text_position_ = 0;
607 text_line_ = 2;
608 } else if (value == 0x6B || value == 0x6D || value == 0x6E ||
609 value == 0x77 || value == 0x78 || value == 0x79 ||
610 value == 0x7A) {
611 skip_next = true;
612
613 continue;
614 } else if (value == 0x6C) // BCD numbers.
615 {
616 DrawCharacterToPreview('0');
617 skip_next = true;
618
619 continue;
620 } else if (value == 0x6A) {
621 // Includes parentheses to be longer, since player names can be up to 6
622 // characters.
623 DrawStringToPreview("(NAME)");
624 } else if (value >= DICTOFF && value < (DICTOFF + 97)) {
625 auto dictionaryEntry = GetDictionaryFromID(value - DICTOFF);
626 DrawCharacterToPreview(dictionaryEntry.Data);
627 }
628 }

681absl::Status MessageEditor::Save() {
682 std::vector<uint8_t> backup = rom()->vector();
683
684 for (int i = 0; i < 100; i++) {
685 RETURN_IF_ERROR(rom()->Write(kCharactersWidth + i, width_array[i]));
686 }
687
688 int pos = kTextData;
689 bool in_second_bank = false;
690
691 for (const auto& message : list_of_texts_) {
692 for (const auto value : message.Data) {
693 RETURN_IF_ERROR(rom()->Write(pos, value));
694
695 if (value == kBlockTerminator) {
696 // Make sure we didn't go over the space available in the first block.
697 // 0x7FFF available.
698 if ((!in_second_bank & pos) > kTextDataEnd) {
699 return absl::InternalError(DisplayTextOverflowError(pos, true));
700 }
701
702 // Switch to the second block.
703 pos = kTextData2 - 1;
704 in_second_bank = true;
705 }
706
707 pos++;
708 }
709
710 RETURN_IF_ERROR(
711 rom()->Write(pos++, MESSAGETERMINATOR)); // , true, "Terminator text"
712 }
713
714 // Verify that we didn't go over the space available for the second block.
715 // 0x14BF available.
716 if ((in_second_bank & pos) > kTextData2End) {
717 // rom()->data() = backup;
718 return absl::InternalError(DisplayTextOverflowError(pos, false));
719 }
720
721 RETURN_IF_ERROR(rom()->Write(pos, 0xFF)); // , true, "End of text"
722
723 return absl::OkStatus();
724}

726std::string MessageEditor::DisplayTextOverflowError(int pos, bool bank) {
727 int space = bank ? kTextDataEnd - kTextData : kTextData2End - kTextData2;
728 std::string bankSTR = bank ? "1st" : "2nd";
729 std::string posSTR =
730 bank ? absl::StrFormat("%X4", pos & 0xFFFF)
731 : absl::StrFormat("%X4", (pos - kTextData2) & 0xFFFF);
732 std::string message = absl::StrFormat(
733 "There is too much text data in the %s block to save.\n"
734 "Available: %X4 | Used: %s",
735 bankSTR, space, posSTR);
736 return message;
737}
