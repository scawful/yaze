45absl::StatusOr<Bytes> Rom::Load2BppGraphics() {
46 Bytes sheet;
47 const uint8_t sheets[] = {113, 114, 218, 219, 220, 221};
48
49 for (const auto& sheet_id : sheets) {
50 auto offset = GetGraphicsAddress(data(), sheet_id,
51 version_constants().kOverworldGfxPtr1,
52 version_constants().kOverworldGfxPtr2,
53 version_constants().kOverworldGfxPtr3);
54 ASSIGN_OR_RETURN(auto decomp_sheet,
55 gfx::lc_lz2::DecompressV2(data(), offset))
56 auto converted_sheet = gfx::SnesTo8bppSheet(decomp_sheet, 2);
57 for (const auto& each_pixel : converted_sheet) {
58 sheet.push_back(each_pixel);
59 }
60 }
61 return sheet;
62}

144absl::Status Rom::LoadFromFile(const std::string& filename, bool z3_load) {
145 std::string full_filename = std::filesystem::absolute(filename).string();
146 if (full_filename.empty()) {
147 return absl::InvalidArgumentError(
148 "Could not load ROM: parameter `filename` is empty.");
149 }
150 // Set filename
151 filename_ = full_filename;
152
153 // Open file
154 std::ifstream file(filename_, std::ios::binary);
155 if (!file.is_open()) {
156 return absl::InternalError(
157 absl::StrCat("Could not open ROM file: ", filename_));
158 }
159
160 // Get file size and resize rom_data_
161 try {
162 size_ = std::filesystem::file_size(filename_);
163 } catch (const std::filesystem::filesystem_error& e) {
164 // Try to get the file size from the open file stream
165 file.seekg(0, std::ios::end);
166 if (!file) {
167 return absl::InternalError(absl::StrCat(
168 "Could not get file size: ", filename_, " - ", e.what()));
169 }
170 size_ = file.tellg();
171 }
172 rom_data_.resize(size_);
173
174 // Read file into rom_data_
175 file.read(reinterpret_cast<char*>(rom_data_.data()), size_);
176
177 // Check if the sROM has a header
178 constexpr size_t baseROMSize = 1048576; // 1MB
179 constexpr size_t headerSize = 0x200; // 512 bytes
180 if (size_ % baseROMSize == headerSize) {
181 std::cout << "ROM has a header" << std::endl;
182 has_header_ = true;
183 }
184
185 // Remove header if present
186 if (has_header_) {
187 auto header =
188 std::vector<uchar>(rom_data_.begin(), rom_data_.begin() + 0x200);
189 rom_data_.erase(rom_data_.begin(), rom_data_.begin() + 0x200);
190 size_ -= 0x200;
191 }
192
193 // Close file
194 file.close();
195
196 // Load Zelda 3 specific data if requested
197 if (z3_load) {
198 // Copy ROM title
199 constexpr uint32_t kTitleStringOffset = 0x7FC0;
200 constexpr uint32_t kTitleStringLength = 20;
201 std::copy(rom_data_.begin() + kTitleStringOffset,
202 rom_data_.begin() + kTitleStringOffset + kTitleStringLength,
203 title_.begin());
204 if (rom_data_[kTitleStringOffset + 0x19] == 0) {
205 version_ = Z3_Version::JP;
206 } else {
207 version_ = Z3_Version::US;
208 }
209 RETURN_IF_ERROR(gfx::LoadAllPalettes(rom_data_, palette_groups_));
210 LoadGfxGroups();
211 }
212
213 // Expand the ROM data to 2MB without changing the data in the first 1MB
214 rom_data_.resize(baseROMSize * 2);
215 size_ = baseROMSize * 2;
216
217 // Set up the resource labels
218 std::string resource_label_filename = absl::StrFormat("%s.labels", filename);
219 resource_label_manager_.LoadLabels(resource_label_filename);
220
221 // Set is_loaded_ flag and return success
222 is_loaded_ = true;
223 return absl::OkStatus();
224}

270absl::Status Rom::SaveToFile(bool backup, bool save_new, std::string filename) {
271 absl::Status non_firing_status;
272 if (rom_data_.empty()) {
273 return absl::InternalError("ROM data is empty.");
274 }
275
276 // Check if filename is empty
277 if (filename == "") {
278 filename = filename_;
279 }
280
281 // Check if backup is enabled
282 if (backup) {
283 // Create a backup file with timestamp in its name
284 auto now = std::chrono::system_clock::now();
285 auto now_c = std::chrono::system_clock::to_time_t(now);
286 std::string backup_filename =
287 absl::StrCat(filename, "_backup_", std::ctime(&now_c));
288
289 // Remove newline character from ctime()
290 backup_filename.erase(
291 std::remove(backup_filename.begin(), backup_filename.end(), '\n'),
292 backup_filename.end());
293
294 // Replace spaces with underscores
295 std::replace(backup_filename.begin(), backup_filename.end(), ' ', '_');
296
297 // Now, copy the original file to the backup file
298 try {
299 std::filesystem::copy(filename_, backup_filename,
300 std::filesystem::copy_options::overwrite_existing);
301 } catch (const std::filesystem::filesystem_error& e) {
302 non_firing_status = absl::InternalError(absl::StrCat(
303 "Could not create backup file: ", backup_filename, " - ", e.what()));
304 }
305 }
306
307 // Run the other save functions
308 if (flags()->kSaveAllPalettes) {
309 RETURN_IF_ERROR(SaveAllPalettes());
310 }
311
312 if (flags()->kSaveGfxGroups) {
313 SaveGroupsToRom();
314 }
315
316 if (save_new) {
317 // Create a file of the same name and append the date between the filename
318 // and file extension
319 auto now = std::chrono::system_clock::now();
320 auto now_c = std::chrono::system_clock::to_time_t(now);
321 auto filename_no_ext = filename.substr(0, filename.find_last_of("."));
322 std::cout << filename_no_ext << std::endl;
323 filename = absl::StrCat(filename_no_ext, "_", std::ctime(&now_c));
324 // Remove spaces from new_filename and replace with _
325 filename.erase(std::remove(filename.begin(), filename.end(), ' '),
326 filename.end());
327 // Remove newline character from ctime()
328 filename.erase(std::remove(filename.begin(), filename.end(), '\n'),
329 filename.end());
330 // Add the file extension back to the new_filename
331 filename = filename + ".sfc";
332 std::cout << filename << std::endl;
333 }
334
335 // Open the file that we know exists for writing
336 std::ofstream file(filename.data(), std::ios::binary | std::ios::app);
337 if (!file) {
338 // Create the file if it does not exist
339 file.open(filename.data(), std::ios::binary);
340 if (!file) {
341 return absl::InternalError(
342 absl::StrCat("Could not open or create ROM file: ", filename));
343 }
344 }
345
346 // Save the data to the file
347 try {
348 file.write(
349 static_cast<const char*>(static_cast<const void*>(rom_data_.data())),
350 rom_data_.size());
351 } catch (const std::ofstream::failure& e) {
352 return absl::InternalError(absl::StrCat(
353 "Error while writing to ROM file: ", filename, " - ", e.what()));
354 }
355
356 // Check for write errors
357 if (!file) {
358 return absl::InternalError(
359 absl::StrCat("Error while writing to ROM file: ", filename));
360 }
361
362 if (!non_firing_status.ok()) {
363 return non_firing_status;
364 }
365
366 return absl::OkStatus();
367}
