39absl::StatusOr<std::vector<uint8_t>> Rom::Load2BppGraphics() {
40 std::vector<uint8_t> sheet;
41 const uint8_t sheets[] = {113, 114, 218, 219, 220, 221};
42
43 for (const auto &sheet_id : sheets) {
44 auto offset = GetGraphicsAddress(data(), sheet_id,
45 version_constants().kOverworldGfxPtr1,
46 version_constants().kOverworldGfxPtr2,
47 version_constants().kOverworldGfxPtr3);
48 ASSIGN_OR_RETURN(auto decomp_sheet,
49 gfx::lc_lz2::DecompressV2(data(), offset))
50 auto converted_sheet = gfx::SnesTo8bppSheet(decomp_sheet, 2);
51 for (const auto &each_pixel : converted_sheet) {
52 sheet.push_back(each_pixel);
53 }
54 }
55 return sheet;

77absl::Status Rom::LoadAllGraphicsData(bool defer_render) {
78 std::vector<uint8_t> sheet;
79 bool bpp3 = false;
80
81 for (uint32_t i = 0; i < kNumGfxSheets; i++) {
82 if (i >= 115 && i <= 126) { // uncompressed sheets
83 sheet.resize(Uncompressed3BPPSize);
84 auto offset =
85 GetGraphicsAddress(data(), i, version_constants().kOverworldGfxPtr1,
86 version_constants().kOverworldGfxPtr2,
87 version_constants().kOverworldGfxPtr3);
88 for (int j = 0; j < Uncompressed3BPPSize; j++) {
89 sheet[j] = rom_data_[j + offset];
90 }
91 bpp3 = true;
92 } else if (i == 113 || i == 114 || i >= 218) {
93 bpp3 = false;
94 } else {
95 auto offset =
96 GetGraphicsAddress(data(), i, version_constants().kOverworldGfxPtr1,
97 version_constants().kOverworldGfxPtr2,
98 version_constants().kOverworldGfxPtr3);
99 ASSIGN_OR_RETURN(sheet,
100 gfx::lc_lz2::DecompressV2(rom_data_.data(), offset))
101 bpp3 = true;
102 }
103
104 if (bpp3) {
105 auto converted_sheet = gfx::SnesTo8bppSheet(sheet, 3);
106 graphics_sheets_[i].Create(gfx::kTilesheetWidth, gfx::kTilesheetHeight,
107 gfx::kTilesheetDepth, converted_sheet);
108 if (graphics_sheets_[i].is_active()) {
109 if (i > 115) {
110 // Apply sprites palette
111 RETURN_IF_ERROR(graphics_sheets_[i].ApplyPaletteWithTransparent(
112 palette_groups_.global_sprites[0], 0));
113 } else {
114 RETURN_IF_ERROR(graphics_sheets_[i].ApplyPaletteWithTransparent(
115 palette_groups_.dungeon_main[0], 0));
116 }
117 }
118
119 if (!defer_render) {
120 graphics_sheets_[i].CreateTexture(Renderer::GetInstance().renderer());
121 }
122
123 for (int j = 0; j < graphics_sheets_[i].size(); ++j) {
124 graphics_buffer_.push_back(graphics_sheets_[i].at(j));
125 }
126
127 } else {
128 for (int j = 0; j < graphics_sheets_[0].size(); ++j) {
129 graphics_buffer_.push_back(0xFF);
130 }
131 }
132 }
133 return absl::OkStatus();

136absl::Status Rom::LoadFromFile(const std::string &filename, bool z3_load) {
137 if (filename.empty()) {
138 return absl::InvalidArgumentError(
139 "Could not load ROM: parameter `filename` is empty.");
140 }
141 std::string full_filename = std::filesystem::absolute(filename).string();
142 filename_ = full_filename;
143
144 std::ifstream file(filename_, std::ios::binary);
145 if (!file.is_open()) {
146 return absl::NotFoundError(
147 absl::StrCat("Could not open ROM file: ", filename_));
148 }
149
150 // Get file size and resize rom_data_
151 try {
152 size_ = std::filesystem::file_size(filename_);
153 } catch (const std::filesystem::filesystem_error &e) {
154 // Try to get the file size from the open file stream
155 file.seekg(0, std::ios::end);
156 if (!file) {
157 return absl::InternalError(absl::StrCat(
158 "Could not get file size: ", filename_, " - ", e.what()));
159 }
160 size_ = file.tellg();
161 }
162 rom_data_.resize(size_);
163
164 // Read file into rom_data_
165 file.read(reinterpret_cast<char *>(rom_data_.data()), size_);
166 file.close();
167
168 // Set is_loaded_ flag and return success
169 is_loaded_ = true;
170
171 if (z3_load) {
172 RETURN_IF_ERROR(LoadZelda3());
173 }
174
175 // Set up the resource labels
176 std::string resource_label_filename = absl::StrFormat("%s.labels", filename);
177 resource_label_manager_.LoadLabels(resource_label_filename);
178 return absl::OkStatus();
179}

199absl::Status Rom::LoadZelda3() {
200 // Check if the ROM has a header
201 constexpr size_t kBaseRomSize = 1048576; // 1MB
202 constexpr size_t kHeaderSize = 0x200; // 512 bytes
203 if (size_ % kBaseRomSize == kHeaderSize) {
204 auto header =
205 std::vector<uchar>(rom_data_.begin(), rom_data_.begin() + kHeaderSize);
206 rom_data_.erase(rom_data_.begin(), rom_data_.begin() + kHeaderSize);
207 size_ -= 0x200;
208 }
209
210 // Copy ROM title
211 constexpr uint32_t kTitleStringOffset = 0x7FC0;
212 constexpr uint32_t kTitleStringLength = 20;
213 std::copy(rom_data_.begin() + kTitleStringOffset,
214 rom_data_.begin() + kTitleStringOffset + kTitleStringLength,
215 title_.begin());
216 if (rom_data_[kTitleStringOffset + 0x19] == 0) {
217 version_ = Z3_Version::JP;
218 } else {
219 version_ = Z3_Version::US;
220 }
221
222 // Load additional resources
223 RETURN_IF_ERROR(gfx::LoadAllPalettes(rom_data_, palette_groups_));
224 // TODO Load gfx groups or expanded ZS values
225 RETURN_IF_ERROR(LoadGfxGroups());
226
227 // Expand the ROM data to 2MB without changing the data in the first 1MB
228 rom_data_.resize(kBaseRomSize * 2);
229 size_ = kBaseRomSize * 2;
230
231 return absl::OkStatus();
232}

245absl::Status Rom::SaveToFile(bool backup, bool save_new, std::string filename) {
246 absl::Status non_firing_status;
247 if (rom_data_.empty()) {
248 return absl::InternalError("ROM data is empty.");
249 }
250
251 // Check if filename is empty
252 if (filename == "") {
253 filename = filename_;
254 }
255
256 // Check if backup is enabled
257 if (backup) {
258 // Create a backup file with timestamp in its name
259 auto now = std::chrono::system_clock::now();
260 auto now_c = std::chrono::system_clock::to_time_t(now);
261 std::string backup_filename =
262 absl::StrCat(filename, "_backup_", std::ctime(&now_c));
263
264 // Remove newline character from ctime()
265 backup_filename.erase(
266 std::remove(backup_filename.begin(), backup_filename.end(), '\n'),
267 backup_filename.end());
268
269 // Replace spaces with underscores
270 std::replace(backup_filename.begin(), backup_filename.end(), ' ', '_');
271
272 // Now, copy the original file to the backup file
273 try {
274 std::filesystem::copy(filename_, backup_filename,
275 std::filesystem::copy_options::overwrite_existing);
276 } catch (const std::filesystem::filesystem_error &e) {
277 non_firing_status = absl::InternalError(absl::StrCat(
278 "Could not create backup file: ", backup_filename, " - ", e.what()));
279 }
280 }
281
282 // Run the other save functions
283 if (flags()->kSaveAllPalettes) {
284 RETURN_IF_ERROR(SaveAllPalettes());
285 }
286
287 if (flags()->kSaveGfxGroups) {
288 RETURN_IF_ERROR(SaveGroupsToRom());
289 }
290
291 if (save_new) {
292 // Create a file of the same name and append the date between the filename
293 // and file extension
294 auto now = std::chrono::system_clock::now();
295 auto now_c = std::chrono::system_clock::to_time_t(now);
296 auto filename_no_ext = filename.substr(0, filename.find_last_of("."));
297 std::cout << filename_no_ext << std::endl;
298 filename = absl::StrCat(filename_no_ext, "_", std::ctime(&now_c));
299 // Remove spaces from new_filename and replace with _
300 filename.erase(std::remove(filename.begin(), filename.end(), ' '),
301 filename.end());
302 // Remove newline character from ctime()
303 filename.erase(std::remove(filename.begin(), filename.end(), '\n'),
304 filename.end());
305 // Add the file extension back to the new_filename
306 filename = filename + ".sfc";
307 std::cout << filename << std::endl;
308 }
309
310 // Open the file that we know exists for writing
311 std::ofstream file(filename.data(), std::ios::binary | std::ios::app);
312 if (!file) {
313 // Create the file if it does not exist
314 file.open(filename.data(), std::ios::binary);
315 if (!file) {
316 return absl::InternalError(
317 absl::StrCat("Could not open or create ROM file: ", filename));
318 }
319 }
320
321 // Save the data to the file
322 try {
323 file.write(
324 static_cast<const char *>(static_cast<const void *>(rom_data_.data())),
325 rom_data_.size());
326 } catch (const std::ofstream::failure &e) {
327 return absl::InternalError(absl::StrCat(
328 "Error while writing to ROM file: ", filename, " - ", e.what()));
329 }
330
331 // Check for write errors
332 if (!file) {
333 return absl::InternalError(
334 absl::StrCat("Error while writing to ROM file: ", filename));
335 }
336
337 if (!non_firing_status.ok()) {
338 return non_firing_status;
339 }
340
341 return absl::OkStatus();
342}
