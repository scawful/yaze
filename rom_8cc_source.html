38absl::StatusOr<std::vector<uint8_t>> Load2BppGraphics(const Rom &rom) {
39 std::vector<uint8_t> sheet;
40 const uint8_t sheets[] = {113, 114, 218, 219, 220, 221};
41
42 for (const auto &sheet_id : sheets) {
43 auto offset = GetGraphicsAddress(rom.data(), sheet_id,
44 rom.version_constants().kOverworldGfxPtr1,
45 rom.version_constants().kOverworldGfxPtr2,
46 rom.version_constants().kOverworldGfxPtr3);
47 ASSIGN_OR_RETURN(auto decomp_sheet,
48 gfx::lc_lz2::DecompressV2(rom.data(), offset))
49 auto converted_sheet = gfx::SnesTo8bppSheet(decomp_sheet, 2);
50 for (const auto &each_pixel : converted_sheet) {
51 sheet.push_back(each_pixel);
52 }
53 }
54 return sheet;
55}

77absl::StatusOr<std::array<gfx::Bitmap, kNumGfxSheets>> LoadAllGraphicsData(
78 Rom &rom, bool defer_render) {
79 std::array<gfx::Bitmap, kNumGfxSheets> graphics_sheets;
80 std::vector<uint8_t> sheet;
81 bool bpp3 = false;
82
83 for (uint32_t i = 0; i < kNumGfxSheets; i++) {
84 if (i >= 115 && i <= 126) { // uncompressed sheets
85 sheet.resize(Uncompressed3BPPSize);
86 auto offset = GetGraphicsAddress(
87 rom.data(), i, rom.version_constants().kOverworldGfxPtr1,
88 rom.version_constants().kOverworldGfxPtr2,
89 rom.version_constants().kOverworldGfxPtr3);
90 std::copy(rom.data() + offset, rom.data() + offset + Uncompressed3BPPSize,

92 bpp3 = true;
93 } else if (i == 113 || i == 114 || i >= 218) {
94 bpp3 = false;
95 } else {

98 rom.version_constants().kOverworldGfxPtr2,
99 rom.version_constants().kOverworldGfxPtr3);
100 ASSIGN_OR_RETURN(sheet, gfx::lc_lz2::DecompressV2(rom.data(), offset))
101 bpp3 = true;
102 }
103
104 if (bpp3) {
105 auto converted_sheet = gfx::SnesTo8bppSheet(sheet, 3);
106 graphics_sheets[i].Create(gfx::kTilesheetWidth, gfx::kTilesheetHeight,
107 gfx::kTilesheetDepth, converted_sheet);
108 if (graphics_sheets[i].is_active()) {
109 if (i > 115) {
110 // Apply sprites palette

113 } else {
114 RETURN_IF_ERROR(graphics_sheets[i].ApplyPaletteWithTransparent(
115 rom.palette_group().dungeon_main[0], 0));

117 }
118

120 graphics_sheets[i].CreateTexture(Renderer::GetInstance().renderer());
121 }
122
123 for (int j = 0; j < graphics_sheets[i].size(); ++j) {
124 rom.mutable_graphics_buffer()->push_back(graphics_sheets[i].at(j));
125 }
126
127 } else {
128 for (int j = 0; j < graphics_sheets[0].size(); ++j) {
129 rom.mutable_graphics_buffer()->push_back(0xFF);
130 }
131 }
132 }

172absl::Status Rom::LoadFromFile(const std::string &filename, bool z3_load) {
173 if (filename.empty()) {
174 return absl::InvalidArgumentError(
175 "Could not load ROM: parameter `filename` is empty.");
176 }
177 std::string full_filename = std::filesystem::absolute(filename).string();
178 filename_ = full_filename;
179
180 std::ifstream file(filename_, std::ios::binary);
181 if (!file.is_open()) {
182 return absl::NotFoundError(
183 absl::StrCat("Could not open ROM file: ", filename_));
184 }
185
186 // Get file size and resize rom_data_
187 try {
188 size_ = std::filesystem::file_size(filename_);
189 } catch (const std::filesystem::filesystem_error &e) {
190 // Try to get the file size from the open file stream
191 file.seekg(0, std::ios::end);
192 if (!file) {
193 return absl::InternalError(absl::StrCat(
194 "Could not get file size: ", filename_, " - ", e.what()));
195 }
196 size_ = file.tellg();
197 }
198 rom_data_.resize(size_);
199
200 // Read file into rom_data_
201 file.read(reinterpret_cast<char *>(rom_data_.data()), size_);
202 file.close();
203
204 // Set is_loaded_ flag and return success
205 is_loaded_ = true;
206
207 if (z3_load) {
208 RETURN_IF_ERROR(LoadZelda3());
209 }
210
211 // Set up the resource labels
212 std::string resource_label_filename = absl::StrFormat("%s.labels", filename);
213 resource_label_manager_.LoadLabels(resource_label_filename);
214 return absl::OkStatus();
215}

246absl::Status Rom::LoadZelda3() {
247 // Check if the ROM has a header
248 constexpr size_t kBaseRomSize = 1048576; // 1MB
249 constexpr size_t kHeaderSize = 0x200; // 512 bytes
250 if (size_ % kBaseRomSize == kHeaderSize) {
251 auto header =
252 std::vector<uchar>(rom_data_.begin(), rom_data_.begin() + kHeaderSize);
253 rom_data_.erase(rom_data_.begin(), rom_data_.begin() + kHeaderSize);
254 size_ -= 0x200;
255 }
256
257 // Copy ROM title
258 constexpr uint32_t kTitleStringOffset = 0x7FC0;
259 constexpr uint32_t kTitleStringLength = 20;
260 title_.resize(kTitleStringLength);
261 std::copy(rom_data_.begin() + kTitleStringOffset,
262 rom_data_.begin() + kTitleStringOffset + kTitleStringLength,
263 title_.begin());
264 if (rom_data_[kTitleStringOffset + 0x19] == 0) {
265 version_ = Z3_Version::JP;
266 } else {
267 version_ = Z3_Version::US;
268 }
269
270 // Load additional resources
271 RETURN_IF_ERROR(gfx::LoadAllPalettes(rom_data_, palette_groups_));
272 // TODO Load gfx groups or expanded ZS values
273 RETURN_IF_ERROR(LoadGfxGroups());
274
275 // Expand the ROM data to 2MB without changing the data in the first 1MB
276 rom_data_.resize(kBaseRomSize * 2);
277 size_ = kBaseRomSize * 2;
278
279 return absl::OkStatus();
280}

282absl::Status Rom::SaveToFile(bool backup, bool save_new, std::string filename) {
283 absl::Status non_firing_status;
284 if (rom_data_.empty()) {
285 return absl::InternalError("ROM data is empty.");
286 }
287
288 // Check if filename is empty
289 if (filename == "") {
290 filename = filename_;
291 }
292
293 // Check if backup is enabled
294 if (backup) {
295 // Create a backup file with timestamp in its name
296 auto now = std::chrono::system_clock::now();
297 auto now_c = std::chrono::system_clock::to_time_t(now);
298 std::string backup_filename =
299 absl::StrCat(filename, "_backup_", std::ctime(&now_c));
300
301 // Remove newline character from ctime()
302 backup_filename.erase(
303 std::remove(backup_filename.begin(), backup_filename.end(), '\n'),
304 backup_filename.end());
305
306 // Replace spaces with underscores
307 std::replace(backup_filename.begin(), backup_filename.end(), ' ', '_');
308
309 // Now, copy the original file to the backup file
310 try {
311 std::filesystem::copy(filename_, backup_filename,
312 std::filesystem::copy_options::overwrite_existing);
313 } catch (const std::filesystem::filesystem_error &e) {
314 non_firing_status = absl::InternalError(absl::StrCat(
315 "Could not create backup file: ", backup_filename, " - ", e.what()));
316 }
317 }
318
319 // Run the other save functions
320 if (core::ExperimentFlags::get().kSaveAllPalettes)
321 RETURN_IF_ERROR(SaveAllPalettes());
322 if (core::ExperimentFlags::get().kSaveGfxGroups)
323 RETURN_IF_ERROR(SaveGroupsToRom());
324
325 if (save_new) {
326 // Create a file of the same name and append the date between the filename
327 // and file extension
328 auto now = std::chrono::system_clock::now();
329 auto now_c = std::chrono::system_clock::to_time_t(now);
330 auto filename_no_ext = filename.substr(0, filename.find_last_of("."));
331 std::cout << filename_no_ext << std::endl;
332 filename = absl::StrCat(filename_no_ext, "_", std::ctime(&now_c));
333 // Remove spaces from new_filename and replace with _
334 filename.erase(std::remove(filename.begin(), filename.end(), ' '),
335 filename.end());
336 // Remove newline character from ctime()
337 filename.erase(std::remove(filename.begin(), filename.end(), '\n'),
338 filename.end());
339 // Add the file extension back to the new_filename
340 filename = filename + ".sfc";
341 std::cout << filename << std::endl;
342 }
343
344 // Open the file that we know exists for writing
345 std::ofstream file(filename.data(), std::ios::binary | std::ios::app);
346 if (!file) {
347 // Create the file if it does not exist
348 file.open(filename.data(), std::ios::binary);
349 if (!file) {
350 return absl::InternalError(
351 absl::StrCat("Could not open or create ROM file: ", filename));
352 }
353 }
354
355 // Save the data to the file
356 try {
357 file.write(
358 static_cast<const char *>(static_cast<const void *>(rom_data_.data())),
359 rom_data_.size());
360 } catch (const std::ofstream::failure &e) {
361 return absl::InternalError(absl::StrCat(
362 "Error while writing to ROM file: ", filename, " - ", e.what()));
363 }
364
365 // Check for write errors
366 if (!file) {
367 return absl::InternalError(
368 absl::StrCat("Error while writing to ROM file: ", filename));
369 }
370
371 if (!non_firing_status.ok()) {
372 return non_firing_status;
373 }
374
375 return absl::OkStatus();
376}
