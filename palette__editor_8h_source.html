33 // Check size and remove the oldest if necessary
34 if (recentChanges.size() >= maxHistorySize) {
35 recentChanges.pop_front();

38 // Push the new change

42
43 // Get recent changes for display in the palette editor

48 // Restore the original color
49 gfx::SnesColor RestoreOriginalColor(const std::string& groupName,
50 size_t paletteIndex,
51 size_t colorIndex) const {
52 for (const auto& change : recentChanges) {
53 if (change.group_name == groupName &&
54 change.palette_index == paletteIndex &&
55 change.color_index == colorIndex) {
56 return change.original_color;
57 }
58 }
59 // Handle error or return default (this is just an example,
60 // handle as appropriate for your application)
61 return gfx::SnesColor();
62 }
63
64 auto size() const { return recentChanges.size(); }
65
66 gfx::SnesColor& GetModifiedColor(size_t index) {
67 return recentChanges[index].new_color;

71 }
