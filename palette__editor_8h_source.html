36class PaletteEditorHistory {
37 public:
38 // Record a change in the palette editor
39 void RecordChange(const std::string& groupName, size_t paletteIndex,
40 size_t colorIndex, const gfx::SnesColor& originalColor,
41 const gfx::SnesColor& newColor) {
42 // Check size and remove the oldest if necessary
43 if (recentChanges.size() >= maxHistorySize) {
44 recentChanges.pop_front();
45 }
46
47 // Push the new change
48 recentChanges.push_back(
49 {groupName, paletteIndex, colorIndex, originalColor, newColor});
50 }
51
52 // Get recent changes for display in the palette editor
53 const std::deque<PaletteChange>& GetRecentChanges() const {
54 return recentChanges;
55 }
56
57 // Restore the original color
58 gfx::SnesColor GetOriginalColor(const std::string& groupName,
59 size_t paletteIndex,
60 size_t colorIndex) const {
61 for (const auto& change : recentChanges) {
62 if (change.group_name == groupName &&
63 change.palette_index == paletteIndex &&
64 change.color_index == colorIndex) {
65 return change.original_color;
66 }
67 }
68 // Handle error or return default (this is just an example,
69 // handle as appropriate for your application)
70 return gfx::SnesColor();
71 }
72
73 private:
74 std::deque<PaletteChange> recentChanges;
75 static const size_t maxHistorySize = 50; // or any other number you deem fit
76};
