43 // Check size and remove the oldest if necessary
44 if (recentChanges.size() >= maxHistorySize) {
45 recentChanges.pop_front();

47
48 // Push the new change
49 recentChanges.push_back(

53 // Get recent changes for display in the palette editor

56 }
57
58 // Restore the original color
59 gfx::SnesColor GetOriginalColor(const std::string& groupName,
60 size_t paletteIndex,
61 size_t colorIndex) const {
62 for (const auto& change : recentChanges) {
63 if (change.group_name == groupName &&
64 change.palette_index == paletteIndex &&
65 change.color_index == colorIndex) {
66 return change.original_color;
67 }
68 }
69 // Handle error or return default (this is just an example,
70 // handle as appropriate for your application)
71 return gfx::SnesColor();
72 }
