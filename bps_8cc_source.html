81void encode(uint64_t data, std::vector<uint8_t> &output) {
82 while (true) {
83 uint8_t x = data & 0x7f;
84 data >>= 7;
85 if (data == 0) {
86 output.push_back(0x80 | x);
87 break;
88 }
89 output.push_back(x);
90 data--;
91 }
92}

94uint64_t decode(const std::vector<uint8_t> &input, size_t &offset) {
95 uint64_t data = 0;
96 uint64_t shift = 1;
97 while (true) {
98 uint8_t x = input[offset++];
99 data += (x & 0x7f) * shift;
100 if (x & 0x80) break;
101 shift <<= 7;
102 data += shift;
103 }
104 return data;
105}

109void CreateBpsPatch(const std::vector<uint8_t> &source,
110 const std::vector<uint8_t> &target,
111 std::vector<uint8_t> &patch) {
112 patch.clear();
113 patch.insert(patch.end(), {'B', 'P', 'S', '1'});
114
115 encode(source.size(), patch);
116 encode(target.size(), patch);
117 encode(0, patch); // No metadata
118
119 size_t source_offset = 0;
120 size_t target_offset = 0;
121 int64_t source_rel_offset = 0;
122 int64_t target_rel_offset = 0;
123
124 while (target_offset < target.size()) {
125 if (source_offset < source.size() &&
126 source[source_offset] == target[target_offset]) {
127 size_t length = 0;
128 while (source_offset + length < source.size() &&
129 target_offset + length < target.size() &&
130 source[source_offset + length] == target[target_offset + length]) {
131 length++;
132 }
133 encode((length - 1) << 2 | 0, patch); // SourceRead
134 source_offset += length;
135 target_offset += length;
136 } else {
137 size_t length = 0;
138 while (
139 target_offset + length < target.size() &&
140 (source_offset + length >= source.size() ||
141 source[source_offset + length] != target[target_offset + length])) {
142 length++;
143 }
144 if (length > 0) {
145 encode((length - 1) << 2 | 1, patch); // TargetRead
146 for (size_t i = 0; i < length; i++) {
147 patch.push_back(target[target_offset + i]);
148 }
149 target_offset += length;
150 }
151 }
152
153 // SourceCopy
154 if (source_offset < source.size()) {
155 size_t length = 0;
156 int64_t offset = source_offset - source_rel_offset;
157 while (source_offset + length < source.size() &&
158 target_offset + length < target.size() &&
159 source[source_offset + length] == target[target_offset + length]) {
160 length++;
161 }
162 if (length > 0) {
163 encode((length - 1) << 2 | 2, patch);
164 encode((offset < 0 ? 1 : 0) | (abs(offset) << 1), patch);
165 source_offset += length;
166 target_offset += length;
167 source_rel_offset = source_offset;
168 }
169 }
170
171 // TargetCopy
172 if (target_offset > 0) {
173 size_t length = 0;
174 int64_t offset = target_offset - target_rel_offset;
175 while (target_offset + length < target.size() &&
176 target[target_offset - 1] == target[target_offset + length]) {
177 length++;
178 }
179 if (length > 0) {
180 encode((length - 1) << 2 | 3, patch);
181 encode((offset < 0 ? 1 : 0) | (abs(offset) << 1), patch);
182 target_offset += length;
183 target_rel_offset = target_offset;
184 }
185 }
186 }
187
188 patch.resize(patch.size() + 12); // Make space for the checksums
189 uint32_t source_checksum = crc32(source);
190 uint32_t target_checksum = crc32(target);
191 uint32_t patch_checksum = crc32(patch);
192
193 memcpy(patch.data() + patch.size() - 12, &source_checksum, sizeof(uint32_t));
194 memcpy(patch.data() + patch.size() - 8, &target_checksum, sizeof(uint32_t));
195 memcpy(patch.data() + patch.size() - 4, &patch_checksum, sizeof(uint32_t));
196}

198void ApplyBpsPatch(const std::vector<uint8_t> &source,
199 const std::vector<uint8_t> &patch,
200 std::vector<uint8_t> &target) {
201 if (patch.size() < 4 || patch[0] != 'B' || patch[1] != 'P' ||
202 patch[2] != 'S' || patch[3] != '1') {
203 throw std::runtime_error("Invalid patch format");
204 }
205
206 size_t patch_offset = 4;
207 uint64_t target_size = decode(patch, patch_offset);
208 uint64_t metadata_size = decode(patch, patch_offset);
209 patch_offset += metadata_size;
210
211 target.resize(target_size);
212 size_t source_offset = 0;
213 size_t target_offset = 0;
214 int64_t source_rel_offset = 0;
215 int64_t target_rel_offset = 0;
216
217 while (patch_offset < patch.size() - 12) {
218 uint64_t data = decode(patch, patch_offset);
219 uint64_t command = data & 3;
220 uint64_t length = (data >> 2) + 1;
221
222 switch (command) {
223 case 0: // SourceRead
224 while (length--) {
225 target[target_offset++] = source[source_offset++];
226 }
227 break;
228 case 1: // TargetRead
229 while (length--) {
230 target[target_offset++] = patch[patch_offset++];
231 }
232 break;
233 case 2: // SourceCopy
234 {
235 int64_t offsetData = decode(patch, patch_offset);
236 source_rel_offset += (offsetData & 1 ? -1 : +1) * (offsetData >> 1);
237 while (length--) {
238 target[target_offset++] = source[source_rel_offset++];
239 }
240 } break;
241 case 3: // TargetCopy
242 {
243 uint64_t offsetData = decode(patch, patch_offset);
244 target_rel_offset += (offsetData & 1 ? -1 : +1) * (offsetData >> 1);
245 while (length--) {
246 target[target_offset++] = target[target_rel_offset++];
247 }
248 }
249 default:
250 throw std::runtime_error("Invalid patch command");
251 }
252 }
253
254 uint32_t source_checksum;
255 uint32_t target_checksum;
256 uint32_t patch_checksum;
257 memcpy(&source_checksum, patch.data() + patch.size() - 12, sizeof(uint32_t));
258 memcpy(&target_checksum, patch.data() + patch.size() - 8, sizeof(uint32_t));
259 memcpy(&patch_checksum, patch.data() + patch.size() - 4, sizeof(uint32_t));
260
261 if (source_checksum != crc32(source) || target_checksum != crc32(target) ||
262 patch_checksum !=
263 crc32(std::vector<uint8_t>(patch.begin(), patch.end() - 4))) {
264 throw std::runtime_error("Checksum mismatch");
265 }
266}
