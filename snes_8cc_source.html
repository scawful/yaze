116void SNES::RunCycle() {
117 cycles_ += 2;
118
119 // check for h/v timer irq's
120 bool condition = ((v_irq_enabled_ || h_irq_enabled_) &&
121 (memory_.v_pos() == v_timer_ || !v_irq_enabled_) &&
122 (memory_.h_pos() == h_timer_ || !h_irq_enabled_));
123
124 if (!irq_condition_ && condition) {
125 in_irq_ = true;
126 cpu_.SetIrq(true);
127 }
128 irq_condition_ = condition;
129
130 // increment position; must come after irq checks! (hagane, cybernator)
131 memory_.set_h_pos(memory_.h_pos() + 2);
132
133 // handle positional stuff
134 if (memory_.h_pos() == next_horiz_event) {
135 switch (memory_.h_pos()) {
136 case 16: {
137 next_horiz_event = 512;
138 if (memory_.v_pos() == 0) memory_.init_hdma_request();
139 } break;
140 case 512: {
141 next_horiz_event = 1104;
142 // render the line halfway of the screen for better compatibility
143 if (!in_vblank_ && memory_.v_pos() > 0) ppu_.RunLine(memory_.v_pos());
144 } break;
145 case 1104: {
146 if (!in_vblank_) memory_.run_hdma_request();
147 if (!memory_.pal_timing()) {
148 // line 240 of odd frame with no interlace is 4 cycles shorter
149 next_horiz_event = (memory_.v_pos() == 240 && !ppu_.even_frame &&
150 !ppu_.frame_interlace)
151 ? 1360
152 : 1364;
153 } else {
154 // line 311 of odd frame with interlace is 4 cycles longer
155 next_horiz_event = (memory_.v_pos() != 311 || ppu_.even_frame ||
156 !ppu_.frame_interlace)
157 ? 1364
158 : 1368;
159 }
160 } break;
161 case 1360:
162 case 1364:
163 case 1368: { // this is the end (of the h-line)
164 next_horiz_event = 16;
165
166 memory_.set_h_pos(0);
167 memory_.set_v_pos(memory_.v_pos() + 1);
168 if (!memory_.pal_timing()) {
169 // even interlace frame is 263 lines
170 if ((memory_.v_pos() == 262 &&
171 (!ppu_.frame_interlace || !ppu_.even_frame)) ||
172 memory_.v_pos() == 263) {
173 memory_.set_v_pos(0);
174 frames_++;
175 }
176 } else {
177 // even interlace frame is 313 lines
178 if ((memory_.v_pos() == 312 &&
179 (!ppu_.frame_interlace || !ppu_.even_frame)) ||
180 memory_.v_pos() == 313) {
181 memory_.set_v_pos(0);
182 frames_++;
183 }
184 }
185
186 // end of hblank, do most memory_.v_pos()-tests
187 bool starting_vblank = false;
188 if (memory_.v_pos() == 0) {
189 // end of vblank
190 in_vblank_ = false;
191 in_nmi_ = false;
192 ppu_.HandleFrameStart();
193 } else if (memory_.v_pos() == 225) {
194 // ask the ppu if we start vblank now or at memory_.v_pos() 240
195 // (overscan)
196 starting_vblank = !ppu_.CheckOverscan();
197 } else if (memory_.v_pos() == 240) {
198 // if we are not yet in vblank, we had an overscan frame, set
199 // starting_vblank
200 if (!in_vblank_) starting_vblank = true;
201 }
202 if (starting_vblank) {
203 // catch up the apu at end of emulated frame (we end frame @ start of
204 // vblank)
205 CatchUpApu();
206 // notify dsp of frame-end, because sometimes dma will extend much
207 // further past vblank (or even into the next frame) Megaman X2
208 // (titlescreen animation), Tales of Phantasia (game demo), Actraiser
209 // 2 (fade-in @ bootup)
210 apu_.dsp().NewFrame();
211 // we are starting vblank
212 ppu_.HandleVblank();
213 in_vblank_ = true;
214 in_nmi_ = true;
215 if (auto_joy_read_) {
216 // TODO: this starts a little after start of vblank
217 auto_joy_timer_ = 4224;
218 HandleInput();
219 }
220 if (nmi_enabled_) {
221 cpu_.Nmi();
222 }
223 }
224 } break;
225 }
226 }
227 // handle auto_joy_read_-timer
228 if (auto_joy_timer_ > 0) auto_joy_timer_ -= 2;
229}

268uint8_t SNES::ReadReg(uint16_t adr) {
269 switch (adr) {
270 case 0x4210: {
271 uint8_t val = 0x2; // CPU version (4 bit)
272 val |= in_nmi_ << 7;
273 in_nmi_ = false;
274 return val | (memory_.open_bus() & 0x70);
275 }
276 case 0x4211: {
277 uint8_t val = in_irq_ << 7;
278 in_irq_ = false;
279 cpu_.SetIrq(false);
280 return val | (memory_.open_bus() & 0x7f);
281 }
282 case 0x4212: {
283 uint8_t val = (auto_joy_timer_ > 0);
284 val |= (memory_.h_pos() < 4 || memory_.h_pos() >= 1096) << 6;
285 val |= in_vblank_ << 7;
286 return val | (memory_.open_bus() & 0x3e);
287 }
288 case 0x4213: {
289 return ppu_latch_ << 7; // IO-port
290 }
291 case 0x4214: {
292 return divide_result_ & 0xff;
293 }
294 case 0x4215: {
295 return divide_result_ >> 8;
296 }
297 case 0x4216: {
298 return multiply_result_ & 0xff;
299 }
300 case 0x4217: {
301 return multiply_result_ >> 8;
302 }
303 case 0x4218:
304 case 0x421a:
305 case 0x421c:
306 case 0x421e: {
307 return port_auto_read_[(adr - 0x4218) / 2] & 0xff;
308 }
309 case 0x4219:
310 case 0x421b:
311 case 0x421d:
312 case 0x421f: {
313 return port_auto_read_[(adr - 0x4219) / 2] >> 8;
314 }
315 default: {
316 return memory_.open_bus();
317 }
318 }

321uint8_t SNES::Rread(uint32_t adr) {
322 uint8_t bank = adr >> 16;
323 adr &= 0xffff;
324 if (bank == 0x7e || bank == 0x7f) {
325 return ram[((bank & 1) << 16) | adr]; // ram
326 }
327 if (bank < 0x40 || (bank >= 0x80 && bank < 0xc0)) {
328 if (adr < 0x2000) {
329 return ram[adr]; // ram mirror
330 }
331 if (adr >= 0x2100 && adr < 0x2200) {
332 return ReadBBus(adr & 0xff); // B-bus
333 }
334 if (adr == 0x4016) {
335 return input_read(&input1) | (memory_.open_bus() & 0xfc);
336 }
337 if (adr == 0x4017) {
338 return input_read(&input2) | (memory_.open_bus() & 0xe0) | 0x1c;
339 }
340 if (adr >= 0x4200 && adr < 0x4220) {
341 return ReadReg(adr); // internal registers
342 }
343 if (adr >= 0x4300 && adr < 0x4380) {
344 return memory::dma::Read(&memory_, adr); // dma registers
345 }
346 }
347 // read from cart
348 return memory_.cart_read(bank, adr);
349}

357void SNES::WriteBBus(uint8_t adr, uint8_t val) {
358 if (adr < 0x40) {
359 ppu_.Write(adr, val);
360 return;
361 }
362 if (adr < 0x80) {
363 CatchUpApu(); // catch up the apu before writing
364 apu_.in_ports_[adr & 0x3] = val;
365 return;
366 }
367 switch (adr) {
368 case 0x80: {
369 ram[ram_adr_++] = val;
370 ram_adr_ &= 0x1ffff;
371 break;
372 }
373 case 0x81: {
374 ram_adr_ = (ram_adr_ & 0x1ff00) | val;
375 break;
376 }
377 case 0x82: {
378 ram_adr_ = (ram_adr_ & 0x100ff) | (val << 8);
379 break;
380 }
381 case 0x83: {
382 ram_adr_ = (ram_adr_ & 0x0ffff) | ((val & 1) << 16);
383 break;
384 }
385 }

388void SNES::WriteReg(uint16_t adr, uint8_t val) {
389 switch (adr) {
390 case 0x4200: {
391 auto_joy_read_ = val & 0x1;
392 if (!auto_joy_read_) auto_joy_timer_ = 0;
393 h_irq_enabled_ = val & 0x10;
394 v_irq_enabled_ = val & 0x20;
395 if (!h_irq_enabled_ && !v_irq_enabled_) {
396 in_irq_ = false;
397 cpu_.SetIrq(false);
398 }
399 // if nmi is enabled while in_nmi_ is still set, immediately generate nmi
400 if (!nmi_enabled_ && (val & 0x80) && in_nmi_) {
401 cpu_.Nmi();
402 }
403 nmi_enabled_ = val & 0x80;
404 cpu_.set_int_delay(true);
405 break;
406 }
407 case 0x4201: {
408 if (!(val & 0x80) && ppu_latch_) {
409 // latch the ppu
410 ppu_.LatchHV();
411 }
412 ppu_latch_ = val & 0x80;
413 break;
414 }
415 case 0x4202: {
416 multiply_a_ = val;
417 break;
418 }
419 case 0x4203: {
420 multiply_result_ = multiply_a_ * val;
421 break;
422 }
423 case 0x4204: {
424 divide_a_ = (divide_a_ & 0xff00) | val;
425 break;
426 }
427 case 0x4205: {
428 divide_a_ = (divide_a_ & 0x00ff) | (val << 8);
429 break;
430 }
431 case 0x4206: {
432 if (val == 0) {
433 divide_result_ = 0xffff;
434 multiply_result_ = divide_a_;
435 } else {
436 divide_result_ = divide_a_ / val;
437 multiply_result_ = divide_a_ % val;
438 }
439 break;
440 }
441 case 0x4207: {
442 h_timer_ = (h_timer_ & 0x100) | val;
443 break;
444 }
445 case 0x4208: {
446 h_timer_ = (h_timer_ & 0x0ff) | ((val & 1) << 8);
447 break;
448 }
449 case 0x4209: {
450 v_timer_ = (v_timer_ & 0x100) | val;
451 break;
452 }
453 case 0x420a: {
454 v_timer_ = (v_timer_ & 0x0ff) | ((val & 1) << 8);
455 break;
456 }
457 case 0x420b: {
458 memory::dma::StartDma(&memory_, val, false);
459 break;
460 }
461 case 0x420c: {
462 memory::dma::StartDma(&memory_, val, true);
463 break;
464 }
465 case 0x420d: {
466 fast_mem_ = val & 0x1;
467 break;
468 }
469 default: {
470 break;
471 }
472 }

475void SNES::Write(uint32_t adr, uint8_t val) {
476 memory_.set_open_bus(val);
477 uint8_t bank = adr >> 16;
478 adr &= 0xffff;
479 if (bank == 0x7e || bank == 0x7f) {
480 ram[((bank & 1) << 16) | adr] = val; // ram
481 }
482 if (bank < 0x40 || (bank >= 0x80 && bank < 0xc0)) {
483 if (adr < 0x2000) {
484 ram[adr] = val; // ram mirror
485 }
486 if (adr >= 0x2100 && adr < 0x2200) {
487 WriteBBus(adr & 0xff, val); // B-bus
488 }
489 if (adr == 0x4016) {
490 input_latch(&input1, val & 1); // input latch
491 input_latch(&input2, val & 1);
492 }
493 if (adr >= 0x4200 && adr < 0x4220) {
494 WriteReg(adr, val); // internal registers
495 }
496 if (adr >= 0x4300 && adr < 0x4380) {
497 memory::dma::Write(&memory_, adr, val); // dma registers
498 }
499 }
500
501 // write to cart
502 memory_.cart_write(bank, adr, val);
503}

505int SNES::GetAccessTime(uint32_t adr) {
506 uint8_t bank = adr >> 16;
507 adr &= 0xffff;
508 if ((bank < 0x40 || (bank >= 0x80 && bank < 0xc0)) && adr < 0x8000) {
509 // 00-3f,80-bf:0-7fff
510 if (adr < 0x2000 || adr >= 0x6000) return 8; // 0-1fff, 6000-7fff
511 if (adr < 0x4000 || adr >= 0x4200) return 6; // 2000-3fff, 4200-5fff
512 return 12; // 4000-41ff
513 }
514 // 40-7f,co-ff:0000-ffff, 00-3f,80-bf:8000-ffff
515 return (fast_mem_ && bank >= 0x80) ? 6
516 : 8; // depends on setting in banks 80+
517}
