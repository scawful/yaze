221 void AddFile(const std::string& file_path) {
222 // Add a file to the list, avoiding duplicates
223 auto it = std::find(recent_files_.begin(), recent_files_.end(), file_path);
224 if (it == recent_files_.end()) {
225 recent_files_.push_back(file_path);
226 }
227 }
228
229 void Save() {
230 std::ofstream file(filename_);
231 if (!file.is_open()) {
232 return; // Handle the error appropriately
233 }
234
235 for (const auto& file_path : recent_files_) {
236 file << file_path << std::endl;
237 }
238 }
239
240 void Load() {
241 std::ifstream file(filename_);
242 if (!file.is_open()) {
243 return;
244 }
245
246 recent_files_.clear();
247 std::string line;
248 while (std::getline(file, line)) {
249 if (!line.empty()) {
250 recent_files_.push_back(line);
251 }
252 }
253 }
254
255 const std::vector<std::string>& GetRecentFiles() const {
256 return recent_files_;
257 }

259 private:
260 std::string filename_;
261 std::vector<std::string> recent_files_;
