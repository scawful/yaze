158

161

163 SavePaletteBackup(bitmap->palette());
164
165 // Apply new palette
166 if (palette_index >= 0 && palette_index < 8) {
167 bitmap->SetPaletteWithTransparent(selected_palette, palette_index);
168 } else {
169 bitmap->SetPalette(selected_palette);
170 }

173
174 current_group_index_ = group_index;
175 current_palette_index_ = palette_index;
176
177 util::logf("Applied ROM palette: %s (index %d)",
178 palette_group_names_[group_index].c_str(), palette_index);
179 return true;

365void EnhancedPaletteEditor::DrawPaletteAnalysis(const gfx::SnesPalette& palette) {
366 ImGui::Text("Palette Information:");
367 ImGui::Text("Size: %zu colors", palette.size());
368
369 // Color distribution analysis
370 std::map<uint16_t, int> color_frequency;
371 for (int i = 0; i < static_cast<int>(palette.size()); i++) {
372 color_frequency[palette[i].snes()]++;
373 }
374
375 ImGui::Text("Unique Colors: %zu", color_frequency.size());
376
377 if (color_frequency.size() < palette.size()) {
378 ImGui::TextColored(ImVec4(1, 1, 0, 1), "Warning: Duplicate colors detected!");
379
380 if (ImGui::TreeNode("Duplicate Colors")) {
381 for (const auto& [snes_color, count] : color_frequency) {
382 if (count > 1) {
383 ImVec4 display_color = gfx::SnesColor(snes_color).rgb();
384 ImGui::ColorButton(("##dup" + std::to_string(snes_color)).c_str(),
385 display_color, ImGuiColorEditFlags_NoTooltip, ImVec2(16, 16));
386 ImGui::SameLine();
387 ImGui::Text("0x%04X appears %d times", snes_color, count);
388 }
389 }
390 ImGui::TreePop();
391 }
392 }
393
394 // Brightness analysis
395 float total_brightness = 0.0f;
396 float min_brightness = 1.0f;
397 float max_brightness = 0.0f;
398
399 for (int i = 0; i < static_cast<int>(palette.size()); i++) {
400 ImVec4 color = palette[i].rgb();
401 float brightness = (color.x + color.y + color.z) / 3.0f;
402 total_brightness += brightness;
403 min_brightness = std::min(min_brightness, brightness);
404 max_brightness = std::max(max_brightness, brightness);
405 }
406
407 float avg_brightness = total_brightness / palette.size();
408
409 ImGui::Separator();
410 ImGui::Text("Brightness Analysis:");
411 ImGui::Text("Average: %.2f", avg_brightness);
412 ImGui::Text("Range: %.2f - %.2f", min_brightness, max_brightness);
413
414 // Show brightness as progress bar
415 ImGui::Text("Brightness Distribution:");
416 ImGui::ProgressBar(avg_brightness, ImVec2(-1, 0), "Avg");
417}
