41// Overall test results summary
42struct TestResults {
43 std::vector<TestResult> individual_results;
44 size_t total_tests = 0;
45 size_t passed_tests = 0;
46 size_t failed_tests = 0;
47 size_t skipped_tests = 0;
48 std::chrono::milliseconds total_duration{0};
49
50 void AddResult(const TestResult& result) {
51 individual_results.push_back(result);
52 total_tests++;
53 switch (result.status) {
54 case TestStatus::kPassed:
55 passed_tests++;
56 break;
57 case TestStatus::kFailed:
58 failed_tests++;
59 break;
60 case TestStatus::kSkipped:
61 skipped_tests++;
62 break;
63 default:
64 break;
65 }
66 total_duration += result.duration;
67 }
68
69 void Clear() {
70 individual_results.clear();
71 total_tests = passed_tests = failed_tests = skipped_tests = 0;
72 total_duration = std::chrono::milliseconds{0};
73 }
74
75 float GetPassRate() const {
76 return total_tests > 0 ? static_cast<float>(passed_tests) / total_tests
77 : 0.0f;
78 }
79};
