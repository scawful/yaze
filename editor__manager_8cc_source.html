721absl::Status EditorManager::OpenProject() {
722 RETURN_IF_ERROR(rom()->LoadFromFile(current_project_.rom_filename_));
723
724 if (!rom()->resource_label()->LoadLabels(current_project_.labels_filename_)) {
725 return absl::InternalError(
726 "Could not load labels file, update your project file.");
727 }
728
729 static RecentFilesManager manager("recent_files.txt");
730 manager.Load();
731 manager.AddFile(current_project_.filepath + "/" + current_project_.name +
732 ".yaze");
733 manager.Save();
734
735 assembly_editor_.OpenFolder(current_project_.code_folder_);
736
737 current_project_.project_opened_ = true;
738
739 return absl::OkStatus();
740}
