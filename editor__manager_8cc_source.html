43void EditorManager::LoadUserSettings() {
44 try {
45 auto data = core::LoadConfigFile(settings_filename_);
46 if (!data.empty()) {
47 std::istringstream ss(data);
48 std::string line;
49 while (std::getline(ss, line)) {
50 auto eq = line.find('=');
51 if (eq == std::string::npos) continue;
52 auto key = line.substr(0, eq);
53 auto val = line.substr(eq + 1);
54 if (key == "font_global_scale") font_global_scale_ = std::stof(val);
55 if (key == "autosave_enabled") autosave_enabled_ = (val == "1");
56 if (key == "autosave_interval_secs") autosave_interval_secs_ = std::stof(val);
57 }
58 ImGui::GetIO().FontGlobalScale = font_global_scale_;
59 }
60 } catch (...) {
61 }
62}

86void EditorManager::SaveWorkspacePreset(const std::string &name) {
87 if (name.empty()) return;
88 std::string ini_name = absl::StrCat("yaze_workspace_", name, ".ini");
89 ImGui::SaveIniSettingsToDisk(ini_name.c_str());
90 // Update index
91 RefreshWorkspacePresets();
92 if (std::find(workspace_presets_.begin(), workspace_presets_.end(), name) == workspace_presets_.end()) {
93 workspace_presets_.push_back(name);
94 std::ostringstream ss;
95 for (auto &n : workspace_presets_) ss << n << "\n";
96 core::SaveFile("workspace_presets.txt", ss.str());
97 }
98 last_workspace_preset_ = name;
99}

379absl::Status EditorManager::Update() {
380 popup_manager_->DrawPopups();
381 ExecuteShortcuts(context_.shortcut_manager);
382 toast_manager_.Draw();
383
384 // Autosave timer
385 if (autosave_enabled_ && current_rom_ && current_rom_->dirty()) {
386 autosave_timer_ += ImGui::GetIO().DeltaTime;
387 if (autosave_timer_ >= autosave_interval_secs_) {
388 autosave_timer_ = 0.0f;
389 Rom::SaveSettings s;
390 s.backup = true;
391 s.save_new = false;
392 auto st = current_rom_->SaveToFile(s);
393 if (st.ok()) {
394 toast_manager_.Show("Autosave completed", editor::ToastType::kSuccess);
395 } else {
396 toast_manager_.Show(std::string(st.message()), editor::ToastType::kError, 5.0f);
397 }
398 }
399 } else {
400 autosave_timer_ = 0.0f;
401 }
402
403 if (show_homepage_) {
404 ImGui::Begin("Home", &show_homepage_);
405 DrawHomepage();
406 ImGui::End();
407 }
408
409 if (!current_editor_set_) {
410 return absl::OkStatus();
411 }
412 for (auto editor : current_editor_set_->active_editors_) {
413 if (*editor->active()) {
414 if (editor->type() == EditorType::kOverworld) {
415 auto &overworld_editor = static_cast<OverworldEditor &>(*editor);
416 if (overworld_editor.jump_to_tab() != -1) {
417 current_editor_set_->dungeon_editor_.set_active(true);
418 // Set the dungeon editor to the jump to tab
419 current_editor_set_->dungeon_editor_.add_room(
420 overworld_editor.jump_to_tab());
421 overworld_editor.jump_to_tab_ = -1;
422 }
423 }
424
425 if (ImGui::Begin(GetEditorName(editor->type()).c_str(),
426 editor->active())) {
427 current_editor_ = editor;
428 status_ = editor->Update();
429 }
430 ImGui::End();
431 }
432 }
433 return absl::OkStatus();
434}
