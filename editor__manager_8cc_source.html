210 current_editor_ = &sprite_editor_;
211 status_ = sprite_editor_.Update();

213 }

218 assembly_editor_.UpdateCodeView();
219 EndTabItem();
220 }
221 break;
222 case EditorType::kSettings:
223 if (BeginTabItem("Settings", &open)) {
224 current_editor_ = &settings_editor_;
225 status_ = settings_editor_.Update();
226 EndTabItem();
227 }
228 break;
229 case EditorType::kMessage:
230 if (BeginTabItem("Message", &open)) {
231 current_editor_ = &message_editor_;
232 status_ = message_editor_.Update();
233 EndTabItem();
234 }
235 break;
236 default:
237 break;
238 }
239 if (!open) {
240 active_editors_.erase(
241 std::remove(active_editors_.begin(), active_editors_.end(), editor),

254void EditorManager::ManageKeyboardShortcuts() {
255 bool ctrl_or_super = (GetIO().KeyCtrl || GetIO().KeySuper);
256
257 // If CMD + R is pressed, reload the top result of recent files
258 if (IsKeyDown(ImGuiKey_R) && ctrl_or_super) {
259 static RecentFilesManager manager("recent_files.txt");
260 manager.Load();
261 if (!manager.GetRecentFiles().empty()) {
262 auto front = manager.GetRecentFiles().front();
263 std::cout << "Reloading: " << front << std::endl;
264 OpenRomOrProject(front);
265 }
266 }
267
268 if (IsKeyDown(ImGuiKey_F1)) {
269 about_ = true;
270 }
271
272 // If CMD + Q is pressed, quit the application
273 if (IsKeyDown(ImGuiKey_Q) && ctrl_or_super) {
274 quit_ = true;
275 }
276
277 // If CMD + O is pressed, open a file dialog
278 if (IsKeyDown(ImGuiKey_O) && ctrl_or_super) {
279 LoadRom();
280 }
281
282 // If CMD + S is pressed, save the current ROM
283 if (IsKeyDown(ImGuiKey_S) && ctrl_or_super) {
284 SaveRom();
285 }
286
287 if (IsKeyDown(ImGuiKey_X) && ctrl_or_super) {
288 status_ = current_editor_->Cut();
289 }
290
291 if (IsKeyDown(ImGuiKey_C) && ctrl_or_super) {
292 status_ = current_editor_->Copy();
293 }
294
295 if (IsKeyDown(ImGuiKey_V) && ctrl_or_super) {
296 status_ = current_editor_->Paste();
297 }
298
299 if (IsKeyDown(ImGuiKey_Z) && ctrl_or_super) {
300 status_ = current_editor_->Undo();
301 }
302
303 if (IsKeyDown(ImGuiKey_Y) && ctrl_or_super) {
304 status_ = current_editor_->Redo();
305 }
306
307 if (IsKeyDown(ImGuiKey_F) && ctrl_or_super) {
308 status_ = current_editor_->Find();
