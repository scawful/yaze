284absl::Status EditorManager::Update() {
285 popup_manager_->DrawPopups();
286 ExecuteShortcuts(context_.shortcut_manager);
287
288 if (current_editor_set_) {
289 for (auto editor : current_editor_set_->active_editors_) {
290 if (*editor->active()) {
291 if (editor->type() == EditorType::kOverworld) {
292 auto &overworld_editor = static_cast<OverworldEditor &>(*editor);
293 if (overworld_editor.jump_to_tab() != -1) {
294 current_editor_set_->dungeon_editor_.set_active(true);
295 // Set the dungeon editor to the jump to tab
296 current_editor_set_->dungeon_editor_.add_room(
297 overworld_editor.jump_to_tab());
298 overworld_editor.jump_to_tab_ = -1;
299 }
300 }
301
302 if (ImGui::Begin(GetEditorName(editor->type()).c_str(),
303 editor->active())) {
304 current_editor_ = editor;
305 status_ = editor->Update();
306 }
307 ImGui::End();
308 }
309 }
310
311 static uint64_t last_cleanup_time = 0;
312 uint64_t current_time = SDL_GetTicks64();
313
314 // Clean up unused textures every 5 seconds
315 if (current_time - last_cleanup_time > 5000) {
316 current_editor_set_->CleanupUnusedTextures(current_time, 5000);
317 last_cleanup_time = current_time;
318 }
319 }
320
321 if (show_homepage_) {
322 ImGui::Begin("Home", &show_homepage_);
323 DrawHomepage();
324 ImGui::End();
325 }
326 return absl::OkStatus();
327}

528absl::Status EditorManager::LoadRom() {
529 auto file_name = FileDialogWrapper::ShowOpenFileDialog();
530 auto new_rom = std::make_unique<Rom>();
531 RETURN_IF_ERROR(new_rom->LoadFromFile(file_name));
532
533 current_rom_ = new_rom.get();
534 roms_.push_back(std::move(new_rom));
535
536 // Create new editor set for this ROM
537 auto editor_set = std::make_unique<EditorSet>(current_rom_);
538 for (auto *editor : editor_set->active_editors_) {
539 editor->set_context(&context_);
540 }
541 current_editor_set_ = editor_set.get();
542 editor_sets_[current_rom_] = std::move(editor_set);
543
544 static RecentFilesManager manager("recent_files.txt");
545 manager.Load();
546 manager.AddFile(file_name);
547 manager.Save();
548 RETURN_IF_ERROR(LoadAssets());
549
550 return absl::OkStatus();
551}

555 return absl::FailedPreconditionError("No ROM or editor set loaded");
556 }
557 current_editor_set_->overworld_editor_.Initialize();
558 current_editor_set_->message_editor_.Initialize();
559

561 ASSIGN_OR_RETURN(*sheet_manager.mutable_gfx_sheets(),
562 LoadAllGraphicsData(*current_rom_));
563 RETURN_IF_ERROR(current_editor_set_->overworld_editor_.Load());
564 RETURN_IF_ERROR(current_editor_set_->dungeon_editor_.Load());
565 return absl::OkStatus();

568absl::Status EditorManager::SaveRom() {
569 if (!current_rom_ || !current_editor_set_) {
570 return absl::FailedPreconditionError("No ROM or editor set loaded");
571 }
572
573 if (core::FeatureFlags::get().kSaveDungeonMaps) {
574 RETURN_IF_ERROR(current_editor_set_->screen_editor_.SaveDungeonMaps());
575 }
576
577 RETURN_IF_ERROR(current_editor_set_->overworld_editor_.Save());
578
579 if (core::FeatureFlags::get().kSaveGraphicsSheet)
580 RETURN_IF_ERROR(SaveAllGraphicsData(

582
583 return current_rom_->SaveToFile(backup_rom_, save_new_auto_);

586absl::Status EditorManager::OpenRomOrProject(const std::string &filename) {
587 if (absl::StrContains(filename, ".yaze")) {
588 RETURN_IF_ERROR(current_project_.Open(filename));
589 RETURN_IF_ERROR(OpenProject());
590 } else {
591 auto new_rom = std::make_unique<Rom>();
592 RETURN_IF_ERROR(new_rom->LoadFromFile(filename));
593 current_rom_ = new_rom.get();
594 SharedRom::set_rom(current_rom_);
595 roms_.push_back(std::move(new_rom));
596
597 // Create new editor set for this ROM
598 auto editor_set = std::make_unique<EditorSet>(current_rom_);
599 for (auto *editor : editor_set->active_editors_) {
600 editor->set_context(&context_);
601 }
602 current_editor_set_ = editor_set.get();
603 editor_sets_[current_rom_] = std::move(editor_set);
604 RETURN_IF_ERROR(LoadAssets());
605 }
606 return absl::OkStatus();
607}
