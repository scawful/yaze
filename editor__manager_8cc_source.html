55void EditorManager::LoadUserSettings() {
56 try {
57 auto data = core::LoadConfigFile(settings_filename_);
58 if (!data.empty()) {
59 std::istringstream ss(data);
60 std::string line;
61 while (std::getline(ss, line)) {
62 auto eq = line.find('=');
63 if (eq == std::string::npos) continue;
64 auto key = line.substr(0, eq);
65 auto val = line.substr(eq + 1);
66 if (key == "font_global_scale") font_global_scale_ = std::stof(val);
67 if (key == "autosave_enabled") autosave_enabled_ = (val == "1");
68 if (key == "autosave_interval_secs") autosave_interval_secs_ = std::stof(val);
69 }
70 ImGui::GetIO().FontGlobalScale = font_global_scale_;
71 }
72 } catch (...) {
73 }
74}

84void EditorManager::RefreshWorkspacePresets() {
85 // Safe clearing with error handling
86 try {
87 // Create a new vector instead of clearing to avoid corruption
88 std::vector<std::string> new_presets;
89
90 // Try to read a simple index file of presets
91 try {
92 auto data = core::LoadConfigFile("workspace_presets.txt");
93 if (!data.empty()) {
94 std::istringstream ss(data);
95 std::string name;
96 while (std::getline(ss, name)) {
97 // Trim whitespace and validate
98 name.erase(0, name.find_first_not_of(" \t\r\n"));
99 name.erase(name.find_last_not_of(" \t\r\n") + 1);
100 if (!name.empty() && name.length() < 256) { // Reasonable length limit
101 new_presets.emplace_back(std::move(name));
102 }
103 }
104 }
105 } catch (const std::exception& e) {
106 util::logf("Warning: Failed to load workspace presets: %s", e.what());
107 }
108
109 // Safely replace the vector
110 workspace_presets_ = std::move(new_presets);
111 workspace_presets_loaded_ = true;
112
113 } catch (const std::exception& e) {
114 util::logf("Error in RefreshWorkspacePresets: %s", e.what());
115 // Ensure we have a valid empty vector
116 workspace_presets_ = std::vector<std::string>();
117 workspace_presets_loaded_ = true; // Mark as loaded even if empty to avoid retry
118 }
119}

121void EditorManager::SaveWorkspacePreset(const std::string &name) {
122 if (name.empty()) return;
123 std::string ini_name = absl::StrCat("yaze_workspace_", name, ".ini");
124 ImGui::SaveIniSettingsToDisk(ini_name.c_str());
125
126 // Ensure presets are loaded before updating
127 if (!workspace_presets_loaded_) {
128 RefreshWorkspacePresets();
129 }
130
131 // Update index
132 if (std::find(workspace_presets_.begin(), workspace_presets_.end(), name) == workspace_presets_.end()) {
133 workspace_presets_.emplace_back(name);
134 try {
135 std::ostringstream ss;
136 for (const auto &n : workspace_presets_) ss << n << "\n";
137 core::SaveFile("workspace_presets.txt", ss.str());
138 } catch (const std::exception& e) {
139 util::logf("Warning: Failed to save workspace presets: %s", e.what());
140 }
141 }
142 last_workspace_preset_ = name;
143}

706absl::Status EditorManager::Update() {
707 popup_manager_->DrawPopups();
708 ExecuteShortcuts(context_.shortcut_manager);
709 toast_manager_.Draw();
710
711 // Draw background grid effects for the entire viewport
712 if (ImGui::GetCurrentContext()) {
713 ImDrawList* bg_draw_list = ImGui::GetBackgroundDrawList();
714 const ImGuiViewport* viewport = ImGui::GetMainViewport();
715
716 auto& theme_manager = gui::ThemeManager::Get();
717 auto current_theme = theme_manager.GetCurrentTheme();
718 auto& bg_renderer = gui::BackgroundRenderer::Get();
719
720 // Draw grid covering the entire main viewport
721 ImVec2 grid_pos = viewport->WorkPos;
722 ImVec2 grid_size = viewport->WorkSize;
723 bg_renderer.RenderDockingBackground(bg_draw_list, grid_pos, grid_size, current_theme.primary);
724 }
725
726 // Ensure TestManager always has the current ROM
727 static Rom* last_test_rom = nullptr;
728 if (last_test_rom != current_rom_) {
729 util::logf("EditorManager::Update - ROM changed, updating TestManager: %p -> %p",
730 (void*)last_test_rom, (void*)current_rom_);
731 test::TestManager::Get().SetCurrentRom(current_rom_);
732 last_test_rom = current_rom_;
733 }
734
735 // Autosave timer
736 if (autosave_enabled_ && current_rom_ && current_rom_->dirty()) {
737 autosave_timer_ += ImGui::GetIO().DeltaTime;
738 if (autosave_timer_ >= autosave_interval_secs_) {
739 autosave_timer_ = 0.0f;
740 Rom::SaveSettings s;
741 s.backup = true;
742 s.save_new = false;
743 auto st = current_rom_->SaveToFile(s);
744 if (st.ok()) {
745 toast_manager_.Show("Autosave completed", editor::ToastType::kSuccess);
746 } else {
747 toast_manager_.Show(std::string(st.message()), editor::ToastType::kError, 5.0f);
748 }
749 }
750 } else {
751 autosave_timer_ = 0.0f;
752 }
753
754
755 // Check if ROM is loaded before allowing editor updates
756 if (!current_editor_set_) {
757 // Show welcome screen when no session is active
758 if (sessions_.empty()) {
759 DrawWelcomeScreen();
760 }
761 return absl::OkStatus();
762 }
763
764 // Check if current ROM is valid
765 if (!current_rom_) {
766 DrawWelcomeScreen();
767 return absl::OkStatus();
768 }
769
770 // Check if any editors are active across ALL sessions
771 bool any_editor_active = false;
772 for (const auto& session : sessions_) {
773 if (!session.rom.is_loaded()) continue;
774 for (auto editor : session.editors.active_editors_) {
775 if (*editor->active()) {
776 any_editor_active = true;
777 break;
778 }
779 }
780 if (any_editor_active) break;
781 }
782
783 // Show welcome screen if no editors are active (ROM loaded but editors not opened)
784 if (!any_editor_active) {
785 DrawWelcomeScreen();
786 return absl::OkStatus();
787 }
788
789 // Iterate through ALL sessions to support multi-session docking
790 for (size_t session_idx = 0; session_idx < sessions_.size(); ++session_idx) {
791 auto& session = sessions_[session_idx];
792 if (!session.rom.is_loaded()) continue; // Skip sessions with invalid ROMs
793
794 for (auto editor : session.editors.active_editors_) {
795 if (*editor->active()) {
796 if (editor->type() == EditorType::kOverworld) {
797 auto &overworld_editor = static_cast<OverworldEditor &>(*editor);
798 if (overworld_editor.jump_to_tab() != -1) {
799 session.editors.dungeon_editor_.set_active(true);
800 // Set the dungeon editor to the jump to tab
801 session.editors.dungeon_editor_.add_room(overworld_editor.jump_to_tab());
802 overworld_editor.jump_to_tab_ = -1;
803 }
804 }
805
806 // Generate unique window titles for multi-session support
807 std::string window_title = GenerateUniqueEditorTitle(editor->type(), session_idx);
808
809 if (ImGui::Begin(window_title.c_str(), editor->active())) {
810 // Temporarily switch context for this editor's update
811 Rom* prev_rom = current_rom_;
812 EditorSet* prev_editor_set = current_editor_set_;
813
814 current_rom_ = &session.rom;
815 current_editor_set_ = &session.editors;
816 current_editor_ = editor;
817
818 status_ = editor->Update();
819
820 // Restore context
821 current_rom_ = prev_rom;
822 current_editor_set_ = prev_editor_set;
823 }
824 ImGui::End();
825 }
826 }
827 }
828 return absl::OkStatus();
829}

1285absl::Status EditorManager::LoadRom() {
1286 auto file_name = FileDialogWrapper::ShowOpenFileDialog();
1287 if (file_name.empty()) {
1288 return absl::OkStatus();
1289 }
1290
1291 // Check for duplicate sessions
1292 if (HasDuplicateSession(file_name)) {
1293 toast_manager_.Show("ROM already open in another session", editor::ToastType::kWarning);
1294 return absl::OkStatus();
1295 }
1296
1297 Rom temp_rom;
1298 RETURN_IF_ERROR(temp_rom.LoadFromFile(file_name));
1299
1300 // Check if there's an empty session we can populate instead of creating new one
1301 RomSession* target_session = nullptr;
1302 for (auto& session : sessions_) {
1303 if (!session.rom.is_loaded()) {
1304 target_session = &session;
1305 util::logf("Found empty session to populate with ROM: %s", file_name.c_str());
1306 break;
1307 }
1308 }
1309
1310 if (target_session) {
1311 // Populate existing empty session
1312 target_session->rom = std::move(temp_rom);
1313 target_session->filepath = file_name;
1314 current_rom_ = &target_session->rom;
1315 current_editor_set_ = &target_session->editors;
1316 } else {
1317 // Create new session only if no empty ones exist
1318 sessions_.emplace_back(std::move(temp_rom));
1319 RomSession &session = sessions_.back();
1320 session.filepath = file_name; // Store filepath for duplicate detection
1321
1322 // Wire editor contexts
1323 for (auto *editor : session.editors.active_editors_) {
1324 editor->set_context(&context_);
1325 }
1326 current_rom_ = &session.rom;
1327 current_editor_set_ = &session.editors;
1328 }
1329
1330 // Update test manager with current ROM for ROM-dependent tests
1331 util::logf("EditorManager: Setting ROM in TestManager - %p ('%s')",
1332 (void*)current_rom_, current_rom_ ? current_rom_->title().c_str() : "null");
1333 test::TestManager::Get().SetCurrentRom(current_rom_);
1334
1335 static core::RecentFilesManager manager("recent_files.txt");
1336 manager.Load();
1337 manager.AddFile(file_name);
1338 manager.Save();
1339 RETURN_IF_ERROR(LoadAssets());
1340
1341 return absl::OkStatus();
1342}
