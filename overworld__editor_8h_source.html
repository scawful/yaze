74 public:
75 absl::Status LoadEntranceTileTypes(Rom& rom) {
76 int offset_low = kEntranceTileTypePtrLow;
77 int offset_high = kEntranceTileTypePtrHigh;
78
79 for (int i = 0; i < kNumEntranceTileTypes; i++) {
80 // Load entrance tile types
81 ASSIGN_OR_RETURN(auto value_low, rom.ReadWord(offset_low + i));
82 entrance_tile_types_low_.push_back(value_low);
83 ASSIGN_OR_RETURN(auto value_high, rom.ReadWord(offset_high + i));
84 entrance_tile_types_low_.push_back(value_high);
85 }
86
87 return absl::OkStatus();
88 }
89
90 private:
91 std::vector<uint16_t> entrance_tile_types_low_;
92 std::vector<uint16_t> entrance_tile_types_high_;
