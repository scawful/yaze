64class EntranceContext {
65 public:
66 absl::Status LoadEntranceTileTypes(Rom& rom) {
67 int offset_low = 0xDB8BF;
68 int offset_high = 0xDB917;
69
70 for (int i = 0; i < 0x2C; i++) {
71 // Load entrance tile types
72 ASSIGN_OR_RETURN(auto value_low, rom.ReadWord(offset_low + i));
73 entrance_tile_types_low_.push_back(value_low);
74 ASSIGN_OR_RETURN(auto value_high, rom.ReadWord(offset_high + i));
75 entrance_tile_types_low_.push_back(value_high);
76 }
77
78 return absl::OkStatus();
79 }
80
81 private:
82 std::vector<uint16_t> entrance_tile_types_low_;
83 std::vector<uint16_t> entrance_tile_types_high_;
84};
