65 absl::Status LoadEntranceTileTypes(Rom& rom) {
66 int offset_low = kEntranceTileTypePtrLow;
67 int offset_high = kEntranceTileTypePtrHigh;
68
69 for (int i = 0; i < kNumEntranceTileTypes; i++) {
70 // Load entrance tile types
71 ASSIGN_OR_RETURN(auto value_low, rom.ReadWord(offset_low + i));
72 entrance_tile_types_low_.push_back(value_low);
73 ASSIGN_OR_RETURN(auto value_high, rom.ReadWord(offset_high + i));
74 entrance_tile_types_low_.push_back(value_high);
75 }
76
77 return absl::OkStatus();
78 }

80 private:
