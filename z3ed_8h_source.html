169 RETURN_IF_ERROR(rom_.LoadFromFile(arg_vec[0]))
170
171 std::stringstream ss(arg_vec[1].data());
172 uint32_t offset;
173 ss >> std::hex >> offset;
174 uint32_t length = 0x01;
175 if (!arg_vec[2].empty()) {
176 length = std::stoi(arg_vec[2]);
177 }
178
179 if (length > 1) {
180 auto returned_bytes_status = rom_.ReadByteVector(offset, length);
181 if (!returned_bytes_status.ok()) {
182 return returned_bytes_status.status();
183 }
184 auto returned_bytes = returned_bytes_status.value();
185 for (const auto& each : returned_bytes) {
186 std::cout << each;
187 }
188 std::cout << std::endl;
189 } else {
190 auto byte = rom_.ReadByte(offset);
191 std::cout << std::hex << byte.value() << std::endl;
192 }
193
194 return absl::OkStatus();
195 }
