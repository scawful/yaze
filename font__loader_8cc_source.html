50absl::Status LoadFont(const FontConfig& font_config) {
51 ImGuiIO& imgui_io = ImGui::GetIO();
52 std::string actual_font_path = SetFontPath(font_config.font_path);
53 // Check if the file exists with std library first, since ImGui IO will assert
54 // if the file does not exist
55 if (!std::filesystem::exists(actual_font_path)) {
56 return absl::InternalError(
57 absl::StrFormat("Font file %s does not exist", actual_font_path));
58 }
59
60 if (!imgui_io.Fonts->AddFontFromFileTTF(actual_font_path.data(),
61 font_config.font_size)) {
62 return absl::InternalError(
63 absl::StrFormat("Failed to load font from %s", actual_font_path));
64 }
65 return absl::OkStatus();
66}

144 std::string WideToUtf8(const std::wstring& wstr) {
145 if (wstr.empty()) return std::string();
146
147 int size_needed = WideCharToMultiByte(CP_UTF8, 0, &wstr[0], (int)wstr.size(), NULL, 0, NULL, NULL);
148 if (size_needed <= 0) return std::string();
149
150 std::string result(size_needed, 0);
151 WideCharToMultiByte(CP_UTF8, 0, &wstr[0], (int)wstr.size(), &result[0], size_needed, NULL, NULL);
152 return result;
153 }

155 // Helper function to get Windows fonts directory
156 std::string GetWindowsFontsDirectory() {
157 wchar_t* fontsPath = nullptr;
158 HRESULT hr = SHGetKnownFolderPath(FOLDERID_Fonts, 0, NULL, &fontsPath);
159
160 if (SUCCEEDED(hr) && fontsPath) {
161 std::string result = WideToUtf8(fontsPath) + "\\";
162 CoTaskMemFree(fontsPath);
163 return result;
164 }
165
166 // Fallback to default path
167 return "C:\\Windows\\Fonts\\";
168 }

184void LoadSystemFonts() {
185 ImGuiIO& imgui_io = ImGui::GetIO();
186
187 // Get the Windows fonts directory
188 std::string fontsDir = GetWindowsFontsDirectory();
189
190 // List of essential Windows fonts to load
191 static const std::vector<std::string> essentialFonts = {
192 "arial.ttf",
193 "arialbd.ttf",
194 "times.ttf",
195 "timesbd.ttf",
196 "cour.ttf",
197 "courbd.ttf",
198 "verdana.ttf",
199 "verdanab.ttf",
200 "tahoma.ttf",
201 "tahomabd.ttf",
202 "segoeui.ttf",
203 "segoeuib.ttf"
204 };
205
206 // Load essential fonts
207 for (const auto& fontName : essentialFonts) {
208 std::string fontPath = fontsDir + fontName;
209
210 if (FontFileExists(fontPath)) {
211 try {
212 // Load the font
213 ImFont* font = imgui_io.Fonts->AddFontFromFileTTF(fontPath.c_str(), 16.0F);
214 if (font) {
215 // Merge icon fonts if available
216 static const ImWchar icons_ranges[] = {ICON_MIN_MD, 0xf900, 0};
217 ImFontConfig icons_config{};
218 icons_config.MergeMode = true;
219 icons_config.GlyphOffset.y = 5.0F;
220 icons_config.GlyphMinAdvanceX = 13.0F;
221 icons_config.PixelSnapH = true;
222
223 std::string iconFontPath = SetFontPath(FONT_ICON_FILE_NAME_MD);
224 if (FontFileExists(iconFontPath)) {
225 imgui_io.Fonts->AddFontFromFileTTF(iconFontPath.c_str(), ICON_FONT_SIZE,
226 &icons_config, icons_ranges);
227 }
228 }
229 } catch (...) {
230 // Silently continue if font loading fails
231 continue;
232 }
233 }
234 }
235
236 // Try to load additional fonts from registry (safer approach)
237 HKEY hKey = nullptr;
238 LONG result = RegOpenKeyExA(HKEY_LOCAL_MACHINE,
239 "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Fonts",
240 0, KEY_READ, &hKey);
241
242 if (result == ERROR_SUCCESS && hKey) {
243 DWORD valueCount = 0;
244 DWORD maxValueNameSize = 0;
245 DWORD maxValueDataSize = 0;
246
247 // Get registry info
248 result = RegQueryInfoKeyA(hKey, nullptr, nullptr, nullptr, nullptr, nullptr,
249 nullptr, &valueCount, &maxValueNameSize,
250 &maxValueDataSize, nullptr, nullptr);
251
252 if (result == ERROR_SUCCESS && valueCount > 0) {
253 // Allocate buffers with proper size limits
254 const size_t maxNameSize = std::min(static_cast<size_t>(maxValueNameSize) + 1, size_t(1024));
255 const size_t maxDataSize = std::min(static_cast<size_t>(maxValueDataSize) + 1, size_t(4096));
256
257 std::vector<char> valueName(maxNameSize);
258 std::vector<BYTE> valueData(maxDataSize);
259
260 // Enumerate font entries (limit to prevent excessive loading)
261 const DWORD maxFontsToLoad = std::min(valueCount, DWORD(50));
262
263 for (DWORD i = 0; i < maxFontsToLoad; i++) {
264 DWORD valueNameSize = static_cast<DWORD>(maxNameSize);
265 DWORD valueDataSize = static_cast<DWORD>(maxDataSize);
266 DWORD valueType = 0;
267
268 result = RegEnumValueA(hKey, i, valueName.data(), &valueNameSize,
269 nullptr, &valueType, valueData.data(), &valueDataSize);
270
271 if (result == ERROR_SUCCESS && valueType == REG_SZ && valueDataSize > 0) {
272 // Ensure null termination
273 valueName[valueNameSize] = '\0';
274 valueData[valueDataSize] = '\0';
275
276 std::string fontPath(reinterpret_cast<char*>(valueData.data()));
277
278 // Normalize the font path
279 if (!fontPath.empty()) {
280 // If it's a relative path, prepend the fonts directory
281 if (fontPath.find(':') == std::string::npos) {
282 fontPath = fontsDir + fontPath;
283 }
284
285 // Check if it's a TTF file and exists
286 std::string lowerPath = fontPath;
287 std::transform(lowerPath.begin(), lowerPath.end(), lowerPath.begin(), ::tolower);
288
289 if ((lowerPath.find(".ttf") != std::string::npos ||
290 lowerPath.find(".otf") != std::string::npos) &&
291 FontFileExists(fontPath)) {
292 try {
293 imgui_io.Fonts->AddFontFromFileTTF(fontPath.c_str(), 16.0F);
294 } catch (...) {
295 // Continue if font loading fails
296 continue;
297 }
298 }
299 }
300 }
301 }
302 }
303
304 RegCloseKey(hKey);
305 }
