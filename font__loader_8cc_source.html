47absl::Status LoadFont(const FontConfig& font_config) {
48 ImGuiIO& io = ImGui::GetIO();
49 std::string actual_font_path = SetFontPath(font_config.font_path);
50 // Check if the file exists with std library first, since ImGui IO will assert
51 // if the file does not exist
52 if (!std::filesystem::exists(actual_font_path)) {
53 return absl::InternalError(
54 absl::StrFormat("Font file %s does not exist", actual_font_path));
55 }
56
57 if (!io.Fonts->AddFontFromFileTTF(actual_font_path.data(), font_config.font_size)) {
58 return absl::InternalError(
59 absl::StrFormat("Failed to load font from %s", actual_font_path));
60 }
61 return absl::OkStatus();
62}

161void LoadSystemFonts() {
162 HKEY hKey;
163 std::vector<std::string> fontPaths;
164
165 // Open the registry key where fonts are listed
166 if (RegOpenKeyEx(
167 HKEY_LOCAL_MACHINE,
168 TEXT("SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Fonts"), 0,
169 KEY_READ, &hKey) == ERROR_SUCCESS) {
170 DWORD valueCount;
171 DWORD maxValueNameSize;
172 DWORD maxValueDataSize;
173
174 // Query the number of entries and the maximum size of the names and values
175 RegQueryInfoKey(hKey, NULL, NULL, NULL, NULL, NULL, NULL, &valueCount,
176 &maxValueNameSize, &maxValueDataSize, NULL, NULL);
177
178 char *valueName = new char[maxValueNameSize + 1]; // +1 for null terminator
179 BYTE *valueData = new BYTE[maxValueDataSize + 1]; // +1 for null terminator
180
181 // Enumerate all font entries
182 for (DWORD i = 0; i < valueCount; i++) {
183 DWORD valueNameSize = maxValueNameSize + 1; // +1 for null terminator
184 DWORD valueDataSize = maxValueDataSize + 1; // +1 for null terminator
185 DWORD valueType;
186
187 // Clear buffers
188 memset(valueName, 0, valueNameSize);
189 memset(valueData, 0, valueDataSize);
190
191 // Get the font name and file path
192 if (RegEnumValue(hKey, i, valueName, &valueNameSize, NULL, &valueType,
193 valueData, &valueDataSize) == ERROR_SUCCESS) {
194 if (valueType == REG_SZ) {
195 // Add the font file path to the vector
196 std::string fontPath(reinterpret_cast<char *>(valueData),
197 valueDataSize);
198
199 fontPaths.push_back(fontPath);
200 }
201 }
202 }
203
204 delete[] valueName;
205 delete[] valueData;
206
207 RegCloseKey(hKey);
208 }
