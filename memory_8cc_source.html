15void MemoryImpl::Initialize(const std::vector<uint8_t>& romData, bool verbose) {
16 verbose_ = verbose;
17 type_ = 1;
18
19 auto location = 0x7FC0; // GetHeaderOffset();
20 romSize = 0x400 << romData[location + 0x17];
21 sramSize = 0x400 << romData[location + 0x18];
22 rom_.resize(romSize);
23
24 // Copy memory into rom_
25 for (size_t i = 0; i < romSize; i++) {
26 rom_[i] = romData[i];
27 }
28 ram_.resize(sramSize);
29 for (size_t i = 0; i < sramSize; i++) {
30 ram_[i] = 0;
31 }
32
33 // Clear memory
34 memory_.resize(0x1000000); // 16 MB
35 std::fill(memory_.begin(), memory_.end(), 0);
36
37 // Load ROM data into memory based on LoROM mapping
38 size_t romSize = romData.size();
39 size_t romAddress = 0;
40 const size_t ROM_CHUNK_SIZE = 0x8000; // 32 KB
41 for (size_t bank = 0x00; bank <= 0x3F; ++bank) {
42 for (size_t offset = 0x8000; offset <= 0xFFFF; offset += ROM_CHUNK_SIZE) {
43 if (romAddress < romSize) {
44 std::copy(romData.begin() + romAddress,
45 romData.begin() + romAddress + ROM_CHUNK_SIZE,
46 memory_.begin() + (bank << 16) + offset);
47 romAddress += ROM_CHUNK_SIZE;
48 }
49 }
50 }

54memory::RomInfo MemoryImpl::ReadRomHeader() {
55 memory::RomInfo rom_info;
56
57 uint32_t offset = GetHeaderOffset();
58
59 // Read cartridge title
60 char title[22];
61 for (int i = 0; i < 21; ++i) {

69 rom_info.romSpeed = (memory::RomSpeed)(romSpeedAndMapMode & 0x07);
70 rom_info.bankSize = (memory::BankSize)((romSpeedAndMapMode >> 5) & 0x01);
71

73 rom_info.romType = (memory::RomType)ReadByte(offset + 0x16);
74

80

83

135uint8_t MemoryImpl::cart_readLorom(uint8_t bank, uint16_t adr) {
136 if (((bank >= 0x70 && bank < 0x7e) || bank >= 0xf0) && adr < 0x8000 &&
137 sramSize > 0) {
138 // banks 70-7e and f0-ff, adr 0000-7fff
139 return ram_[(((bank & 0xf) << 15) | adr) & (sramSize - 1)];
140 }
141 bank &= 0x7f;
142 if (adr >= 0x8000 || bank >= 0x40) {
143 // adr 8000-ffff in all banks or all addresses in banks 40-7f and c0-ff
144 return rom_[((bank << 15) | (adr & 0x7fff)) & (romSize - 1)];
145 }
146 return open_bus_;
147}

170uint8_t MemoryImpl::cart_readExHirom(uint8_t bank, uint16_t adr) {
171 if ((bank & 0x7f) < 0x40 && adr >= 0x6000 && adr < 0x8000 && sramSize > 0) {
172 // banks 00-3f and 80-bf, adr 6000-7fff
173 return ram_[(((bank & 0x3f) << 13) | (adr & 0x1fff)) & (sramSize - 1)];
174 }
175 bool secondHalf = bank < 0x80;
176 bank &= 0x7f;
177 if (adr >= 0x8000 || bank >= 0x40) {
178 // adr 8000-ffff in all banks or all addresses in banks 40-7f and c0-ff
179 return rom_[(((bank & 0x3f) << 16) | (secondHalf ? 0x400000 : 0) | adr) &
180 (romSize - 1)];
181 }

193uint32_t MemoryImpl::GetMappedAddress(uint32_t address) const {
194 uint8_t bank = address >> 16;
195 uint32_t offset = address & 0xFFFF;
196
197 if (bank <= 0x3F) {
198 if (address <= 0x1FFF) {
199 return (0x7E << 16) + offset; // Shadow RAM
200 } else if (address <= 0x5FFF) {
201 return (bank << 16) + (offset - 0x2000) + 0x2000; // Hardware Registers
202 } else if (address <= 0x7FFF) {
203 return offset - 0x6000 + 0x6000; // Expansion RAM
204 } else {
205 // Return lorom mapping
206 return (bank << 16) + (offset - 0x8000) + 0x8000; // ROM
207 }
208 } else if (bank == 0x7D) {
209 return offset + 0x7D0000; // SRAM
210 } else if (bank == 0x7E || bank == 0x7F) {
211 return offset + 0x7E0000; // System RAM
212 } else if (bank >= 0x80) {
213 // Handle HiROM and mirrored areas
214 }
215
216 return address; // Return the original address if no mapping is defined
217}
