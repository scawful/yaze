86 // Add minimal SNES header at 0x7FC0 (LoROM)
87 const size_t header_offset = 0x7FC0;
88 if (size > header_offset + 32) {
89 // ROM title
90 std::string title = "YAZE TEST ROM ";
91 std::copy(title.begin(), title.end(), rom_data.begin() + header_offset);
92
93 // Map mode (LoROM)
94 rom_data[header_offset + 21] = 0x20;
95
96 // ROM size (1MB)
97 rom_data[header_offset + 23] = 0x0A;
98
99 // Calculate and set checksum
100 uint16_t checksum = 0;
101 for (size_t i = 0; i < size; ++i) {
102 if (i < header_offset + 28 || i > header_offset + 31) {
103 checksum += rom_data[i];
104 }
105 }
106
107 uint16_t checksum_complement = checksum ^ 0xFFFF;
108 rom_data[header_offset + 28] = checksum_complement & 0xFF;
109 rom_data[header_offset + 29] = (checksum_complement >> 8) & 0xFF;
110 rom_data[header_offset + 30] = checksum & 0xFF;
111 rom_data[header_offset + 31] = (checksum >> 8) & 0xFF;
112 }
