281 absl::StatusOr<gfx::Tile16> ReadTile16(uint32_t tile16_id) {
282 // Skip 8 bytes per tile.
283 auto tpos = kTile16Ptr + (tile16_id * 0x08);
284 gfx::Tile16 tile16;
285 ASSIGN_OR_RETURN(auto new_tile0, ReadWord(tpos))
286 tile16.tile0_ = gfx::WordToTileInfo(new_tile0);
287 tpos += 2;
288 ASSIGN_OR_RETURN(auto new_tile1, ReadWord(tpos))
289 tile16.tile1_ = gfx::WordToTileInfo(new_tile1);
290 tpos += 2;
291 ASSIGN_OR_RETURN(auto new_tile2, ReadWord(tpos))
292 tile16.tile2_ = gfx::WordToTileInfo(new_tile2);
293 tpos += 2;
294 ASSIGN_OR_RETURN(auto new_tile3, ReadWord(tpos))
295 tile16.tile3_ = gfx::WordToTileInfo(new_tile3);
296 return tile16;
297 }

299 absl::Status WriteTile16(int tile16_id, const gfx::Tile16& tile) {
300 // Skip 8 bytes per tile.
301 auto tpos = kTile16Ptr + (tile16_id * 0x08);
302 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile0_)));
303 tpos += 2;
304 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile1_)));
305 tpos += 2;
306 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile2_)));
307 tpos += 2;
308 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile3_)));
309 return absl::OkStatus();
310 }

323 absl::Status WriteByte(int addr, uint8_t value) {
324 RETURN_IF_ERROR(ReadWritePreconditions());
325 if (addr >= static_cast<int>(rom_data_.size())) {
326 return absl::OutOfRangeError(absl::StrFormat(
327 "Attempt to write byte %#02x value failed, address %d out of range",
328 value, addr));
329 }
330 rom_data_[addr] = value;
331 core::logf("WriteByte: %#06X: %s", addr, core::HexByte(value).data());
332 return absl::OkStatus();
333 }
334
335 absl::Status WriteWord(int addr, uint16_t value) {
336 RETURN_IF_ERROR(ReadWritePreconditions());
337 if (addr + 1 >= static_cast<int>(rom_data_.size())) {
338 return absl::OutOfRangeError(absl::StrFormat(
339 "Attempt to write word %#04x value failed, address %d out of range",
340 value, addr));
341 }
342 rom_data_[addr] = (uint8_t)(value & 0xFF);
343 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
344 core::logf("WriteWord: %#06X: %s", addr, core::HexWord(value).data());
345 return absl::OkStatus();
346 }

351 return absl::OutOfRangeError(absl::StrFormat(
352 "Attempt to write short %#04x value failed, address %d out of range",
353 value, addr));
354 }
355 rom_data_[addr] = (uint8_t)(value & 0xFF);
356 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);

358 return absl::OkStatus();
359 }

361 absl::Status WriteLong(uint32_t addr, uint32_t value) {
362 RETURN_IF_ERROR(ReadWritePreconditions());
363 if (addr + 2 >= static_cast<uint32_t>(rom_data_.size())) {
364 return absl::OutOfRangeError(absl::StrFormat(
365 "Attempt to write long %#06x value failed, address %d out of range",

370 rom_data_[addr + 2] = (uint8_t)((value >> 16) & 0xFF);

375 absl::Status WriteVector(int addr, std::vector<uint8_t> data) {
376 if (addr + static_cast<int>(data.size()) >
377 static_cast<int>(rom_data_.size())) {
378 return absl::InvalidArgumentError(absl::StrFormat(
379 "Attempt to write vector value failed, address %d out of range",
380 addr));
381 }
382 for (int i = 0; i < static_cast<int>(data.size()); i++) {
383 rom_data_[addr + i] = data[i];

385 core::logf("WriteVector: %#06X: %s", addr, core::HexByte(data[0]).data());
386 return absl::OkStatus();

389 absl::Status WriteColor(uint32_t address, const gfx::SnesColor& color) {
390 uint16_t bgr = ((color.snes() >> 10) & 0x1F) |
391 ((color.snes() & 0x1F) << 10) | (color.snes() & 0x7C00);
392
393 // Write the 16-bit color value to the ROM at the specified address
394 core::logf("WriteColor: %#06X: %s", address, core::HexWord(bgr).data());
395 return WriteShort(address, bgr);

406 template <typename T, typename... Args>
407 absl::Status ReadTransaction(T& var, int address, Args&&... args) {
408 absl::Status status = ReadHelper<T>(var, address);
409 if (!status.ok()) {
410 return status;
411 }
412
413 if constexpr (sizeof...(args) > 0) {
414 status = ReadTransaction(std::forward<Args>(args)...);
415 }
416
417 return status;

475 struct WriteAction {
476 int address;
477 std::variant<int, uint8_t, uint16_t, short, std::vector<uint8_t>,
478 gfx::SnesColor, std::vector<gfx::SnesColor>>
479 value;
480 };
481
482 private:
483 virtual absl::Status WriteHelper(const WriteAction& action) {
484 if (std::holds_alternative<uint8_t>(action.value)) {
485 return Write(action.address, std::get<uint8_t>(action.value));
486 } else if (std::holds_alternative<uint16_t>(action.value) ||
487 std::holds_alternative<short>(action.value)) {
488 return WriteShort(action.address, std::get<uint16_t>(action.value));
489 } else if (std::holds_alternative<std::vector<uint8_t>>(action.value)) {
490 return WriteVector(action.address,
491 std::get<std::vector<uint8_t>>(action.value));
492 } else if (std::holds_alternative<gfx::SnesColor>(action.value)) {
493 return WriteColor(action.address, std::get<gfx::SnesColor>(action.value));
494 } else if (std::holds_alternative<std::vector<gfx::SnesColor>>(
495 action.value)) {
496 return absl::UnimplementedError(
497 "WriteHelper: std::vector<gfx::SnesColor>");
498 }
499 auto error_message = absl::StrFormat("Invalid write argument type: %s",
500 typeid(action.value).name());
501 return absl::InvalidArgumentError(error_message);
502 }

504 template <typename T>
505 absl::Status ReadHelper(T& var, int address) {
506 if constexpr (std::is_same_v<T, uint8_t>) {
507 ASSIGN_OR_RETURN(auto result, ReadByte(address));
508 var = result;
509 } else if constexpr (std::is_same_v<T, uint16_t>) {
510 ASSIGN_OR_RETURN(auto result, ReadWord(address));
511 var = result;
512 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
513 ASSIGN_OR_RETURN(auto result, ReadByteVector(address, var.size()));
514 var = result;
515 }
516 return absl::OkStatus();
517 }
