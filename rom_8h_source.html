248 absl::StatusOr<gfx::Tile16> ReadTile16(uint32_t tile16_id) {
249 // Skip 8 bytes per tile.
250 auto tpos = 0x78000 + (tile16_id * 0x08);
251 gfx::Tile16 tile16;
252 ASSIGN_OR_RETURN(auto new_tile0, ReadWord(tpos))
253 tile16.tile0_ = gfx::WordToTileInfo(new_tile0);
254 tpos += 2;
255 ASSIGN_OR_RETURN(auto new_tile1, ReadWord(tpos))
256 tile16.tile1_ = gfx::WordToTileInfo(new_tile1);
257 tpos += 2;
258 ASSIGN_OR_RETURN(auto new_tile2, ReadWord(tpos))
259 tile16.tile2_ = gfx::WordToTileInfo(new_tile2);
260 tpos += 2;
261 ASSIGN_OR_RETURN(auto new_tile3, ReadWord(tpos))
262 tile16.tile3_ = gfx::WordToTileInfo(new_tile3);
263 return tile16;
264 }

274 tpos += 2;
275 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile3_)));

280 absl::Status Write(int addr, int value) {
281 if (addr >= rom_data_.size()) {
282 return absl::InvalidArgumentError(absl::StrFormat(
283 "Attempt to write %d value failed, address %d out of range", value,
284 addr));
285 }

289
290 absl::Status WriteByte(int addr, uint8_t value) {
291 if (addr >= rom_data_.size()) {
292 return absl::InvalidArgumentError(absl::StrFormat(
293 "Attempt to write byte %#02x value failed, address %d out of range",
294 value, addr));
295 }
296 rom_data_[addr] = value;
297 std::string log_str = absl::StrFormat("WriteByte: %#06X: %s", addr,
298 core::UppercaseHexByte(value).data());
299 core::Logger::log(log_str);
300 return absl::OkStatus();
301 }
302
303 absl::Status WriteWord(int addr, uint16_t value) {
304 if (addr + 1 >= rom_data_.size()) {
305 return absl::InvalidArgumentError(absl::StrFormat(
306 "Attempt to write word %#04x value failed, address %d out of range",
307 value, addr));
308 }
309 rom_data_[addr] = (uint8_t)(value & 0xFF);
310 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
311 core::Logger::log(absl::StrFormat("WriteWord: %#06X: %s", addr,
312 core::UppercaseHexWord(value)));
313 return absl::OkStatus();
314 }
315
316 absl::Status WriteShort(int addr, uint16_t value) {
317 if (addr + 1 >= rom_data_.size()) {
318 return absl::InvalidArgumentError(absl::StrFormat(
319 "Attempt to write short %#04x value failed, address %d out of range",
320 value, addr));
321 }
322 rom_data_[addr] = (uint8_t)(value & 0xFF);
323 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
324 core::Logger::log(absl::StrFormat("WriteShort: %#06X: %s", addr,
325 core::UppercaseHexWord(value)));
326 return absl::OkStatus();
327 }
328
329 absl::Status WriteLong(uint32_t addr, uint32_t value) {
330 if (addr + 2 >= rom_data_.size()) {
331 return absl::InvalidArgumentError(absl::StrFormat(
332 "Attempt to write long %#06x value failed, address %d out of range",
333 value, addr));
334 }
335 rom_data_[addr] = (uint8_t)(value & 0xFF);
336 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
337 rom_data_[addr + 2] = (uint8_t)((value >> 16) & 0xFF);
338 core::Logger::log(absl::StrFormat("WriteLong: %#06X: %s", addr,
339 core::UppercaseHexLong(value)));
340 return absl::OkStatus();
341 }
342
343 absl::Status WriteVector(int addr, std::vector<uint8_t> data) {
344 if (addr + data.size() > rom_data_.size()) {
345 return absl::InvalidArgumentError(absl::StrFormat(
346 "Attempt to write vector value failed, address %d out of range",
347 addr));
348 }
349 for (int i = 0; i < data.size(); i++) {
350 rom_data_[addr + i] = data[i];
351 }
352 core::Logger::log(absl::StrFormat("WriteVector: %#06X: %s", addr,
353 core::UppercaseHexByte(data[0])));
354 return absl::OkStatus();
355 }

357 absl::Status WriteColor(uint32_t address, const gfx::SnesColor& color) {
358 uint16_t bgr = ((color.snes() >> 10) & 0x1F) |
359 ((color.snes() & 0x1F) << 10) | (color.snes() & 0x7C00);
360
361 // Write the 16-bit color value to the ROM at the specified address
362 core::Logger::log(absl::StrFormat("WriteColor: %#06X: %s", address,
363 core::UppercaseHexWord(bgr)));
364 return WriteShort(address, bgr);
365 }

466 absl::Status WriteHelper(const WriteAction& action) {
467 if (std::holds_alternative<uint8_t>(action.value)) {
468 return Write(action.address, std::get<uint8_t>(action.value));
469 } else if (std::holds_alternative<uint16_t>(action.value) ||
470 std::holds_alternative<short>(action.value)) {
471 return WriteShort(action.address, std::get<uint16_t>(action.value));
472 } else if (std::holds_alternative<std::vector<uint8_t>>(action.value)) {
473 return WriteVector(action.address,
474 std::get<std::vector<uint8_t>>(action.value));
475 } else if (std::holds_alternative<gfx::SnesColor>(action.value)) {
476 return WriteColor(action.address, std::get<gfx::SnesColor>(action.value));
477 } else if (std::holds_alternative<std::vector<gfx::SnesColor>>(
478 action.value)) {
479 return absl::UnimplementedError(
480 "WriteHelper: std::vector<gfx::SnesColor>");
481 }
482 auto error_message = absl::StrFormat("Invalid write argument type: %s",
483 typeid(action.value).name());
484 throw std::runtime_error(error_message);
485 return absl::InvalidArgumentError(error_message);
486 }

488 template <typename T>
489 absl::Status ReadHelper(T& var, int address) {
490 if constexpr (std::is_same_v<T, uint8_t>) {
491 ASSIGN_OR_RETURN(auto result, ReadByte(address));
492 var = result;
493 } else if constexpr (std::is_same_v<T, uint16_t>) {
494 ASSIGN_OR_RETURN(auto result, ReadWord(address));
495 var = result;
496 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
497 ASSIGN_OR_RETURN(auto result, ReadByteVector(address, var.size()));
498 var = result;
499 }
500 return absl::OkStatus();
501 }
