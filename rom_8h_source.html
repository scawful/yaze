146 template <typename... Args>
147 absl::Status RunTransaction(Args... args) {
148 absl::Status status;
149 // Fold expression to apply the Write function on each argument
150 ((status = WriteHelper(args)), ...);
151 return status;
152 }
153
154 absl::Status WriteHelper(const WriteAction& action) {
155 if (std::holds_alternative<uint8_t>(action.value)) {
156 return Write(action.address, std::get<uint8_t>(action.value));
157 } else if (std::holds_alternative<uint16_t>(action.value) ||
158 std::holds_alternative<short>(action.value)) {
159 return WriteShort(action.address, std::get<uint16_t>(action.value));
160 } else if (std::holds_alternative<std::vector<uint8_t>>(action.value)) {
161 return WriteVector(action.address,
162 std::get<std::vector<uint8_t>>(action.value));
163 } else if (std::holds_alternative<gfx::SnesColor>(action.value)) {
164 return WriteColor(action.address, std::get<gfx::SnesColor>(action.value));
165 } else if (std::holds_alternative<std::vector<gfx::SnesColor>>(
166 action.value)) {
167 return absl::UnimplementedError(
168 "WriteHelper: std::vector<gfx::SnesColor>");
169 }
170 auto error_message = absl::StrFormat("Invalid write argument type: %s",
171 typeid(action.value).name());
172 throw std::runtime_error(error_message);
173 return absl::InvalidArgumentError(error_message);
174 }
175
176 template <typename T, typename... Args>
177 absl::Status ReadTransaction(T& var, int address, Args&&... args) {
178 absl::Status status = ReadHelper<T>(var, address);
179 if (!status.ok()) {
180 return status;
181 }
182
183 if constexpr (sizeof...(args) > 0) {
184 status = ReadTransaction(std::forward<Args>(args)...);
185 }
186
187 return status;
188 }

190 template <typename T>
191 absl::Status ReadHelper(T& var, int address) {
192 if constexpr (std::is_same_v<T, uint8_t>) {
193 ASSIGN_OR_RETURN(auto result, ReadByte(address));
194 var = result;
195 } else if constexpr (std::is_same_v<T, uint16_t>) {
196 ASSIGN_OR_RETURN(auto result, ReadWord(address));
197 var = result;
198 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
199 ASSIGN_OR_RETURN(auto result, ReadByteVector(address, var.size()));
200 var = result;
201 }
202 return absl::OkStatus();
203 }

352 absl::StatusOr<gfx::Tile16> ReadTile16(uint32_t tile16_id) {
353 // Skip 8 bytes per tile.
354 auto tpos = 0x78000 + (tile16_id * 0x08);
355 gfx::Tile16 tile16;
356 ASSIGN_OR_RETURN(auto new_tile0, ReadWord(tpos))
357 tile16.tile0_ = gfx::WordToTileInfo(new_tile0);
358 tpos += 2;
359 ASSIGN_OR_RETURN(auto new_tile1, ReadWord(tpos))
360 tile16.tile1_ = gfx::WordToTileInfo(new_tile1);
361 tpos += 2;
362 ASSIGN_OR_RETURN(auto new_tile2, ReadWord(tpos))
363 tile16.tile2_ = gfx::WordToTileInfo(new_tile2);
364 tpos += 2;
365 ASSIGN_OR_RETURN(auto new_tile3, ReadWord(tpos))
366 tile16.tile3_ = gfx::WordToTileInfo(new_tile3);
367 return tile16;
368 }

370 absl::Status WriteTile16(int tile16_id, const gfx::Tile16& tile) {
371 // Skip 8 bytes per tile.
372 auto tpos = 0x78000 + (tile16_id * 0x08);
373 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile0_)));
374 tpos += 2;
375 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile1_)));
376 tpos += 2;
377 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile2_)));
378 tpos += 2;
379 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile3_)));
380 return absl::OkStatus();
381 }

394 absl::Status WriteByte(int addr, uint8_t value) {
395 if (addr >= rom_data_.size()) {
396 return absl::InvalidArgumentError(absl::StrFormat(
397 "Attempt to write byte %#02x value failed, address %d out of range",
398 value, addr));
399 }
400 rom_data_[addr] = value;
401 std::string log_str = absl::StrFormat("WriteByte: %#06X: %s", addr,
402 core::UppercaseHexByte(value).data());
403 core::Logger::log(log_str);
404 return absl::OkStatus();
405 }
406
407 absl::Status WriteWord(int addr, uint16_t value) {
408 if (addr + 1 >= rom_data_.size()) {
409 return absl::InvalidArgumentError(absl::StrFormat(
410 "Attempt to write word %#04x value failed, address %d out of range",
411 value, addr));
412 }
413 rom_data_[addr] = (uint8_t)(value & 0xFF);
414 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
415 core::Logger::log(absl::StrFormat("WriteWord: %#06X: %s", addr,
416 core::UppercaseHexWord(value)));
417 return absl::OkStatus();
418 }
419
420 absl::Status WriteShort(int addr, uint16_t value) {
421 if (addr + 1 >= rom_data_.size()) {
422 return absl::InvalidArgumentError(absl::StrFormat(
423 "Attempt to write short %#04x value failed, address %d out of range",
424 value, addr));
425 }
426 rom_data_[addr] = (uint8_t)(value & 0xFF);
427 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
428 core::Logger::log(absl::StrFormat("WriteShort: %#06X: %s", addr,
429 core::UppercaseHexWord(value)));
430 return absl::OkStatus();
431 }
432
433 absl::Status WriteLong(uint32_t addr, uint32_t value) {
434 if (addr + 2 >= rom_data_.size()) {
435 return absl::InvalidArgumentError(absl::StrFormat(
436 "Attempt to write long %#06x value failed, address %d out of range",
437 value, addr));
438 }
439 rom_data_[addr] = (uint8_t)(value & 0xFF);
440 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
441 rom_data_[addr + 2] = (uint8_t)((value >> 16) & 0xFF);
442 core::Logger::log(absl::StrFormat("WriteLong: %#06X: %s", addr,
443 core::UppercaseHexLong(value)));
444 return absl::OkStatus();
445 }
446
447 absl::Status WriteVector(int addr, std::vector<uint8_t> data) {
448 if (addr + data.size() > rom_data_.size()) {
449 return absl::InvalidArgumentError(absl::StrFormat(
450 "Attempt to write vector value failed, address %d out of range",
451 addr));
452 }
453 for (int i = 0; i < data.size(); i++) {
454 rom_data_[addr + i] = data[i];
455 }
456 core::Logger::log(absl::StrFormat("WriteVector: %#06X: %s", addr,
457 core::UppercaseHexByte(data[0])));
458 return absl::OkStatus();
459 }

461 absl::Status WriteColor(uint32_t address, const gfx::SnesColor& color) {
462 uint16_t bgr = ((color.snes() >> 10) & 0x1F) |

598
599 int gfxPointer = (rom_data_[gfx_groups_pointer + 1] << 8) +
600 rom_data_[gfx_groups_pointer];
601 gfxPointer = core::SnesToPc(gfxPointer);
602
603 for (int i = 0; i < 37; i++) {
604 for (int j = 0; j < 8; j++) {
605 main_blockset_ids[i][j] = rom_data_[gfxPointer + (i * 8) + j];
606 }
607 }
608

610 for (int j = 0; j < 4; j++) {
611 room_blockset_ids[i][j] =
612 rom_data_[core::entrance_gfx_group + (i * 4) + j];

614 }

619 rom_data_[version_constants().kSpriteBlocksetPointer + (i * 4) + j];
