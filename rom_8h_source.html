137 absl::StatusOr<uint8_t> ReadByte(int offset) {
138 RETURN_IF_ERROR(ReadWritePreconditions());

145 absl::StatusOr<uint16_t> ReadWord(int offset) {
146 RETURN_IF_ERROR(ReadWritePreconditions());
147 if (offset + 1 >= static_cast<int>(rom_data_.size())) {
148 return absl::FailedPreconditionError("Offset out of range");
149 }
150 auto result = (uint16_t)(rom_data_[offset] | (rom_data_[offset + 1] << 8));

157 return absl::OutOfRangeError("Offset out of range");

159 auto result = (uint32_t)(rom_data_[offset] | (rom_data_[offset + 1] << 8) |

164 absl::StatusOr<std::vector<uint8_t>> ReadByteVector(uint32_t offset,
165 uint32_t length) const {
166 RETURN_IF_ERROR(ReadWritePreconditions());

168 return absl::OutOfRangeError("Offset and length out of range");
169 }

177 absl::StatusOr<gfx::Tile16> ReadTile16(uint32_t tile16_id) {
178 // Skip 8 bytes per tile.
179 auto tpos = kTile16Ptr + (tile16_id * 0x08);
180 gfx::Tile16 tile16 = {};
181 ASSIGN_OR_RETURN(auto new_tile0, ReadWord(tpos))
182 tile16.tile0_ = gfx::WordToTileInfo(new_tile0);
183 tpos += 2;
184 ASSIGN_OR_RETURN(auto new_tile1, ReadWord(tpos))
185 tile16.tile1_ = gfx::WordToTileInfo(new_tile1);
186 tpos += 2;
187 ASSIGN_OR_RETURN(auto new_tile2, ReadWord(tpos))
188 tile16.tile2_ = gfx::WordToTileInfo(new_tile2);
189 tpos += 2;
190 ASSIGN_OR_RETURN(auto new_tile3, ReadWord(tpos))
191 tile16.tile3_ = gfx::WordToTileInfo(new_tile3);
192 return tile16;
193 }

195 absl::Status WriteTile16(int tile16_id, const gfx::Tile16& tile) {
196 // Skip 8 bytes per tile.
197 auto tpos = kTile16Ptr + (tile16_id * 0x08);
198 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile0_)));
199 tpos += 2;
200 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile1_)));
201 tpos += 2;
202 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile2_)));
203 tpos += 2;
204 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile3_)));
205 return absl::OkStatus();
206 }

208 absl::Status WriteByte(int addr, uint8_t value) {
209 RETURN_IF_ERROR(ReadWritePreconditions());
210 if (addr >= static_cast<int>(rom_data_.size())) {
211 return absl::OutOfRangeError(absl::StrFormat(
212 "Attempt to write byte %#02x value failed, address %d out of range",
213 value, addr));
214 }
215 rom_data_[addr] = value;
216 core::logf("WriteByte: %#06X: %s", addr, core::HexByte(value).data());
217 return absl::OkStatus();
218 }
219
220 absl::Status WriteWord(int addr, uint16_t value) {
221 RETURN_IF_ERROR(ReadWritePreconditions());
222 if (addr + 1 >= static_cast<int>(rom_data_.size())) {
223 return absl::OutOfRangeError(absl::StrFormat(
224 "Attempt to write word %#04x value failed, address %d out of range",
225 value, addr));
226 }
227 rom_data_[addr] = (uint8_t)(value & 0xFF);
228 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
229 core::logf("WriteWord: %#06X: %s", addr, core::HexWord(value).data());
230 return absl::OkStatus();
231 }

246 absl::Status WriteLong(uint32_t addr, uint32_t value) {
247 RETURN_IF_ERROR(ReadWritePreconditions());
248 if (addr + 2 >= static_cast<uint32_t>(rom_data_.size())) {
249 return absl::OutOfRangeError(absl::StrFormat(

260 absl::Status WriteVector(int addr, std::vector<uint8_t> data) {
261 if (addr + static_cast<int>(data.size()) >
262 static_cast<int>(rom_data_.size())) {
263 return absl::InvalidArgumentError(absl::StrFormat(
264 "Attempt to write vector value failed, address %d out of range",
265 addr));
266 }
267 for (int i = 0; i < static_cast<int>(data.size()); i++) {
268 rom_data_[addr + i] = data[i];
269 }
270 core::logf("WriteVector: %#06X: %s", addr, core::HexByte(data[0]).data());
271 return absl::OkStatus();
272 }

274 absl::Status WriteColor(uint32_t address, const gfx::SnesColor& color) {
275 uint16_t bgr = ((color.snes() >> 10) & 0x1F) |
276 ((color.snes() & 0x1F) << 10) | (color.snes() & 0x7C00);
277
278 // Write the 16-bit color value to the ROM at the specified address
279 core::logf("WriteColor: %#06X: %s", address, core::HexWord(bgr).data());
280 return WriteShort(address, bgr);

351 struct WriteAction {
352 int address;
353 std::variant<int, uint8_t, uint16_t, short, std::vector<uint8_t>,
354 gfx::SnesColor, std::vector<gfx::SnesColor>>
355 value;
356 };
357
358 private:
359 virtual absl::Status WriteHelper(const WriteAction& action) {
360 if (std::holds_alternative<uint8_t>(action.value)) {
361 return WriteByte(action.address, std::get<uint8_t>(action.value));
362 } else if (std::holds_alternative<uint16_t>(action.value) ||
363 std::holds_alternative<short>(action.value)) {
364 return WriteShort(action.address, std::get<uint16_t>(action.value));
365 } else if (std::holds_alternative<std::vector<uint8_t>>(action.value)) {
366 return WriteVector(action.address,
367 std::get<std::vector<uint8_t>>(action.value));
368 } else if (std::holds_alternative<gfx::SnesColor>(action.value)) {
369 return WriteColor(action.address, std::get<gfx::SnesColor>(action.value));
370 } else if (std::holds_alternative<std::vector<gfx::SnesColor>>(
371 action.value)) {
372 return absl::UnimplementedError(
373 "WriteHelper: std::vector<gfx::SnesColor>");
374 }
375 auto error_message = absl::StrFormat("Invalid write argument type: %s",
376 typeid(action.value).name());
377 return absl::InvalidArgumentError(error_message);
378 }

380 template <typename T>
381 absl::Status ReadHelper(T& var, int address) {
382 if constexpr (std::is_same_v<T, uint8_t>) {
383 ASSIGN_OR_RETURN(auto result, ReadByte(address));
384 var = result;
385 } else if constexpr (std::is_same_v<T, uint16_t>) {
386 ASSIGN_OR_RETURN(auto result, ReadWord(address));
387 var = result;
388 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
389 ASSIGN_OR_RETURN(auto result, ReadByteVector(address, var.size()));
390 var = result;
391 }
392 return absl::OkStatus();
393 }
