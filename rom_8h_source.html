270 absl::StatusOr<gfx::Tile16> ReadTile16(uint32_t tile16_id) {
271 // Skip 8 bytes per tile.
272 auto tpos = 0x78000 + (tile16_id * 0x08);
273 gfx::Tile16 tile16;
274 ASSIGN_OR_RETURN(auto new_tile0, ReadWord(tpos))
275 tile16.tile0_ = gfx::WordToTileInfo(new_tile0);
276 tpos += 2;
277 ASSIGN_OR_RETURN(auto new_tile1, ReadWord(tpos))
278 tile16.tile1_ = gfx::WordToTileInfo(new_tile1);
279 tpos += 2;

281 tile16.tile2_ = gfx::WordToTileInfo(new_tile2);
282 tpos += 2;

325 absl::Status WriteWord(int addr, uint16_t value) {
326 if (addr + 1 >= rom_data_.size()) {
327 return absl::InvalidArgumentError(absl::StrFormat(
328 "Attempt to write word %#04x value failed, address %d out of range",
329 value, addr));

332 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
333 core::Logger::log(absl::StrFormat("WriteWord: %#06X: %s", addr,
334 core::UppercaseHexWord(value)));

548 absl::Status WriteHelper(const WriteAction& action) {
549 if (std::holds_alternative<uint8_t>(action.value)) {
550 return Write(action.address, std::get<uint8_t>(action.value));
551 } else if (std::holds_alternative<uint16_t>(action.value) ||
552 std::holds_alternative<short>(action.value)) {
553 return WriteShort(action.address, std::get<uint16_t>(action.value));
554 } else if (std::holds_alternative<std::vector<uint8_t>>(action.value)) {
555 return WriteVector(action.address,
556 std::get<std::vector<uint8_t>>(action.value));
557 } else if (std::holds_alternative<gfx::SnesColor>(action.value)) {
558 return WriteColor(action.address, std::get<gfx::SnesColor>(action.value));
559 } else if (std::holds_alternative<std::vector<gfx::SnesColor>>(
560 action.value)) {
561 return absl::UnimplementedError(
562 "WriteHelper: std::vector<gfx::SnesColor>");
563 }
564 auto error_message = absl::StrFormat("Invalid write argument type: %s",
565 typeid(action.value).name());
566 throw std::runtime_error(error_message);
567 return absl::InvalidArgumentError(error_message);
568 }

570 template <typename T>
571 absl::Status ReadHelper(T& var, int address) {
572 if constexpr (std::is_same_v<T, uint8_t>) {
573 ASSIGN_OR_RETURN(auto result, ReadByte(address));
574 var = result;
575 } else if constexpr (std::is_same_v<T, uint16_t>) {
576 ASSIGN_OR_RETURN(auto result, ReadWord(address));
577 var = result;
578 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
579 ASSIGN_OR_RETURN(auto result, ReadByteVector(address, var.size()));
580 var = result;
581 }
582 return absl::OkStatus();
583 }
