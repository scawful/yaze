218 absl::StatusOr<uint16_t> ReadWord(int offset) {
219 RETURN_IF_ERROR(ReadWritePreconditions());
220 if (offset + 1 >= static_cast<int>(rom_data_.size())) {
221 return absl::FailedPreconditionError("Offset out of range");
222 }
223 auto result = (uint16_t)(rom_data_[offset] | (rom_data_[offset + 1] << 8));
224 return result;
225 }

227 absl::StatusOr<uint32_t> ReadLong(int offset) {
228 RETURN_IF_ERROR(ReadWritePreconditions());
229 if (offset + 2 >= static_cast<int>(rom_data_.size())) {
230 return absl::OutOfRangeError("Offset out of range");
231 }
232 auto result = (uint32_t)(rom_data_[offset] | (rom_data_[offset + 1] << 8) |
233 (rom_data_[offset + 2] << 16));
234 return result;

250 absl::StatusOr<gfx::Tile16> ReadTile16(uint32_t tile16_id) {
251 // Skip 8 bytes per tile.
252 auto tpos = kTile16Ptr + (tile16_id * 0x08);
253 gfx::Tile16 tile16 = {};
254 ASSIGN_OR_RETURN(auto new_tile0, ReadWord(tpos))
255 tile16.tile0_ = gfx::WordToTileInfo(new_tile0);
256 tpos += 2;
257 ASSIGN_OR_RETURN(auto new_tile1, ReadWord(tpos))
258 tile16.tile1_ = gfx::WordToTileInfo(new_tile1);
259 tpos += 2;
260 ASSIGN_OR_RETURN(auto new_tile2, ReadWord(tpos))
261 tile16.tile2_ = gfx::WordToTileInfo(new_tile2);
262 tpos += 2;
263 ASSIGN_OR_RETURN(auto new_tile3, ReadWord(tpos))
264 tile16.tile3_ = gfx::WordToTileInfo(new_tile3);
265 return tile16;
266 }

268 absl::Status WriteTile16(int tile16_id, const gfx::Tile16& tile) {
269 // Skip 8 bytes per tile.
270 auto tpos = kTile16Ptr + (tile16_id * 0x08);
271 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile0_)));
272 tpos += 2;

274 tpos += 2;
275 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile2_)));
276 tpos += 2;
277 RETURN_IF_ERROR(WriteShort(tpos, gfx::TileInfoToWord(tile.tile3_)));
278 return absl::OkStatus();
279 }

281 absl::Status WriteByte(int addr, uint8_t value) {
282 RETURN_IF_ERROR(ReadWritePreconditions());
283 if (addr >= static_cast<int>(rom_data_.size())) {
284 return absl::OutOfRangeError(absl::StrFormat(
285 "Attempt to write byte %#02x value failed, address %d out of range",
286 value, addr));
287 }
288 rom_data_[addr] = value;
289 core::logf("WriteByte: %#06X: %s", addr, core::HexByte(value).data());
290 return absl::OkStatus();
291 }
292
293 absl::Status WriteWord(int addr, uint16_t value) {
294 RETURN_IF_ERROR(ReadWritePreconditions());
295 if (addr + 1 >= static_cast<int>(rom_data_.size())) {
296 return absl::OutOfRangeError(absl::StrFormat(
297 "Attempt to write word %#04x value failed, address %d out of range",
298 value, addr));
299 }
300 rom_data_[addr] = (uint8_t)(value & 0xFF);
301 rom_data_[addr + 1] = (uint8_t)((value >> 8) & 0xFF);
302 core::logf("WriteWord: %#06X: %s", addr, core::HexWord(value).data());
303 return absl::OkStatus();
304 }

333 absl::Status WriteVector(int addr, std::vector<uint8_t> data) {
334 if (addr + static_cast<int>(data.size()) >
335 static_cast<int>(rom_data_.size())) {
336 return absl::InvalidArgumentError(absl::StrFormat(
337 "Attempt to write vector value failed, address %d out of range",
338 addr));
339 }
340 for (int i = 0; i < static_cast<int>(data.size()); i++) {
341 rom_data_[addr + i] = data[i];

347 absl::Status WriteColor(uint32_t address, const gfx::SnesColor& color) {
348 uint16_t bgr = ((color.snes() >> 10) & 0x1F) |
349 ((color.snes() & 0x1F) << 10) | (color.snes() & 0x7C00);
350
351 // Write the 16-bit color value to the ROM at the specified address
352 core::logf("WriteColor: %#06X: %s", address, core::HexWord(bgr).data());
353 return WriteShort(address, bgr);

357 absl::Status WriteTransaction(Args... args) {
358 absl::Status status;
359 // Fold expression to apply the Write function on each argument
360 ((status = WriteHelper(args)), ...);

364 template <typename T, typename... Args>
365 absl::Status ReadTransaction(T& var, int address, Args&&... args) {
366 absl::Status status = ReadHelper<T>(var, address);
367 if (!status.ok()) {
368 return status;
369 }
370
371 if constexpr (sizeof...(args) > 0) {
372 status = ReadTransaction(std::forward<Args>(args)...);
373 }
374
375 return status;

424 struct WriteAction {
425 int address;
426 std::variant<int, uint8_t, uint16_t, short, std::vector<uint8_t>,
427 gfx::SnesColor, std::vector<gfx::SnesColor>>
428 value;
429 };
430
431 private:
432 virtual absl::Status WriteHelper(const WriteAction& action) {
433 if (std::holds_alternative<uint8_t>(action.value)) {
434 return WriteByte(action.address, std::get<uint8_t>(action.value));
435 } else if (std::holds_alternative<uint16_t>(action.value) ||
436 std::holds_alternative<short>(action.value)) {
437 return WriteShort(action.address, std::get<uint16_t>(action.value));
438 } else if (std::holds_alternative<std::vector<uint8_t>>(action.value)) {
439 return WriteVector(action.address,
440 std::get<std::vector<uint8_t>>(action.value));
441 } else if (std::holds_alternative<gfx::SnesColor>(action.value)) {
442 return WriteColor(action.address, std::get<gfx::SnesColor>(action.value));
443 } else if (std::holds_alternative<std::vector<gfx::SnesColor>>(
444 action.value)) {
445 return absl::UnimplementedError(
446 "WriteHelper: std::vector<gfx::SnesColor>");
447 }
448 auto error_message = absl::StrFormat("Invalid write argument type: %s",
449 typeid(action.value).name());
450 return absl::InvalidArgumentError(error_message);
451 }

453 template <typename T>
454 absl::Status ReadHelper(T& var, int address) {
455 if constexpr (std::is_same_v<T, uint8_t>) {
456 ASSIGN_OR_RETURN(auto result, ReadByte(address));
457 var = result;
458 } else if constexpr (std::is_same_v<T, uint16_t>) {
459 ASSIGN_OR_RETURN(auto result, ReadWord(address));
460 var = result;
461 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
462 ASSIGN_OR_RETURN(auto result, ReadByteVector(address, var.size()));
463 var = result;
464 }
465 return absl::OkStatus();
466 }
