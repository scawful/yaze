170std::vector<std::string> yaze::core::FileDialogWrapper::GetSubdirectoriesInFolder(
171 const std::string& folder) {
172 std::vector<std::string> subdirectories;
173 NSFileManager* fileManager = [NSFileManager defaultManager];
174 NSDirectoryEnumerator* enumerator =
175 [fileManager enumeratorAtPath:[NSString stringWithUTF8String:folder.c_str()]];
176 NSString* file;
177 while (file = [enumerator nextObject]) {
178 if ([file hasPrefix:@"."]) {
179 continue;
180 }
181 BOOL isDirectory;
182 NSString* path =
183 [NSString stringWithFormat:@"%@/%@", [NSString stringWithUTF8String:folder.c_str()], file];
184 [fileManager fileExistsAtPath:path isDirectory:&isDirectory];
185 if (isDirectory) {
186 subdirectories.push_back(std::string([file UTF8String]));
187 }
188 }
189 return subdirectories;
190}
