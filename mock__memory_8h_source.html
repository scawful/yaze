121 // 02MB = 0x200000
122 void Initialize(const std::vector<uint8_t>& romData) {
123 // 16 MB, simplifying the memory layout for testing
124 memory_.resize(0x1000000);
125
126 // Clear memory
127 std::fill(memory_.begin(), memory_.end(), 0);
128
129 // Load ROM data into mock memory
130 size_t romSize = romData.size();
131 size_t romAddress = 0;
132 const size_t ROM_CHUNK_SIZE = 0x8000; // 32 KB
133 for (size_t bank = 0x00; bank <= 0xBF; bank += 0x80) {
134 for (size_t offset = 0x8000; offset <= 0xFFFF; offset += ROM_CHUNK_SIZE) {
135 if (romAddress < romSize) {
136 std::copy(romData.begin() + romAddress,
137 romData.begin() + romAddress + ROM_CHUNK_SIZE,
138 memory_.begin() + (bank << 16) + offset);
139 romAddress += ROM_CHUNK_SIZE;
140 }
141 }
142 }
143 }
