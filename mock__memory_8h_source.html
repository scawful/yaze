115 // Load ROM data into mock memory
116 size_t romSize = romData.size();
117 size_t romAddress = 0;
118 const size_t ROM_CHUNK_SIZE = 0x8000; // 32 KB
119 for (size_t bank = 0x00; bank <= 0xBF; bank += 0x80) {
120 for (size_t offset = 0x8000; offset <= 0xFFFF; offset += ROM_CHUNK_SIZE) {
121 if (romAddress < romSize) {
122 std::copy(romData.begin() + romAddress,
123 romData.begin() + romAddress + ROM_CHUNK_SIZE,
124 memory_.begin() + (bank << 16) + offset);
