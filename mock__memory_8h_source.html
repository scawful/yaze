120 // 02MB = 0x200000
121 void Initialize(const std::vector<uint8_t>& romData) {
122 // 16 MB, simplifying the memory layout for testing
123 memory_.resize(0x1000000);
124
125 // Clear memory
126 std::fill(memory_.begin(), memory_.end(), 0);
127
128 // Load ROM data into mock memory
129 size_t romSize = romData.size();
130 size_t romAddress = 0;
131 const size_t ROM_CHUNK_SIZE = 0x8000; // 32 KB
132 for (size_t bank = 0x00; bank <= 0xBF; bank += 0x80) {
133 for (size_t offset = 0x8000; offset <= 0xFFFF; offset += ROM_CHUNK_SIZE) {
134 if (romAddress < romSize) {
135 std::copy(romData.begin() + romAddress,
136 romData.begin() + romAddress + ROM_CHUNK_SIZE,
137 memory_.begin() + (bank << 16) + offset);
138 romAddress += ROM_CHUNK_SIZE;
139 }
140 }
141 }
