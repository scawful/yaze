172 if (is_running_) {

174 }
175

234absl::Status TestManager::RunTestSuite(const std::string& suite_name) {
235 if (is_running_) {
236 return absl::FailedPreconditionError("Tests are already running");
237 }
238
239 auto it = suite_lookup_.find(suite_name);
240 if (it == suite_lookup_.end()) {
241 return absl::NotFoundError("Test suite not found: " + suite_name);
242 }
243
244 is_running_ = true;
245 progress_ = 0.0f;
246 last_results_.Clear();
247 current_test_name_ = suite_name;
248
249 auto status = ExecuteTestSuite(it->second);
250
251 is_running_ = false;
252 current_test_name_.clear();
253 progress_ = 1.0f;
254
255 return status;
256}

1311std::string TestManager::GenerateTestRomFilename(
1312 const std::string& base_name) {
1313 // Generate filename with timestamp
1314 auto now = std::chrono::system_clock::now();
1315 auto time_t = std::chrono::system_clock::to_time_t(now);
1316 auto local_time = *std::localtime(&time_t);
1317
1318 std::string timestamp =
1319 absl::StrFormat("%04d%02d%02d_%02d%02d%02d", local_time.tm_year + 1900,
1320 local_time.tm_mon + 1, local_time.tm_mday,
1321 local_time.tm_hour, local_time.tm_min, local_time.tm_sec);
1322
1323 std::string base_filename = base_name;
1324 // Remove any path and extension
1325 auto last_slash = base_filename.find_last_of("/\\");
1326 if (last_slash != std::string::npos) {
1327 base_filename = base_filename.substr(last_slash + 1);
1328 }
1329 auto last_dot = base_filename.find_last_of('.');
1330 if (last_dot != std::string::npos) {
1331 base_filename = base_filename.substr(0, last_dot);
1332 }
1333
1334 return absl::StrFormat("%s_test_%s.sfc", base_filename.c_str(),
1335 timestamp.c_str());
1336}
