131 // Performance measurement helpers
132 struct PerformanceMetrics {
133 std::chrono::milliseconds render_time;
134 size_t objects_rendered;
135 size_t memory_used;
136 size_t cache_hits;
137 size_t cache_misses;

140 PerformanceMetrics MeasureRenderPerformance(const std::vector<RoomObject>& objects,
141 const gfx::SnesPalette& palette) {
142 auto start_time = std::chrono::high_resolution_clock::now();
143
144 auto stats_before = object_renderer_->GetPerformanceStats();
145
146 auto result = object_renderer_->RenderObjects(objects, palette);
147
148 auto end_time = std::chrono::high_resolution_clock::now();
149 auto stats_after = object_renderer_->GetPerformanceStats();
150
151 PerformanceMetrics metrics;
152 metrics.render_time = std::chrono::duration_cast<std::chrono::milliseconds>(
153 end_time - start_time);
154 metrics.objects_rendered = objects.size();
155 metrics.cache_hits = stats_after.cache_hits - stats_before.cache_hits;
156 metrics.cache_misses = stats_after.cache_misses - stats_before.cache_misses;
157 metrics.memory_used = object_renderer_->GetMemoryUsage();
158
159 return metrics;
160 }
