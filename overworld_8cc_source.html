29absl::flat_hash_map<int, MapData> parseFile(const std::string &filename) {
30 absl::flat_hash_map<int, MapData> resultMap;
31
32 std::ifstream file(filename);
33 if (!file.is_open()) {
34 std::cerr << "Failed to open file: " << filename << std::endl;
35 return resultMap;
36 }
37
38 std::string line;
39 int currentKey;
40 bool isHigh = true;
41
42 while (getline(file, line)) {
43 // Skip empty or whitespace-only lines
44 if (line.find_first_not_of(" \t\r\n") == std::string::npos) {
45 continue;
46 }
47
48 // If the line starts with "MAPDTH" or "MAPDTL", extract the ID.
49 if (line.find("MAPDTH") == 0) {
50 auto num_str = line.substr(6); // Extract ID after "MAPDTH"
51 currentKey = std::stoi(num_str);
52 isHigh = true;
53 } else if (line.find("MAPDTL") == 0) {
54 auto num_str = line.substr(6); // Extract ID after "MAPDTH"
55 currentKey = std::stoi(num_str);
56 isHigh = false;
57 } else {
58 // Check if the currentKey is already in the map. If not, initialize it.
59 if (resultMap.find(currentKey) == resultMap.end()) {
60 resultMap[currentKey] = MapData();
61 }
62
63 // Split the line by commas and convert to uint8_t.
64 std::stringstream ss(line);
65 std::string valueStr;
66 while (getline(ss, valueStr, ',')) {
67 uint8_t value = std::stoi(valueStr, nullptr, 16);
68 if (isHigh) {
69 resultMap[currentKey].highData.emplace_back(value);
70 } else {
71 resultMap[currentKey].lowData.emplace_back(value);
72 }
73 }
74 }
75 }
76
77 return resultMap;
78}

600
601 auto compareArray = [](const std::vector<uint8_t> &array1,
602 const std::vector<uint8_t> &array2) -> bool {
603 if (array1.size() != array2.size()) {
604 return false;
605 }
606
607 for (size_t i = 0; i < array1.size(); i++) {
608 if (array1[i] != array2[i]) {
609 return false;
610 }
611 }
612
613 return true;
614 };
615
616 for (int j = 0; j < i; j++) {
617 if (compareArray(a, map_data_p1[j])) {
618 // Reuse pointer id j for P1 (a)
619 map_pointers1_id[i] = j;
620 }
621
622 if (compareArray(b, map_data_p2[j])) {
623 map_pointers2_id[i] = j;
624 // Reuse pointer id j for P2 (b)
625 }
626 }
627
628 if (map_pointers1_id[i] == -1) {
629 // Save compressed data and pointer for map1
630 std::copy(a.begin(), a.end(), map_data_p1[i].begin());
631 int snes_pos = core::PcToSnes(pos);
632 map_pointers1[i] = snes_pos;
633 core::Logger::log("Saving map pointers1 and compressed data for map " +
634 core::UppercaseHexByte(i) + " at " +
635 core::UppercaseHexLong(snes_pos));
636 RETURN_IF_ERROR(
637 rom()->WriteLong(kCompressedAllMap32PointersLow + (3 * i), snes_pos));
638 RETURN_IF_ERROR(rom()->WriteVector(pos, a));
639 pos += size_a;
640 } else {
641 // Save pointer for map1
642 int snes_pos = map_pointers1[map_pointers1_id[i]];
643 core::Logger::log("Saving map pointers1 for map " +
644 core::UppercaseHexByte(i) + " at " +
645 core::UppercaseHexLong(snes_pos));
646 RETURN_IF_ERROR(
647 rom()->WriteLong(kCompressedAllMap32PointersLow + (3 * i), snes_pos));
648 }
649
650 if ((pos + b.size()) >= 0x5FE70 && (pos + b.size()) <= 0x60000) {
651 pos = 0x60000;
652 }
653
654 if ((pos + b.size()) >= 0x6411F && (pos + b.size()) <= 0x70000) {
655 core::Logger::log("Pos set to overflow region for map " +
656 core::UppercaseHexByte(i) + " at " +
657 core::UppercaseHexLong(pos));
658 pos = OverworldMapDataOverflow;
659 }
660
661 if (map_pointers2_id[i] == -1) {
662 // Save compressed data and pointer for map2
663 std::copy(b.begin(), b.end(), map_data_p2[i].begin());
664 int snes_pos = core::PcToSnes(pos);
665 map_pointers2[i] = snes_pos;
666 core::Logger::log("Saving map pointers2 and compressed data for map " +
667 core::UppercaseHexByte(i) + " at " +
668 core::UppercaseHexLong(snes_pos));
669 RETURN_IF_ERROR(rom()->WriteLong(
670 kCompressedAllMap32PointersHigh + (3 * i), snes_pos));
671 RETURN_IF_ERROR(rom()->WriteVector(pos, b));
672 pos += size_b;
673 } else {
674 // Save pointer for map2
675 int snes_pos = map_pointers2[map_pointers2_id[i]];
676 core::Logger::log("Saving map pointers2 for map " +
677 core::UppercaseHexByte(i) + " at " +
678 core::UppercaseHexLong(snes_pos));
679 RETURN_IF_ERROR(rom()->WriteLong(
680 kCompressedAllMap32PointersHigh + (3 * i), snes_pos));
681 }
682 }
683
684 // Check if too many maps data
685 if (pos > 0x137FFF) {
686 std::cerr << "Too many maps data " << std::hex << pos << std::endl;
687 return absl::AbortedError("Too many maps data " + std::to_string(pos));
688 }
689
690 // Save large maps
691 RETURN_IF_ERROR(SaveLargeMaps())
692
693 return absl::OkStatus();
694}

1055absl::Status Overworld::CreateTile32Tilemap() {
1056 tiles32_unique_.clear();
1057 tiles32_list_.clear();
1058
1059 // Get all tiles16 and packs them into tiles32
1060 std::vector<uint64_t> all_tile_16 = GetAllTile16(map_tiles_);
1061
1062 // Convert to set then back to vector
1063 std::set<uint64_t> unique_tiles_set(all_tile_16.begin(), all_tile_16.end());
1064
1065 std::vector<uint64_t> unique_tiles(all_tile_16);
1066 unique_tiles.assign(unique_tiles_set.begin(), unique_tiles_set.end());
1067
1068 // Create the indexed tiles list
1069 std::unordered_map<uint64_t, uint16_t> all_tiles_indexed;
1070 for (size_t tile32_id = 0; tile32_id < unique_tiles.size(); tile32_id++) {
1071 all_tiles_indexed.insert(
1072 {unique_tiles[tile32_id], static_cast<uint16_t>(tile32_id)});
1073 }
1074
1075 // Add all tiles32 from all maps.
1076 // Convert all tiles32 non-unique IDs into unique array of IDs.
1077 for (int j = 0; j < NumberOfMap32; j++) {
1078 tiles32_list_.emplace_back(all_tiles_indexed[all_tile_16[j]]);
1079 }
1080
1081 // Create the unique tiles list
1082 for (int i = 0; i < unique_tiles.size(); ++i) {
1083 // static_cast<uint16_t>(tile)
1084 tiles32_unique_.emplace_back(gfx::Tile32(unique_tiles[i]));
1085 }
1086
1087 while (tiles32_unique_.size() % 4 != 0) {
1088 gfx::Tile32 padding_tile(0, 0, 0, 0);
1089 tiles32_unique_.emplace_back(padding_tile.GetPackedValue());
1090 }
1091
1092 if (tiles32_unique_.size() > LimitOfMap32) {
1093 return absl::InternalError(absl::StrFormat(
1094 "Number of unique Tiles32: %d Out of: %d\nUnique Tile32 count exceed "
1095 "the limit\nThe ROM Has not been saved\nYou can fill maps with grass "
1096 "tiles to free some space\nOr use the option Clear DW Tiles in the "
1097 "Overworld Menu",
1098 unique_tiles.size(), LimitOfMap32));
1099 }
1100
1101 if (flags()->kLogToConsole) {
1102 std::cout << "Number of unique Tiles32: " << tiles32_unique_.size()
1103 << " Saved:" << tiles32_unique_.size()
1104 << " Out of: " << LimitOfMap32 << std::endl;
1105 }
1106
1107 int v = tiles32_unique_.size();
1108 for (int i = v; i < LimitOfMap32; i++) {
1109 gfx::Tile32 padding_tile(420, 420, 420, 420);
1110 tiles32_unique_.emplace_back(padding_tile.GetPackedValue());
1111 }
1112
1113 return absl::OkStatus();
1114}
