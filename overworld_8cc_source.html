125 // Loop through each quadrant of the 32x32 pixel tile.

127 // Generate the 16-bit tile for the current quadrant of the current
128 // 32x32 pixel tile.
129 ASSIGN_OR_RETURN(
130 uint16_t tl,
131 GetTile16ForTile32(i, k, (int)Dimension::map32TilesTL, map32address));
132 ASSIGN_OR_RETURN(

135 ASSIGN_OR_RETURN(

138 ASSIGN_OR_RETURN(
139 uint16_t br,
140 GetTile16ForTile32(i, k, (int)Dimension::map32TilesBR, map32address));

142 // Add the generated 16-bit tiles to the tiles32 vector.
143 tiles32_unique_.emplace_back(gfx::Tile32(tl, tr, bl, br));
144 }
145 }
146
147 map_tiles_.light_world.resize(0x200);
148 map_tiles_.dark_world.resize(0x200);
149 map_tiles_.special_world.resize(0x200);
150 for (int i = 0; i < 0x200; i++) {

414 ASSIGN_OR_RETURN(uint32_t pointer,
415 rom()->ReadLong(zelda3::kOverworldItemsAddress));
416 uint32_t pointer_pc = core::SnesToPc(pointer); // 1BC2F9 -> 0DC2F9
417 for (int i = 0; i < 128; i++) {
418 ASSIGN_OR_RETURN(uint16_t word_address,
419 rom()->ReadWord(pointer_pc + i * 2));
420 uint32_t addr = (pointer & 0xFF0000) | word_address; // 1B F9 3C
421 addr = core::SnesToPc(addr);
422
423 if (overworld_maps_[i].is_large_map()) {
424 if (overworld_maps_[i].parent() != (uint8_t)i) {
425 continue;
426 }
427 }
428
429 while (true) {
430 ASSIGN_OR_RETURN(uint8_t b1, rom()->ReadByte(addr));
431 ASSIGN_OR_RETURN(uint8_t b2, rom()->ReadByte(addr + 1));
432 ASSIGN_OR_RETURN(uint8_t b3, rom()->ReadByte(addr + 2));
433
434 if (b1 == 0xFF && b2 == 0xFF) {
435 break;
436 }
437
438 int p = (((b2 & 0x1F) << 8) + b1) >> 1;
439
440 int x = p % 64;
441 int y = p >> 6;
442
443 int fakeID = i;
444 if (fakeID >= 64) {
445 fakeID -= 64;
446 }
447
448 int sy = fakeID / 8;
449 int sx = fakeID - (sy * 8);
450
451 all_items_.emplace_back(b3, (uint16_t)i, (x * 16) + (sx * 512),
452 (y * 16) + (sy * 512), false);
453 auto size = all_items_.size();
454
455 all_items_[size - 1].game_x_ = (uint8_t)x;
456 all_items_[size - 1].game_y_ = (uint8_t)y;

533absl::Status Overworld::SaveOverworldMaps() {
534 core::logf("Saving Overworld Maps");
535
536 // Initialize map pointers
537 std::fill(map_pointers1_id.begin(), map_pointers1_id.end(), -1);
538 std::fill(map_pointers2_id.begin(), map_pointers2_id.end(), -1);
539
540 // Compress and save each map
541 int pos = kOverworldCompressedMapPos;
542 for (int i = 0; i < kNumOverworldMaps; i++) {
543 std::vector<uint8_t> single_map_1(512);
544 std::vector<uint8_t> single_map_2(512);
545
546 // Copy tiles32 data to single_map_1 and single_map_2
547 int npos = 0;
548 for (int y = 0; y < 16; y++) {
549 for (int x = 0; x < 16; x++) {
550 auto packed = tiles32_list_[npos + (i * 256)];
551 single_map_1[npos] = packed & 0xFF; // Lower 8 bits
552 single_map_2[npos] = (packed >> 8) & 0xFF; // Next 8 bits
553 npos++;
554 }
555 }
556
557 int size_a, size_b;
558 // Compress single_map_1 and single_map_2
559 auto a = gfx::HyruleMagicCompress(single_map_1.data(), 256, &size_a, 1);
560 auto b = gfx::HyruleMagicCompress(single_map_2.data(), 256, &size_b, 1);
561 if (a.empty() || b.empty()) {
562 return absl::AbortedError("Error compressing map gfx.");
563 }
564
565 // Save compressed data and pointers
566 map_data_p1[i] = std::vector<uint8_t>(size_a);
567 map_data_p2[i] = std::vector<uint8_t>(size_b);
568
569 if ((pos + size_a) >= 0x5FE70 && (pos + size_a) <= 0x60000) {
570 pos = 0x60000;
571 }
572
573 if ((pos + size_a) >= 0x6411F && (pos + size_a) <= 0x70000) {
574 core::logf("Pos set to overflow region for map %s at %s",
575 std::to_string(i), core::HexLong(pos));
576 pos = kOverworldMapDataOverflow; // 0x0F8780;
577 }
578
579 const auto compare_array = [](const std::vector<uint8_t> &array1,
580 const std::vector<uint8_t> &array2) -> bool {
581 if (array1.size() != array2.size()) {
582 return false;
583 }
584
585 for (size_t i = 0; i < array1.size(); i++) {
586 if (array1[i] != array2[i]) {
587 return false;
588 }
589 }
590
591 return true;
592 };
593
594 for (int j = 0; j < i; j++) {
595 if (compare_array(a, map_data_p1[j])) {
596 // Reuse pointer id j for P1 (a)
597 map_pointers1_id[i] = j;
598 }
599
600 if (compare_array(b, map_data_p2[j])) {
601 map_pointers2_id[i] = j;
602 // Reuse pointer id j for P2 (b)
603 }
604 }
605
606 if (map_pointers1_id[i] == -1) {
607 // Save compressed data and pointer for map1
608 std::copy(a.begin(), a.end(), map_data_p1[i].begin());
609 int snes_pos = core::PcToSnes(pos);
610 map_pointers1[i] = snes_pos;
611 core::logf("Saving map pointers1 and compressed data for map %s at %s",
612 core::HexByte(i), core::HexLong(snes_pos));
613 RETURN_IF_ERROR(rom()->WriteLong(
614 rom()->version_constants().kCompressedAllMap32PointersLow + (3 * i),
615 snes_pos));
616 RETURN_IF_ERROR(rom()->WriteVector(pos, a));
617 pos += size_a;
618 } else {
619 // Save pointer for map1
620 int snes_pos = map_pointers1[map_pointers1_id[i]];
621 core::logf("Saving map pointers1 for map %s at %s", core::HexByte(i),
622 core::HexLong(snes_pos));
623 RETURN_IF_ERROR(rom()->WriteLong(
624 rom()->version_constants().kCompressedAllMap32PointersLow + (3 * i),
625 snes_pos));
626 }
627
628 if ((pos + b.size()) >= 0x5FE70 && (pos + b.size()) <= 0x60000) {
629 pos = 0x60000;
630 }
631
632 if ((pos + b.size()) >= 0x6411F && (pos + b.size()) <= 0x70000) {
633 core::logf("Pos set to overflow region for map %s at %s",
634 core::HexByte(i), core::HexLong(pos));
635 pos = kOverworldMapDataOverflow;
636 }
637
638 if (map_pointers2_id[i] == -1) {
639 // Save compressed data and pointer for map2
640 std::copy(b.begin(), b.end(), map_data_p2[i].begin());
641 int snes_pos = core::PcToSnes(pos);
642 map_pointers2[i] = snes_pos;
643 core::logf("Saving map pointers2 and compressed data for map %s at %s",
644 core::HexByte(i), core::HexLong(snes_pos));
645 RETURN_IF_ERROR(rom()->WriteLong(
646 rom()->version_constants().kCompressedAllMap32PointersHigh + (3 * i),
647 snes_pos));
648 RETURN_IF_ERROR(rom()->WriteVector(pos, b));
649 pos += size_b;
650 } else {
651 // Save pointer for map2
652 int snes_pos = map_pointers2[map_pointers2_id[i]];
653 core::logf("Saving map pointers2 for map %s at %s", core::HexByte(i),
654 core::HexLong(snes_pos));
655 RETURN_IF_ERROR(rom()->WriteLong(
656 rom()->version_constants().kCompressedAllMap32PointersHigh + (3 * i),
657 snes_pos));
658 }
659 }
660
661 // Check if too many maps data
662 if (pos > kOverworldCompressedOverflowPos) {
663 core::logf("Too many maps data %s", core::HexLong(pos));
664 return absl::AbortedError("Too many maps data " + std::to_string(pos));
665 }
666
667 RETURN_IF_ERROR(SaveLargeMaps())
668 return absl::OkStatus();
669}

992std::vector<uint64_t> GetAllTile16(OverworldMapTiles &map_tiles_) {
993 std::vector<uint64_t> all_tile_16; // Ensure it's 64 bits
994
995 int sx = 0;
996 int sy = 0;
997 int c = 0;

1000 if (i < kDarkWorldMapIdStart) {
1001 tiles_used = map_tiles_.light_world;
1002 } else if (i < kSpecialWorldMapIdStart && i >= kDarkWorldMapIdStart) {
1003 tiles_used = map_tiles_.dark_world;
1004 } else {
1005 tiles_used = map_tiles_.special_world;
1006 }
1007
1008 for (int y = 0; y < 32; y += 2) {
1009 for (int x = 0; x < 32; x += 2) {
1010 gfx::Tile32 current_tile(
1011 tiles_used[x + (sx * 32)][y + (sy * 32)],
1012 tiles_used[x + 1 + (sx * 32)][y + (sy * 32)],
1013 tiles_used[x + (sx * 32)][y + 1 + (sy * 32)],
1014 tiles_used[x + 1 + (sx * 32)][y + 1 + (sy * 32)]);
1015
1016 all_tile_16.emplace_back(current_tile.GetPackedValue());
1017 }
1018 }
1019
1020 sx++;
1021 if (sx >= 8) {
1022 sy++;
1023 sx = 0;
1024 }
1025
1026 c++;
1027 if (c >= 64) {
1028 sx = 0;
1029 sy = 0;
1030 c = 0;
1031 }
1032 }
1033
1034 return all_tile_16;
1035}

1038absl::Status Overworld::CreateTile32Tilemap() {
1039 tiles32_unique_.clear();
1040 tiles32_list_.clear();
1041
1042 // Get all tiles16 and packs them into tiles32
1043 std::vector<uint64_t> all_tile_16 = GetAllTile16(map_tiles_);
1044
1045 // Convert to set then back to vector
1046 std::set<uint64_t> unique_tiles_set(all_tile_16.begin(), all_tile_16.end());
1047
1048 std::vector<uint64_t> unique_tiles(all_tile_16);
1049 unique_tiles.assign(unique_tiles_set.begin(), unique_tiles_set.end());
1050
1051 // Create the indexed tiles list
1052 std::unordered_map<uint64_t, uint16_t> all_tiles_indexed;
1053 for (size_t tile32_id = 0; tile32_id < unique_tiles.size(); tile32_id++) {
1054 all_tiles_indexed.insert(
1055 {unique_tiles[tile32_id], static_cast<uint16_t>(tile32_id)});
1056 }
1057
1058 // Add all tiles32 from all maps.
1059 // Convert all tiles32 non-unique IDs into unique array of IDs.
1060 for (int j = 0; j < NumberOfMap32; j++) {
1061 tiles32_list_.emplace_back(all_tiles_indexed[all_tile_16[j]]);
1062 }
1063
1064 // Create the unique tiles list
1065 for (size_t i = 0; i < unique_tiles.size(); ++i) {
1066 tiles32_unique_.emplace_back(gfx::Tile32(unique_tiles[i]));
1067 }
1068
1069 while (tiles32_unique_.size() % 4 != 0) {
1070 gfx::Tile32 padding_tile(0, 0, 0, 0);
1071 tiles32_unique_.emplace_back(padding_tile.GetPackedValue());
1072 }
1073
1074 if (tiles32_unique_.size() > LimitOfMap32) {
1075 return absl::InternalError(absl::StrFormat(
1076 "Number of unique Tiles32: %d Out of: %d\nUnique Tile32 count exceed "
1077 "the limit\nThe ROM Has not been saved\nYou can fill maps with grass "
1078 "tiles to free some space\nOr use the option Clear DW Tiles in the "
1079 "Overworld Menu",
1080 unique_tiles.size(), LimitOfMap32));
1081 }
1082
1083 if (core::ExperimentFlags::get().kLogToConsole) {
1084 std::cout << "Number of unique Tiles32: " << tiles32_unique_.size()
1085 << " Saved:" << tiles32_unique_.size()
1086 << " Out of: " << LimitOfMap32 << std::endl;
1087 }
1088
1089 int v = tiles32_unique_.size();
1090 for (int i = v; i < LimitOfMap32; i++) {
1091 gfx::Tile32 padding_tile(420, 420, 420, 420);
1092 tiles32_unique_.emplace_back(padding_tile.GetPackedValue());
1093 }
1094
1095 return absl::OkStatus();
1096}
