161absl::Status ScreenEditor::LoadDungeonMaps() {
162 std::vector<std::array<uint8_t, 25>> current_floor_rooms_d;
163 std::vector<std::array<uint8_t, 25>> current_floor_gfx_d;
164 int total_floors_d;
165 uint8_t nbr_floor_d;
166 uint8_t nbr_basement_d;
167
168 for (int d = 0; d < 14; d++) {
169 current_floor_rooms_d.clear();
170 current_floor_gfx_d.clear();
171 ASSIGN_OR_RETURN(int ptr,
172 rom()->ReadWord(zelda3::kDungeonMapRoomsPtr + (d * 2)));
173 ASSIGN_OR_RETURN(int ptr_gfx,
174 rom()->ReadWord(zelda3::kDungeonMapGfxPtr + (d * 2)));
175 ptr |= 0x0A0000; // Add bank to the short ptr
176 ptr_gfx |= 0x0A0000; // Add bank to the short ptr
177 int pc_ptr = SnesToPc(ptr); // Contains data for the next 25 rooms
178 int pc_ptr_gfx = SnesToPc(ptr_gfx); // Contains data for the next 25 rooms
179
180 ASSIGN_OR_RETURN(uint16_t boss_room_d,
181 rom()->ReadWord(zelda3::kDungeonMapBossRooms + (d * 2)));
182
183 ASSIGN_OR_RETURN(nbr_basement_d,
184 rom()->ReadByte(zelda3::kDungeonMapFloors + (d * 2)));
185 nbr_basement_d &= 0x0F;
186
187 ASSIGN_OR_RETURN(nbr_floor_d,
188 rom()->ReadByte(zelda3::kDungeonMapFloors + (d * 2)));
189 nbr_floor_d &= 0xF0;
190 nbr_floor_d = nbr_floor_d >> 4;
191
192 total_floors_d = nbr_basement_d + nbr_floor_d;
193
194 dungeon_map_labels_.emplace_back();
195
196 // for each floor in the dungeon
197 for (int i = 0; i < total_floors_d; i++) {
198 dungeon_map_labels_[d].emplace_back();
199
200 std::array<uint8_t, 25> rdata;
201 std::array<uint8_t, 25> gdata;
202
203 // for each room on the floor
204 for (int j = 0; j < 25; j++) {
205 gdata[j] = 0xFF;
206 rdata[j] = rom()->data()[pc_ptr + j + (i * 25)]; // Set the rooms
207
208 if (rdata[j] == 0x0F) {
209 gdata[j] = 0xFF;
210 } else {
211 gdata[j] = rom()->data()[pc_ptr_gfx++];
212 }
213
214 std::string label = util::HexByte(rdata[j]);
215 dungeon_map_labels_[d][i][j] = label;
216 }
217
218 current_floor_gfx_d.push_back(gdata); // Add new floor gfx data
219 current_floor_rooms_d.push_back(rdata); // Add new floor data
220 }
221
222 dungeon_maps_.emplace_back(boss_room_d, nbr_floor_d, nbr_basement_d,
223 current_floor_rooms_d, current_floor_gfx_d);
224 }
225
226 return absl::OkStatus();
227}

229absl::Status ScreenEditor::SaveDungeonMaps() {
230 for (int d = 0; d < 14; d++) {
231 int ptr = zelda3::kDungeonMapRoomsPtr + (d * 2);
232 int ptr_gfx = zelda3::kDungeonMapGfxPtr + (d * 2);
233 int pc_ptr = SnesToPc(ptr);
234 int pc_ptr_gfx = SnesToPc(ptr_gfx);
235
236 const int nbr_floors = dungeon_maps_[d].nbr_of_floor;
237 const int nbr_basements = dungeon_maps_[d].nbr_of_basement;
238 for (int i = 0; i < nbr_floors + nbr_basements; i++) {
239 for (int j = 0; j < 25; j++) {
240 RETURN_IF_ERROR(rom()->WriteByte(pc_ptr + j + (i * 25),
241 dungeon_maps_[d].floor_rooms[i][j]));
242 RETURN_IF_ERROR(rom()->WriteByte(pc_ptr_gfx + j + (i * 25),
243 dungeon_maps_[d].floor_gfx[i][j]));
244 pc_ptr_gfx++;
245 }
246 }
247 }
248
249 return absl::OkStatus();

252absl::Status ScreenEditor::LoadDungeonMapTile16(
253 const std::vector<uint8_t> &gfx_data, bool bin_mode) {
254 tile16_blockset_.tile_size = {16, 16};
255 tile16_blockset_.map_size = {186, 186};
256 tile16_blockset_.atlas.Create(256, 192, 8, gfx_data);
257 tile16_sheet_.Init(256, 192, gfx::TileType::Tile16);
258
259 for (int i = 0; i < 186; i++) {
260 int addr = zelda3::kDungeonMapTile16;
261 if (rom()->data()[zelda3::kDungeonMapExpCheck] != 0xB9) {
262 addr = zelda3::kDungeonMapTile16Expanded;
263 }
264
265 ASSIGN_OR_RETURN(auto tl, rom()->ReadWord(addr + (i * 8)));
266 gfx::TileInfo t1 = gfx::WordToTileInfo(tl); // Top left
267
268 ASSIGN_OR_RETURN(auto tr, rom()->ReadWord(addr + 2 + (i * 8)));
269 gfx::TileInfo t2 = gfx::WordToTileInfo(tr); // Top right
270
271 ASSIGN_OR_RETURN(auto bl, rom()->ReadWord(addr + 4 + (i * 8)));
272 gfx::TileInfo t3 = gfx::WordToTileInfo(bl); // Bottom left
273
274 ASSIGN_OR_RETURN(auto br, rom()->ReadWord(addr + 6 + (i * 8)));
275 gfx::TileInfo t4 = gfx::WordToTileInfo(br); // Bottom right
276
277 int sheet_offset = 212;
278 if (bin_mode) {
279 sheet_offset = 0;
280 }
281 ComposeTile16(tile16_blockset_, gfx_data, t1, t2, t3, t4, sheet_offset);
282 tile16_sheet_.ComposeTile16(gfx_data, t1, t2, t3, t4, sheet_offset);
283 }
284
285 tile16_sheet_.mutable_bitmap()->SetPalette(
286 *rom()->mutable_dungeon_palette(3));
287 tile16_blockset_.atlas.SetPalette(*rom()->mutable_dungeon_palette(3));
288 Renderer::Get().RenderBitmap(&tile16_blockset_.atlas);
289 Renderer::Get().RenderBitmap(&*tile16_sheet_.mutable_bitmap().get());
290
291 for (int i = 0; i < tile16_sheet_.num_tiles(); ++i) {
292 auto tile = tile16_sheet_.GetTile16(i);
293 tile16_individual_[i] = tile;
294 tile16_individual_[i].SetPalette(*rom()->mutable_dungeon_palette(3));
295 Renderer::Get().RenderBitmap(&tile16_individual_[i]);
296 }
297
298 return absl::OkStatus();
299}

557void ScreenEditor::LoadBinaryGfx() {
558 std::string bin_file = core::FileDialogWrapper::ShowOpenFileDialog();
559 if (!bin_file.empty()) {
560 std::ifstream file(bin_file, std::ios::binary);
561 if (file.is_open()) {
562 // Read the gfx data into a buffer
563 std::vector<uint8_t> bin_data((std::istreambuf_iterator<char>(file)),
564 std::istreambuf_iterator<char>());
565 auto converted_bin = gfx::SnesTo8bppSheet(bin_data, 4, 4);
566 tile16_sheet_.clear();
567 if (LoadDungeonMapTile16(converted_bin, true).ok()) {
568 sheets_.clear();
569 std::vector<std::vector<uint8_t>> gfx_sheets;
570 for (int i = 0; i < 4; i++) {
571 gfx_sheets.emplace_back(converted_bin.begin() + (i * 0x1000),
572 converted_bin.begin() + ((i + 1) * 0x1000));
573 sheets_.emplace(i, gfx::Bitmap(128, 32, 8, gfx_sheets[i]));
574 sheets_[i].SetPalette(*rom()->mutable_dungeon_palette(3));
575 Renderer::Get().RenderBitmap(&sheets_[i]);
576 }
577 binary_gfx_loaded_ = true;
578 } else {
579 status_ = absl::InternalError("Failed to load dungeon map tile16");
580 }
581 file.close();
582 }
583 }
