84 std::string OptimizeMessageForDictionary(std::string messageString) {
85 std::stringstream protons;
86 bool command = false;
87 for (const auto& c : messageString) {
88 if (c == '[') {
89 command = true;
90 } else if (c == ']') {
91 command = false;
92 }
93
94 protons << c;
95 if (command) {
96 protons << CHEESE;

123 TextElement() = default;
124 TextElement(uint8_t id, std::string token, bool arg,
125 std::string description) {
126 ID = id;
127 Token = token;
128 if (arg) {
129 GenericToken = absl::StrFormat("[%s:##]", Token);
130 } else {
131 GenericToken = absl::StrFormat("[%s]", Token);
132 }
133 HasArgument = arg;
134 Description = description;
135 Pattern =
136 arg ? "\\[" + Token + ":?([0-9A-F]{1,2})\\]" : "\\[" + Token + "\\]";
137 Pattern = absl::StrReplaceAll(Pattern, {{"[", "\\["}, {"]", "\\]"}});
138 StrictPattern = absl::StrCat("^", Pattern, "$");
139 StrictPattern = "^" + Pattern + "$";
140 }
141
142 std::string GetParameterizedToken(uint8_t value = 0) {
143 if (HasArgument) {
144 return absl::StrFormat("[%s:%02X]", Token, value);
145 } else {
146 return absl::StrFormat("[%s]", Token);
147 }
148 }
149
150 std::string ToString() {
151 return absl::StrFormat("%s %s", GenericToken, Description);
152 }
153
154 std::smatch MatchMe(std::string dfrag) const {
155 std::regex pattern(StrictPattern);
156 std::smatch match;
157 std::regex_match(dfrag, match, pattern);
158 return match;
159 }
160
161 bool Empty() { return ID == 0; }
162
163 // Comparison operator
164 bool operator==(const TextElement& other) const { return ID == other.ID; }
