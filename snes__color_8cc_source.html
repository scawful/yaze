33uint16_t ConvertRgbToSnes(const snes_color& color) {
34 uint16_t red = color.red / 8;
35 uint16_t green = color.green / 8;
36 uint16_t blue = color.blue / 8;
37 return (blue * SNES_BLUE_SHIFT) + (green * SNES_GREEN_SHIFT) + red;
38}
