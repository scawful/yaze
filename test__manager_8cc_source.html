178absl::Status TestManager::RunTestSuite(const std::string& suite_name) {
179 if (is_running_) {
180 return absl::FailedPreconditionError("Tests are already running");
181 }
182
183 auto it = suite_lookup_.find(suite_name);
184 if (it == suite_lookup_.end()) {
185 return absl::NotFoundError("Test suite not found: " + suite_name);
186 }
187
188 is_running_ = true;
189 progress_ = 0.0f;
190 last_results_.Clear();
191 current_test_name_ = suite_name;
192
193 auto status = ExecuteTestSuite(it->second);
194
195 is_running_ = false;
196 current_test_name_.clear();
197 progress_ = 1.0f;
198
199 return status;
200}

1134std::string TestManager::GenerateTestRomFilename(const std::string& base_name) {
1135 // Generate filename with timestamp
1136 auto now = std::chrono::system_clock::now();
1137 auto time_t = std::chrono::system_clock::to_time_t(now);
1138 auto local_time = *std::localtime(&time_t);
1139
1140 std::string timestamp = absl::StrFormat("%04d%02d%02d_%02d%02d%02d",
1141 local_time.tm_year + 1900,
1142 local_time.tm_mon + 1,
1143 local_time.tm_mday,
1144 local_time.tm_hour,
1145 local_time.tm_min,
1146 local_time.tm_sec);
1147
1148 std::string base_filename = base_name;
1149 // Remove any path and extension
1150 auto last_slash = base_filename.find_last_of("/\\");
1151 if (last_slash != std::string::npos) {
1152 base_filename = base_filename.substr(last_slash + 1);
1153 }
1154 auto last_dot = base_filename.find_last_of('.');
1155 if (last_dot != std::string::npos) {
1156 base_filename = base_filename.substr(0, last_dot);
1157 }
1158
1159 return absl::StrFormat("%s_test_%s.sfc", base_filename.c_str(), timestamp.c_str());
1160}

1251absl::Status TestManager::TestRomDataIntegrity(Rom* rom) {
1252 if (!rom || !rom->is_loaded()) {
1253 return absl::FailedPreconditionError("No ROM loaded for testing");
1254 }
1255
1256 // Use TestRomWithCopy for integrity testing (read-only but uses copy for safety)
1257 return TestRomWithCopy(rom, [](Rom* test_rom) -> absl::Status {
1258 util::logf("Testing ROM data integrity on copy: %s", test_rom->title().c_str());
1259
1260 // Perform data integrity checks on the copy
1261 // This validates ROM structure, checksums, etc. without affecting original
1262
1263 // Basic ROM structure validation
1264 if (test_rom->size() < 0x100000) { // 1MB minimum for ALTTP
1265 return absl::FailedPreconditionError("ROM file too small for A Link to the Past");
