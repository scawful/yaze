21void Spc700::MOVS(uint16_t address) {
22 static int movs_log = 0;
23 switch (bstep) {
24 case 0:
25 this->adr = address; // Save address for bstep=1
26 read(this->adr);
27 bstep++;
28 break;
29 case 1:
30 write(this->adr, A); // Use saved address
31 if (this->adr == 0x00F4 && movs_log++ < 10) {
32 LOG_DEBUG("SPC", "MOVS wrote A=$%02X to F4!", A);
33 }
34 bstep = 0;
35 break;
36 }
37}

79void Spc700::ADC(uint16_t adr) {
80 uint8_t value = read(adr);
81 uint16_t result = A + value + PSW.C;
82 PSW.V = ((A ^ result) & (adr ^ result) & 0x80);
83 PSW.C = (result > 0xFF);
84 PSW.H = ((A ^ adr ^ result) & 0x10);
85 A = result & 0xFF;
86 PSW.Z = ((A & 0xFF) == 0);
87 PSW.N = (A & 0x80);
88}

90void Spc700::ADCM(uint16_t& dest, uint8_t operand) {
91 uint8_t applyOn = read(dest);
92 int result = applyOn + operand + PSW.C;
93 PSW.V = (applyOn & 0x80) == (operand & 0x80) &&
94 (operand & 0x80) != (result & 0x80);
95 PSW.H = ((applyOn & 0xf) + (operand & 0xf) + PSW.C) > 0xf;
96 PSW.C = result > 0xff;
97 write(dest, result);
98 PSW.Z = ((result & 0xFF) == 0);
99 PSW.N = (result & 0x80);
