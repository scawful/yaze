90void Snes::RunFrame() {
91 // Debug: Log every 60th frame
92 static int frame_log_count = 0;
93 if (frame_log_count % 60 == 0) {
94 LOG_INFO("SNES", "Frame %d: CPU=$%02X:%04X vblank=%d frames_=%d",
95 frame_log_count, cpu_.PB, cpu_.PC, in_vblank_, frames_);
96 }
97 frame_log_count++;
98
99 // Debug: Log vblank loop entry
100 static int vblank_loop_count = 0;
101 if (in_vblank_ && vblank_loop_count++ < 10) {
102 LOG_INFO("SNES", "RunFrame: Entering vblank loop (in_vblank_=true)");
103 }
104
105 while (in_vblank_) {
106 cpu_.RunOpcode();
107 }
108
109 uint32_t frame = frames_;
110
111 // Debug: Log active frame loop entry
112 static int active_loop_count = 0;
113 if (!in_vblank_ && active_loop_count++ < 10) {
114 LOG_INFO("SNES", "RunFrame: Entering active frame loop (in_vblank_=false, frame=%d, frames_=%d)",
115 frame, frames_);
116 }
117
118 while (!in_vblank_ && frame == frames_) {
119 cpu_.RunOpcode();
120 }
121}

129void Snes::HandleInput() {
130 memset(port_auto_read_, 0, sizeof(port_auto_read_));
131 // latch controllers
132 input_latch(&input1, true);
133 input_latch(&input2, true);
134 input_latch(&input1, false);
135 input_latch(&input2, false);
136 for (int i = 0; i < 16; i++) {
137 uint8_t val = input_read(&input1);
138 port_auto_read_[0] |= ((val & 1) << (15 - i));
139 port_auto_read_[2] |= (((val >> 1) & 1) << (15 - i));
140 val = input_read(&input2);
141 port_auto_read_[1] |= ((val & 1) << (15 - i));
142 port_auto_read_[3] |= (((val >> 1) & 1) << (15 - i));
143 }

146void Snes::RunCycle() {
147 cycles_ += 2;
148
149 // check for h/v timer irq's
150 bool condition = ((v_irq_enabled_ || h_irq_enabled_) &&
151 (memory_.v_pos() == v_timer_ || !v_irq_enabled_) &&
152 (memory_.h_pos() == h_timer_ || !h_irq_enabled_));
153
154 if (!irq_condition_ && condition) {
155 in_irq_ = true;
156 cpu_.SetIrq(true);
157 }
158 irq_condition_ = condition;
159
160 // increment position; must come after irq checks! (hagane, cybernator)
161 memory_.set_h_pos(memory_.h_pos() + 2);
162
163 // handle positional stuff
164 if (memory_.h_pos() == next_horiz_event) {
165 switch (memory_.h_pos()) {
166 case 16: {
167 next_horiz_event = 512;
168 if (memory_.v_pos() == 0) memory_.init_hdma_request();
169 } break;
170 case 512: {
171 next_horiz_event = 1104;
172 // render the line halfway of the screen for better compatibility
173 if (!in_vblank_ && memory_.v_pos() > 0) ppu_.RunLine(memory_.v_pos());
174 } break;
175 case 1104: {
176 if (!in_vblank_) memory_.run_hdma_request();
177 if (!memory_.pal_timing()) {
178 // line 240 of odd frame with no interlace is 4 cycles shorter
179 next_horiz_event = (memory_.v_pos() == 240 && !ppu_.even_frame &&
180 !ppu_.frame_interlace)
181 ? 1360
182 : 1364;
183 } else {
184 // line 311 of odd frame with interlace is 4 cycles longer
185 next_horiz_event = (memory_.v_pos() != 311 || ppu_.even_frame ||
186 !ppu_.frame_interlace)
187 ? 1364
188 : 1368;
189 }
190 } break;
191 case 1360:
192 case 1364:
193 case 1368: { // this is the end (of the h-line)
194 next_horiz_event = 16;
195
196 memory_.set_h_pos(0);
197 memory_.set_v_pos(memory_.v_pos() + 1);
198 if (!memory_.pal_timing()) {
199 // even interlace frame is 263 lines
200 if ((memory_.v_pos() == 262 &&
201 (!ppu_.frame_interlace || !ppu_.even_frame)) ||
202 memory_.v_pos() == 263) {
203 memory_.set_v_pos(0);
204 frames_++;
205 }
206 } else {
207 // even interlace frame is 313 lines
208 if ((memory_.v_pos() == 312 &&
209 (!ppu_.frame_interlace || !ppu_.even_frame)) ||
210 memory_.v_pos() == 313) {
211 memory_.set_v_pos(0);
212 frames_++;
213 }
214 }
215
216 // end of hblank, do most memory_.v_pos()-tests
217 bool starting_vblank = false;
218 if (memory_.v_pos() == 0) {
219 // end of vblank
220 static int vblank_end_count = 0;
221 if (vblank_end_count++ < 10) {
222 LOG_INFO("SNES", "VBlank END - v_pos=0, setting in_vblank_=false at frame %d", frames_);
223 }
224 in_vblank_ = false;
225 in_nmi_ = false;
226 ppu_.HandleFrameStart();
227 } else if (memory_.v_pos() == 225) {
228 // ask the ppu if we start vblank now or at memory_.v_pos() 240
229 // (overscan)
230 starting_vblank = !ppu_.CheckOverscan();
231 } else if (memory_.v_pos() == 240) {
232 // if we are not yet in vblank, we had an overscan frame, set
233 // starting_vblank
234 if (!in_vblank_) starting_vblank = true;
235 }
236 if (starting_vblank) {
237 // catch up the apu at end of emulated frame (we end frame @ start of
238 // vblank)
239 CatchUpApu();
240 // notify dsp of frame-end, because sometimes dma will extend much
241 // further past vblank (or even into the next frame) Megaman X2
242 // (titlescreen animation), Tales of Phantasia (game demo), Actraiser
243 // 2 (fade-in @ bootup)
244 apu_.dsp().NewFrame();
245 // we are starting vblank
246 ppu_.HandleVblank();
247
248 static int vblank_start_count = 0;
249 if (vblank_start_count++ < 10) {
250 LOG_INFO("SNES", "VBlank START - v_pos=%d, setting in_vblank_=true at frame %d",
251 memory_.v_pos(), frames_);
252 }
253
254 in_vblank_ = true;
255 in_nmi_ = true;
256 if (auto_joy_read_) {
257 // TODO: this starts a little after start of vblank
258 auto_joy_timer_ = 4224;
259 HandleInput();
260 }
261 static int nmi_log_count = 0;
262 if (nmi_log_count++ < 10) {
263 LOG_INFO("SNES", "VBlank NMI check: nmi_enabled_=%d, calling Nmi()=%s",
264 nmi_enabled_, nmi_enabled_ ? "YES" : "NO");
265 }
266 if (nmi_enabled_) {
267 cpu_.Nmi();
268 }
269 }
270 } break;
271 }
272 }
273 // handle auto_joy_read_-timer
274 if (auto_joy_timer_ > 0) auto_joy_timer_ -= 2;
275}

277void Snes::RunCycles(int cycles) {
278 if (memory_.h_pos() + cycles >= 536 && memory_.h_pos() < 536) {
279 // if we go past 536, add 40 cycles for dram refersh
280 cycles += 40;
281 }
282 for (int i = 0; i < cycles; i += 2) {
283 RunCycle();
284 }

287void Snes::SyncCycles(bool start, int sync_cycles) {
288 int count = 0;
289 if (start) {
290 sync_cycle_ = cycles_;
291 count = sync_cycles - (cycles_ % sync_cycles);

293 count = sync_cycles - ((cycles_ - sync_cycle_) % sync_cycles);
294 }
295 RunCycles(count);

298uint8_t Snes::ReadBBus(uint8_t adr) {
299 if (adr < 0x40) {
300 return ppu_.Read(adr, ppu_latch_);
301 }
302 if (adr < 0x80) {
303 CatchUpApu(); // catch up the apu before reading
304 uint8_t val = apu_.out_ports_[adr & 0x3];
305 // Log port reads when value changes or during critical phase
306 static int cpu_port_read_count = 0;
307 static uint8_t last_f4 = 0xFF, last_f5 = 0xFF;
308 bool value_changed = ((adr & 0x3) == 0 && val != last_f4) || ((adr & 0x3) == 1 && val != last_f5);
309 if (value_changed || cpu_port_read_count++ < 5) {
310 LOG_INFO("SNES", "CPU read APU port $21%02X (F%d) = $%02X at PC=$%02X:%04X [AFTER CatchUp: APU_cycles=%llu CPU_cycles=%llu]",
311 0x40 + (adr & 0x3), (adr & 0x3) + 4, val, cpu_.PB, cpu_.PC, apu_.GetCycles(), cycles_);
312 if ((adr & 0x3) == 0) last_f4 = val;
313 if ((adr & 0x3) == 1) last_f5 = val;
314 }
315 return val;
316 }
317 if (adr == 0x80) {
318 uint8_t ret = ram[ram_adr_++];
319 ram_adr_ &= 0x1ffff;
320 return ret;
321 }
322 return memory_.open_bus();
323}

325uint8_t Snes::ReadReg(uint16_t adr) {
326 switch (adr) {
327 case 0x4210: {
328 uint8_t val = 0x2; // CPU version (4 bit)
329 val |= in_nmi_ << 7;
330 in_nmi_ = false;
331 return val | (memory_.open_bus() & 0x70);
332 }
333 case 0x4211: {
334 uint8_t val = in_irq_ << 7;
335 in_irq_ = false;
336 cpu_.SetIrq(false);
337 return val | (memory_.open_bus() & 0x7f);
338 }
339 case 0x4212: {
340 uint8_t val = (auto_joy_timer_ > 0);
341 val |= (memory_.h_pos() < 4 || memory_.h_pos() >= 1096) << 6;
342 val |= in_vblank_ << 7;
343 return val | (memory_.open_bus() & 0x3e);
344 }
345 case 0x4213: {
346 return ppu_latch_ << 7; // IO-port
347 }
348 case 0x4214: {
349 return divide_result_ & 0xff;
350 }
351 case 0x4215: {
352 return divide_result_ >> 8;
353 }
354 case 0x4216: {
355 return multiply_result_ & 0xff;
356 }
357 case 0x4217: {
358 return multiply_result_ >> 8;
359 }
360 case 0x4218:
361 case 0x421a:
362 case 0x421c:
363 case 0x421e: {
364 return port_auto_read_[(adr - 0x4218) / 2] & 0xff;
365 }
366 case 0x4219:
367 case 0x421b:
368 case 0x421d:
369 case 0x421f: {
370 return port_auto_read_[(adr - 0x4219) / 2] >> 8;
371 }
372 default: {
373 return memory_.open_bus();
374 }
375 }
376}

378uint8_t Snes::Rread(uint32_t adr) {
379 uint8_t bank = adr >> 16;
380 adr &= 0xffff;
381 if (bank == 0x7e || bank == 0x7f) {
382 return ram[((bank & 1) << 16) | adr]; // ram
383 }
384 if (bank < 0x40 || (bank >= 0x80 && bank < 0xc0)) {
385 if (adr < 0x2000) {
386 return ram[adr]; // ram mirror
387 }
388 if (adr >= 0x2100 && adr < 0x2200) {
389 return ReadBBus(adr & 0xff); // B-bus
390 }
391 if (adr == 0x4016) {
392 return input_read(&input1) | (memory_.open_bus() & 0xfc);
393 }
394 if (adr == 0x4017) {
395 return input_read(&input2) | (memory_.open_bus() & 0xe0) | 0x1c;
396 }
397 if (adr >= 0x4200 && adr < 0x4220) {
398 return ReadReg(adr); // internal registers
399 }
400 if (adr >= 0x4300 && adr < 0x4380) {
401 return ReadDma(&memory_, adr); // dma registers
402 }
403 }
404 // read from cart
405 return memory_.cart_read(bank, adr);
406}

414void Snes::WriteBBus(uint8_t adr, uint8_t val) {
415 if (adr < 0x40) {
416 ppu_.Write(adr, val);
417 return;
418 }
419 if (adr < 0x80) {
420 CatchUpApu(); // catch up the apu before writing
421 apu_.in_ports_[adr & 0x3] = val;
422 static int cpu_port_write_count = 0;
423 if (cpu_port_write_count++ < 10) { // Reduced to prevent crash
424 LOG_INFO("SNES", "CPU wrote APU port $21%02X (F%d) = $%02X at PC=$%02X:%04X",
425 0x40 + (adr & 0x3), (adr & 0x3) + 4, val, cpu_.PB, cpu_.PC);
426 }
427
428 // NOTE: Auto-reset disabled - relying on complete IPL ROM with counter protocol
429 // The IPL ROM will handle multi-upload sequences via its transfer loop
430
431 return;
432 }
433 switch (adr) {
434 case 0x80: {
435 ram[ram_adr_++] = val;
436 ram_adr_ &= 0x1ffff;
437 break;
438 }
439 case 0x81: {
440 ram_adr_ = (ram_adr_ & 0x1ff00) | val;
441 break;
442 }
443 case 0x82: {
444 ram_adr_ = (ram_adr_ & 0x100ff) | (val << 8);
445 break;
446 }
447 case 0x83: {
448 ram_adr_ = (ram_adr_ & 0x0ffff) | ((val & 1) << 16);
449 break;
450 }
451 }

454void Snes::WriteReg(uint16_t adr, uint8_t val) {
455 switch (adr) {
456 case 0x4200: {
457 // Log ALL writes to $4200 unconditionally
458 static int write_4200_count = 0;
459 if (write_4200_count++ < 20) {
460 LOG_INFO("SNES", "Write $%02X to $4200 at PC=$%02X:%04X (NMI=%d IRQ_H=%d IRQ_V=%d JOY=%d)",
461 val, cpu_.PB, cpu_.PC, (val & 0x80) ? 1 : 0, (val & 0x10) ? 1 : 0,
462 (val & 0x20) ? 1 : 0, (val & 0x01) ? 1 : 0);
463 }
464
465 auto_joy_read_ = val & 0x1;
466 if (!auto_joy_read_) auto_joy_timer_ = 0;
467 h_irq_enabled_ = val & 0x10;
468 v_irq_enabled_ = val & 0x20;
469 if (!h_irq_enabled_ && !v_irq_enabled_) {
470 in_irq_ = false;
471 cpu_.SetIrq(false);
472 }
473 // if nmi is enabled while in_nmi_ is still set, immediately generate nmi
474 if (!nmi_enabled_ && (val & 0x80) && in_nmi_) {
475 cpu_.Nmi();
476 }
477 bool old_nmi = nmi_enabled_;
478 nmi_enabled_ = val & 0x80;
479 if (old_nmi != nmi_enabled_) {
480 LOG_INFO("SNES", ">>> NMI enabled CHANGED: %d -> %d <<<",
481 old_nmi, nmi_enabled_);
482 }
483 cpu_.set_int_delay(true);
484 break;
485 }
486 case 0x4201: {
487 if (!(val & 0x80) && ppu_latch_) {
488 // latch the ppu
489 ppu_.LatchHV();
490 }
491 ppu_latch_ = val & 0x80;
492 break;
493 }
494 case 0x4202: {
495 multiply_a_ = val;
496 break;
497 }
498 case 0x4203: {
499 multiply_result_ = multiply_a_ * val;
500 break;
501 }
502 case 0x4204: {
503 divide_a_ = (divide_a_ & 0xff00) | val;
504 break;
505 }
506 case 0x4205: {
507 divide_a_ = (divide_a_ & 0x00ff) | (val << 8);
508 break;
509 }
510 case 0x4206: {
511 if (val == 0) {
512 divide_result_ = 0xffff;
513 multiply_result_ = divide_a_;
514 } else {
515 divide_result_ = divide_a_ / val;
516 multiply_result_ = divide_a_ % val;
517 }
518 break;
519 }
520 case 0x4207: {
521 h_timer_ = (h_timer_ & 0x100) | val;
522 break;
523 }
524 case 0x4208: {
525 h_timer_ = (h_timer_ & 0x0ff) | ((val & 1) << 8);
526 break;
527 }
528 case 0x4209: {
529 v_timer_ = (v_timer_ & 0x100) | val;
530 break;
531 }
532 case 0x420a: {
533 v_timer_ = (v_timer_ & 0x0ff) | ((val & 1) << 8);
534 break;
535 }
536 case 0x420b: {
537 StartDma(&memory_, val, false);
538 break;
539 }
540 case 0x420c: {
541 StartDma(&memory_, val, true);
542 break;
543 }
544 case 0x420d: {
545 fast_mem_ = val & 0x1;
546 break;
547 }
548 default: {
549 break;
550 }
551 }

554void Snes::Write(uint32_t adr, uint8_t val) {
555 memory_.set_open_bus(val);
556 uint8_t bank = adr >> 16;
557 adr &= 0xffff;
558 if (bank == 0x7e || bank == 0x7f) {
559 ram[((bank & 1) << 16) | adr] = val; // ram
560 }
561 if (bank < 0x40 || (bank >= 0x80 && bank < 0xc0)) {
562 if (adr < 0x2000) {
563 ram[adr] = val; // ram mirror
564 }
565 if (adr >= 0x2100 && adr < 0x2200) {
566 WriteBBus(adr & 0xff, val); // B-bus
567 }
568 if (adr == 0x4016) {
569 input_latch(&input1, val & 1); // input latch
570 input_latch(&input2, val & 1);
571 }
572 if (adr >= 0x4200 && adr < 0x4220) {
573 WriteReg(adr, val); // internal registers
574 }
575 if (adr >= 0x4300 && adr < 0x4380) {
576 WriteDma(&memory_, adr, val); // dma registers
577 }
578 }
579
580 // write to cart
581 memory_.cart_write(bank, adr, val);
582}

584int Snes::GetAccessTime(uint32_t adr) {
585 uint8_t bank = adr >> 16;
586 adr &= 0xffff;
587 if ((bank < 0x40 || (bank >= 0x80 && bank < 0xc0)) && adr < 0x8000) {
588 // 00-3f,80-bf:0-7fff
589 if (adr < 0x2000 || adr >= 0x6000) return 8; // 0-1fff, 6000-7fff
590 if (adr < 0x4000 || adr >= 0x4200) return 6; // 2000-3fff, 4200-5fff
591 return 12; // 4000-41ff
592 }
593 // 40-7f,co-ff:0000-ffff, 00-3f,80-bf:8000-ffff
594 return (fast_mem_ && bank >= 0x80) ? 6
595 : 8; // depends on setting in banks 80+
596}

631void Snes::loadState(const std::string& path) {
632 std::ifstream file(path, std::ios::binary);
633 if (!file) {
634 return;
635 }
636
637 uint32_t version;
638 READ_STATE(file, version);
639 if (version != 1) {
640 return;
641 }
642
643 // SNES state
644 READ_STATE(file, ram);
645 READ_STATE(file, ram_adr_);
646 READ_STATE(file, cycles_);
647 READ_STATE(file, sync_cycle_);
648 READ_STATE(file, apu_catchup_cycles_);
649 READ_STATE(file, h_irq_enabled_);
650 READ_STATE(file, v_irq_enabled_);
651 READ_STATE(file, nmi_enabled_);
652 READ_STATE(file, h_timer_);
653 READ_STATE(file, v_timer_);
654 READ_STATE(file, in_nmi_);
655 READ_STATE(file, irq_condition_);
656 READ_STATE(file, in_irq_);
657 READ_STATE(file, in_vblank_);
658 READ_STATE(file, port_auto_read_);
659 READ_STATE(file, auto_joy_read_);
660 READ_STATE(file, auto_joy_timer_);
661 READ_STATE(file, ppu_latch_);
662 READ_STATE(file, multiply_a_);
663 READ_STATE(file, multiply_result_);
664 READ_STATE(file, divide_a_);
665 READ_STATE(file, divide_result_);
666 READ_STATE(file, fast_mem_);
667 READ_STATE(file, next_horiz_event);
668
669 // CPU state
670 READ_STATE(file, cpu_);
671
672 // PPU state
673 READ_STATE(file, ppu_);
674
675 // APU state
676 READ_STATE(file, apu_);
677}
