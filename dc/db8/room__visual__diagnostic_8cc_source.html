277void RoomVisualDiagnostic::DrawTileDecoder(const std::vector<uint8_t>& gfx16_data, int tile_id) {
278 if (gfx16_data.empty()) {
279 ImGui::TextColored(ImVec4(1, 0, 0, 1), "No graphics data");
280 return;
281 }
282
283 // Calculate tile offset in graphics data
284 int tx = (tile_id / 16 * 512) + ((tile_id & 0xF) * 4);
285
286 ImGui::Text("Decoding tile 0x%03X", tile_id);
287 ImGui::Text("Offset in gfx16_data: 0x%X (%d)", tx, tx);
288
289 if (tx < 0 || tx + 32 > gfx16_data.size()) {
290 ImGui::TextColored(ImVec4(1, 0, 0, 1), "Tile offset out of range!");
291 return;
292 }
293
294 // Show raw tile data
295 ImGui::Text("Raw 4bpp data (32 bytes):");
296 for (int i = 0; i < 32; i += 8) {
297 ImGui::Text("%02X %02X %02X %02X %02X %02X %02X %02X",
298 gfx16_data[tx + i], gfx16_data[tx + i + 1],
299 gfx16_data[tx + i + 2], gfx16_data[tx + i + 3],
300 gfx16_data[tx + i + 4], gfx16_data[tx + i + 5],
301 gfx16_data[tx + i + 6], gfx16_data[tx + i + 7]);
302 }
303
304 // Decode and visualize 8x8 tile
305 ImGui::Separator();
306 ImGui::Text("Decoded 8x8 pixel values (palette 0):");
307
308 // Decode the tile manually to show what pixels are produced
309 for (int y = 0; y < 8; y++) {
310 std::string line;
311 for (int x = 0; x < 8; x++) {
312 int yl = (y / 8) * 64 + (y % 8) * 8;
313 int xl = x / 2;
314
315 if (tx + yl + xl < gfx16_data.size()) {
316 uint8_t pixel_byte = gfx16_data[tx + yl + xl];
317 uint8_t pixel_val = (x % 2 == 0) ? (pixel_byte >> 4) : (pixel_byte & 0x0F);
318 line += absl::StrFormat("%X", pixel_val);
319 } else {
320 line += "?";
321 }
322 }
323 ImGui::Text("%s", line.c_str());
324 }
