707absl::Status ModernCLI::Run(int argc, char* argv[]) {
708 if (argc < 2) {
709 ShowHelp();
710 return absl::OkStatus();
711 }
712
713 std::vector<std::string> args;
714 args.reserve(argc - 1);
715 for (int i = 1; i < argc; ++i) {
716 args.emplace_back(argv[i]);
717 }
718
719 // Handle "help <category>" command
720 if (args.size() >= 1 && args[0] == "help") {
721 if (args.size() == 1) {
722 ShowHelp();
723 return absl::OkStatus();
724 }
725 ShowCategoryHelp(args[1]);
726 return absl::OkStatus();
727 }
728
729 const CommandInfo* command_info = nullptr;
730 size_t consumed_tokens = 0;
731
732 if (args.size() >= 2) {
733 std::string candidate = absl::StrCat(args[0], " ", args[1]);
734 auto it = commands_.find(candidate);
735 if (it != commands_.end()) {
736 command_info = &it->second;
737 consumed_tokens = 2;
738 }
739 }
740
741 if (command_info == nullptr && !args.empty()) {
742 auto it = commands_.find(args[0]);
743 if (it != commands_.end()) {
744 command_info = &it->second;
745 consumed_tokens = 1;
746 }
747 }
748
749 if (command_info == nullptr) {
750 ShowHelp();
751 std::string joined = args.empty() ? std::string() : absl::StrJoin(args, " ");
752 return absl::NotFoundError(
753 absl::StrCat("Unknown command: ", joined.empty() ? "<none>" : joined));
754 }
755
756 std::vector<std::string> command_args(args.begin() + consumed_tokens, args.end());
757 return command_info->handler(command_args);
758}

1005absl::Status ModernCLI::HandleChatEntryCommand(
1006 const std::vector<std::string>& args) {
1007 std::string mode = "simple";
1008 std::optional<std::string> prompt;
1009 std::vector<std::string> forwarded;
1010
1011 for (size_t i = 0; i < args.size(); ++i) {
1012 const std::string& token = args[i];
1013 if (absl::StartsWith(token, "--mode=")) {
1014 mode = absl::AsciiStrToLower(token.substr(7));
1015 continue;
1016 }
1017 if (token == "--mode" && i + 1 < args.size()) {
1018 mode = absl::AsciiStrToLower(args[i + 1]);
1019 ++i;
1020 continue;
1021 }
1022 if (absl::StartsWith(token, "--prompt=")) {
1023 prompt = token.substr(9);
1024 continue;
1025 }
1026 if (token == "--prompt" && i + 1 < args.size()) {
1027 prompt = args[i + 1];
1028 ++i;
1029 continue;
1030 }
1031 if (token == "--quiet" || token == "-q") {
1032 absl::SetFlag(&FLAGS_quiet, true);
1033 continue;
1034 }
1035 if (!absl::StartsWith(token, "--") && !prompt.has_value()) {
1036 prompt = token;
1037 continue;
1038 }
1039 forwarded.push_back(token);
1040 }
1041
1042 const std::string normalized_mode = absl::AsciiStrToLower(mode);
1043
1044 auto has_batch_file = [&forwarded]() {
1045 for (const auto& token : forwarded) {
1046 if (absl::StartsWith(token, "--file") || token == "--file") {
1047 return true;
1048 }
1049 }
1050 return false;
1051 };
1052
1053 std::vector<std::string> agent_args;
1054 if (normalized_mode == "gui" || normalized_mode == "visual" ||
1055 normalized_mode == "tui") {
1056 if (prompt.has_value()) {
1057 return absl::InvalidArgumentError(
1058 "GUI chat mode launches the interactive TUI and does not accept a --prompt value.");
1059 }
1060 agent_args.push_back("chat");
1061 } else if (normalized_mode == "test" || normalized_mode == "qa") {
1062 if (prompt.has_value()) {
1063 return absl::InvalidArgumentError(
1064 "Test conversation mode does not accept an inline prompt.");
1065 }
1066 agent_args.push_back("test-conversation");
1067 } else {
1068 if (normalized_mode == "batch" && !has_batch_file()) {
1069 return absl::InvalidArgumentError(
1070 "Batch chat mode requires a --file=<path> option.");
1071 }
1072 agent_args.push_back("simple-chat");
1073 if (prompt.has_value()) {
1074 agent_args.push_back(*prompt);
1075 }
1076 }
1077
1078 agent_args.insert(agent_args.end(), forwarded.begin(), forwarded.end());
1079 return HandleAgentCommand(agent_args);
1080}
