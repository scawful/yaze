250absl::Status ModernCLI::Run(int argc, char* argv[]) {
251 if (argc < 2) {
252 ShowHelp();
253 return absl::OkStatus();
254 }
255
256 std::vector<std::string> args;
257 args.reserve(argc - 1);
258 for (int i = 1; i < argc; ++i) {
259 args.emplace_back(argv[i]);
260 }
261
262 const CommandInfo* command_info = nullptr;
263 size_t consumed_tokens = 0;
264
265 if (args.size() >= 2) {
266 std::string candidate = absl::StrCat(args[0], " ", args[1]);
267 auto it = commands_.find(candidate);
268 if (it != commands_.end()) {
269 command_info = &it->second;
270 consumed_tokens = 2;
271 }
272 }
273
274 if (command_info == nullptr && !args.empty()) {
275 auto it = commands_.find(args[0]);
276 if (it != commands_.end()) {
277 command_info = &it->second;
278 consumed_tokens = 1;
279 }
280 }
281
282 if (command_info == nullptr) {
283 ShowHelp();
284 std::string joined = args.empty() ? std::string() : absl::StrJoin(args, " ");
285 return absl::NotFoundError(
286 absl::StrCat("Unknown command: ", joined.empty() ? "<none>" : joined));
287 }
288
289 std::vector<std::string> command_args(args.begin() + consumed_tokens, args.end());
290 return command_info->handler(command_args);
291}

325absl::Status ModernCLI::HandleExtractSymbolsCommand(const std::vector<std::string>& args) {
326 // Use the AsarWrapper to extract symbols
327 yaze::app::core::AsarWrapper wrapper;
328 RETURN_IF_ERROR(wrapper.Initialize());
329
330 auto symbols_result = wrapper.ExtractSymbols(args[0]);
331 if (!symbols_result.ok()) {
332 return symbols_result.status();
333 }
334
335 const auto& symbols = symbols_result.value();
336 std::cout << "🏷️ Extracted " << symbols.size() << " symbols from " << args[0] << ":" << std::endl;
337 std::cout << std::endl;
338
339 for (const auto& symbol : symbols) {
340 std::cout << absl::StrFormat(" %-20s @ $%06X", symbol.name, symbol.address) << std::endl;
341 }
342
343 return absl::OkStatus();
344}
