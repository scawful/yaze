323absl::Status ModernCLI::Run(int argc, char* argv[]) {
324 if (argc < 2) {
325 ShowHelp();
326 return absl::OkStatus();
327 }
328
329 std::vector<std::string> args;
330 args.reserve(argc - 1);
331 for (int i = 1; i < argc; ++i) {
332 args.emplace_back(argv[i]);
333 }
334
335 const CommandInfo* command_info = nullptr;
336 size_t consumed_tokens = 0;
337
338 if (args.size() >= 2) {
339 std::string candidate = absl::StrCat(args[0], " ", args[1]);
340 auto it = commands_.find(candidate);
341 if (it != commands_.end()) {
342 command_info = &it->second;
343 consumed_tokens = 2;
344 }
345 }
346
347 if (command_info == nullptr && !args.empty()) {
348 auto it = commands_.find(args[0]);
349 if (it != commands_.end()) {
350 command_info = &it->second;
351 consumed_tokens = 1;
352 }
353 }
354
355 if (command_info == nullptr) {
356 ShowHelp();
357 std::string joined = args.empty() ? std::string() : absl::StrJoin(args, " ");
358 return absl::NotFoundError(
359 absl::StrCat("Unknown command: ", joined.empty() ? "<none>" : joined));
360 }
361
362 std::vector<std::string> command_args(args.begin() + consumed_tokens, args.end());
363 return command_info->handler(command_args);
364}

398absl::Status ModernCLI::HandleExtractSymbolsCommand(const std::vector<std::string>& args) {
399 // Use the AsarWrapper to extract symbols
400 yaze::core::AsarWrapper wrapper;
401 RETURN_IF_ERROR(wrapper.Initialize());
402
403 auto symbols_result = wrapper.ExtractSymbols(args[0]);
404 if (!symbols_result.ok()) {
405 return symbols_result.status();
406 }
407
408 const auto& symbols = symbols_result.value();
409 std::cout << "🏷️ Extracted " << symbols.size() << " symbols from " << args[0] << ":" << std::endl;
410 std::cout << std::endl;
411
412 for (const auto& symbol : symbols) {
413 std::cout << absl::StrFormat(" %-20s @ $%06X", symbol.name, symbol.address) << std::endl;
414 }
415
416 return absl::OkStatus();
417}
