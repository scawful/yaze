330class ReadFromRom : public CommandHandler {
331 public:
332 absl::Status Run(const std::vector<std::string>& arg_vec) override {
333 RETURN_IF_ERROR(rom_.LoadFromFile(arg_vec[0]))
334
335 std::stringstream ss(arg_vec[1].data());
336 uint32_t offset;
337 ss >> std::hex >> offset;
338 uint32_t length = 0x01;
339 if (!arg_vec[2].empty()) {
340 length = std::stoi(arg_vec[2]);
341 }
342
343 if (length > 1) {
344 auto returned_bytes_status = rom_.ReadByteVector(offset, length);
345 if (!returned_bytes_status.ok()) {
346 return returned_bytes_status.status();
347 }
348 auto returned_bytes = returned_bytes_status.value();
349 for (const auto& each : returned_bytes) {
350 std::cout << each;
351 }
352 std::cout << std::endl;
353 } else {
354 auto byte = rom_.ReadByte(offset);
355 std::cout << std::hex << byte.value() << std::endl;
356 }
357
358 return absl::OkStatus();
359 }
