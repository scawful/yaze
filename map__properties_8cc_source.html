179void MapPropertiesSystem::DrawCustomBackgroundColorEditor(
180 int current_map, bool& show_custom_bg_color_editor) {
181 if (!overworld_->is_loaded()) {
182 Text("No overworld loaded");
183 return;
184 }
185
186 static uint8_t asm_version =
187 (*rom_)[zelda3::OverworldCustomASMHasBeenApplied];
188 if (asm_version < 2) {
189 Text("Custom background colors require ZSCustomOverworld v2+");
190 return;
191 }
192
193 Text("Custom Background Color Editor");
194 Separator();
195
196 // Enable/disable area-specific background color
197 static bool use_area_specific_bg_color = false;
198 if (ImGui::Checkbox("Use Area-Specific Background Color",
199 &use_area_specific_bg_color)) {
200 // Update ROM data
201 (*rom_)[zelda3::OverworldCustomAreaSpecificBGEnabled] =
202 use_area_specific_bg_color ? 1 : 0;
203 }
204
205 if (use_area_specific_bg_color) {
206 // Get current color
207 uint16_t current_color =
208 overworld_->overworld_map(current_map)->area_specific_bg_color();
209 gfx::SnesColor snes_color(current_color);
210
211 // Convert to ImVec4 for color picker
212 ImVec4 color_vec = gui::ConvertSnesColorToImVec4(snes_color);
213
214 if (ImGui::ColorPicker4(
215 "Background Color", (float*)&color_vec,
216 ImGuiColorEditFlags_DisplayRGB | ImGuiColorEditFlags_DisplayHex)) {
217 // Convert back to SNES color and update
218 gfx::SnesColor new_snes_color = gui::ConvertImVec4ToSnesColor(color_vec);
219 overworld_->mutable_overworld_map(current_map)
220 ->set_area_specific_bg_color(new_snes_color.snes());
221
222 // Update ROM
223 int rom_address =
224 zelda3::OverworldCustomAreaSpecificBGPalette + (current_map * 2);
225 (*rom_)[rom_address] = new_snes_color.snes() & 0xFF;
226 (*rom_)[rom_address + 1] = (new_snes_color.snes() >> 8) & 0xFF;
227 }
228
229 Text("SNES Color: 0x%04X", current_color);
230 }

1052void MapPropertiesSystem::DrawOverlayPreviewOnMap(int current_map,
1053 int current_world,
1054 bool show_overlay_preview) {
1055 gfx::ScopedTimer timer("map_properties_draw_overlay_preview");
1056
1057 if (!show_overlay_preview || !maps_bmp_ || !canvas_)
1058 return;
1059
1060 // Get subscreen overlay information based on ROM version and map type
1061 uint16_t overlay_id = 0x00FF;
1062 bool has_subscreen_overlay = false;
1063
1064 static uint8_t asm_version =
1065 (*rom_)[zelda3::OverworldCustomASMHasBeenApplied];
1066 bool is_special_overworld_map = (current_map >= 0x80 && current_map < 0xA0);
1067
1068 if (is_special_overworld_map) {
1069 // Special overworld maps (0x80-0x9F) do not support subscreen overlays
1070 return;
1071 }
1072
1073 // Light World (0x00-0x3F) and Dark World (0x40-0x7F) maps support subscreen overlays for all versions
1074 overlay_id = overworld_->overworld_map(current_map)->subscreen_overlay();
1075 has_subscreen_overlay = (overlay_id != 0x00FF);
1076
1077 if (!has_subscreen_overlay)
1078 return;
1079
1080 // Map subscreen overlay ID to special area map for bitmap
1081 int overlay_map_index = -1;
1082 if (overlay_id >= 0x80 && overlay_id < 0xA0) {
1083 overlay_map_index = overlay_id;
1084 }
1085
1086 if (overlay_map_index < 0 || overlay_map_index >= zelda3::kNumOverworldMaps)
1087 return;
1088
1089 // Get the subscreen overlay map's bitmap
1090 const auto& overlay_bitmap = (*maps_bmp_)[overlay_map_index];
1091 if (!overlay_bitmap.is_active())
1092 return;
1093
1094 // Calculate position for subscreen overlay preview on the current map
1095 int current_map_x = current_map % 8;
1096 int current_map_y = current_map / 8;
1097 if (current_world == 1) {
1098 current_map_x = (current_map - 0x40) % 8;
1099 current_map_y = (current_map - 0x40) / 8;
1100 } else if (current_world == 2) {
1101 current_map_x = (current_map - 0x80) % 8;
1102 current_map_y = (current_map - 0x80) / 8;
1103 }
1104
1105 int scale = static_cast<int>(canvas_->global_scale());
1106 int map_x = current_map_x * kOverworldMapSize * scale;
1107 int map_y = current_map_y * kOverworldMapSize * scale;
1108
1109 // Determine if this is a background or foreground subscreen overlay
1110 bool is_background_overlay =
1111 (overlay_id == 0x0095 || overlay_id == 0x0096 || overlay_id == 0x009C);
1112
1113 // Set alpha for semi-transparent preview
1114 ImU32 overlay_color =
1115 is_background_overlay ? IM_COL32(255, 255, 255, 128)
1116 : // Background subscreen overlays - lighter
1117 IM_COL32(255, 255, 255,
1118 180); // Foreground subscreen overlays - more opaque
1119
1120 // Draw the subscreen overlay bitmap with semi-transparency
1121 canvas_->draw_list()->AddImage(
1122 (ImTextureID)(intptr_t)overlay_bitmap.texture(), ImVec2(map_x, map_y),
1123 ImVec2(map_x + kOverworldMapSize * scale,
1124 map_y + kOverworldMapSize * scale),
1125 ImVec2(0, 0), ImVec2(1, 1), overlay_color);
1126}
