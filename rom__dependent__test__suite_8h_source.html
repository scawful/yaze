92 auto start_time = std::chrono::steady_clock::now();
93
94 TestResult result;
95 result.name = "ROM_Header_Validation_Test";
96 result.suite_name = GetName();
97 result.category = GetCategory();
98 result.timestamp = start_time;
99
100 if (!test_header_validation_) {
101 result.status = TestStatus::kSkipped;
102 result.error_message = "Header validation disabled in configuration";
103 } else {
104 try {
105 std::string title = rom->title();
106 size_t size = rom->size();
107
108 // Basic validation
109 bool valid_title = !title.empty() && title != "ZELDA3" && title.length() <= 21;
110 bool valid_size = size >= 1024*1024 && size <= 8*1024*1024; // 1MB to 8MB
111
112 if (valid_title && valid_size) {
113 result.status = TestStatus::kPassed;
114 result.error_message = absl::StrFormat(
115 "ROM header valid: '%s' (%zu bytes)", title.c_str(), size);
116 } else {
117 result.status = TestStatus::kFailed;
118 result.error_message = absl::StrFormat(
119 "ROM header validation failed: title='%s' size=%zu", title.c_str(), size);
120 }
121 } catch (const std::exception& e) {
122 result.status = TestStatus::kFailed;
123 result.error_message = "Header validation failed: " + std::string(e.what());
124 }
125 }
126
127 auto end_time = std::chrono::steady_clock::now();
128 result.duration = std::chrono::duration_cast<std::chrono::milliseconds>(
129 end_time - start_time);
