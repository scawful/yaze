115 // Copy the tile16 data into individual tiles.
116 const auto tile16_data = overworld_.tile16_blockset_data();
117
118 std::vector<std::future<void>> futures;
119 // Loop through the tiles and copy their pixel data into separate vectors
120 for (uint i = 0; i < kNumTile16Individual; i++) {
121 futures.push_back(std::async(
122 std::launch::async,
123 [&](int index) {
124 std::vector<uint8_t> tile_data(16 * 16, 0x00);
125 for (int ty = 0; ty < 16; ty++) {
126 for (int tx = 0; tx < 16; tx++) {
127 int position = tx + (ty * 0x10);
128 uint8_t value =
129 tile16_data[(index % 8 * 16) + (index / 8 * 16 * 0x80) +
130 (ty * 0x80) + tx];
131 tile_data[position] = value;
132 }
133 }
134 tile16_individual_[index].set_data(tile_data);
135 },
136 i));
137 }
138
139 for (auto &future : futures) {
140 future.get();
141 }
142
143 // Render the bitmaps of each tile.
144 for (uint id = 0; id < kNumTile16Individual; id++) {
145 RETURN_IF_ERROR(tile16_individual_[id].ApplyPalette(palette_));
146 Renderer::GetInstance().UpdateBitmap(&tile16_individual_[id]);
147 }
