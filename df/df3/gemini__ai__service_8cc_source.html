166 const std::string& response_body) {
167#ifdef YAZE_WITH_JSON
168 AgentResponse agent_response;
169
170 try {
171 nlohmann::json response_json = nlohmann::json::parse(response_body);
172
173 // Navigate Gemini's response structure
174 if (!response_json.contains("candidates") ||
175 response_json["candidates"].empty()) {
176 return absl::InternalError("❌ No candidates in Gemini response");
177 }
178
179 for (const auto& candidate : response_json["candidates"]) {
180 if (!candidate.contains("content") ||
181 !candidate["content"].contains("parts")) {
182 continue;
183 }
184
185 for (const auto& part : candidate["content"]["parts"]) {
186 if (!part.contains("text")) {
187 continue;
188 }
189
190 std::string text_content = part["text"].get<std::string>();
191
192 // Strip markdown code blocks if present (```json ... ```)
193 text_content = std::string(absl::StripAsciiWhitespace(text_content));
194 if (absl::StartsWith(text_content, "```json")) {
195 text_content = text_content.substr(7); // Remove ```json
196 } else if (absl::StartsWith(text_content, "```")) {
197 text_content = text_content.substr(3); // Remove ```
198 }
199 if (absl::EndsWith(text_content, "```")) {
200 text_content = text_content.substr(0, text_content.length() - 3);
201 }
202 text_content = std::string(absl::StripAsciiWhitespace(text_content));
203
204 // Parse as JSON object
205 try {
206 nlohmann::json response_json = nlohmann::json::parse(text_content);
207 if (response_json.contains("text_response") &&
208 response_json["text_response"].is_string()) {
209 agent_response.text_response =
210 response_json["text_response"].get<std::string>();
211 }
212 if (response_json.contains("reasoning") &&
213 response_json["reasoning"].is_string()) {
214 agent_response.reasoning =

217 if (response_json.contains("tool_calls") &&
