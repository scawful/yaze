145absl::StatusOr<std::vector<std::string>> GeminiAIService::ParseGeminiResponse(
146 const std::string& response_body) {
147#ifdef YAZE_WITH_JSON
148 std::vector<std::string> commands;
149
150 try {
151 nlohmann::json response_json = nlohmann::json::parse(response_body);
152
153 // Navigate Gemini's response structure
154 if (!response_json.contains("candidates") ||
155 response_json["candidates"].empty()) {
156 return absl::InternalError("❌ No candidates in Gemini response");
157 }
158
159 for (const auto& candidate : response_json["candidates"]) {
160 if (!candidate.contains("content") ||
161 !candidate["content"].contains("parts")) {
162 continue;
163 }
164
165 for (const auto& part : candidate["content"]["parts"]) {
166 if (!part.contains("text")) {
167 continue;
168 }
169
170 std::string text_content = part["text"].get<std::string>();
171
172 // Strip markdown code blocks if present (```json ... ```)
173 text_content = std::string(absl::StripAsciiWhitespace(text_content));
174 if (absl::StartsWith(text_content, "```json")) {
175 text_content = text_content.substr(7); // Remove ```json
176 } else if (absl::StartsWith(text_content, "```")) {
177 text_content = text_content.substr(3); // Remove ```
178 }
179 if (absl::EndsWith(text_content, "```")) {
180 text_content = text_content.substr(0, text_content.length() - 3);
181 }
182 text_content = std::string(absl::StripAsciiWhitespace(text_content));
183
184 // Parse as JSON array
185 try {
186 nlohmann::json commands_array = nlohmann::json::parse(text_content);
187
188 if (commands_array.is_array()) {
189 for (const auto& cmd : commands_array) {
190 if (cmd.is_string()) {
191 std::string command = cmd.get<std::string>();
192 // Remove "z3ed " prefix if LLM included it
193 if (absl::StartsWith(command, "z3ed ")) {
194 command = command.substr(5);
195 }
196 commands.push_back(command);
197 }
198 }
199 }
200 } catch (const nlohmann::json::exception& inner_e) {
201 // Fallback: Try to extract commands line by line
202 std::vector<std::string> lines = absl::StrSplit(text_content, '\n');
203 for (const auto& line : lines) {
204 std::string trimmed = std::string(absl::StripAsciiWhitespace(line));
205 if (!trimmed.empty() &&
206 (absl::StartsWith(trimmed, "z3ed ") ||
207 absl::StartsWith(trimmed, "palette ") ||
208 absl::StartsWith(trimmed, "overworld ") ||
209 absl::StartsWith(trimmed, "sprite ") ||
210 absl::StartsWith(trimmed, "dungeon "))) {
211 if (absl::StartsWith(trimmed, "z3ed ")) {
212 trimmed = trimmed.substr(5);
213 }
214 commands.push_back(trimmed);
215 }
216 }
217 }
218 }
219 }
220 } catch (const nlohmann::json::exception& e) {
221 return absl::InternalError(
222 absl::StrCat("❌ Failed to parse Gemini response: ", e.what()));
223 }
224
225 if (commands.empty()) {
226 return absl::InternalError(
227 "❌ No valid commands extracted from Gemini response\n"
228 " Raw response: " + response_body);
229 }
230
231 return commands;
232#else
233 return absl::UnimplementedError("JSON support required");
234#endif
235}
