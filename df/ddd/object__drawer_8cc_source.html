17absl::Status ObjectDrawer::DrawObject(const RoomObject& object,
18 gfx::BackgroundBuffer& bg1,
19 gfx::BackgroundBuffer& bg2,
20 const gfx::PaletteGroup& palette_group) {
21 if (!rom_ || !rom_->is_loaded()) {
22 return absl::FailedPreconditionError("ROM not loaded");
23 }
24
25 if (!routines_initialized_) {
26 return absl::FailedPreconditionError("Draw routines not initialized");
27 }
28
29 // Ensure object has tiles loaded
30 auto mutable_obj = const_cast<RoomObject&>(object);
31 LOG_DEBUG("ObjectDrawer", "Setting ROM for object ID=0x%02X", object.id_);
32 mutable_obj.set_rom(rom_);
33 LOG_DEBUG("ObjectDrawer", "Calling EnsureTilesLoaded for object ID=0x%02X", object.id_);
34 mutable_obj.EnsureTilesLoaded();
35
36 // Check if tiles were actually loaded on the mutable object
37 LOG_DEBUG("ObjectDrawer", "After EnsureTilesLoaded: mutable object has %zu tiles",
38 mutable_obj.tiles().size());
39
40 // Select buffer based on layer
41 auto& target_bg = (object.layer_ == RoomObject::LayerType::BG2) ? bg2 : bg1;
42 LOG_DEBUG("ObjectDrawer", "Object ID=0x%02X using %s buffer (layer=%d)",
43 object.id_, (object.layer_ == RoomObject::LayerType::BG2) ? "BG2" : "BG1", object.layer_);
44
45 // Skip objects that don't have tiles loaded - check mutable object
46 if (mutable_obj.tiles().empty()) {
47 LOG_DEBUG("ObjectDrawer", "Object ID=0x%02X has no tiles loaded, skipping", object.id_);
48 return absl::OkStatus();
49 }
50
51 LOG_DEBUG("ObjectDrawer", "Object ID=0x%02X has %zu tiles, proceeding with drawing",
52 object.id_, mutable_obj.tiles().size());
53
54 // Look up draw routine for this object
55 int routine_id = GetDrawRoutineId(object.id_);
56
57 if (routine_id < 0 || routine_id >= static_cast<int>(draw_routines_.size())) {
58 // Fallback to simple 1x1 drawing using first 8x8 tile
59 if (!mutable_obj.tiles().empty()) {
60 const auto& tile16 = mutable_obj.tiles()[0];
61 WriteTile8(target_bg, object.x_, object.y_, tile16.tile0_);
62 }
63 return absl::OkStatus();
64 }
65
66 // Execute the appropriate draw routine
67 draw_routines_[routine_id](this, object, target_bg, mutable_obj.tiles());
68
69 return absl::OkStatus();
70}

72absl::Status ObjectDrawer::DrawObjectList(
73 const std::vector<RoomObject>& objects,
74 gfx::BackgroundBuffer& bg1,
75 gfx::BackgroundBuffer& bg2,
76 const gfx::PaletteGroup& palette_group) {
77
78 LOG_DEBUG("ObjectDrawer", "Drawing %zu objects", objects.size());
79
80 int drawn_count = 0;
81 int skipped_count = 0;
82
83 for (const auto& object : objects) {
84 auto status = DrawObject(object, bg1, bg2, palette_group);
85 if (status.ok()) {
86 drawn_count++;
87 } else {
88 skipped_count++;
89 // Only print errors that aren't "no tiles" (which is common and expected)
90 if (status.code() != absl::StatusCode::kOk) {
91 // Skip silently - many objects don't have tiles loaded yet
92 }
93 }
94 }
95
96 // Only log if there are failures
97 if (skipped_count > 0) {
98 LOG_DEBUG("ObjectDrawer", "Drew %d objects, skipped %d", drawn_count, skipped_count);
99 }
100
101 return absl::OkStatus();
102}

244void ObjectDrawer::DrawRightwards2x2_1to15or32(const RoomObject& obj, gfx::BackgroundBuffer& bg,
245 const std::vector<gfx::Tile16>& tiles) {
246 // Pattern: Draws 2x2 tiles rightward (object 0x00)
247 // Size byte determines how many times to repeat (1-15 or 32)
248 int size = obj.size_;
249 if (size == 0) size = 32; // Special case for object 0x00
250
251 for (int s = 0; s < size; s++) {
252 if (tiles.size() >= 1) {
253 // Draw 2x2 pattern using 8x8 tiles from the first Tile16
254 const auto& tile16 = tiles[0];
255 WriteTile8(bg, obj.x_ + (s * 2), obj.y_, tile16.tile0_); // Top-left
256 WriteTile8(bg, obj.x_ + (s * 2) + 1, obj.y_, tile16.tile1_); // Top-right
257 WriteTile8(bg, obj.x_ + (s * 2), obj.y_ + 1, tile16.tile2_); // Bottom-left
258 WriteTile8(bg, obj.x_ + (s * 2) + 1, obj.y_ + 1, tile16.tile3_); // Bottom-right
259 }
260 }

331void ObjectDrawer::DrawDiagonalAcute_1to16(const RoomObject& obj, gfx::BackgroundBuffer& bg,
332 const std::vector<gfx::Tile16>& tiles) {
333 // Pattern: Diagonal line going down-right (/) (object 0x09)
334 int size = obj.size_ & 0x0F;
335
336 for (int s = 0; s < size + 6; s++) {
337 if (tiles.size() >= 1) {
338 // Use first tile16 for diagonal pattern
339 const auto& tile16 = tiles[0];
340 for (int i = 0; i < 5; i++) {
341 // Cycle through the 4 tiles in the tile16
342 const gfx::TileInfo& tile_info = (i % 4 == 0) ? tile16.tile0_ :
343 (i % 4 == 1) ? tile16.tile1_ :
344 (i % 4 == 2) ? tile16.tile2_ : tile16.tile3_;
345 WriteTile8(bg, obj.x_ + s, obj.y_ + (i - s), tile_info);
346 }
347 }
348 }

467void ObjectDrawer::DrawRightwards4x4_1to16(const RoomObject& obj, gfx::BackgroundBuffer& bg,
468 const std::vector<gfx::Tile16>& tiles) {
469 // Pattern: 4x4 block rightward (object 0x33)
470 int size = obj.size_ & 0x0F;
471
472 for (int s = 0; s < size; s++) {
473 if (tiles.size() >= 1) {
474 // Use first tile16 for 4x4 pattern (repeat the 2x2 pattern)
475 const auto& tile16 = tiles[0];
476 // Draw 2x2 pattern repeated to make 4x4
477 WriteTile8(bg, obj.x_ + (s * 4), obj.y_, tile16.tile0_); // Top-left
478 WriteTile8(bg, obj.x_ + (s * 4) + 1, obj.y_, tile16.tile1_); // Top-right
479 WriteTile8(bg, obj.x_ + (s * 4), obj.y_ + 1, tile16.tile2_); // Bottom-left
480 WriteTile8(bg, obj.x_ + (s * 4) + 1, obj.y_ + 1, tile16.tile3_); // Bottom-right
481 WriteTile8(bg, obj.x_ + (s * 4) + 2, obj.y_, tile16.tile0_); // Top-left (repeat)

664void ObjectDrawer::WriteTile8(gfx::BackgroundBuffer& bg, int tile_x, int tile_y,
665 const gfx::TileInfo& tile_info) {
666 LOG_DEBUG("ObjectDrawer", "Writing 8x8 tile at tile pos (%d,%d) to bitmap", tile_x, tile_y);
667
668 // Draw directly to bitmap instead of tile buffer to avoid being overwritten
669 auto& bitmap = bg.bitmap();
670 LOG_DEBUG("ObjectDrawer", "Bitmap status: active=%d, width=%d, height=%d, surface=%p",
671 bitmap.is_active(), bitmap.width(), bitmap.height(), bitmap.surface());
672 if (!bitmap.is_active() || bitmap.width() == 0) {
673 LOG_DEBUG("ObjectDrawer", "Bitmap not ready: active=%d, width=%d", bitmap.is_active(), bitmap.width());
674 return; // Bitmap not ready
675 }
676
677 // Get graphics data - prefer room-specific buffer if available
678 const uint8_t* gfx_data = nullptr;
679
680 if (room_gfx_buffer_) {
681 // Use room-specific graphics buffer (current_gfx16_)
682 gfx_data = room_gfx_buffer_;
683 } else if (rom_ && rom_->is_loaded()) {
684 // Fallback to ROM graphics buffer
685 auto rom_gfx = rom_->mutable_graphics_buffer();
686 if (rom_gfx && !rom_gfx->empty()) {
687 gfx_data = rom_gfx->data();
688 }
689 }
690
691 if (!gfx_data) {
692 LOG_DEBUG("ObjectDrawer", "ERROR: No graphics data available");
693 return;
694 }
695
696 // Draw single 8x8 tile directly to bitmap
697 DrawTileToBitmap(bitmap, tile_info, tile_x * 8, tile_y * 8, gfx_data);
698}

705void ObjectDrawer::DrawTileToBitmap(gfx::Bitmap& bitmap, const gfx::TileInfo& tile_info,
706 int pixel_x, int pixel_y, const uint8_t* tiledata) {
707 // Draw an 8x8 tile directly to bitmap at pixel coordinates
708 if (!tiledata) return;
709
710 // DEBUG: Check if bitmap is valid
711 if (!bitmap.is_active() || bitmap.width() == 0 || bitmap.height() == 0) {
712 LOG_DEBUG("ObjectDrawer", "ERROR: Invalid bitmap - active=%d, size=%dx%d",
713 bitmap.is_active(), bitmap.width(), bitmap.height());
714 return;
715 }
716
717 // Calculate tile position in graphics sheet (128 pixels wide)
718 int tile_sheet_x = (tile_info.id_ % 16) * 8; // 16 tiles per row
719 int tile_sheet_y = (tile_info.id_ / 16) * 8; // Each row is 16 tiles
720
721 // Clamp palette to valid range
722 uint8_t palette_id = tile_info.palette_ & 0x0F;
723 if (palette_id > 10) palette_id = palette_id % 11;
724 uint8_t palette_offset = palette_id * 8; // 3BPP: 8 colors per palette
725
726 // Force a visible palette for debugging
727 if (palette_id == 0) {
728 palette_id = 1; // Use palette 1 instead of 0
729 palette_offset = palette_id * 8;
730 }
731
732 // Draw 8x8 pixels
733 for (int py = 0; py < 8; py++) {
734 for (int px = 0; px < 8; px++) {
735 // Apply mirroring
736 int src_x = tile_info.horizontal_mirror_ ? (7 - px) : px;
737 int src_y = tile_info.vertical_mirror_ ? (7 - py) : py;
738
739 // Read pixel from graphics sheet
740 int src_index = (tile_sheet_y + src_y) * 128 + (tile_sheet_x + src_x);
741 uint8_t pixel_index = tiledata[src_index];
742
743 // Apply palette and write to bitmap
744 uint8_t final_color = pixel_index + palette_offset;
745 int dest_x = pixel_x + px;
746 int dest_y = pixel_y + py;
747
748 if (dest_x >= 0 && dest_x < bitmap.width() && dest_y >= 0 && dest_y < bitmap.height()) {
749 int dest_index = dest_y * bitmap.width() + dest_x;
750 if (dest_index >= 0 && dest_index < static_cast<int>(bitmap.mutable_data().size())) {
751 bitmap.mutable_data()[dest_index] = final_color;
752
753 // Debug first pixel of each tile
754 if (py == 0 && px == 0) {
755 LOG_DEBUG("ObjectDrawer", "Tile ID=0x%02X at (%d,%d): palette=%d, pixel_index=%d, final_color=%d",
756 tile_info.id_, pixel_x, pixel_y, palette_id, pixel_index, final_color);
757 }
758 }
759 }
760 }
761 }
