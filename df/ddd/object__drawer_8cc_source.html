13absl::Status ObjectDrawer::DrawObject(const RoomObject& object,
14 gfx::BackgroundBuffer& bg1,
15 gfx::BackgroundBuffer& bg2) {
16 if (!rom_ || !rom_->is_loaded()) {
17 return absl::FailedPreconditionError("ROM not loaded");
18 }
19
20 // Ensure object has tiles loaded
21 auto mutable_obj = const_cast<RoomObject&>(object);
22 mutable_obj.set_rom(rom_);
23 mutable_obj.EnsureTilesLoaded();
24
25 // Get tiles - silently skip objects that can't load tiles
26 if (object.tiles().empty()) {
27 // Many objects may not have tiles loaded yet - this is normal
28 // Just skip them rather than failing the whole draw operation
29 return absl::OkStatus();
30 }
31
32 const auto& tile = object.tiles()[0]; // Base tile for object
33
34 // Select buffer based on layer
35 auto& target_bg = (object.layer_ == RoomObject::LayerType::BG2) ? bg2 : bg1;
36
37 // Dispatch to pattern-specific drawing based on object ID
38 // This is reverse-engineered from the game's drawing routines
39
40 if (object.id_ == 0x34) {
41 // Object 0x34: 1x1 solid block (simplest)
42 Draw1x1Solid(object, target_bg, tile);
43 }
44 else if (object.id_ >= 0x00 && object.id_ <= 0x08) {
45 // Objects 0x00-0x08: Rightward 2x2 patterns
46 DrawRightwards2x2(object, target_bg, tile);
47 }
48 else if (object.id_ >= 0x60 && object.id_ <= 0x68) {
49 // Objects 0x60-0x68: Downward 2x2 patterns
50 DrawDownwards2x2(object, target_bg, tile);
51 }
52 else if (object.id_ >= 0x09 && object.id_ <= 0x14) {
53 // Objects 0x09-0x14: Diagonal acute patterns
54 DrawDiagonalAcute(object, target_bg, tile);
55 }
56 else if (object.id_ >= 0x15 && object.id_ <= 0x20) {
57 // Objects 0x15-0x20: Diagonal grave patterns
58 DrawDiagonalGrave(object, target_bg, tile);
59 }
60 else if (object.id_ == 0x33 || (object.id_ >= 0x70 && object.id_ <= 0x71)) {
61 // 4x4 block objects
62 Draw4x4Block(object, target_bg, tile);
63 }
64 else {
65 // Default: Draw as simple 1x1 at position
66 Draw1x1Solid(object, target_bg, tile);
67 }
68
69 return absl::OkStatus();
70}

72absl::Status ObjectDrawer::DrawObjectList(
73 const std::vector<RoomObject>& objects,
74 gfx::BackgroundBuffer& bg1,
75 gfx::BackgroundBuffer& bg2) {
76
77 int drawn_count = 0;
78 int skipped_count = 0;
79
80 for (const auto& object : objects) {
81 auto status = DrawObject(object, bg1, bg2);
82 if (status.ok()) {
83 drawn_count++;
84 } else {
85 skipped_count++;
86 // Only print errors that aren't "no tiles" (which is common and expected)
87 if (status.code() != absl::StatusCode::kOk) {
88 // Skip silently - many objects don't have tiles loaded yet
89 }
90 }
91 }
92
93 if (drawn_count > 0 || skipped_count > 0) {
94 printf("[ObjectDrawer] Drew %d objects, skipped %d\n", drawn_count, skipped_count);
95 }
96
97 return absl::OkStatus();
98}

111void ObjectDrawer::DrawRightwards2x2(const RoomObject& obj,
112 gfx::BackgroundBuffer& bg,
113 const gfx::Tile16& tile) {
114 // Pattern: Draws 2x2 tiles rightward
115 // Size byte determines how many times to repeat
116 int repeat_count = (obj.size_ & 0x0F) + 1; // Low nibble = width
117
118 for (int i = 0; i < repeat_count; i++) {
119 // Each iteration draws a 2x2 tile16
120 int tile_x = obj.x_ + (i * 2); // Each tile16 is 2x2 8x8 tiles
121 int tile_y = obj.y_;
122
123 WriteTile16(bg, tile_x, tile_y, tile);
124 }
125}

188void ObjectDrawer::WriteTile16(gfx::BackgroundBuffer& bg, int tile_x, int tile_y,
189 const gfx::Tile16& tile) {
190 // A Tile16 is 2x2 8x8 tiles, so we write 4 tile entries
191
192 // Top-left (tile0)
193 if (IsValidTilePosition(tile_x, tile_y)) {
194 bg.SetTileAt(tile_x, tile_y, gfx::TileInfoToWord(tile.tile0_));
195 }
196
197 // Top-right (tile1)
198 if (IsValidTilePosition(tile_x + 1, tile_y)) {
199 bg.SetTileAt(tile_x + 1, tile_y, gfx::TileInfoToWord(tile.tile1_));
200 }
201
202 // Bottom-left (tile2)
203 if (IsValidTilePosition(tile_x, tile_y + 1)) {
204 bg.SetTileAt(tile_x, tile_y + 1, gfx::TileInfoToWord(tile.tile2_));
205 }
206
207 // Bottom-right (tile3)
208 if (IsValidTilePosition(tile_x + 1, tile_y + 1)) {
209 bg.SetTileAt(tile_x + 1, tile_y + 1, gfx::TileInfoToWord(tile.tile3_));
210 }
211}
