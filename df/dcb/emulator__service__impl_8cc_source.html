14emu::input::SnesButton ToSnesButton(Button button) {
15 using emu::input::SnesButton;
16 switch (button) {
17 case A: return SnesButton::A;
18 case B: return SnesButton::B;
19 case X: return SnesButton::X;
20 case Y: return SnesButton::Y;
21 case L: return SnesButton::L;
22 case R: return SnesButton::R;
23 case SELECT: return SnesButton::SELECT;
24 case START: return SnesButton::START;
25 case UP: return SnesButton::UP;
26 case DOWN: return SnesButton::DOWN;
27 case LEFT: return SnesButton::LEFT;
