185absl::Status OverworldFindTile::Run(const std::vector<std::string>& arg_vec) {
186 std::unordered_map<std::string, std::string> options;
187 std::vector<std::string> positional;
188 options.reserve(arg_vec.size());
189
190 for (size_t i = 0; i < arg_vec.size(); ++i) {
191 const std::string& token = arg_vec[i];
192 if (absl::StartsWith(token, "--")) {
193 std::string key;
194 std::string value;
195 auto eq_pos = token.find('=');
196 if (eq_pos != std::string::npos) {
197 key = token.substr(2, eq_pos - 2);
198 value = token.substr(eq_pos + 1);
199 } else {
200 key = token.substr(2);
201 if (i + 1 >= arg_vec.size()) {
202 return absl::InvalidArgumentError(
203 absl::StrCat("Missing value for --", key, "\n", kFindTileUsage));
204 }
205 value = arg_vec[++i];
206 }
207 if (value.empty()) {
208 return absl::InvalidArgumentError(
209 absl::StrCat("Missing value for --", key, "\n", kFindTileUsage));
210 }
211 options[key] = value;
212 } else {
213 positional.push_back(token);
214 }
215 }
216
217 if (!positional.empty()) {
218 return absl::InvalidArgumentError(
219 absl::StrCat("Unexpected positional arguments: ",
220 absl::StrJoin(positional, ", "), "\n", kFindTileUsage));
221 }
222
223 auto tile_it = options.find("tile");
224 if (tile_it == options.end()) {
225 return absl::InvalidArgumentError(
226 absl::StrCat("Missing required --tile argument\n", kFindTileUsage));
227 }
228
229 ASSIGN_OR_RETURN(int tile_value, ParseNumeric(tile_it->second));
230 if (tile_value < 0 || tile_value > 0xFFFF) {
231 return absl::InvalidArgumentError(
232 absl::StrCat("Tile ID must be between 0x0000 and 0xFFFF (got ",
233 tile_it->second, ")"));
234 }
235 const uint16_t tile_id = static_cast<uint16_t>(tile_value);
236
237 std::optional<int> map_filter;
238 if (auto map_it = options.find("map"); map_it != options.end()) {
239 ASSIGN_OR_RETURN(int map_value, ParseNumeric(map_it->second));
240 if (map_value < 0 || map_value >= 0xA0) {
241 return absl::InvalidArgumentError(
242 absl::StrCat("Map ID out of range: ", map_it->second));
243 }
244 map_filter = map_value;
245 }
246
247 std::optional<int> world_filter;
248 if (auto world_it = options.find("world"); world_it != options.end()) {
249 ASSIGN_OR_RETURN(int parsed_world, WorldFromString(world_it->second));
250 world_filter = parsed_world;
251 }
252
253 if (map_filter.has_value()) {
254 ASSIGN_OR_RETURN(int inferred_world, WorldFromMapId(*map_filter));
255 if (world_filter.has_value() && inferred_world != *world_filter) {
256 return absl::InvalidArgumentError(
257 absl::StrCat("Map 0x",
258 absl::StrFormat("%02X", *map_filter),
259 " belongs to the ", WorldName(inferred_world),
260 " World but --world requested ", WorldName(*world_filter)));
261 }
262 if (!world_filter.has_value()) {
263 world_filter = inferred_world;
264 }
265 }
266
267 std::string format = "text";
268 if (auto format_it = options.find("format"); format_it != options.end()) {
269 format = absl::AsciiStrToLower(format_it->second);
270 if (format != "text" && format != "json") {
271 return absl::InvalidArgumentError(
272 absl::StrCat("Unsupported format: ", format_it->second));
273 }
274 }
275
276 std::string rom_file = absl::GetFlag(FLAGS_rom);
277 if (auto rom_it = options.find("rom"); rom_it != options.end()) {
278 rom_file = rom_it->second;
279 }
280
281 if (rom_file.empty()) {
282 return absl::InvalidArgumentError(
283 "ROM file must be provided via --rom flag.");
284 }
285
286 auto load_status = rom_.LoadFromFile(rom_file);
287 if (!load_status.ok()) {
288 return load_status;
289 }
290 if (!rom_.is_loaded()) {
291 return absl::AbortedError("Failed to load ROM.");
292 }

295 auto ow_status = overworld.Load(&rom_);
296 if (!ow_status.ok()) {
297 return ow_status;

300 struct TileMatch {
301 int map_id;
302 int world;
303 int local_x;
304 int local_y;
305 int global_x;
306 int global_y;
307 };
308
309 std::vector<int> worlds_to_search;
310 if (world_filter.has_value()) {
311 worlds_to_search.push_back(*world_filter);
312 } else {
313 worlds_to_search = {0, 1, 2};
314 }
315
316 std::vector<TileMatch> matches;
317
318 for (int world : worlds_to_search) {
319 int world_start = 0;
320 int world_maps = 0;
321 switch (world) {
322 case 0:
323 world_start = 0x00;
324 world_maps = 0x40;
325 break;
326 case 1:
327 world_start = 0x40;
328 world_maps = 0x40;
329 break;
330 case 2:
331 world_start = 0x80;
332 world_maps = 0x20;
333 break;
334 default:
335 return absl::InvalidArgumentError(
336 absl::StrCat("Unknown world index: ", world));
337 }
338
339 overworld.set_current_world(world);
340
341 for (int local_map = 0; local_map < world_maps; ++local_map) {
342 int map_id = world_start + local_map;
343 if (map_filter.has_value() && map_id != *map_filter) {
344 continue;
345 }
346
347 int map_x_index = local_map % 8;
348 int map_y_index = local_map / 8;
349
350 int global_x_start = map_x_index * 32;
351 int global_y_start = map_y_index * 32;
352
353 for (int local_y = 0; local_y < 32; ++local_y) {
354 for (int local_x = 0; local_x < 32; ++local_x) {
355 int global_x = global_x_start + local_x;
356 int global_y = global_y_start + local_y;
357
358 uint16_t tile = overworld.GetTile(global_x, global_y);
359 if (tile == tile_id) {
360 matches.push_back({map_id, world, local_x, local_y, global_x,
361 global_y});
362 }
363 }
364 }
365 }
366 }
367
368 if (format == "json") {
369 std::cout << "{\n";
370 std::cout << absl::StrFormat(
371 " \"tile\": \"0x%04X\",\n", tile_id);
372 std::cout << absl::StrFormat(
373 " \"match_count\": %zu,\n", matches.size());
374 std::cout << " \"matches\": [\n";
375 for (size_t i = 0; i < matches.size(); ++i) {
376 const auto& match = matches[i];
377 std::cout << absl::StrFormat(
378 " {\"map\": \"0x%02X\", \"world\": \"%s\", "
379 "\"local\": {\"x\": %d, \"y\": %d}, "
380 "\"global\": {\"x\": %d, \"y\": %d}}%s\n",
381 match.map_id, WorldName(match.world), match.local_x, match.local_y,
382 match.global_x, match.global_y,
383 (i + 1 == matches.size()) ? "" : ",");
384 }
385 std::cout << " ]\n";
386 std::cout << "}\n";
387 } else {
388 std::cout << absl::StrFormat(
389 "🔎 Tile 0x%04X → %zu match(es)\n", tile_id, matches.size());
390 if (matches.empty()) {
391 std::cout << " No matches found." << std::endl;
392 return absl::OkStatus();
393 }
394
395 for (const auto& match : matches) {
396 std::cout << absl::StrFormat(
397 " • Map 0x%02X (%s World) local(%2d,%2d) global(%3d,%3d)\n",
398 match.map_id, WorldName(match.world), match.local_x, match.local_y,
399 match.global_x, match.global_y);
400 }
401 }
402
403 return absl::OkStatus();
404}
