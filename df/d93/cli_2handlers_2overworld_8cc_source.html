84 x = std::stoi(arg_vec[++i]);

86 y = std::stoi(arg_vec[++i]);

88 tile_id = std::stoi(arg_vec[++i], nullptr, 16);

92 if (map_id == -1 || x == -1 || y == -1 || tile_id == -1) {

94 }

100

102 if (!load_status.ok()) {
103 return load_status;
104 }
105 if (!rom_.is_loaded()) {
106 return absl::AbortedError("Failed to load ROM.");
107 }
108

110 auto status = overworld.Load(&rom_);
111 if (!status.ok()) {
112 return status;

129 if (!save_status.ok()) {
130 return save_status;
131 }
132

134 << " to: 0x" << std::hex << tile_id << std::dec << std::endl;

294 std::cout << "}\n";
295 } else {
296 std::cout << absl::StrFormat(

300 return absl::OkStatus();
301 }
302

304 std::cout << absl::StrFormat(

308 match.global_x, match.global_y);
