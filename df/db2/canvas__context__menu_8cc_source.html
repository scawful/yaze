71void CanvasContextMenu::Render(const std::string& context_id, const ImVec2& mouse_pos, Rom* rom,
72 const gfx::Bitmap* bitmap, const gfx::SnesPalette* palette,
73 const std::function<void(Command, const CanvasConfig&)>& command_handler,
74 CanvasConfig current_config) {
75 if (!enabled_) return;
76
77 // Context menu (under default mouse threshold)
78 if (ImVec2 drag_delta = ImGui::GetMouseDragDelta(ImGuiMouseButton_Right);
79 enable_context_menu_ && drag_delta.x == 0.0F && drag_delta.y == 0.0F) {
80 ImGui::OpenPopupOnItemClick(context_id.c_str(), ImGuiPopupFlags_MouseButtonRight);
81 }
82
83 // Contents of the Context Menu
84 if (ImGui::BeginPopup(context_id.c_str())) {
85 // Render usage-specific menu first
86 RenderUsageSpecificMenu();
87
88 // Add separator if there are usage-specific items
89 if (!usage_specific_items_[current_usage_].empty()) {
90 ImGui::Separator();
91 }
92
93 // Render view controls
94 RenderViewControlsMenu(command_handler, current_config);
95 ImGui::Separator();
96
97 // Render canvas properties
98 RenderCanvasPropertiesMenu(command_handler, current_config);
99 ImGui::Separator();
100
101 // Render bitmap operations if bitmap is available
102 if (bitmap) {
103 RenderBitmapOperationsMenu(const_cast<gfx::Bitmap*>(bitmap));
104 ImGui::Separator();
105 }
106
107 // Render palette operations if bitmap is available
108 if (bitmap) {
109 RenderPaletteOperationsMenu(rom, const_cast<gfx::Bitmap*>(bitmap));
110 ImGui::Separator();
111 }
112
113 if (bitmap) {
114 RenderBppOperationsMenu(bitmap);
115 ImGui::Separator();
116 }
117
118 RenderPerformanceMenu();
119 ImGui::Separator();
120
121 RenderGridControlsMenu(command_handler, current_config);
122 ImGui::Separator();
123
124 RenderScalingControlsMenu(command_handler, current_config);
125
126 // Render global menu items
127 if (!global_items_.empty()) {
128 ImGui::Separator();
129 RenderMenuSection("Custom Actions", global_items_);
130 }
131
132 ImGui::EndPopup();
133 }

164void CanvasContextMenu::RenderMenuItem(const ContextMenuItem& item) {
165 if (!item.visible_condition()) {
166 return;
167 }
168
169 if (!item.enabled_condition()) {
170 ImGui::BeginDisabled();
171 }
172
173 if (item.subitems.empty()) {
174 // Simple menu item
175 if (ImGui::MenuItem(item.label.c_str(),
176 item.shortcut.empty() ? nullptr : item.shortcut.c_str())) {
177 item.callback();
178 }
179 } else {
180 // Menu with subitems
181 if (ImGui::BeginMenu(item.label.c_str())) {
182 for (const auto& subitem : item.subitems) {
183 RenderMenuItem(subitem);
184 }
185 ImGui::EndMenu();
186 }
187 }
188
189 if (!item.enabled_condition()) {
190 ImGui::EndDisabled();
191 }
192
193 if (item.separator_after) {
194 ImGui::Separator();
195 }
