65int CustomFormatString(char* buf, size_t buf_size, const char* fmt, ...) {
66 va_list args;
67 va_start(args, fmt);
68#ifdef IMGUI_USE_STB_SPRINTF
69 int w = stbsp_vsnprintf(buf, (int)buf_size, fmt, args);
70#else
71 int w = vsnprintf(buf, buf_size, fmt, args);
72#endif
73 va_end(args);
74 if (buf == nullptr) return w;
75 if (w == -1 || w >= (int)buf_size) w = (int)buf_size - 1;
76 buf[w] = 0;
77 return w;
78}

321absl::Status PaletteEditor::HandleColorPopup(gfx::SnesPalette& palette, int i,
322 int j, int n) {
323 auto col = gfx::ToFloatArray(palette[n]);
324 auto original_color = palette[n];
325 if (gui::SnesColorEdit4("Edit Color", &palette[n], kColorPopupFlags)) {
326 history_.RecordChange(/*group_name=*/std::string(kPaletteGroupNames[i]),
327 /*palette_index=*/j, /*color_index=*/n,
328 original_color, palette[n]);
329 palette[n].set_modified(true);
330 }
331
332 if (Button("Copy as..", ImVec2(-1, 0))) OpenPopup("Copy");
333 if (BeginPopup("Copy")) {
334 int cr = F32_TO_INT8_SAT(col[0]);
335 int cg = F32_TO_INT8_SAT(col[1]);
336 int cb = F32_TO_INT8_SAT(col[2]);
337 char buf[64];
338
339 CustomFormatString(buf, IM_ARRAYSIZE(buf), "(%.3ff, %.3ff, %.3ff)", col[0],
340 col[1], col[2]);
341 if (Selectable(buf)) SetClipboardText(buf);
342
343 CustomFormatString(buf, IM_ARRAYSIZE(buf), "(%d,%d,%d)", cr, cg, cb);
344 if (Selectable(buf)) SetClipboardText(buf);
345
346 CustomFormatString(buf, IM_ARRAYSIZE(buf), "#%02X%02X%02X", cr, cg, cb);
347 if (Selectable(buf)) SetClipboardText(buf);
348
349 // SNES Format
350 CustomFormatString(buf, IM_ARRAYSIZE(buf), "$%04X",
351 ConvertRGBtoSNES(ImVec4(col[0], col[1], col[2], 1.0f)));
352 if (Selectable(buf)) SetClipboardText(buf);
353
354 EndPopup();
355 }
356
357 EndPopup();
358 return absl::OkStatus();
