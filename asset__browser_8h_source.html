21// Extra functions to add deletion support to ImGuiSelectionBasicStorage
22struct ExampleSelectionWithDeletion : ImGuiSelectionBasicStorage {
23 // Find which item should be Focused after deletion.
24 // Call _before_ item submission. Retunr an index in the before-deletion item
25 // list, your item loop should call SetKeyboardFocusHere() on it. The
26 // subsequent ApplyDeletionPostLoop() code will use it to apply Selection.
27 // - We cannot provide this logic in core Dear ImGui because we don't have
28 // access to selection data.
29 // - We don't actually manipulate the ImVector<> here, only in
30 // ApplyDeletionPostLoop(), but using similar API for consistency and
31 // flexibility.
32 // - Important: Deletion only works if the underlying ImGuiID for your items
33 // are stable: aka not depend on their index, but on e.g. item id/ptr.
34 // FIXME-MULTISELECT: Doesn't take account of the possibility focus target
35 // will be moved during deletion. Need refocus or scroll offset.
36 int ApplyDeletionPreLoop(ImGuiMultiSelectIO* ms_io, int items_count) {
37 if (Size == 0) return -1;
38
39 // If focused item is not selected...
40 const int focused_idx =
41 (int)ms_io->NavIdItem; // Index of currently focused item
42 if (ms_io->NavIdSelected ==
43 false) // This is merely a shortcut, ==
44 // Contains(adapter->IndexToStorage(items, focused_idx))
45 {
46 ms_io->RangeSrcReset =
47 true; // Request to recover RangeSrc from NavId next frame. Would be
48 // ok to reset even when NavIdSelected==true, but it would take
49 // an extra frame to recover RangeSrc when deleting a selected
50 // item.
51 return focused_idx; // Request to focus same item after deletion.
52 }
53
54 // If focused item is selected: land on first unselected item after focused
55 // item.
56 for (int idx = focused_idx + 1; idx < items_count; idx++)
57 if (!Contains(GetStorageIdFromIndex(idx))) return idx;
58
59 // If focused item is selected: otherwise return last unselected item before
60 // focused item.
61 for (int idx = IM_MIN(focused_idx, items_count) - 1; idx >= 0; idx--)
62 if (!Contains(GetStorageIdFromIndex(idx))) return idx;
63
64 return -1;
65 }
66
67 // Rewrite item list (delete items) + update selection.
68 // - Call after EndMultiSelect()
69 // - We cannot provide this logic in core Dear ImGui because we don't have
70 // access to your items, nor to selection data.
71 template <typename ITEM_TYPE>
72 void ApplyDeletionPostLoop(ImGuiMultiSelectIO* ms_io,
73 ImVector<ITEM_TYPE>& items,
74 int item_curr_idx_to_select) {
75 // Rewrite item list (delete items) + convert old selection index (before
76 // deletion) to new selection index (after selection). If NavId was not part
77 // of selection, we will stay on same item.
78 ImVector<ITEM_TYPE> new_items;
79 new_items.reserve(items.Size - Size);
80 int item_next_idx_to_select = -1;
81 for (int idx = 0; idx < items.Size; idx++) {
82 if (!Contains(GetStorageIdFromIndex(idx)))
83 new_items.push_back(items[idx]);
84 if (item_curr_idx_to_select == idx)
85 item_next_idx_to_select = new_items.Size - 1;
86 }
87 items.swap(new_items);
88
89 // Update selection
90 Clear();
91 if (item_next_idx_to_select != -1 && ms_io->NavIdSelected)
92 SetItemSelected(GetStorageIdFromIndex(item_next_idx_to_select), true);
93 }
94};

107 static void SortWithSortSpecs(ImGuiTableSortSpecs* sort_specs,
108 AssetObject* items, int items_count) {
109 // Store in variable accessible by the sort function.
110 s_current_sort_specs = sort_specs;
111 if (items_count > 1)
112 qsort(items, (size_t)items_count, sizeof(items[0]),
113 AssetObject::CompareWithSortSpecs);
114 s_current_sort_specs = NULL;
115 }

117 // Compare function to be used by qsort()
118 static int IMGUI_CDECL CompareWithSortSpecs(const void* lhs,
119 const void* rhs) {
120 const AssetObject* a = (const AssetObject*)lhs;
121 const AssetObject* b = (const AssetObject*)rhs;
122 for (int n = 0; n < s_current_sort_specs->SpecsCount; n++) {
123 const ImGuiTableColumnSortSpecs* sort_spec =
124 &s_current_sort_specs->Specs[n];
125 int delta = 0;
126 if (sort_spec->ColumnIndex == 0)
127 delta = ((int)a->ID - (int)b->ID);
128 else if (sort_spec->ColumnIndex == 1)
129 delta = (a->Type - b->Type);
130 if (delta > 0)
131 return (sort_spec->SortDirection == ImGuiSortDirection_Ascending) ? +1
132 : -1;
133 if (delta < 0)
134 return (sort_spec->SortDirection == ImGuiSortDirection_Ascending) ? -1
135 : +1;
136 }
137 return ((int)a->ID - (int)b->ID);
138 }
