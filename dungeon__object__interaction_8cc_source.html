8void DungeonObjectInteraction::HandleCanvasMouseInput() {
9 const ImGuiIO& io = ImGui::GetIO();
10
11 // Check if mouse is over the canvas
12 if (!canvas_->IsMouseHovering()) {
13 return;
14 }
15
16 // Get mouse position relative to canvas
17 ImVec2 mouse_pos = io.MousePos;
18 ImVec2 canvas_pos = canvas_->zero_point();
19 ImVec2 canvas_size = canvas_->canvas_size();
20
21 // Convert to canvas coordinates
22 ImVec2 canvas_mouse_pos =
23 ImVec2(mouse_pos.x - canvas_pos.x, mouse_pos.y - canvas_pos.y);
24
25 // Handle mouse clicks
26 if (ImGui::IsMouseClicked(ImGuiMouseButton_Left)) {
27 if (ImGui::IsKeyDown(ImGuiKey_LeftCtrl) ||
28 ImGui::IsKeyDown(ImGuiKey_RightCtrl)) {
29 // Start selection box
30 is_selecting_ = true;
31 select_start_pos_ = canvas_mouse_pos;
32 select_current_pos_ = canvas_mouse_pos;
33 selected_objects_.clear();
34 } else {
35 // Start dragging or place object
36 if (object_loaded_) {
37 // Convert canvas coordinates to room coordinates
38 auto [room_x, room_y] =
39 CanvasToRoomCoordinates(static_cast<int>(canvas_mouse_pos.x),
40 static_cast<int>(canvas_mouse_pos.y));
41 PlaceObjectAtPosition(room_x, room_y);
42 } else {
43 // Start dragging existing objects
44 is_dragging_ = true;
45 drag_start_pos_ = canvas_mouse_pos;
46 drag_current_pos_ = canvas_mouse_pos;
47 }
48 }
49 }
50
51 // Handle mouse drag
52 if (is_selecting_ && ImGui::IsMouseDragging(ImGuiMouseButton_Left)) {
53 select_current_pos_ = canvas_mouse_pos;
54 UpdateSelectedObjects();
55 }
56
57 if (is_dragging_ && ImGui::IsMouseDragging(ImGuiMouseButton_Left)) {
58 drag_current_pos_ = canvas_mouse_pos;
59 DrawDragPreview();
60 }
61
62 // Handle mouse release
63 if (ImGui::IsMouseReleased(ImGuiMouseButton_Left)) {
64 if (is_selecting_) {
65 is_selecting_ = false;
66 UpdateSelectedObjects();
67 }
68 if (is_dragging_) {
69 is_dragging_ = false;
70 // TODO: Apply drag transformation to selected objects
71 }
72 }
73}

85void DungeonObjectInteraction::DrawObjectSelectRect() {
86 if (!canvas_->IsMouseHovering()) return;
87
88 const ImGuiIO& io = ImGui::GetIO();
89 const ImVec2 canvas_pos = canvas_->zero_point();
90 const ImVec2 mouse_pos =
91 ImVec2(io.MousePos.x - canvas_pos.x, io.MousePos.y - canvas_pos.y);
92
93 static bool dragging = false;
94 static ImVec2 drag_start_pos;
95
96 // Right click to start object selection
97 if (ImGui::IsMouseClicked(ImGuiMouseButton_Right) && !object_loaded_) {
98 drag_start_pos = mouse_pos;
99 object_select_start_ = mouse_pos;
100 selected_object_indices_.clear();
101 object_select_active_ = false;
102 dragging = false;
103 }
104
105 // Right drag to create selection rectangle
106 if (ImGui::IsMouseDragging(ImGuiMouseButton_Right) && !object_loaded_) {
107 object_select_end_ = mouse_pos;
108 dragging = true;
109
110 // Draw selection rectangle
111 ImVec2 start =
112 ImVec2(canvas_pos.x + std::min(drag_start_pos.x, mouse_pos.x),
113 canvas_pos.y + std::min(drag_start_pos.y, mouse_pos.y));
114 ImVec2 end = ImVec2(canvas_pos.x + std::max(drag_start_pos.x, mouse_pos.x),
115 canvas_pos.y + std::max(drag_start_pos.y, mouse_pos.y));
116
117 ImDrawList* draw_list = ImGui::GetWindowDrawList();
118 draw_list->AddRect(start, end, IM_COL32(255, 255, 0, 255), 0.0f, 0, 2.0f);
119 draw_list->AddRectFilled(start, end, IM_COL32(255, 255, 0, 32));
120 }
121
122 // Complete selection on mouse release
123 if (dragging && !ImGui::IsMouseDown(ImGuiMouseButton_Right)) {
124 dragging = false;
125 object_select_active_ = true;
126 SelectObjectsInRect();
127 }

130void DungeonObjectInteraction::SelectObjectsInRect() {
131 if (!rooms_ || current_room_id_ < 0 || current_room_id_ >= 296) return;
132
133 auto& room = (*rooms_)[current_room_id_];
134 selected_object_indices_.clear();
135
136 // Calculate selection bounds in room coordinates
137 auto [start_room_x, start_room_y] = CanvasToRoomCoordinates(
138 static_cast<int>(std::min(object_select_start_.x, object_select_end_.x)),
139 static_cast<int>(std::min(object_select_start_.y, object_select_end_.y)));
140 auto [end_room_x, end_room_y] = CanvasToRoomCoordinates(
141 static_cast<int>(std::max(object_select_start_.x, object_select_end_.x)),
142 static_cast<int>(std::max(object_select_start_.y, object_select_end_.y)));
143
144 // Find objects within selection rectangle
145 const auto& objects = room.GetTileObjects();
146 for (size_t i = 0; i < objects.size(); ++i) {
147 const auto& object = objects[i];
148 if (object.x_ >= start_room_x && object.x_ <= end_room_x &&
149 object.y_ >= start_room_y && object.y_ <= end_room_y) {
150 selected_object_indices_.push_back(i);
151 }
152 }
153
154 // Highlight selected objects
155 if (!selected_object_indices_.empty()) {
156 for (size_t index : selected_object_indices_) {
157 if (index < objects.size()) {
158 const auto& object = objects[index];
159 auto [canvas_x, canvas_y] =
160 RoomToCanvasCoordinates(object.x_, object.y_);
161
162 // Draw selection highlight
163 ImDrawList* draw_list = ImGui::GetWindowDrawList();
164 ImVec2 canvas_pos = canvas_->zero_point();
165 ImVec2 obj_start(canvas_pos.x + canvas_x - 2,
166 canvas_pos.y + canvas_y - 2);
167 ImVec2 obj_end(canvas_pos.x + canvas_x + 18,
168 canvas_pos.y + canvas_y + 18);
169 draw_list->AddRect(obj_start, obj_end, IM_COL32(0, 255, 255, 255), 0.0f,
170 0, 2.0f);
171 }
172 }
173 }

254bool DungeonObjectInteraction::IsObjectInSelectBox(
255 const zelda3::RoomObject& object) const {
256 if (!is_selecting_) return false;
257
258 // Convert object position to canvas coordinates
259 auto [canvas_x, canvas_y] = RoomToCanvasCoordinates(object.x_, object.y_);
260
261 // Calculate select box bounds
262 float min_x = std::min(select_start_pos_.x, select_current_pos_.x);
263 float max_x = std::max(select_start_pos_.x, select_current_pos_.x);
264 float min_y = std::min(select_start_pos_.y, select_current_pos_.y);
265 float max_y = std::max(select_start_pos_.y, select_current_pos_.y);
266
267 // Check if object is within select box
268 return (canvas_x >= min_x && canvas_x <= max_x && canvas_y >= min_y &&
269 canvas_y <= max_y);
