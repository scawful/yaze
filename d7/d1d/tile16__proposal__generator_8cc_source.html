73absl::StatusOr<Tile16Change> Tile16ProposalGenerator::ParseSetTileCommand(
74 const std::string& command,
75 Rom* rom) {
76
77 // Expected format: "overworld set-tile --map 0 --x 10 --y 20 --tile 0x02E"
78 std::vector<std::string> parts = absl::StrSplit(command, ' ');
79
80 if (parts.size() < 10) {
81 return absl::InvalidArgumentError(
82 absl::StrCat("Invalid command format: ", command));
83 }
84
85 if (parts[0] != "overworld" || parts[1] != "set-tile") {
86 return absl::InvalidArgumentError(
87 absl::StrCat("Not a set-tile command: ", command));
88 }
89
90 Tile16Change change;
91
92 // Parse arguments
93 for (size_t i = 2; i < parts.size(); i += 2) {
94 if (i + 1 >= parts.size()) break;
95
96 const std::string& flag = parts[i];
97 const std::string& value = parts[i + 1];
98
99 if (flag == "--map") {
100 change.map_id = std::stoi(value);
101 } else if (flag == "--x") {
102 change.x = std::stoi(value);
103 } else if (flag == "--y") {
104 change.y = std::stoi(value);
105 } else if (flag == "--tile") {
106 // Parse as hex (both 0x prefix and plain hex)
107 change.new_tile = static_cast<uint16_t>(std::stoi(value, nullptr, 16));
108 }
109 }
110
111 // Load the ROM to get the old tile value
112 if (rom && rom->is_loaded()) {
113 zelda3::Overworld overworld(rom);
114 auto status = overworld.Load(rom);
115 if (!status.ok()) {
116 return status;
117 }
118
119 // Set the correct world based on map_id
120 if (change.map_id < 0x40) {
121 overworld.set_current_world(0); // Light World
122 } else if (change.map_id < 0x80) {
123 overworld.set_current_world(1); // Dark World
124 } else {
125 overworld.set_current_world(2); // Special World
126 }
127
128 change.old_tile = overworld.GetTile(change.x, change.y);
129 } else {
130 change.old_tile = 0x0000; // Unknown
131 }
132
133 return change;
134}

136absl::StatusOr<Tile16Proposal> Tile16ProposalGenerator::GenerateFromCommands(
137 const std::string& prompt,
138 const std::vector<std::string>& commands,
139 const std::string& ai_service,
140 Rom* rom) {
141
142 Tile16Proposal proposal;
143 proposal.id = GenerateProposalId();
144 proposal.prompt = prompt;
145 proposal.ai_service = ai_service;
146 proposal.created_at = std::chrono::system_clock::now();
147 proposal.status = Tile16Proposal::Status::PENDING;
148
149 // Parse each command
150 for (const auto& command : commands) {
151 // Skip empty commands or comments
152 if (command.empty() || command[0] == '#') {
153 continue;
154 }
155
156 // Check if it's a set-tile command
157 if (absl::StrContains(command, "overworld set-tile")) {
158 auto change_or = ParseSetTileCommand(command, rom);
159 if (change_or.ok()) {
160 proposal.changes.push_back(change_or.value());
161 } else {
162 return change_or.status();
163 }
164 }
165 // TODO: Add support for other command types (set-area, replace-tile, etc.)
166 }
167
168 if (proposal.changes.empty()) {
169 return absl::InvalidArgumentError(
170 "No valid tile16 changes found in commands");
171 }
172
173 proposal.reasoning = absl::StrCat(
174 "Generated ", proposal.changes.size(), " tile16 changes from prompt");
175
176 return proposal;
177}

179absl::Status Tile16ProposalGenerator::ApplyProposal(
180 const Tile16Proposal& proposal,
181 Rom* rom) {
182
183 if (!rom || !rom->is_loaded()) {
184 return absl::FailedPreconditionError("ROM not loaded");
185 }
186
187 zelda3::Overworld overworld(rom);
188 auto status = overworld.Load(rom);
189 if (!status.ok()) {
190 return status;
191 }
192
193 // Apply each change
194 for (const auto& change : proposal.changes) {
195 // Set the correct world
196 if (change.map_id < 0x40) {
197 overworld.set_current_world(0); // Light World
198 } else if (change.map_id < 0x80) {
199 overworld.set_current_world(1); // Dark World
200 } else {
201 overworld.set_current_world(2); // Special World
202 }
203
204 // Apply the tile change
205 overworld.SetTile(change.x, change.y, change.new_tile);
206 }
207
208 // Note: We don't save to disk here - that's the caller's responsibility
209 // This allows for sandbox testing before committing
210
211 return absl::OkStatus();
212}
