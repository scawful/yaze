309 auto parse_int = [](absl::string_view value,
310 const char* flag) -> absl::StatusOr<int> {
311 int result = 0;
312 if (!absl::SimpleAtoi(value, &result)) {
313 return absl::InvalidArgumentError(
314 absl::StrCat(flag, " requires an integer value"));
315 }
316 if (result <= 0 || result > 65535) {
317 return absl::InvalidArgumentError(
318 absl::StrCat(flag, " must be between 1 and 65535"));
319 }
320 return result;
321 };
322
323 for (size_t i = 0; i < args.size(); ++i) {
324 const std::string& token = args[i];
325
326 if (token == "--ci-mode" || token == "--ci") {
327 options.ci_mode = true;
328 continue;
329 }
330
331 if (token == "--host" && i + 1 < args.size()) {
332 options.host = args[++i];
333 continue;
334 }
335 if (absl::StartsWith(token, "--host=")) {
336 options.host = token.substr(7);
337 continue;
338 }
339
340 if (token == "--port" && i + 1 < args.size()) {
341 ASSIGN_OR_RETURN(options.port, parse_int(args[++i], "--port"));
342 continue;
343 }
344 if (absl::StartsWith(token, "--port=")) {
345 ASSIGN_OR_RETURN(options.port,
346 parse_int(token.substr(7), "--port"));
347 continue;
348 }
349
350 if ((token == "--format" || token == "--output") &&
351 i + 1 < args.size()) {
352 options.output_format = absl::AsciiStrToLower(args[++i]);
353 continue;
354 }
355 if (absl::StartsWith(token, "--format=") ||
356 absl::StartsWith(token, "--output=")) {
357 options.output_format =
358 absl::AsciiStrToLower(token.substr(token.find('=') + 1));
359 continue;
360 }
361
362 if (token == "--param" && i + 1 < args.size()) {
363 std::string pair = args[++i];
364 auto eq = pair.find('=');
365 if (eq == std::string::npos) {
366 return absl::InvalidArgumentError(
367 "--param expects KEY=VALUE format");
368 }
369 options.parameters[pair.substr(0, eq)] = pair.substr(eq + 1);
370 continue;
371 }
372 if (absl::StartsWith(token, "--param=")) {
373 std::string pair = token.substr(8);
374 auto eq = pair.find('=');
375 if (eq == std::string::npos) {
376 return absl::InvalidArgumentError(
377 "--param expects KEY=VALUE format");
378 }
379 options.parameters[pair.substr(0, eq)] = pair.substr(eq + 1);
380 continue;
381 }
382
383 if (token == "--script" && i + 1 < args.size()) {
384 options.script_path = args[++i];
385 continue;
386 }
387 if (absl::StartsWith(token, "--script=")) {
388 options.script_path = token.substr(9);
389 continue;
390 }
391
392 if (absl::StartsWith(token, "--")) {
393 return absl::InvalidArgumentError(
394 absl::StrCat("Unknown flag for agent test replay: ", token));
395 }
396
397 if (options.script_path.empty()) {
398 options.script_path = token;
399 continue;
400 }
401
402 return absl::InvalidArgumentError(
403 absl::StrCat("Unexpected argument: ", token));
404 }
405
406 if (options.script_path.empty()) {
407 return absl::InvalidArgumentError(
408 "Usage: agent test replay <script.json> [--ci-mode] [--host <host>] "
409 "[--port <port>] [--format text|json] [--param KEY=VALUE]");
410 }
411
412 if (options.output_format != "text" && options.output_format != "json") {
413 return absl::InvalidArgumentError(
414 "--format must be either 'text' or 'json'");
415 }
416
417 return options;
418}

748 absl::StatusOr<SuiteRunOptions> ParseSuiteRunArgs(
749 const std::vector<std::string>& args) {
750 SuiteRunOptions options;
751
752 auto parse_int = [](absl::string_view value,
753 const char* flag) -> absl::StatusOr<int> {
754 int result = 0;
755 if (!absl::SimpleAtoi(value, &result)) {
756 return absl::InvalidArgumentError(
757 absl::StrCat(flag, " requires an integer value"));
758 }
759 if (result <= 0 || result > 65535) {
760 return absl::InvalidArgumentError(
761 absl::StrCat(flag, " must be between 1 and 65535"));
762 }
763 return result;
764 };
765
766 for (size_t i = 0; i < args.size(); ++i) {
767 const std::string& token = args[i];
768
769 if (token == "--ci-mode" || token == "--ci") {
770 options.ci_mode = true;
771 options.stop_on_failure = true;
772 continue;
773 }
774 if (token == "--stop-on-failure") {
775 options.stop_on_failure = true;
776 continue;
777 }
778
779 if ((token == "--host" || token == "-H") && i + 1 < args.size()) {
780 options.host = args[++i];
781 continue;
782 }
783 if (absl::StartsWith(token, "--host=")) {
784 options.host = token.substr(7);
785 continue;
786 }
787
788 if ((token == "--port" || token == "-p") && i + 1 < args.size()) {
789 ASSIGN_OR_RETURN(options.port, parse_int(args[++i], "--port"));
790 continue;
791 }
792 if (absl::StartsWith(token, "--port=")) {
793 ASSIGN_OR_RETURN(options.port,
794 parse_int(token.substr(7), "--port"));
795 continue;
796 }
797
798 if ((token == "--format" || token == "--output") &&
799 i + 1 < args.size()) {
800 options.output_format = absl::AsciiStrToLower(args[++i]);
801 continue;
802 }
803 if (absl::StartsWith(token, "--format=") ||
804 absl::StartsWith(token, "--output=")) {
805 options.output_format = absl::AsciiStrToLower(
806 token.substr(token.find('=') + 1));
807 continue;
808 }
809
810 if ((token == "--group" || token == "-g") && i + 1 < args.size()) {
811 AppendCsvList(args[++i], &options.group_filters);
812 continue;
813 }
814 if (absl::StartsWith(token, "--group=")) {
815 AppendCsvList(token.substr(8), &options.group_filters);
816 continue;
817 }
818
819 if ((token == "--tag" || token == "-t") && i + 1 < args.size()) {
820 AppendCsvList(args[++i], &options.tag_filters);
821 continue;
822 }
823 if (absl::StartsWith(token, "--tag=")) {
824 AppendCsvList(token.substr(6), &options.tag_filters);
825 continue;
826 }
827
828 if (token == "--param" && i + 1 < args.size()) {
829 std::string pair = args[++i];
830 auto eq = pair.find('=');
831 if (eq == std::string::npos) {
832 return absl::InvalidArgumentError(
833 "--param expects KEY=VALUE format");
834 }
835 options.parameter_overrides[pair.substr(0, eq)] = pair.substr(eq + 1);
836 continue;
837 }
838 if (absl::StartsWith(token, "--param=")) {
839 std::string pair = token.substr(8);
840 auto eq = pair.find('=');
841 if (eq == std::string::npos) {
842 return absl::InvalidArgumentError(
843 "--param expects KEY=VALUE format");
844 }
845 options.parameter_overrides[pair.substr(0, eq)] = pair.substr(eq + 1);
846 continue;
847 }
848
849 if ((token == "--retries" || token == "--retry") &&
850 i + 1 < args.size()) {
851 int value = 0;
852 if (!absl::SimpleAtoi(args[++i], &value) || value < 0) {
853 return absl::InvalidArgumentError(
854 "--retries expects a non-negative integer");
855 }
856 options.retry_override = value;
857 continue;
858 }
859 if (absl::StartsWith(token, "--retries=") ||
860 absl::StartsWith(token, "--retry=")) {
861 std::string value = token.substr(token.find('=') + 1);
862 int retries = 0;
863 if (!absl::SimpleAtoi(value, &retries) || retries < 0) {
864 return absl::InvalidArgumentError(
865 "--retries expects a non-negative integer");
866 }
867 options.retry_override = retries;
868 continue;
869 }
870
871 if ((token == "--junit" || token == "--junit-output") &&
872 i + 1 < args.size()) {
873 options.junit_output_path = args[++i];
874 continue;
875 }
876 if (absl::StartsWith(token, "--junit=") ||
877 absl::StartsWith(token, "--junit-output=")) {
878 options.junit_output_path = token.substr(token.find('=') + 1);
879 continue;
880 }
881
882 if (token == "--suite" && i + 1 < args.size()) {
883 options.suite_path = args[++i];
884 continue;
885 }
886 if (absl::StartsWith(token, "--suite=")) {
887 options.suite_path = token.substr(8);
888 continue;
889 }
890
891 if (!absl::StartsWith(token, "--") && options.suite_path.empty()) {
892 options.suite_path = token;
893 continue;
894 }
895
896 if (!absl::StartsWith(token, "--")) {
897 return absl::InvalidArgumentError(
898 absl::StrCat("Unexpected argument: ", token));
899 }
900
901 return absl::InvalidArgumentError(
902 absl::StrCat("Unknown flag for agent test suite run: ", token));
903 }
904
905 if (options.suite_path.empty()) {
906 return absl::InvalidArgumentError(
907 "Usage: agent test suite run <suite.yaml> [--group <name>] [--tag "
908 "<tag>] [--ci-mode] [--format text|json] [--junit <path>]"
909 " [--param KEY=VALUE] [--retries N]");
910 }
911
912 options.output_format = absl::AsciiStrToLower(options.output_format);
913 if (options.output_format != "text" && options.output_format != "json") {
914 return absl::InvalidArgumentError(
915 "--format must be either 'text' or 'json'");
916 }
917
918 return options;
919 }

975 void AddResult(TestSuiteRunSummary* summary, TestCaseRunResult result) {
976 switch (result.outcome) {
977 case TestCaseOutcome::kPassed:
978 summary->passed++;
979 break;
980 case TestCaseOutcome::kFailed:
981 summary->failed++;
982 break;
983 case TestCaseOutcome::kError:
984 summary->errors++;
985 break;
986 case TestCaseOutcome::kSkipped:
987 summary->skipped++;
988 break;
989 }
990 summary->results.push_back(std::move(result));

993 TestCaseRunResult ExecuteTestCase(
994 GuiAutomationClient* client, const TestSuiteDefinition& suite,
995 const TestGroupDefinition& group, const TestCaseDefinition& test,
996 const SuiteRunOptions& options, int max_attempts) {
997 TestCaseRunResult result;
998 result.test = &test;
999 result.group = &group;
1000 result.outcome = TestCaseOutcome::kError;
1001 result.start_time = absl::Now();
1002
1003 std::map<std::string, std::string> parameters = MergeParameters(test, options);
1004
1005 for (int attempt = 1; attempt <= max_attempts; ++attempt) {
1006 ++result.attempts;
1007 result.retries = attempt - 1;
1008
1009 absl::StatusOr<ReplayTestResult> replay =
1010 client->ReplayTest(test.script_path, options.ci_mode, parameters);
1011
1012 if (!replay.ok()) {
1013 result.outcome = TestCaseOutcome::kError;
1014 result.message = replay.status().message();
1015 break;
1016 }
1017
1018 result.replay_session_id = replay->replay_session_id;
1019 result.assertions = replay->assertions;
1020 result.logs = replay->logs;
1021 result.message = replay->message;
1022
1023 if (replay->success) {
1024 result.outcome = TestCaseOutcome::kPassed;
1025 break;
1026 }
1027
1028 result.outcome = TestCaseOutcome::kFailed;
1029 if (attempt < max_attempts) {
1030 continue;
1031 }
1032 break;
1033 }
1034
1035 result.duration = absl::Now() - result.start_time;
1036 if (result.outcome == TestCaseOutcome::kPassed && result.message.empty()) {
1037 result.message = "Test passed";
1038 }
1039 return result;
1040 }

1050 absl::StatusOr<TestSuiteRunSummary> ExecuteTestSuite(
1051 GuiAutomationClient* client, const TestSuiteDefinition& suite,
1052 const SuiteRunOptions& options) {
1053 TestSuiteRunSummary summary;
1054 summary.suite = &suite;
1055 summary.started_at = absl::Now();
1056
1057 int max_attempts = DetermineMaxAttempts(suite, options);
1058 std::unordered_map<std::string, bool> group_success;
1059 bool interrupted = false;
1060
1061 for (const auto& group : suite.groups) {
1062 bool group_selected = ShouldRunGroup(group, options);
1063 if (!group_selected) {
1064 group_success[group.name] = false;
1065 continue;
1066 }
1067
1068 bool dependencies_met = true;
1069 std::vector<std::string> unmet_dependencies;
1070 for (const std::string& dependency : group.depends_on) {
1071 auto it = group_success.find(dependency);
1072 if (it == group_success.end() || !it->second) {
1073 dependencies_met = false;
1074 unmet_dependencies.push_back(dependency);
1075 }
1076 }
1077
1078 if (!dependencies_met) {
1079 for (const auto& test : group.tests) {
1080 if (!ShouldRunTest(test, options)) {
1081 continue;
1082 }
1083 TestCaseRunResult skipped;
1084 skipped.test = &test;
1085 skipped.group = &group;
1086 skipped.outcome = TestCaseOutcome::kSkipped;
1087 skipped.message =
1088 absl::StrCat("Skipped because dependencies not satisfied: ",
1089 JoinStrings(unmet_dependencies, ", "));
1090 AddResult(&summary, std::move(skipped));
1091 }
1092 group_success[group.name] = false;
1093 continue;
1094 }
1095
1096 bool group_passed = true;
1097
1098 for (const auto& test : group.tests) {
1099 if (!ShouldRunTest(test, options)) {
1100 TestCaseRunResult skipped;
1101 skipped.test = &test;
1102 skipped.group = &group;
1103 skipped.outcome = TestCaseOutcome::kSkipped;
1104 skipped.message = "Skipped by CLI filter";
1105 AddResult(&summary, std::move(skipped));
1106 continue;
1107 }
1108
1109 if (interrupted) {
1110 TestCaseRunResult skipped;
1111 skipped.test = &test;
1112 skipped.group = &group;
1113 skipped.outcome = TestCaseOutcome::kSkipped;
1114 skipped.message =
1115 "Skipped because stop-on-failure condition was triggered";
1116 AddResult(&summary, std::move(skipped));
1117 continue;
1118 }
1119
1120 TestCaseRunResult result =
1121 ExecuteTestCase(client, suite, group, test, options, max_attempts);
1122 AddResult(&summary, std::move(result));
1123
1124 const auto& stored = summary.results.back();
1125 if (stored.outcome == TestCaseOutcome::kFailed ||
1126 stored.outcome == TestCaseOutcome::kError) {
1127 group_passed = false;
1128 if (options.stop_on_failure) {
1129 interrupted = true;
1130 }
1131 }
1132 }
1133
1134 group_success[group.name] = group_passed;
1135 }
1136
1137 summary.total_duration = absl::Now() - summary.started_at;
1138 if (summary.results.empty()) {
1139 return absl::InvalidArgumentError(
1140 "No tests were executed. Adjust filters or suite definition.");
1141 }
1142
1143 return summary;
1144 }
