231void MockRom::SetMockRoomData(int room_id, const std::vector<uint8_t>& data) {
232 mock_room_data_[room_id] = data;
233
234 if (room_header_table_pc_ == 0 || room_header_data_base_pc_ == 0) {
235 return;
236 }
237
238 uint32_t header_offset = room_header_data_base_pc_ + kRoomHeaderStride * static_cast<uint32_t>(room_id);
239 EnsureBufferCapacity(header_offset + static_cast<uint32_t>(data.size()));
240
241 std::memcpy(backing_buffer_.data() + header_offset, data.data(), data.size());
242 std::memcpy(mutable_data() + header_offset, data.data(), data.size());
243
244 uint32_t snes_offset = PcToSnes(header_offset);
245 uint32_t pointer_entry = room_header_table_pc_ + static_cast<uint32_t>(room_id) * 2;
246 EnsureBufferCapacity(pointer_entry + 2);
247 backing_buffer_[pointer_entry] = static_cast<uint8_t>(snes_offset & 0xFF);
248 backing_buffer_[pointer_entry + 1] = static_cast<uint8_t>((snes_offset >> 8) & 0xFF);
249 mutable_data()[pointer_entry] = backing_buffer_[pointer_entry];
250 mutable_data()[pointer_entry + 1] = backing_buffer_[pointer_entry + 1];

295void MockRom::InitializeMemoryLayout() {

304
305 EnsureBufferCapacity(room_header_table_pc_ + 2);
306 EnsureBufferCapacity(room_object_table_pc_ + 3);

308 uint32_t header_table_snes = PcToSnes(room_header_table_pc_);
309 EnsureBufferCapacity(kRoomHeaderPointer + 3);
