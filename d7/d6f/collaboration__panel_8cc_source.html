151

153 // Filter by search

155 std::string search_lower = search_filter_;

157 std::transform(search_lower.begin(), search_lower.end(), search_lower.begin(), ::tolower);
158 std::transform(sender_lower.begin(), sender_lower.end(), sender_lower.begin(), ::tolower);
159

161 snapshots_[i].snapshot_type.find(search_lower) == std::string::npos) {
162 continue;
163 }

165
166 RenderSnapshotEntry(snapshots_[i], i);
167
168 // Grid layout
169 if ((i + 1) % columns != 0 && i < snapshots_.size() - 1) {

412std::string CollaborationPanel::FormatFileSize(size_t bytes) {
413 const char* units[] = {"B", "KB", "MB", "GB"};
414 int unit_index = 0;
415 double size = static_cast<double>(bytes);
416
417 while (size >= 1024.0 && unit_index < 3) {
418 size /= 1024.0;
419 unit_index++;
420 }
421
422 char buffer[32];
423 snprintf(buffer, sizeof(buffer), "%.1f %s", size, units[unit_index]);
424 return std::string(buffer);
425}
