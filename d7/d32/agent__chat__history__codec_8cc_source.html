44std::optional<cli::agent::ChatMessage::TableData> ParseTableData(
45 const Json& json) {
46 if (!json.is_object()) {
47 return std::nullopt;
48 }
49
50 cli::agent::ChatMessage::TableData table;
51 if (json.contains("headers") && json["headers"].is_array()) {
52 for (const auto& header : json["headers"]) {
53 if (header.is_string()) {
54 table.headers.push_back(header.get<std::string>());
55 }
56 }
57 }
58
59 if (json.contains("rows") && json["rows"].is_array()) {
60 for (const auto& row : json["rows"]) {
61 if (!row.is_array()) {
62 continue;
63 }
64 std::vector<std::string> row_values;
65 for (const auto& value : row) {
66 if (value.is_string()) {
67 row_values.push_back(value.get<std::string>());
68 } else {
69 row_values.push_back(value.dump());
70 }
71 }
72 table.rows.push_back(std::move(row_values));
73 }
74 }
75
76 if (table.headers.empty() && table.rows.empty()) {
77 return std::nullopt;
78 }
79
80 return table;
81}

128absl::StatusOr<AgentChatHistoryCodec::Snapshot> AgentChatHistoryCodec::Load(

130#if defined(YAZE_WITH_JSON)
131 Snapshot snapshot;
132
133 std::ifstream file(path);
134 if (!file.good()) {
135 return snapshot; // Treat missing file as empty history.
136 }
137
138 Json json;
139 try {
140 file >> json;
141 } catch (const std::exception& e) {
142 return absl::InternalError(
143 absl::StrFormat("Failed to parse chat history: %s", e.what()));
144 }
145
146 if (!json.contains("messages") || !json["messages"].is_array()) {
147 return snapshot;
148 }
149
150 for (const auto& item : json["messages"]) {
151 if (!item.is_object()) {
152 continue;
153 }
154
155 cli::agent::ChatMessage message;
156 std::string sender = item.value("sender", "agent");
157 message.sender = sender == "user"
158 ? cli::agent::ChatMessage::Sender::kUser
159 : cli::agent::ChatMessage::Sender::kAgent;
160 message.message = item.value("message", "");
161 message.timestamp = ParseTimestamp(item["timestamp"]);
162
163 if (item.contains("json_pretty") && item["json_pretty"].is_string()) {
164 message.json_pretty = item["json_pretty"].get<std::string>();
165 }
166 if (item.contains("table_data")) {
167 message.table_data = ParseTableData(item["table_data"]);
168 }
169 if (item.contains("metrics") && item["metrics"].is_object()) {
170 cli::agent::ChatMessage::SessionMetrics metrics;
171 const auto& metrics_json = item["metrics"];
172 metrics.turn_index = metrics_json.value("turn_index", 0);
173 metrics.total_user_messages =
174 metrics_json.value("total_user_messages", 0);
175 metrics.total_agent_messages =
176 metrics_json.value("total_agent_messages", 0);
177 metrics.total_tool_calls = metrics_json.value("total_tool_calls", 0);
178 metrics.total_commands = metrics_json.value("total_commands", 0);
179 metrics.total_proposals = metrics_json.value("total_proposals", 0);
180 metrics.total_elapsed_seconds =
181 metrics_json.value("total_elapsed_seconds", 0.0);
182 metrics.average_latency_seconds =
183 metrics_json.value("average_latency_seconds", 0.0);
184 message.metrics = metrics;
185 }
186 if (item.contains("proposal")) {
187 message.proposal = ParseProposal(item["proposal"]);
188 }
189
190 snapshot.history.push_back(std::move(message));
191 }
192
193 if (json.contains("collaboration") && json["collaboration"].is_object()) {
194 const auto& collab_json = json["collaboration"];
195 snapshot.collaboration.active = collab_json.value("active", false);
196 snapshot.collaboration.session_id = collab_json.value("session_id", "");
197 snapshot.collaboration.session_name =
198 collab_json.value("session_name", "");
199 snapshot.collaboration.participants.clear();
200 if (collab_json.contains("participants") &&
201 collab_json["participants"].is_array()) {
202 for (const auto& participant : collab_json["participants"]) {
203 if (participant.is_string()) {
204 snapshot.collaboration.participants.push_back(
205 participant.get<std::string>());
206 }
207 }
208 }
209 if (collab_json.contains("last_synced")) {
210 snapshot.collaboration.last_synced =
211 ParseTimestamp(collab_json["last_synced"]);
212 }
213 if (snapshot.collaboration.session_name.empty() &&
214 !snapshot.collaboration.session_id.empty()) {
215 snapshot.collaboration.session_name =
216 snapshot.collaboration.session_id;
217 }
218 }
219
220 if (json.contains("multimodal") && json["multimodal"].is_object()) {
221 const auto& multimodal_json = json["multimodal"];
222 if (multimodal_json.contains("last_capture_path") &&
223 multimodal_json["last_capture_path"].is_string()) {
224 std::string path_value =
225 multimodal_json["last_capture_path"].get<std::string>();
226 if (!path_value.empty()) {
227 snapshot.multimodal.last_capture_path =
228 std::filesystem::path(path_value);
229 }
230 }
231 snapshot.multimodal.status_message =
232 multimodal_json.value("status_message", "");
233 if (multimodal_json.contains("last_updated")) {
234 snapshot.multimodal.last_updated =
235 ParseTimestamp(multimodal_json["last_updated"]);
236 }
237 }
238
239 return snapshot;
240#else
241 (void)path;

249#if defined(YAZE_WITH_JSON)

273 metrics_json["turn_index"] = metrics.turn_index;
274 metrics_json["total_user_messages"] = metrics.total_user_messages;
275 metrics_json["total_agent_messages"] = metrics.total_agent_messages;
276 metrics_json["total_tool_calls"] = metrics.total_tool_calls;

279 metrics_json["total_elapsed_seconds"] = metrics.total_elapsed_seconds;
280 metrics_json["average_latency_seconds"] =
281 metrics.average_latency_seconds;
282 entry["metrics"] = metrics_json;
283 }
284 if (message.proposal.has_value()) {
285 entry["proposal"] = SerializeProposal(*message.proposal);
286 }
287
288 json["messages"].push_back(std::move(entry));
289 }
290
291 Json collab_json;
292 collab_json["active"] = snapshot.collaboration.active;

294 collab_json["session_name"] = snapshot.collaboration.session_name;
295 collab_json["participants"] = snapshot.collaboration.participants;
296 if (snapshot.collaboration.last_synced != absl::InfinitePast()) {
297 collab_json["last_synced"] = absl::FormatTime(
298 absl::RFC3339_full, snapshot.collaboration.last_synced,
299 absl::UTCTimeZone());

301 json["collaboration"] = std::move(collab_json);
302

305 multimodal_json["last_capture_path"] =

307 } else {
308 multimodal_json["last_capture_path"] = "";
309 }
310 multimodal_json["status_message"] = snapshot.multimodal.status_message;
311 if (snapshot.multimodal.last_updated != absl::InfinitePast()) {
312 multimodal_json["last_updated"] = absl::FormatTime(
313 absl::RFC3339_full, snapshot.multimodal.last_updated,
314 absl::UTCTimeZone());
315 }
316 json["multimodal"] = std::move(multimodal_json);
317
318 std::error_code ec;
319 auto directory = path.parent_path();
320 if (!directory.empty()) {
321 std::filesystem::create_directories(directory, ec);
322 if (ec) {
323 return absl::InternalError(absl::StrFormat(
324 "Unable to create chat history directory: %s", ec.message()));
325 }
326 }
327
328 std::ofstream file(path);
329 if (!file.is_open()) {
330 return absl::InternalError("Cannot write chat history file");
331 }
332
333 file << json.dump(2);
334 return absl::OkStatus();
335#else
336 (void)path;
337 (void)snapshot;
338 return absl::UnimplementedError(
339 "Chat history persistence requires YAZE_WITH_GRPC=ON");
340#endif
341}
