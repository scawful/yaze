30void DisassemblyViewer::RecordInstruction(uint32_t address, uint8_t opcode,
31 const std::vector<uint8_t>& operands,
32 const std::string& mnemonic,
33 const std::string& operand_str) {
34 auto it = instructions_.find(address);
35 if (it != instructions_.end()) {
36 // Instruction already recorded, just increment execution count
37 it->second.execution_count++;
38 } else {
39 // New instruction, add to map
40 DisassemblyEntry entry;
41 entry.address = address;
42 entry.opcode = opcode;
43 entry.operands = operands;
44 entry.mnemonic = mnemonic;
45 entry.operand_str = operand_str;
46 entry.size = 1 + operands.size();
47 entry.execution_count = 1;
48 entry.is_breakpoint = false;
49 entry.is_current_pc = false;
50
51 instructions_[address] = entry;
52 }
53}

138 ImGuiTableFlags_Sortable |

141
142 // Calculate column count based on optional columns

158 ImGui::TableSetupColumn("Operand", ImGuiTableColumnFlags_WidthStretch);
159 if (show_execution_counts_) {

167 ImGuiListClipper clipper;
168 auto sorted_addrs = GetSortedAddresses();
169 clipper.Begin(sorted_addrs.size());

172 for (int row = clipper.DisplayStart; row < clipper.DisplayEnd; row++) {

176 // Skip if doesn't pass filter

227 ImGui::SetTooltip("Click for instruction documentation");
228 }
229
230 // Column 4: Operand (clickable for jump-to-address)
231 ImGui::TableNextColumn();
232 ImGui::TextColored(kColorOperand, "%s", entry.operand_str.c_str());

252 if (auto_scroll_ && scroll_to_address_ != current_pc) {
253 // Find row index of current PC
254 auto it = std::find(sorted_addrs.begin(), sorted_addrs.end(), current_pc);
255 if (it != sorted_addrs.end()) {
256 int row_index = std::distance(sorted_addrs.begin(), it);
257 ImGui::SetScrollY((row_index * ImGui::GetTextLineHeightWithSpacing()) -

261 }
262
263 ImGui::EndTable();

307ImVec4 DisassemblyViewer::GetMnemonicColor(const DisassemblyEntry& entry) const {
308 // Color-code by instruction type
309 const std::string& mnemonic = entry.mnemonic;
310
311 // Branches and jumps
312 if (mnemonic.find('B') == 0 || mnemonic == "JMP" || mnemonic == "JSR" ||
313 mnemonic == "RTL" || mnemonic == "RTS" || mnemonic == "RTI") {
314 return ImVec4(0.8f, 0.4f, 1.0f, 1.0f); // Purple for control flow
315 }
316
317 // Loads

320 }
321
322 // Stores
323 if (mnemonic.find("ST") == 0) {
324 return ImVec4(1.0f, 0.6f, 0.4f, 1.0f); // Orange for stores
325 }
326
327 return kColorMnemonic;

330std::string DisassemblyViewer::FormatHexDump(const DisassemblyEntry& entry) const {
331 std::stringstream ss;
332 ss << std::hex << std::uppercase << std::setfill('0');
333
334 // Opcode
335 ss << std::setw(2) << static_cast<int>(entry.opcode);
336
337 // Operands
338 for (const auto& operand_byte : entry.operands) {
339 ss << " " << std::setw(2) << static_cast<int>(operand_byte);
340 }
341
342 // Pad to consistent width (3 bytes max)
343 for (size_t i = entry.operands.size(); i < 2; i++) {
344 ss << " ";
345 }
346
347 return ss.str();
348}

350bool DisassemblyViewer::PassesFilter(const DisassemblyEntry& entry) const {
351 if (search_filter_[0] == '\0') {
352 return true; // No filter active
353 }
354
355 std::string filter_lower(search_filter_);
356 std::transform(filter_lower.begin(), filter_lower.end(),
357 filter_lower.begin(), ::tolower);
358
359 // Check address
360 std::string addr_str = absl::StrFormat("%06x", entry.address);
361 if (addr_str.find(filter_lower) != std::string::npos) {
362 return true;
363 }
364
365 // Check mnemonic
366 std::string mnemonic_lower = entry.mnemonic;
367 std::transform(mnemonic_lower.begin(), mnemonic_lower.end(),
368 mnemonic_lower.begin(), ::tolower);
369 if (mnemonic_lower.find(filter_lower) != std::string::npos) {
370 return true;
371 }
372
373 // Check operand
374 std::string operand_lower = entry.operand_str;
375 std::transform(operand_lower.begin(), operand_lower.end(),
376 operand_lower.begin(), ::tolower);
377 if (operand_lower.find(filter_lower) != std::string::npos) {
378 return true;
379 }
380
381 return false;
382}

390bool DisassemblyViewer::ExportToFile(const std::string& filepath) const {
391 std::ofstream out(filepath);
392 if (!out.is_open()) {
393 return false;
394 }
395
396 out << "; YAZE Disassembly Export\n";
397 out << "; Total instructions: " << instructions_.size() << "\n";
398 out << "; Generated: " << __DATE__ << " " << __TIME__ << "\n\n";
399
400 auto sorted_addrs = GetSortedAddresses();
401 for (uint32_t addr : sorted_addrs) {
402 const auto& entry = instructions_.at(addr);
403
404 out << absl::StrFormat("$%02X:%04X: %-8s %-6s %-20s ; exec=%llu\n",
405 (addr >> 16) & 0xFF,
406 addr & 0xFFFF,
407 FormatHexDump(entry).c_str(),
408 entry.mnemonic.c_str(),
409 entry.operand_str.c_str(),
410 entry.execution_count);
411 }
412
413 out.close();
414 return true;
415}

423std::vector<uint32_t> DisassemblyViewer::GetSortedAddresses() const {
424 std::vector<uint32_t> addrs;
425 addrs.reserve(instructions_.size());
426
427 for (const auto& [addr, _] : instructions_) {
428 addrs.push_back(addr);

430
431 std::sort(addrs.begin(), addrs.end());
