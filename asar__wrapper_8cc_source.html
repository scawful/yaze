69absl::StatusOr<AsarPatchResult> AsarWrapper::ApplyPatch(
70 const std::string& patch_path,
71 std::vector<uint8_t>& rom_data,
72 const std::vector<std::string>& include_paths) {
73
74 if (!initialized_) {
75 return absl::FailedPreconditionError("Asar not initialized");
76 }
77
78 // Reset previous state
79 Reset();
80
81 AsarPatchResult result;
82 result.success = false;
83
84 // Prepare ROM data
85 int rom_size = static_cast<int>(rom_data.size());
86 int buffer_size = std::max(rom_size, 16 * 1024 * 1024); // At least 16MB buffer
87
88 // Resize ROM data if needed
89 if (rom_data.size() < buffer_size) {
90 rom_data.resize(buffer_size, 0);
91 }
92
93 // Apply the patch
94 bool patch_success = asar_patch(
95 patch_path.c_str(),
96 reinterpret_cast<char*>(rom_data.data()),
97 buffer_size,
98 &rom_size);
99
100 // Process results
101 ProcessErrors();
102 ProcessWarnings();
103
104 result.errors = last_errors_;
105 result.warnings = last_warnings_;
106 result.success = patch_success && last_errors_.empty();
107
108 if (result.success) {
109 // Resize ROM data to actual size
110 rom_data.resize(rom_size);
111 result.rom_size = rom_size;
112
113 // Extract symbols
114 ExtractSymbolsFromLastOperation();
115 result.symbols.reserve(symbol_table_.size());
116 for (const auto& [name, symbol] : symbol_table_) {
117 result.symbols.push_back(symbol);
118 }
119
120 // Calculate CRC32 if available
121 // Note: Asar might provide this, check if function exists
122 result.crc32 = 0; // TODO: Implement CRC32 calculation
123 } else {
124 return absl::InternalError(absl::StrFormat(
125 "Patch failed: %s", absl::StrJoin(last_errors_, "; ")));
126 }
127
128 return result;
129}

131absl::StatusOr<AsarPatchResult> AsarWrapper::ApplyPatchFromString(
132 const std::string& patch_content,
133 std::vector<uint8_t>& rom_data,
134 const std::string& base_path) {
135
136 // Create temporary file for patch content
137 std::string temp_path = "/tmp/yaze_temp_patch.asm";
138 if (!base_path.empty()) {
139 // Ensure directory exists
140 std::filesystem::create_directories(base_path);
141 temp_path = base_path + "/temp_patch.asm";
142 }
143
144 std::ofstream temp_file(temp_path);
145 if (!temp_file) {
146 return absl::InternalError(absl::StrFormat(
147 "Failed to create temporary patch file at: %s", temp_path));
148 }
149
150 temp_file << patch_content;
151 temp_file.close();
152
153 auto result = ApplyPatch(temp_path, rom_data);
154
155 // Clean up temporary file
156 std::remove(temp_path.c_str());
157
158 return result;
159}
