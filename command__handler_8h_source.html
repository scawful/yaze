63class ApplyPatch : public CommandHandler {
64 public:
65 absl::Status handle(const std::vector<std::string>& arg_vec) override {
66 std::string rom_filename = arg_vec[1];
67 std::string patch_filename = arg_vec[2];
68 RETURN_IF_ERROR(rom_.LoadFromFile(rom_filename))
69 auto source = rom_.vector();
70 std::ifstream patch_file(patch_filename, std::ios::binary);
71 std::vector<uint8_t> patch;
72 patch.resize(rom_.size());
73 patch_file.read((char*)patch.data(), patch.size());
74
75 // Apply patch
76 std::vector<uint8_t> patched;
77 app::core::ApplyBpsPatch(source, patch, patched);
78
79 // Save patched file
80 std::ofstream patched_rom("patched.sfc", std::ios::binary);
81 patched_rom.write((char*)patched.data(), patched.size());
82 patched_rom.close();
83 return absl::OkStatus();
84 }
85};
