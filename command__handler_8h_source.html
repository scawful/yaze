74 std::vector<uint8_t> patch;

76 patch_file.read((char*)patch.data(), patch.size());

78 // Apply patch

85 patched_rom.close();

220class ReadFromRom : public CommandHandler {
221 public:
222 absl::Status handle(const std::vector<std::string>& arg_vec) override {
223 RETURN_IF_ERROR(rom_.LoadFromFile(arg_vec[0]))
224
225 std::stringstream ss(arg_vec[1].data());
226 uint32_t offset;
227 ss >> std::hex >> offset;
228 uint32_t length = 0x01;
229 if (!arg_vec[2].empty()) {
230 length = std::stoi(arg_vec[2]);
231 }
232
233 if (length > 1) {
234 auto returned_bytes_status = rom_.ReadByteVector(offset, length);
235 if (!returned_bytes_status.ok()) {
236 return returned_bytes_status.status();
237 }
238 auto returned_bytes = returned_bytes_status.value();
239 for (const auto& each : returned_bytes) {
240 std::cout << each;
241 }
242 std::cout << std::endl;
243 } else {
244 auto byte = rom_.ReadByte(offset);
245 std::cout << std::hex << byte.value() << std::endl;
246 }
247
248 return absl::OkStatus();
249 }

277// Start Emulator on a SNES rom file
278// -emu <rom_file> <optional:num_cpu_cycles>
279class Emulator : public CommandHandler {
280 public:
281 absl::Status handle(const std::vector<std::string>& arg_vec) override {
282 std::string filename = arg_vec[0];
283 RETURN_IF_ERROR(rom_.LoadFromFile(filename))
284
285 bool step = false;
286 if (arg_vec[1].empty()) {
287 snes.SetCpuMode(0);
288 } else {
289 snes.SetCpuMode(1);
290 step = true;
291 }
292
293 snes.SetupMemory(rom_);
294 snes.Init(rom_);
295
296 if (!step) {
297 int i = 0;
298 while (i < 80000) {
299 snes.Run();
300 i++;
301 }
302 } else {
303 // This loop should take in input from the keyboard, such as pressing
304 // space to step through the loop and pressing x to end the execution.
305 bool stepping = true;
306 std::cout << "Press space to step, x to exit" << std::endl;
307 while (stepping) {
308 char input;
309 std::cin.get(input);
310 if (input == 'x') {
311 break;
312 } else {
313 snes.StepRun();
314 }
315 }
316 }
317
318 return absl::OkStatus();
319 }
