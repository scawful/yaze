168 // Generate and push a new ID onto the stack
169 static ImGuiID GetNewID() {
170 static int counter = 1; // Start from 1 to ensure uniqueness
171 ImGuiID child_id = ImGui::GetID((void *)(intptr_t)counter++);
172 idStack.push(child_id);
173 return child_id;
174 }
175
