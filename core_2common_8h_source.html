217inline uint32_t SnesToPc(uint32_t addr) noexcept {
218 if (addr >= kFastRomRegion) {
219 addr -= kFastRomRegion;
220 }
221 uint32_t temp = (addr & 0x7FFF) + ((addr / 2) & 0xFF8000);
222 return (temp + 0x0);
223}

225inline uint32_t PcToSnes(uint32_t addr) {
226 uint8_t *b = reinterpret_cast<uint8_t *>(&addr);
227 b[2] = static_cast<uint8_t>(b[2] * 2);
228
229 if (b[1] >= 0x80) {
230 b[2] += 1;
231 } else {
232 b[1] += 0x80;
233 }
234
235 return addr;
236}
