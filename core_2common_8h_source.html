117inline uint32_t SnesToPc(uint32_t addr) noexcept {
118 if (addr >= kFastRomRegion) {
119 addr -= kFastRomRegion;
120 }
121 uint32_t temp = (addr & 0x7FFF) + ((addr / 2) & 0xFF8000);

125inline uint32_t PcToSnes(uint32_t addr) {
126 uint8_t *b = reinterpret_cast<uint8_t *>(&addr);
127 b[2] = static_cast<uint8_t>(b[2] * 2);
128
129 if (b[1] >= 0x80) {
130 b[2] += 1;
131 } else {
132 b[1] += 0x80;
133 }
134
135 return addr;
136}
