248inline uint32_t SnesToPc(uint32_t addr) noexcept {
249 if (addr >= kFastRomRegion) {
250 addr -= kFastRomRegion;
251 }
252 uint32_t temp = (addr & 0x7FFF) + ((addr / 2) & 0xFF8000);
253 return (temp + 0x0);
254}
255
256inline uint32_t PcToSnes(uint32_t addr) {
257 uint8_t *b = reinterpret_cast<uint8_t *>(&addr);
258 b[2] = static_cast<uint8_t>(b[2] * 2);
259
260 if (b[1] >= 0x80) {
261 b[2] += 1;
262 } else {
263 b[1] += 0x80;
264 }
265
266 return addr;
267}
