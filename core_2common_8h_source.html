195inline uint32_t SnesToPc(uint32_t addr) noexcept {
196 if (addr >= kFastRomRegion) {
197 addr -= kFastRomRegion;
198 }
199 uint32_t temp = (addr & 0x7FFF) + ((addr / 2) & 0xFF8000);
200 return (temp + 0x0);
201}

203inline uint32_t PcToSnes(uint32_t addr) {
204 uint8_t *b = reinterpret_cast<uint8_t *>(&addr);
205 b[2] = static_cast<uint8_t>(b[2] * 2);
206
207 if (b[1] >= 0x80) {
208 b[2] += 1;
209 } else {
210 b[1] += 0x80;
211 }
212
213 return addr;
214}
