242inline uint32_t SnesToPc(uint32_t addr) noexcept {
243 if (addr >= kFastRomRegion) {
244 addr -= kFastRomRegion;
245 }
246 uint32_t temp = (addr & 0x7FFF) + ((addr / 2) & 0xFF8000);
247 return (temp + 0x0);
248}

250inline uint32_t PcToSnes(uint32_t addr) {
251 uint8_t *b = reinterpret_cast<uint8_t *>(&addr);
252 b[2] = static_cast<uint8_t>(b[2] * 2);
253
254 if (b[1] >= 0x80) {
255 b[2] += 1;
256 } else {
257 b[1] += 0x80;
258 }
259
260 return addr;
261}
