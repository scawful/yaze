162 // Generate and push a new ID onto the stack
163 static ImGuiID GetNewID() {
164 static int counter = 1; // Start from 1 to ensure uniqueness
165 ImGuiID child_id = ImGui::GetID((void *)(intptr_t)counter++);
166 idStack.push(child_id);
167 return child_id;
168 }
169
170 // Pop all IDs from the stack (can be called explicitly or upon program exit)
