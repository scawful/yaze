140void AtlasRenderer::RenderBatch(const std::vector<RenderCommand>& render_commands) {
141 if (render_commands.empty()) {
142 return;
143 }
144
145 ScopedTimer timer("atlas_batch_render");
146
147 // Group commands by atlas for efficient rendering
148 std::unordered_map<int, std::vector<const RenderCommand*>> atlas_groups;
149
150 for (const auto& cmd : render_commands) {
151 auto it = atlas_lookup_.find(cmd.atlas_id);
152 if (it != atlas_lookup_.end() && it->second->in_use) {
153 // Find which atlas contains this entry
154 for (size_t i = 0; i < atlases_.size(); ++i) {
155 for (const auto& entry : atlases_[i]->entries) {
156 if (entry.atlas_id == cmd.atlas_id) {
157 atlas_groups[i].push_back(&cmd);
158 break;
159 }
160 }
161 }
162 }
163 }
164
165 // Render each atlas group
166 for (const auto& [atlas_index, commands] : atlas_groups) {
167 if (commands.empty()) continue;
168
169 auto& atlas = *atlases_[atlas_index];
170
171 // Set atlas texture
172 SDL_SetTextureBlendMode(atlas.texture, SDL_BLENDMODE_BLEND);
173
174 // Render all commands for this atlas
175 for (const auto* cmd : commands) {
176 auto it = atlas_lookup_.find(cmd->atlas_id);
177 if (it == atlas_lookup_.end()) continue;
178
179 AtlasEntry* entry = it->second;
180
181 // Calculate destination rectangle
182 SDL_Rect dest_rect = {
183 static_cast<int>(cmd->x),
184 static_cast<int>(cmd->y),
185 static_cast<int>(entry->uv_rect.w * cmd->scale_x),
186 static_cast<int>(entry->uv_rect.h * cmd->scale_y)
187 };
188
189 // Apply rotation if needed
190 if (std::abs(cmd->rotation) > 0.001F) {
191 // For rotation, we'd need to use SDL_RenderCopyEx
192 // This is a simplified version
193 SDL_RenderCopy(renderer_, atlas.texture, &entry->uv_rect, &dest_rect);
194 } else {
195 SDL_RenderCopy(renderer_, atlas.texture, &entry->uv_rect, &dest_rect);
196 }
197 }
198 }
199}

202 const std::unordered_map<BppFormat, std::vector<int>>& bpp_groups) {
203 if (render_commands.empty()) {

205 }

208
209 // Render each BPP group separately for optimal performance
210 for (const auto& [bpp_format, command_indices] : bpp_groups) {
211 if (command_indices.empty()) continue;

213 // Group commands by atlas for this BPP format
214 std::unordered_map<int, std::vector<const RenderCommand*>> atlas_groups;
215
216 for (int cmd_index : command_indices) {
217 if (cmd_index >= 0 && cmd_index < static_cast<int>(render_commands.size())) {
218 const auto& cmd = render_commands[cmd_index];
219 auto it = atlas_lookup_.find(cmd.atlas_id);
220 if (it != atlas_lookup_.end() && it->second->in_use && it->second->bpp_format == bpp_format) {
221 // Find which atlas contains this entry
222 for (size_t i = 0; i < atlases_.size(); ++i) {
223 for (const auto& entry : atlases_[i]->entries) {
224 if (entry.atlas_id == cmd.atlas_id) {
225 atlas_groups[i].push_back(&cmd);
226 break;
227 }
228 }
229 }
230 }
231 }
232 }
233
234 // Render each atlas group for this BPP format
235 for (const auto& [atlas_index, commands] : atlas_groups) {
236 if (commands.empty()) continue;
237
238 auto& atlas = *atlases_[atlas_index];
239
240 // Set atlas texture with BPP-specific blend mode
241 SDL_SetTextureBlendMode(atlas.texture, SDL_BLENDMODE_BLEND);
242
243 // Render all commands for this atlas and BPP format
244 for (const auto* cmd : commands) {
245 auto it = atlas_lookup_.find(cmd->atlas_id);
246 if (it == atlas_lookup_.end()) continue;

248 AtlasEntry* entry = it->second;
249
250 // Calculate destination rectangle

252 static_cast<int>(cmd->x),
253 static_cast<int>(cmd->y),
254 static_cast<int>(entry->uv_rect.w * cmd->scale_x),
255 static_cast<int>(entry->uv_rect.h * cmd->scale_y)

269AtlasStats AtlasRenderer::GetStats() const {
270 AtlasStats stats;
271
272 stats.total_atlases = atlases_.size();
273
274 for (const auto& atlas : atlases_) {
275 stats.total_entries += atlas->entries.size();
276 stats.used_entries += std::count_if(atlas->entries.begin(), atlas->entries.end(),
277 [](const AtlasEntry& entry) { return entry.in_use; });
278
279 // Calculate memory usage (simplified)
280 stats.total_memory += atlas->size * atlas->size * 4; // RGBA8888
281 }
282
283 if (stats.total_entries > 0) {
284 stats.utilization_percent = (static_cast<float>(stats.used_entries) / stats.total_entries) * 100.0F;
285 }
286
287 return stats;
288}

404void AtlasRenderer::RebuildAtlas(Atlas& atlas) {
405 // Clear used regions
406 std::fill(atlas.used_regions.begin(), atlas.used_regions.end(), false);
407
408 // Rebuild atlas texture by copying from source textures
409 SDL_SetRenderTarget(renderer_, atlas.texture);
410 SDL_SetRenderDrawColor(renderer_, 0, 0, 0, 0);
411 SDL_RenderClear(renderer_);
412
413 for (auto& entry : atlas.entries) {
414 if (entry.in_use && entry.texture) {
415 SDL_RenderCopy(renderer_, entry.texture, nullptr, &entry.uv_rect);
416 MarkRegionUsed(atlas, entry.uv_rect, true);
417 }
418 }
419
420 SDL_SetRenderTarget(renderer_, nullptr);
421}

423SDL_Rect AtlasRenderer::FindFreeRegion(Atlas& atlas, int width, int height) {
424 // Simple first-fit algorithm
425 for (int y = 0; y <= atlas.size - height; ++y) {
426 for (int x = 0; x <= atlas.size - width; ++x) {
427 bool can_fit = true;
428
429 // Check if region is free
430 for (int dy = 0; dy < height && can_fit; ++dy) {
431 for (int dx = 0; dx < width && can_fit; ++dx) {
432 int index = (y + dy) * atlas.size + (x + dx);
433 if (index >= static_cast<int>(atlas.used_regions.size()) || atlas.used_regions[index]) {
434 can_fit = false;
435 }
436 }
437 }
438
439 if (can_fit) {
440 return {x, y, width, height};
441 }
442 }
443 }
444
445 return {0, 0, 0, 0}; // No space found
446}

448void AtlasRenderer::MarkRegionUsed(Atlas& atlas, const SDL_Rect& rect, bool used) {
449 for (int y = rect.y; y < rect.y + rect.h; ++y) {
450 for (int x = rect.x; x < rect.x + rect.w; ++x) {
451 int index = y * atlas.size + x;
452 if (index >= 0 && index < static_cast<int>(atlas.used_regions.size())) {
453 atlas.used_regions[index] = used;
454 }
455 }
