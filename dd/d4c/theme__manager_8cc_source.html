260absl::Status ThemeManager::LoadThemeFromFile(const std::string& filepath) {
261 // Try multiple possible paths where theme files might be located
262 std::vector<std::string> possible_paths = {
263 filepath, // Absolute path
264 "assets/themes/" + filepath, // Relative from build dir
265 "../assets/themes/" + filepath, // Relative from bin dir
266 util::GetResourcePath("assets/themes/" + filepath), // Platform-specific resource path
267 };
268
269 std::ifstream file;
270 std::string successful_path;
271
272 for (const auto& path : possible_paths) {
273 file.open(path);
274 if (file.is_open()) {
275 successful_path = path;
276 break;
277 } else {
278 file.clear(); // Clear any error flags before trying next path
279 }
280 }
281
282 if (!file.is_open()) {
283 return absl::InvalidArgumentError(absl::StrFormat("Cannot open theme file: %s (tried %zu paths)",
284 filepath, possible_paths.size()));
285 }
286
287 std::string content((std::istreambuf_iterator<char>(file)),
288 std::istreambuf_iterator<char>());
289 file.close();
290
291 if (content.empty()) {
292 return absl::InvalidArgumentError(absl::StrFormat("Theme file is empty: %s", successful_path));
293 }
294
295 EnhancedTheme theme;
296 auto parse_status = ParseThemeFile(content, theme);
297 if (!parse_status.ok()) {
298 return absl::InvalidArgumentError(absl::StrFormat("Failed to parse theme file %s: %s",
299 successful_path, parse_status.message()));
300 }
301
302 if (theme.name.empty()) {
303 return absl::InvalidArgumentError(absl::StrFormat("Theme file missing name: %s", successful_path));
304 }
305
306 themes_[theme.name] = theme;
307 return absl::OkStatus();
308}

498absl::Status ThemeManager::ParseThemeFile(const std::string& content, EnhancedTheme& theme) {
499 std::istringstream stream(content);
500 std::string line;
501 std::string current_section = "";
502
503 while (std::getline(stream, line)) {
504 // Skip empty lines and comments
505 if (line.empty() || line[0] == '#') continue;
506
507 // Check for section headers [section_name]
508 if (line[0] == '[' && line.back() == ']') {
509 current_section = line.substr(1, line.length() - 2);
510 continue;
511 }
512
513 size_t eq_pos = line.find('=');
514 if (eq_pos == std::string::npos) continue;
515
516 std::string key = line.substr(0, eq_pos);
517 std::string value = line.substr(eq_pos + 1);
518
519 // Trim whitespace and comments
520 key.erase(0, key.find_first_not_of(" \t"));
521 key.erase(key.find_last_not_of(" \t") + 1);
522 value.erase(0, value.find_first_not_of(" \t"));
523
524 // Remove inline comments
525 size_t comment_pos = value.find('#');
526 if (comment_pos != std::string::npos) {
527 value = value.substr(0, comment_pos);
528 }
529 value.erase(value.find_last_not_of(" \t") + 1);
530
531 // Parse based on section
532 if (current_section == "colors") {
533 Color color = ParseColorFromString(value);
534
535 if (key == "primary") theme.primary = color;
536 else if (key == "secondary") theme.secondary = color;
537 else if (key == "accent") theme.accent = color;

629Color ThemeManager::ParseColorFromString(const std::string& color_str) const {
630 std::vector<std::string> components = absl::StrSplit(color_str, ',');
631 if (components.size() != 4) {
632 return RGBA(255, 255, 255, 255); // White fallback
633 }
634
635 try {
636 int r = std::stoi(components[0]);
637 int g = std::stoi(components[1]);
638 int b = std::stoi(components[2]);
639 int a = std::stoi(components[3]);
640 return RGBA(r, g, b, a);
641 } catch (...) {
642 return RGBA(255, 255, 255, 255); // White fallback
643 }

1979 theme_files.push_back(full_path);

1981 }
1982#endif
1983 } catch (const std::exception& e) {
1984 LOG_ERROR("Theme Manager", "Error scanning directory %s", search_path.c_str());
1985 }
1986 }
1987
1988 // Remove duplicates while preserving order
1989 std::vector<std::string> unique_files;
1990 std::set<std::string> seen_basenames;
