14absl::Status GuiAutomationClient::Connect() {
15#ifdef YAZE_WITH_GRPC
16 auto channel = grpc::CreateChannel(server_address_,
17 grpc::InsecureChannelCredentials());
18 if (!channel) {
19 return absl::InternalError("Failed to create gRPC channel");
20 }
21
22 stub_ = yaze::test::ImGuiTestHarness::NewStub(channel);
23 if (!stub_) {
24 return absl::InternalError("Failed to create gRPC stub");
25 }
26
27 // Test connection with a ping
28 auto result = Ping("connection_test");
29 if (!result.ok()) {
30 return absl::UnavailableError(
31 absl::StrFormat("Failed to connect to test harness at %s: %s",
32 server_address_, result.status().message()));
33 }
34
35 connected_ = true;
36 return absl::OkStatus();
37#else
38 return absl::UnimplementedError(
39 "GUI automation requires YAZE_WITH_GRPC=ON at build time");
40#endif
41}

43absl::StatusOr<AutomationResult> GuiAutomationClient::Ping(
44 const std::string& message) {
45#ifdef YAZE_WITH_GRPC
46 if (!stub_) {
47 return absl::FailedPreconditionError("Not connected. Call Connect() first.");
48 }
49
50 yaze::test::PingRequest request;
51 request.set_message(message);
52
53 yaze::test::PingResponse response;
54 grpc::ClientContext context;
55
56 grpc::Status status = stub_->Ping(&context, request, &response);
57
58 if (!status.ok()) {
59 return absl::InternalError(
60 absl::StrFormat("Ping RPC failed: %s", status.error_message()));
61 }
62
63 AutomationResult result;
64 result.success = true;
65 result.message = absl::StrFormat("Server version: %s (timestamp: %lld)",
66 response.yaze_version(),
67 response.timestamp_ms());
68 result.execution_time = std::chrono::milliseconds(0);
69 return result;
70#else
71 return absl::UnimplementedError("gRPC not available");
72#endif
73}
