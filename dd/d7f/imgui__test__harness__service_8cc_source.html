393absl::Status ImGuiTestHarnessServiceImpl::Wait(const WaitRequest* request,
394 WaitResponse* response) {
395 auto start = std::chrono::steady_clock::now();

397#if defined(YAZE_ENABLE_IMGUI_TEST_ENGINE) && YAZE_ENABLE_IMGUI_TEST_ENGINE

413 // Parse condition: "window_visible:Overworld Editor" -> check if window is visible
414 std::string condition = request->condition();
415 size_t colon_pos = condition.find(':');
416
417 if (colon_pos == std::string::npos) {
418 response->set_success(false);
419 response->set_message("Invalid condition format. Use 'type:target' (e.g. 'window_visible:Overworld Editor')");
420 return absl::OkStatus();
421 }
422
423 std::string condition_type = condition.substr(0, colon_pos);
424 std::string condition_target = condition.substr(colon_pos + 1);
425
426 int timeout_ms = request->timeout_ms() > 0 ? request->timeout_ms() : 5000; // Default 5s
427 int poll_interval_ms = request->poll_interval_ms() > 0 ? request->poll_interval_ms() : 100; // Default 100ms
428
429 // Create a dynamic test to poll the condition
430 bool condition_met = false;
431 std::string message;
432
433 auto test_data = std::make_shared<DynamicTestData>();
434 test_data->test_func = [=, &condition_met, &message](ImGuiTestContext* ctx) {
435 try {
436 auto poll_start = std::chrono::steady_clock::now();
437 auto timeout = std::chrono::milliseconds(timeout_ms);
438
439 while (std::chrono::steady_clock::now() - poll_start < timeout) {
440 bool current_state = false;
441
442 // Check the condition type
443 if (condition_type == "window_visible") {
444 ImGuiWindow* window = ImGui::FindWindowByName(condition_target.c_str());
445 current_state = (window != nullptr && !window->Hidden);
446 } else if (condition_type == "element_visible") {
447 ImGuiTestItemInfo item = ctx->ItemInfo(condition_target.c_str());
448 current_state = (item.ID != 0 && item.RectClipped.GetWidth() > 0 && item.RectClipped.GetHeight() > 0);
449 } else if (condition_type == "element_enabled") {
450 ImGuiTestItemInfo item = ctx->ItemInfo(condition_target.c_str());
451 current_state = (item.ID != 0 && !(item.ItemFlags & ImGuiItemFlags_Disabled));
452 } else {
453 message = absl::StrFormat("Unknown condition type: %s", condition_type);
454 condition_met = false;
455 return;
456 }
457
458 if (current_state) {
459 condition_met = true;
460 message = absl::StrFormat("Condition '%s:%s' met after %lld ms",
461 condition_type, condition_target,
462 std::chrono::duration_cast<std::chrono::milliseconds>(
463 std::chrono::steady_clock::now() - poll_start).count());
464 return;
465 }
466
467 // Sleep before next poll
468 std::this_thread::sleep_for(std::chrono::milliseconds(poll_interval_ms));
469 ctx->Yield(); // Let ImGui process events
470 }
471
472 // Timeout reached
473 condition_met = false;
474 message = absl::StrFormat("Condition '%s:%s' not met after %d ms timeout",
475 condition_type, condition_target, timeout_ms);
476 } catch (const std::exception& e) {
477 condition_met = false;
478 message = absl::StrFormat("Wait failed: %s", e.what());
479 }
480 };
481
482 // Register and queue the test
483 std::string test_name = absl::StrFormat("grpc_wait_%lld",
484 std::chrono::system_clock::now().time_since_epoch().count());
485
486 ImGuiTest* test = IM_REGISTER_TEST(engine, "grpc", test_name.c_str());
487 test->TestFunc = RunDynamicTest;
488 test->UserData = test_data.get();
489
490 ImGuiTestEngine_QueueTest(engine, test, ImGuiTestRunFlags_RunFromGui);
491
492 // Wait for test to complete (with extended timeout for the wait itself)
493 auto extended_timeout = std::chrono::milliseconds(timeout_ms + 5000);
494 auto wait_start = std::chrono::steady_clock::now();
495 while (test->Output.Status == ImGuiTestStatus_Queued || test->Output.Status == ImGuiTestStatus_Running) {
496 if (std::chrono::steady_clock::now() - wait_start > extended_timeout) {
497 condition_met = false;
498 message = "Test execution timeout";
499 break;
500 }
501 std::this_thread::sleep_for(std::chrono::milliseconds(10));
502 }
503
504 if (test->Output.Status == ImGuiTestStatus_Success) {
505 // Status already set by test function
506 } else if (test->Output.Status != ImGuiTestStatus_Unknown) {
507 condition_met = false;
508 if (message.empty()) {
509 message = "Test failed";
510 }
511 }
512
513 // Cleanup
514 ImGuiTestEngine_UnregisterTest(engine, test);
515
516#else
517 // ImGuiTestEngine not available - stub implementation
518 bool condition_met = true;
519 std::string message = absl::StrFormat("[STUB] Condition '%s' met (ImGuiTestEngine not available)",
520 request->condition());
521#endif
522
523 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
524 std::chrono::steady_clock::now() - start);
525
526 response->set_success(condition_met);
527 response->set_message(message);
528 response->set_elapsed_ms(elapsed.count());
529
530 return absl::OkStatus();
