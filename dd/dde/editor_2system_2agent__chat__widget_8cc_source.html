69std::optional<ChatMessage::TableData> ParseTableData(const nlohmann::json& json) {
70 if (!json.is_object()) {
71 return std::nullopt;
72 }
73
74 ChatMessage::TableData table;
75 if (json.contains("headers") && json["headers"].is_array()) {
76 for (const auto& header : json["headers"]) {
77 if (header.is_string()) {
78 table.headers.push_back(header.get<std::string>());
79 }
80 }
81 }
82
83 if (json.contains("rows") && json["rows"].is_array()) {
84 for (const auto& row : json["rows"]) {
85 if (!row.is_array()) {
86 continue;
87 }
88 std::vector<std::string> row_values;
89 for (const auto& value : row) {
90 if (value.is_string()) {
91 row_values.push_back(value.get<std::string>());
92 } else {
93 row_values.push_back(value.dump());
94 }
95 }
96 table.rows.push_back(std::move(row_values));
97 }
98 }
99
100 if (table.headers.empty() && table.rows.empty()) {
101 return std::nullopt;
102 }
103
104 return table;
105}

204 if (!directory.empty()) {
205 std::filesystem::create_directories(directory, ec);
206 if (ec) {
207 if (toast_manager_) {
208 toast_manager_->Show(
209 "Unable to prepare chat history directory",
210 ToastType::kError, 5.0f);

212 return;

214 }

218 return;
219 }
220

225 return;

229 for (const auto& item : json["messages"]) {

234 ChatMessage message;

243 message.json_pretty = item["json_pretty"].get<std::string>();

247 message.table_data = ParseTableData(item["table_data"]);

250 if (item.contains("metrics") && item["metrics"].is_object()) {
251 ChatMessage::SessionMetrics metrics;

253 metrics.turn_index = metrics_json.value("turn_index", 0);
254 metrics.total_user_messages =

256 metrics.total_agent_messages =
257 metrics_json.value("total_agent_messages", 0);
258 metrics.total_tool_calls =
259 metrics_json.value("total_tool_calls", 0);
260 metrics.total_commands = metrics_json.value("total_commands", 0);
261 metrics.total_proposals = metrics_json.value("total_proposals", 0);
262 metrics.total_elapsed_seconds =
263 metrics_json.value("total_elapsed_seconds", 0.0);
264 metrics.average_latency_seconds =
265 metrics_json.value("average_latency_seconds", 0.0);
266 message.metrics = metrics;
267 }
268
269 if (item.contains("proposal")) {
270 message.proposal = ParseProposal(item["proposal"]);
271 }
272
273 history.push_back(std::move(message));
274 }
275
276 if (!history.empty()) {
277 agent_service_.ReplaceHistory(std::move(history));
278 last_history_size_ = agent_service_.GetHistory().size();
279 last_proposal_count_ = CountKnownProposals();
280 history_dirty_ = false;
281 last_persist_time_ = absl::Now();
282 if (toast_manager_) {
283 toast_manager_->Show("Restored chat history",
284 ToastType::kInfo, 3.5f);

286 }
287
288 if (json.contains("collaboration") && json["collaboration"].is_object()) {
289 const auto& collab_json = json["collaboration"];
290 collaboration_state_.active = collab_json.value("active", false);
291 collaboration_state_.session_id = collab_json.value("session_id", "");
292 collaboration_state_.participants.clear();
293 if (collab_json.contains("participants") &&
294 collab_json["participants"].is_array()) {
295 for (const auto& participant : collab_json["participants"]) {
296 if (participant.is_string()) {
297 collaboration_state_.participants.push_back(
298 participant.get<std::string>());
299 }
300 }
301 }
302 if (collab_json.contains("last_synced")) {
303 collaboration_state_.last_synced =
304 ParseTimestamp(collab_json["last_synced"]);
305 }
306 }
307
308 if (json.contains("multimodal") && json["multimodal"].is_object()) {
309 const auto& multimodal_json = json["multimodal"];
310 if (multimodal_json.contains("last_capture_path") &&
311 multimodal_json["last_capture_path"].is_string()) {
312 std::string path = multimodal_json["last_capture_path"].get<std::string>();
313 if (!path.empty()) {
314 multimodal_state_.last_capture_path = std::filesystem::path(path);
315 }
316 }
317 multimodal_state_.status_message =
318 multimodal_json.value("status_message", "");
319 if (multimodal_json.contains("last_updated")) {
320 multimodal_state_.last_updated =
321 ParseTimestamp(multimodal_json["last_updated"]);
322 }
323 }
324 } catch (const std::exception& e) {
325 if (toast_manager_) {
326 toast_manager_->Show(
327 absl::StrFormat("Failed to load chat history: %s", e.what()),
328 ToastType::kError, 6.0f);
329 }
330 }

333void AgentChatWidget::PersistHistory() {
334 if (!history_loaded_ || !history_dirty_) {
335 return;
336 }
337
338 const auto& history = agent_service_.GetHistory();
339
340 nlohmann::json json;
341 json["version"] = 2;
342 json["messages"] = nlohmann::json::array();
343
344 for (const auto& message : history) {
345 nlohmann::json entry;
346 entry["sender"] =
347 message.sender == ChatMessage::Sender::kUser ? "user" : "agent";
348 entry["message"] = message.message;
349 entry["timestamp"] = absl::FormatTime(absl::RFC3339_full,
350 message.timestamp,
351 absl::UTCTimeZone());
352
353 if (message.json_pretty.has_value()) {
354 entry["json_pretty"] = *message.json_pretty;
355 }
356 if (message.table_data.has_value()) {
357 entry["table_data"] = SerializeTableData(*message.table_data);
358 }
359 if (message.metrics.has_value()) {
360 const auto& metrics = *message.metrics;
361 nlohmann::json metrics_json;
362 metrics_json["turn_index"] = metrics.turn_index;
363 metrics_json["total_user_messages"] = metrics.total_user_messages;
364 metrics_json["total_agent_messages"] = metrics.total_agent_messages;
365 metrics_json["total_tool_calls"] = metrics.total_tool_calls;
366 metrics_json["total_commands"] = metrics.total_commands;
367 metrics_json["total_proposals"] = metrics.total_proposals;
368 metrics_json["total_elapsed_seconds"] = metrics.total_elapsed_seconds;
369 metrics_json["average_latency_seconds"] =
370 metrics.average_latency_seconds;
371 entry["metrics"] = metrics_json;
372 }
373 if (message.proposal.has_value()) {
374 entry["proposal"] = SerializeProposal(*message.proposal);
375 }
376
377 json["messages"].push_back(std::move(entry));
378 }
379
380 nlohmann::json collab_json;
381 collab_json["active"] = collaboration_state_.active;
382 collab_json["session_id"] = collaboration_state_.session_id;
383 collab_json["participants"] = collaboration_state_.participants;
384 if (collaboration_state_.last_synced != absl::InfinitePast()) {
385 collab_json["last_synced"] = absl::FormatTime(
386 absl::RFC3339_full, collaboration_state_.last_synced,
387 absl::UTCTimeZone());
388 }
389 json["collaboration"] = std::move(collab_json);
390
391 nlohmann::json multimodal_json;
392 if (multimodal_state_.last_capture_path.has_value()) {
393 multimodal_json["last_capture_path"] =
394 multimodal_state_.last_capture_path->string();
395 } else {
396 multimodal_json["last_capture_path"] = "";
397 }
398 multimodal_json["status_message"] = multimodal_state_.status_message;
399 if (multimodal_state_.last_updated != absl::InfinitePast()) {
400 multimodal_json["last_updated"] = absl::FormatTime(
401 absl::RFC3339_full, multimodal_state_.last_updated,
402 absl::UTCTimeZone());
403 }
404 json["multimodal"] = std::move(multimodal_json);
405
406 std::error_code ec;
407 auto directory = history_path_.parent_path();
408 if (!directory.empty()) {
409 std::filesystem::create_directories(directory, ec);
410 if (ec) {
411 if (toast_manager_) {
412 toast_manager_->Show(
413 "Unable to create chat history directory",
414 ToastType::kError, 5.0f);
415 }
416 return;
417 }
418 }
419
420 std::ofstream file(history_path_);
421 if (!file.is_open()) {
422 if (toast_manager_) {
423 toast_manager_->Show("Cannot write chat history",
424 ToastType::kError, 5.0f);
425 }
426 return;
427 }
428
429 file << json.dump(2);
430 history_dirty_ = false;
431 last_persist_time_ = absl::Now();
432}

481void AgentChatWidget::HandleAgentResponse(

483 if (!response.ok()) {
484 if (toast_manager_) {

487 ToastType::kError, 5.0f);

497

501 last_proposal_count_ = std::max(last_proposal_count_, total);
