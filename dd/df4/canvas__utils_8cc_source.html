19float CalculateEffectiveScale(ImVec2 canvas_size, ImVec2 content_size,
20 float global_scale) {

38bool LoadROMPaletteGroups(Rom* rom, CanvasPaletteManager& palette_manager) {

98bool ApplyPaletteGroup(gfx::Bitmap* bitmap,
99 const CanvasPaletteManager& palette_manager,
100 int group_index, int palette_index) {
101 if (!bitmap || group_index < 0 ||
102 group_index >=
103 static_cast<int>(palette_manager.rom_palette_groups.size())) {
104 return false;
105 }
106
107 try {
108 const auto& selected_palette =
109 palette_manager.rom_palette_groups[group_index];
110
111 // Apply the palette based on the index
112 if (palette_index >= 0 && palette_index < 8) {
113 bitmap->SetPaletteWithTransparent(selected_palette, palette_index);
114 } else {
115 bitmap->SetPalette(selected_palette);
116 }
117
118 Renderer::Get().UpdateBitmap(bitmap);
119 LOG_INFO("Canvas", "Applied palette group %d, index %d to bitmap",
120 group_index, palette_index);
121 return true;

123 } catch (const std::exception& e) {
124 LOG_ERROR("Canvas", "Failed to apply palette");
125 return false;
