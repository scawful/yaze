109void SettingsEditor::DrawThemeSettings() {
110 using namespace ImGui;
111
112 auto& theme_manager = gui::ThemeManager::Get();
113
114 Text("%s Theme Management", ICON_MD_PALETTE);
115 Separator();
116
117 // Current theme selection
118 Text("Current Theme:");
119 SameLine();
120 auto current = theme_manager.GetCurrentThemeName();
121 TextColored(ImVec4(0.4f, 0.8f, 1.0f, 1.0f), "%s", current.c_str());
122
123 Spacing();
124
125 // Available themes grid
126 Text("Available Themes:");
127 for (const auto& theme_name : theme_manager.GetAvailableThemes()) {
128 PushID(theme_name.c_str());
129 bool is_current = (theme_name == current);
130
131 if (is_current) {
132 PushStyleColor(ImGuiCol_Button, ImVec4(0.2f, 0.6f, 0.8f, 1.0f));
133 }
134
135 if (Button(theme_name.c_str(), ImVec2(180, 0))) {
136 theme_manager.LoadTheme(theme_name);
137 }
138
139 if (is_current) {
140 PopStyleColor();
141 SameLine();
142 Text(ICON_MD_CHECK);
143 }
144
145 PopID();
146 }
147
148 Separator();
149 Spacing();
150
151 // Theme operations
152 if (CollapsingHeader(ICON_MD_EDIT " Theme Operations")) {
153 TextColored(ImVec4(1.0f, 1.0f, 0.0f, 1.0f),
154 "Theme import/export coming soon");
155 }
