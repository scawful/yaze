56// Overall test results summary
57struct TestResults {
58 std::vector<TestResult> individual_results;
59 size_t total_tests = 0;
60 size_t passed_tests = 0;
61 size_t failed_tests = 0;
62 size_t skipped_tests = 0;
63 std::chrono::milliseconds total_duration{0};
64
65 void AddResult(const TestResult& result) {
66 individual_results.push_back(result);
67 total_tests++;
68 switch (result.status) {
69 case TestStatus::kPassed:
70 passed_tests++;
71 break;
72 case TestStatus::kFailed:
73 failed_tests++;
74 break;
75 case TestStatus::kSkipped:
76 skipped_tests++;
77 break;
78 default:
79 break;
80 }
81 total_duration += result.duration;
82 }
83
84 void Clear() {
85 individual_results.clear();
86 total_tests = passed_tests = failed_tests = skipped_tests = 0;
87 total_duration = std::chrono::milliseconds{0};
88 }
89
90 float GetPassRate() const {
91 return total_tests > 0 ? static_cast<float>(passed_tests) / total_tests
92 : 0.0f;
93 }
94};
