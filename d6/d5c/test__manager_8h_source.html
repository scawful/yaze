59// Overall test results summary
60struct TestResults {
61 std::vector<TestResult> individual_results;
62 size_t total_tests = 0;
63 size_t passed_tests = 0;
64 size_t failed_tests = 0;
65 size_t skipped_tests = 0;
66 std::chrono::milliseconds total_duration{0};
67
68 void AddResult(const TestResult& result) {
69 individual_results.push_back(result);
70 total_tests++;
71 switch (result.status) {
72 case TestStatus::kPassed:
73 passed_tests++;
74 break;
75 case TestStatus::kFailed:
76 failed_tests++;
77 break;
78 case TestStatus::kSkipped:
79 skipped_tests++;
80 break;
81 default:
82 break;
83 }
84 total_duration += result.duration;
85 }
86
87 void Clear() {
88 individual_results.clear();
89 total_tests = passed_tests = failed_tests = skipped_tests = 0;
90 total_duration = std::chrono::milliseconds{0};
91 }
92
93 float GetPassRate() const {
94 return total_tests > 0 ? static_cast<float>(passed_tests) / total_tests
95 : 0.0f;
96 }
97};
