95def parse_block(lines: List[str], start_idx: int) -> int:
96 """Return index of the closing ')' line for a set/list block."""
97 for idx in range(start_idx + 1, len(lines)):
98 if lines[idx].strip().startswith(")"):
99 return idx
100 raise ValueError(f"Unterminated set/list block starting at line {start_idx}")

103def parse_entry(line: str) -> Optional[str]:
104 stripped = line.strip()
105 if not stripped or stripped.startswith("#"):
106 return None
107 # Remove trailing inline comment
108 stripped = stripped.split("#", 1)[0].strip()
109 if not stripped:
110 return None
111 if stripped.startswith("$"):
112 return None
113 return stripped

116def gather_expected_sources(block: CMakeSourceBlock) -> List[str]:
117 entries: Set[str] = set()

138def update_cmake_block(block: CMakeSourceBlock, dry_run: bool) -> bool:
139 cmake_lines = (block.cmake_path.read_text(encoding="utf-8")).splitlines()
140 pattern = re.compile(rf"\s*set\‍(\s*{re.escape(block.variable)}\b")
141
142 start_idx: Optional[int] = None
143 for idx, line in enumerate(cmake_lines):
144 if pattern.match(line):
145 start_idx = idx
146 break
147
148 if start_idx is None:
149 for idx, line in enumerate(cmake_lines):
150 stripped = line.strip()
151 if not stripped.startswith("set("):
152 continue
153 remainder = stripped[4:].strip()
154 if remainder:
155 if remainder.startswith(block.variable):
156 start_idx = idx
157 break
158 continue
159 lookahead = idx + 1
160 while lookahead < len(cmake_lines):
161 next_line = cmake_lines[lookahead].strip()
162 if not next_line or next_line.startswith("#"):
163 lookahead += 1
164 continue
165 if next_line == block.variable:
166 start_idx = idx
167 break
168 if start_idx is not None:
169 break
170
171 if start_idx is None:
172 raise ValueError(f"Could not locate set({block.variable}) in {block.cmake_path}")
173
174 end_idx = parse_block(cmake_lines, start_idx)
175 block_slice = cmake_lines[start_idx + 1 : end_idx]
176
177 prelude: List[str] = []
178 postlude: List[str] = []
179 existing_entries: List[str] = []
180
181 first_entry_idx: Optional[int] = None
182
183 for idx, line in enumerate(block_slice):
184 entry = parse_entry(line)
185 if entry:
186 if entry == block.variable and not existing_entries:
187 prelude.append(line)
188 continue
189 existing_entries.append(entry)
190 if first_entry_idx is None:
191 first_entry_idx = idx
192 else:
193 if first_entry_idx is None:
194 prelude.append(line)
195 else:
196 postlude.append(line)
197
198 expected_entries = gather_expected_sources(block)
199 expected_set = set(expected_entries)
200
201 if set(existing_entries) == expected_set:
202 return False
203
204 indent = " "
205 if first_entry_idx is not None:
206 sample_line = block_slice[first_entry_idx]
207 indent = sample_line[: len(sample_line) - len(sample_line.lstrip())]
208
209 rebuilt_block = prelude + [f"{indent}{entry}" for entry in expected_entries] + postlude
210
211 if dry_run:
212 print(f"[DRY-RUN] Would update {block.cmake_path.relative_to(PROJECT_ROOT)}")
213 return True
214
215 cmake_lines[start_idx + 1 : end_idx] = rebuilt_block
216 block.cmake_path.write_text("\n".join(cmake_lines) + "\n", encoding="utf-8")
217 print(f"Updated {block.cmake_path.relative_to(PROJECT_ROOT)}")
218 missing = sorted(expected_set - set(existing_entries))
219 removed = sorted(set(existing_entries) - expected_set)
220 if missing:
221 print(f" Added: {', '.join(missing)}")
222 if removed:
223 print(f" Removed: {', '.join(removed)}")
224 return True

240def find_insert_index(lines: List[str]) -> int:
241 include_block_start = None
242 for idx, line in enumerate(lines):
243 if line.startswith("#include"):
244 include_block_start = idx
245 break
246
247 if include_block_start is not None:
248 return include_block_start
249
250 # No includes yet; skip leading comments/blank lines
251 index = 0
252 in_block_comment = False
253 while index < len(lines):
254 stripped = lines[index].strip()
255 if not stripped:
256 index += 1
257 continue
258 if stripped.startswith("/*") and not stripped.endswith("*/"):
259 in_block_comment = True
260 index += 1
261 continue
262 if in_block_comment:
263 if "*/" in stripped:
264 in_block_comment = False
265 index += 1
266 continue
267 if stripped.startswith("//"):
268 index += 1
269 continue
270 break
271 return index

274def ensure_self_header_include(source: Path, dry_run: bool) -> bool:
275 if should_ignore_path(source):
276 return False
277
278 header = find_self_header(source)
279 if header is None:
280 return False
281
282 try:
283 lines = source.read_text(encoding="utf-8").splitlines()
284 except UnicodeDecodeError:
285 return False
286
287 header_variants = {
288 header.name,
289 str(header.relative_to(SOURCE_ROOT)).replace("\\", "/"),
290 }
291
292 if has_include(lines, header_variants):
293 return False
294
295 include_line = f'#include "{header.name}"'
296
297 insert_idx = find_insert_index(lines)
298 lines.insert(insert_idx, include_line)
299
300 if dry_run:
301 rel = source.relative_to(PROJECT_ROOT)
302 print(f"[DRY-RUN] Would insert self-header include into {rel}")
303 return True
304
305 source.write_text("\n".join(lines) + "\n", encoding="utf-8")
306 print(f"Inserted self-header include into {source.relative_to(PROJECT_ROOT)}")
307 return True

310def collect_source_files() -> Set[Path]:
311 managed_dirs: Set[Path] = set()
312 for block in CONFIG:
313 for directory in block.directories:
314 managed_dirs.add(directory.path)

316 result: Set[Path] = set()

326def run(dry_run: bool, cmake_only: bool, includes_only: bool) -> int:
327 if cmake_only and includes_only:
328 raise ValueError("Cannot use --cmake-only and --includes-only together")
329
330 changed = False
331
332 if not includes_only:
333 for block in CONFIG:
334 changed |= update_cmake_block(block, dry_run)
335
336 if not cmake_only:
337 for source in collect_source_files():
338 changed |= ensure_self_header_include(source, dry_run)
339
340 if dry_run and not changed:
341 print("No changes required (dry-run)")
342 if not dry_run and not changed:
343 print("No changes required")
344 return 0
