99bool ApplyPaletteGroup(gfx::Bitmap* bitmap,
100 const CanvasPaletteManager& palette_manager,
101 int group_index, int palette_index) {
102 if (!bitmap || group_index < 0 ||
103 group_index >=
104 static_cast<int>(palette_manager.rom_palette_groups.size())) {
105 return false;
106 }
107
108 try {
109 const auto& selected_palette =
110 palette_manager.rom_palette_groups[group_index];
111
112 // Apply the palette based on the index
113 if (palette_index >= 0 && palette_index < 8) {
114 bitmap->SetPaletteWithTransparent(selected_palette, palette_index);
115 } else {
116 bitmap->SetPalette(selected_palette);

119 Renderer::Get().UpdateBitmap(bitmap);
120 util::logf("Canvas: Applied palette group %d, index %d to bitmap",

122 return true;
