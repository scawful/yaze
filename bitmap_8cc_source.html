28 if (!io_ptr) return;

33

44bool ConvertSurfaceToPng(SDL_Surface *surface, std::vector<uint8_t> &buffer) {

50
51 png_infop info_ptr = png_create_info_struct(png_ptr);
52 if (!info_ptr) {
53 png_destroy_write_struct(&png_ptr, (png_infopp)NULL);
54 SDL_Log("Failed to create PNG info struct");
55 return false;
56 }
57
58 if (setjmp(png_jmpbuf(png_ptr))) {
59 png_destroy_write_struct(&png_ptr, &info_ptr);
60 SDL_Log("Error during PNG write");
61 return false;

63
64 png_set_write_fn(png_ptr, &buffer, png_internal::PngWriteCallback, NULL);
65
66 png_colorp pal_ptr;
67
68 /* Prepare chunks */
69 int colortype = PNG_COLOR_MASK_COLOR;
70 int i = 0;
71 SDL_Palette *pal;
72 if (surface->format->BytesPerPixel > 0 &&
73 surface->format->BytesPerPixel <= 8 && (pal = surface->format->palette)) {
74 SDL_Log("Writing PNG image with palette");
75 colortype |= PNG_COLOR_MASK_PALETTE;
76 pal_ptr = (png_colorp)malloc(pal->ncolors * sizeof(png_color));
77 for (i = 0; i < pal->ncolors; i++) {
78 pal_ptr[i].red = pal->colors[i].r;
79 pal_ptr[i].green = pal->colors[i].g;
80 pal_ptr[i].blue = pal->colors[i].b;

83 free(pal_ptr);
84 }
85
86 if (surface->format->Amask) { // Check for alpha channel
87 colortype |= PNG_COLOR_MASK_ALPHA;
88 }
89
90 auto depth = surface->format->BitsPerPixel;
91
92 // Set image attributes.
93 png_set_IHDR(png_ptr, info_ptr, surface->w, surface->h, depth, colortype,
94 PNG_INTERLACE_NONE, PNG_COMPRESSION_TYPE_DEFAULT,
95 PNG_FILTER_TYPE_DEFAULT);
96
97 png_set_bgr(png_ptr);
98
99 // Write the image data.
100 std::vector<png_bytep> row_pointers(surface->h);
101 for (int y = 0; y < surface->h; ++y) {
102 row_pointers[y] = (png_bytep)(surface->pixels) + y * surface->pitch;
103 }
104
105 png_set_rows(png_ptr, info_ptr, row_pointers.data());
106
107 SDL_Log("Writing PNG image...");
108 png_write_png(png_ptr, info_ptr, PNG_TRANSFORM_IDENTITY, NULL);
109 SDL_Log("PNG image write complete");
110
111 png_destroy_write_struct(&png_ptr, &info_ptr);
112
113 return true;
114}
