106 uint16_t GetHeaderOffset() {
107 uint16_t offset;
108 switch (memory_[(0x00 << 16) + 0xFFD5] & 0x07) {
109 case 0: // LoROM
110 offset = 0x7FC0;
111 break;
112 case 1: // HiROM
113 offset = 0xFFC0;
114 break;
115 case 5: // ExHiROM
116 offset = 0x40;
117 break;
118 default:
119 throw std::invalid_argument(
120 "Unable to locate supported ROM mapping mode in the provided ROM "
121 "file. Please try another ROM file.");
122 }
123
124 return offset;
125 }
