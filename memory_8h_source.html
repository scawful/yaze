13// Banks Offset Purpose

22// 8000-FFFF 32k ROM Chunk

168 void Initialize(const std::vector<uint8_t>& romData, bool verbose = false,
169 MemoryMapping mapping = MemoryMapping::SNES_LOROM) {
170 verbose_ = verbose;
171 mapping_ = mapping;
172 if (mapping == MemoryMapping::PC_ADDRESS) {
173 memory_.resize(romData.size());
174 std::copy(romData.begin(), romData.end(), memory_.begin());
175 return;
176 }
177
178 memory_.resize(0x1000000); // 16 MB
179
180 const size_t ROM_CHUNK_SIZE = 0x8000; // 32 KB
181 const size_t SRAM_SIZE = 0x10000; // 64 KB
182 const size_t SYSTEM_RAM_SIZE = 0x20000; // 128 KB
183 const size_t EXPANSION_RAM_SIZE = 0x2000; // 8 KB
184 const size_t HARDWARE_REGISTERS_SIZE = 0x4000; // 16 KB
185
186 // Clear memory
187 std::fill(memory_.begin(), memory_.end(), 0);
188
189 // Load ROM data into memory based on LoROM mapping
190 size_t romSize = romData.size();
191 size_t romAddress = 0;
192 for (size_t bank = 0x00; bank <= 0x3F; ++bank) {
193 for (size_t offset = 0x8000; offset <= 0xFFFF; offset += ROM_CHUNK_SIZE) {
194 if (romAddress < romSize) {
195 std::copy(romData.begin() + romAddress,
196 romData.begin() + romAddress + ROM_CHUNK_SIZE,
197 memory_.begin() + (bank << 16) + offset);
198 romAddress += ROM_CHUNK_SIZE;
199 }
200 }
201 }
202
203 // Initialize SRAM at banks 0x7D and 0xFD
204 std::fill(memory_.begin() + (0x7D << 16), memory_.begin() + (0x7E << 16),
205 0);
206 std::fill(memory_.begin() + (0xFD << 16), memory_.begin() + (0xFE << 16),
207 0);
208
209 // Initialize System RAM at banks 0x7E and 0x7F
210 std::fill(memory_.begin() + (0x7E << 16),
211 memory_.begin() + (0x7E << 16) + SYSTEM_RAM_SIZE, 0);
212
213 // Initialize Shadow RAM at banks 0x00-0x3F and 0x80-0xBF
214 for (size_t bank = 0x00; bank <= 0xBF; bank += 0x80) {
215 std::fill(memory_.begin() + (bank << 16),
216 memory_.begin() + (bank << 16) + 0x2000, 0);
217 }
218
219 // Initialize Hardware Registers at banks 0x00-0x3F and 0x80-0xBF
220 for (size_t bank = 0x00; bank <= 0xBF; bank += 0x80) {
221 std::fill(
222 memory_.begin() + (bank << 16) + 0x2000,
223 memory_.begin() + (bank << 16) + 0x2000 + HARDWARE_REGISTERS_SIZE, 0);
224 }
225
226 // Initialize Expansion RAM at banks 0x00-0x3F and 0x80-0xBF
227 for (size_t bank = 0x00; bank <= 0xBF; bank += 0x80) {
228 std::fill(memory_.begin() + (bank << 16) + 0x6000,
229 memory_.begin() + (bank << 16) + 0x6000 + EXPANSION_RAM_SIZE,
230 0);
231 }
232
233 // Initialize Reset and NMI Vectors at bank 0xFF
234 std::fill(memory_.begin() + (0xFF << 16) + 0xFF00,
235 memory_.begin() + (0xFF << 16) + 0xFFFF + 1, 0);
236
237 // Copy data into rom_ vector
238 rom_.resize(kROMSize);
239 std::copy(memory_.begin() + kROMStart,
240 memory_.begin() + kROMStart + kROMSize, rom_.begin());
241
242 // Copy data into ram_ vector
243 ram_.resize(kRAMSize);
244 std::copy(memory_.begin() + kRAMStart,
245 memory_.begin() + kRAMStart + kRAMSize, ram_.begin());
246
247 // Copy data into vram_ vector
248 vram_.resize(kVRAMSize);
249 std::copy(memory_.begin() + kVRAMStart,
250 memory_.begin() + kVRAMStart + kVRAMSize, vram_.begin());
251
252 // Copy data into oam_ vector
253 oam_.resize(kOAMSize);
254 std::copy(memory_.begin() + kOAMStart,
255 memory_.begin() + kOAMStart + kOAMSize, oam_.begin());
256 }

374 uint32_t GetMappedAddress(uint32_t address) const {
375 uint8_t bank = address >> 16;
376 uint32_t offset = address & 0xFFFF;
377
378 if (mapping_ == MemoryMapping::PC_ADDRESS) {
379 return address;
380 }
381
382 if (bank <= 0x3F) {
383 if (address <= 0x1FFF) {
384 return (0x7E << 16) + offset; // Shadow RAM
385 } else if (address <= 0x5FFF) {
386 return (bank << 16) + (offset - 0x2000) + 0x2000; // Hardware Registers
387 } else if (address <= 0x7FFF) {
388 return offset - 0x6000 + 0x6000; // Expansion RAM
389 } else {
390 // Return lorom mapping
391 return (bank << 16) + (offset - 0x8000) + 0x8000; // ROM
392 }
393 } else if (bank == 0x7D) {
394 return offset + 0x7D0000; // SRAM
395 } else if (bank == 0x7E || bank == 0x7F) {
396 return offset + 0x7E0000; // System RAM
397 } else if (bank >= 0x80) {
398 // Handle HiROM and mirrored areas
399 }
400
401 return address; // Return the original address if no mapping is defined
402 }
