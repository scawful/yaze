105 uint16_t GetHeaderOffset() {
106 uint16_t offset;
107 switch (memory_[(0x00 << 16) + 0xFFD5] & 0x07) {
108 case 0: // LoROM
109 offset = 0x7FC0;
110 break;
111 case 1: // HiROM
112 offset = 0xFFC0;
113 break;
114 case 5: // ExHiROM
115 offset = 0x40;
116 break;
117 default:
118 throw std::invalid_argument(
119 "Unable to locate supported ROM mapping mode in the provided ROM "
120 "file. Please try another ROM file.");
121 }
122
123 return offset;
124 }
