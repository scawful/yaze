122 uint16_t GetHeaderOffset() {
123 uint16_t offset;
124 switch (memory_[(0x00 << 16) + 0xFFD5] & 0x07) {
125 case 0: // LoROM
126 offset = 0x7FC0;
127 break;
128 case 1: // HiROM
129 offset = 0xFFC0;
130 break;
131 case 5: // ExHiROM
132 offset = 0x40;
133 break;
134 default:
135 throw std::invalid_argument(
136 "Unable to locate supported ROM mapping mode in the provided ROM "
137 "file. Please try another ROM file.");
138 }
139
140 return offset;
141 }
