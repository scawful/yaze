47void InputManager::Poll(Snes* snes, int player) {
48 if (!snes || !backend_) return;
49
50 // Poll backend for current controller state
51 ControllerState state = backend_->Poll(player);
52
53 // Update SNES controller state using the hardware button layout
54 // SNES controller bits: 0=B, 1=Y, 2=Select, 3=Start, 4-7=DPad, 8=A, 9=X, 10=L, 11=R
55 for (int i = 0; i < 12; i++) {
56 bool pressed = (state.buttons & (1 << i)) != 0;
57 snes->SetButtonState(player, i, pressed);
