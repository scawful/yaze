211absl::Status AgentEditor::LeaveSession() {
212 if (!in_session_) {
213 return absl::FailedPreconditionError("Not in a session");
214 }
215
216 if (current_mode_ == CollaborationMode::kLocal) {
217 RETURN_IF_ERROR(local_coordinator_->LeaveSession());
218 }
219#ifdef YAZE_WITH_GRPC
220 else if (current_mode_ == CollaborationMode::kNetwork) {
221 if (network_coordinator_) {
222 RETURN_IF_ERROR(network_coordinator_->LeaveSession());
223 }
224 }
225#endif
226
227 // Switch chat back to local history
228 if (chat_widget_) {
229 chat_widget_->SwitchToLocalHistory();
230 }
231
232 in_session_ = false;
233 current_session_id_.clear();
234 current_session_name_.clear();
235 current_participants_.clear();
236
237 if (toast_manager_) {
238 toast_manager_->Show("Left collaboration session", ToastType::kInfo, 3.0f);
239 }
240
241 return absl::OkStatus();
242}

342void AgentEditor::SetupChatWidgetCallbacks() {
343 if (!chat_widget_) {
344 return;
345 }
346
347 AgentChatWidget::CollaborationCallbacks collab_callbacks;
348
349 collab_callbacks.host_session =
350 [this](const std::string& session_name)
351 -> absl::StatusOr<AgentChatWidget::CollaborationCallbacks::SessionContext> {
352 // Use the current mode from the chat widget UI
353 ASSIGN_OR_RETURN(auto session, this->HostSession(session_name, current_mode_));
354
355 AgentChatWidget::CollaborationCallbacks::SessionContext context;
356 context.session_id = session.session_id;
357 context.session_name = session.session_name;
358 context.participants = session.participants;
359 return context;
360 };
361
362 collab_callbacks.join_session =
363 [this](const std::string& session_code)
364 -> absl::StatusOr<AgentChatWidget::CollaborationCallbacks::SessionContext> {
365 ASSIGN_OR_RETURN(auto session, this->JoinSession(session_code, current_mode_));
366
367 AgentChatWidget::CollaborationCallbacks::SessionContext context;
368 context.session_id = session.session_id;
369 context.session_name = session.session_name;
370 context.participants = session.participants;
371 return context;
372 };
373
374 collab_callbacks.leave_session = [this]() {
375 return this->LeaveSession();
376 };
377
378 collab_callbacks.refresh_session =
379 [this]() -> absl::StatusOr<AgentChatWidget::CollaborationCallbacks::SessionContext> {
380 ASSIGN_OR_RETURN(auto session, this->RefreshSession());
381
382 AgentChatWidget::CollaborationCallbacks::SessionContext context;
383 context.session_id = session.session_id;
384 context.session_name = session.session_name;
385 context.participants = session.participants;
386 return context;
387 };
388
389 chat_widget_->SetCollaborationCallbacks(collab_callbacks);
390}
