499absl::Status AgentEditor::LeaveSession() {
500 if (!in_session_) {
501 return absl::FailedPreconditionError("Not in a session");
502 }
503
504 if (current_mode_ == CollaborationMode::kLocal) {
505 RETURN_IF_ERROR(local_coordinator_->LeaveSession());
506 }
507#ifdef YAZE_WITH_GRPC
508 else if (current_mode_ == CollaborationMode::kNetwork) {
509 if (network_coordinator_) {
510 RETURN_IF_ERROR(network_coordinator_->LeaveSession());
511 }
512 }
513#endif
514
515 // Switch chat back to local history
516 if (chat_widget_) {
517 chat_widget_->SwitchToLocalHistory();
518 }
519
520 in_session_ = false;
521 current_session_id_.clear();
522 current_session_name_.clear();
523 current_participants_.clear();
524
525 if (toast_manager_) {
526 toast_manager_->Show("Left collaboration session", ToastType::kInfo, 3.0f);
527 }
528
529 return absl::OkStatus();

630void AgentEditor::SetupChatWidgetCallbacks() {
631 if (!chat_widget_) {
632 return;
633 }
634
635 AgentChatWidget::CollaborationCallbacks collab_callbacks;
636
637 collab_callbacks.host_session =
638 [this](const std::string& session_name)
639 -> absl::StatusOr<AgentChatWidget::CollaborationCallbacks::SessionContext> {
640 // Use the current mode from the chat widget UI
641 ASSIGN_OR_RETURN(auto session, this->HostSession(session_name, current_mode_));
642
643 AgentChatWidget::CollaborationCallbacks::SessionContext context;
644 context.session_id = session.session_id;
645 context.session_name = session.session_name;
646 context.participants = session.participants;
647 return context;
648 };
649
650 collab_callbacks.join_session =
651 [this](const std::string& session_code)
652 -> absl::StatusOr<AgentChatWidget::CollaborationCallbacks::SessionContext> {
653 ASSIGN_OR_RETURN(auto session, this->JoinSession(session_code, current_mode_));
654
655 AgentChatWidget::CollaborationCallbacks::SessionContext context;
656 context.session_id = session.session_id;
657 context.session_name = session.session_name;
658 context.participants = session.participants;
659 return context;
660 };
661
662 collab_callbacks.leave_session = [this]() {
663 return this->LeaveSession();
664 };
665
666 collab_callbacks.refresh_session =
667 [this]() -> absl::StatusOr<AgentChatWidget::CollaborationCallbacks::SessionContext> {
668 ASSIGN_OR_RETURN(auto session, this->RefreshSession());
669
670 AgentChatWidget::CollaborationCallbacks::SessionContext context;
671 context.session_id = session.session_id;
672 context.session_name = session.session_name;
673 context.participants = session.participants;
674 return context;
675 };
676
677 chat_widget_->SetCollaborationCallbacks(collab_callbacks);
678}
