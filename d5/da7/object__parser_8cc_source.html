58absl::StatusOr<ObjectSubtypeInfo> ObjectParser::GetObjectSubtype(int16_t object_id) {
59 ObjectSubtypeInfo info;
60 info.subtype = DetermineSubtype(object_id);
61
62 switch (info.subtype) {
63 case 1: {
64 int index = object_id & 0xFF;
65 info.subtype_ptr = kRoomObjectSubtype1 + (index * 2);
66 info.routine_ptr = kRoomObjectSubtype1 + 0x200 + (index * 2);
67 info.max_tile_count = 8; // Most subtype 1 objects use 8 tiles
68 break;
69 }
70 case 2: {
71 int index = object_id & 0x7F;
72 info.subtype_ptr = kRoomObjectSubtype2 + (index * 2);
73 info.routine_ptr = kRoomObjectSubtype2 + 0x80 + (index * 2);
74 info.max_tile_count = 8;
75 break;
76 }
77 case 3: {
78 int index = object_id & 0xFF;
79 info.subtype_ptr = kRoomObjectSubtype3 + (index * 2);
80 info.routine_ptr = kRoomObjectSubtype3 + 0x100 + (index * 2);
81 info.max_tile_count = 8;
82 break;
83 }
84 default:
85 return absl::InvalidArgumentError(
86 absl::StrFormat("Invalid object subtype for ID: %#04x", object_id));
87 }

92absl::StatusOr<ObjectSizeInfo> ObjectParser::ParseObjectSize(int16_t object_id, uint8_t size_byte) {
93 ObjectSizeInfo info;
94
95 // Extract size bits (0-3 for X, 4-7 for Y)
96 int size_x = size_byte & 0x03;
97 int size_y = (size_byte >> 2) & 0x03;
98
99 info.width_tiles = (size_x + 1) * 2; // Convert to tile count
100 info.height_tiles = (size_y + 1) * 2;
101
102 // Determine orientation based on object ID and size
103 // This is a heuristic based on the object naming patterns
104 if (object_id >= 0x80 && object_id <= 0xFF) {
105 // Objects 0x80-0xFF are typically vertical
106 info.is_horizontal = false;
107 } else {
108 // Objects 0x00-0x7F are typically horizontal
109 info.is_horizontal = true;
110 }
111
112 // Determine if object is repeatable
113 info.is_repeatable = (size_byte != 0);
114 info.repeat_count = size_byte == 0 ? 32 : size_byte;
115
116 return info;
117}

119absl::StatusOr<std::vector<gfx::Tile16>> ObjectParser::ParseSubtype1(int16_t object_id) {
120 int index = object_id & 0xFF;
121 int tile_ptr = kRoomObjectSubtype1 + (index * 2);
122
123 if (tile_ptr + 1 >= (int)rom_->size()) {
124 return absl::OutOfRangeError(
125 absl::StrFormat("Tile pointer out of range: %#06x", tile_ptr));
126 }
127
128 // Read tile data pointer
129 uint8_t low = rom_->data()[tile_ptr];
130 uint8_t high = rom_->data()[tile_ptr + 1];
131 int tile_data_ptr = kRoomObjectTileAddress + ((high << 8) | low);
132
133 // Read 8 tiles (most subtype 1 objects use 8 tiles)
134 return ReadTileData(tile_data_ptr, 8);
135}

173absl::StatusOr<std::vector<gfx::Tile16>> ObjectParser::ReadTileData(int address, int tile_count) {
174 if (address < 0 || address + (tile_count * 8) >= (int)rom_->size()) {
175 return absl::OutOfRangeError(
176 absl::StrFormat("Tile data address out of range: %#06x", address));
177 }
178
179 std::vector<gfx::Tile16> tiles;
180 tiles.reserve(tile_count);
181
182 for (int i = 0; i < tile_count; i++) {
183 int tile_offset = address + (i * 8);
184
185 // Read 4 words (8 bytes) per tile
186 uint16_t w0 = rom_->data()[tile_offset] | (rom_->data()[tile_offset + 1] << 8);
187 uint16_t w1 = rom_->data()[tile_offset + 2] | (rom_->data()[tile_offset + 3] << 8);
188 uint16_t w2 = rom_->data()[tile_offset + 4] | (rom_->data()[tile_offset + 5] << 8);
189 uint16_t w3 = rom_->data()[tile_offset + 6] | (rom_->data()[tile_offset + 7] << 8);
190
191 tiles.emplace_back(
192 gfx::WordToTileInfo(w0),
193 gfx::WordToTileInfo(w1),
194 gfx::WordToTileInfo(w2),
195 gfx::WordToTileInfo(w3)
196 );
197 }
198
199 return tiles;
200}
