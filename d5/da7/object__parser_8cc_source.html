86absl::StatusOr<ObjectSizeInfo> ObjectParser::ParseObjectSize(int16_t object_id, uint8_t size_byte) {
87 ObjectSizeInfo info;
88
89 // Extract size bits (0-3 for X, 4-7 for Y)
90 int size_x = size_byte & 0x03;
91 int size_y = (size_byte >> 2) & 0x03;
92
93 info.width_tiles = (size_x + 1) * 2; // Convert to tile count
94 info.height_tiles = (size_y + 1) * 2;
95
96 // Determine orientation based on object ID and size
97 // This is a heuristic based on the object naming patterns
98 if (object_id >= 0x80 && object_id <= 0xFF) {
99 // Objects 0x80-0xFF are typically vertical
100 info.is_horizontal = false;
101 } else {
102 // Objects 0x00-0x7F are typically horizontal
103 info.is_horizontal = true;
104 }
105
106 // Determine if object is repeatable
107 info.is_repeatable = (size_byte != 0);
108 info.repeat_count = size_byte == 0 ? 32 : size_byte;
109
110 return info;
111}

167absl::StatusOr<std::vector<gfx::Tile16>> ObjectParser::ReadTileData(int address, int tile_count) {
168 if (address < 0 || address + (tile_count * 8) >= (int)rom_->size()) {
169 return absl::OutOfRangeError(
170 absl::StrFormat("Tile data address out of range: %#06x", address));
171 }
172
173 std::vector<gfx::Tile16> tiles;
174 tiles.reserve(tile_count);
175
176 for (int i = 0; i < tile_count; i++) {
177 int tile_offset = address + (i * 8);
178
179 // Read 4 words (8 bytes) per tile
180 uint16_t w0 = rom_->data()[tile_offset] | (rom_->data()[tile_offset + 1] << 8);
181 uint16_t w1 = rom_->data()[tile_offset + 2] | (rom_->data()[tile_offset + 3] << 8);
182 uint16_t w2 = rom_->data()[tile_offset + 4] | (rom_->data()[tile_offset + 5] << 8);
183 uint16_t w3 = rom_->data()[tile_offset + 6] | (rom_->data()[tile_offset + 7] << 8);
184
185 tiles.emplace_back(
186 gfx::WordToTileInfo(w0),
187 gfx::WordToTileInfo(w1),
188 gfx::WordToTileInfo(w2),
189 gfx::WordToTileInfo(w3)
190 );
191 }
192
193 return tiles;
194}
