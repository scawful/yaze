213 return;

252 last_history_size_ = agent_service_.GetHistory().size();
253 last_proposal_count_ = CountKnownProposals();
254 history_dirty_ = false;

259 }

262 collaboration_state_.session_id = snapshot.collaboration.session_id;

1499void AgentChatWidget::PollSharedHistory() {
1500 if (!collaboration_state_.active) {
1501 return; // Not in a collaborative session
1502 }
1503
1504 const absl::Time now = absl::Now();
1505
1506 // Poll every 2 seconds
1507 if (now - last_shared_history_poll_ < absl::Seconds(2)) {
1508 return;
1509 }
1510
1511 last_shared_history_poll_ = now;
1512
1513 // Check if the shared history file has been updated
1514 auto result = AgentChatHistoryCodec::Load(history_path_);
1515 if (!result.ok()) {
1516 return; // File might not exist yet or be temporarily locked
1517 }
1518
1519 const size_t new_size = result->history.size();
1520
1521 // If history has grown, reload it
1522 if (new_size > last_known_history_size_) {
1523 const size_t new_messages = new_size - last_known_history_size_;
1524
1525 agent_service_.ReplaceHistory(std::move(result->history));
1526 last_history_size_ = new_size;
1527 last_known_history_size_ = new_size;
1528
1529 if (toast_manager_) {
1530 toast_manager_->Show(
1531 absl::StrFormat("📬 %zu new message%s from collaborators",
1532 new_messages, new_messages == 1 ? "" : "s"),
1533 ToastType::kInfo, 3.0f);
1534 }
1535 }
