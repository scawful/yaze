40void PerformanceProfiler::EndTimer(const std::string& operation_name) {
41 if (!enabled_ || is_shutting_down_) return;
42
43 auto timer_iter = active_timers_.find(operation_name);
44 if (timer_iter == active_timers_.end()) {
45 // During shutdown, silently ignore missing timers to avoid log spam
46 if (!is_shutting_down_) {
47 LOG_DEBUG("PerformanceProfiler", "EndTimer called for operation '%s' that was not started",
48 operation_name.c_str());
49 }
50 return;
51 }
52
53 auto end_time = std::chrono::high_resolution_clock::now();
54 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(
55 end_time - timer_iter->second).count();
56
57 double duration_ms = duration / 1000.0;
58
59 // Store timing data using memory pool for efficiency
60 operation_times_[operation_name].push_back(static_cast<double>(duration));
61 operation_totals_[operation_name] += duration_ms;
62 operation_counts_[operation_name]++;
63
64 active_timers_.erase(timer_iter);
65}

67PerformanceProfiler::TimingStats PerformanceProfiler::GetStats(
68 const std::string& operation_name) const {
69 TimingStats stats;
70
71 auto times_iter = operation_times_.find(operation_name);
72 auto total_iter = operation_totals_.find(operation_name);
73
74 if (times_iter == operation_times_.end() || times_iter->second.empty()) {
75 return stats;
76 }
77
78 const auto& times = times_iter->second;
79 stats.sample_count = times.size();
80 stats.total_time_ms = (total_iter != operation_totals_.end()) ? total_iter->second : 0.0;
81
82 if (times.empty()) {
83 return stats;
84 }
85
86 // Calculate min, max, and average
87 stats.min_time_us = *std::min_element(times.begin(), times.end());
88 stats.max_time_us = *std::max_element(times.begin(), times.end());
89 stats.avg_time_us = std::accumulate(times.begin(), times.end(), 0.0) / times.size();
90
91 // Calculate median
92 std::vector<double> sorted_times = times;
93 std::sort(sorted_times.begin(), sorted_times.end());
94 stats.median_time_us = PerformanceProfiler::CalculateMedian(sorted_times);
95
96 return stats;
97}

261double PerformanceProfiler::CalculateMedian(std::vector<double> values) {
262 if (values.empty()) {
263 return 0.0;
264 }
265
266 size_t size = values.size();
267 if (size % 2 == 0) {
268 return (values[size / 2 - 1] + values[size / 2]) / 2.0;
269 }
270 return values[size / 2];
271}
