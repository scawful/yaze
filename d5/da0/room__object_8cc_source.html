155void RoomObject::EnsureTilesLoaded() {
156 if (tiles_loaded_) return;
157 if (rom_ == nullptr) return;
158
159 // Try the new parser first - this is more efficient and accurate
160 if (LoadTilesWithParser().ok()) {
161 tiles_loaded_ = true;
162 return;
163 }
164
165 // Fallback to legacy method for compatibility with enhanced validation
166 auto rom_data = rom_->data();
167
168 // Determine which subtype table to use and compute the tile data offset.
169 SubtypeTableInfo sti = GetSubtypeTable(id_);
170 int index = (id_ & sti.index_mask);
171 int tile_ptr = sti.base_ptr + (index * 2);
172
173 // Enhanced bounds checking
174 if (tile_ptr < 0 || tile_ptr + 1 >= (int)rom_->size()) {
175 // Log error but don't crash
176 return;
177 }
178
179 int tile_rel = (int16_t)((rom_data[tile_ptr + 1] << 8) + rom_data[tile_ptr]);
180 int pos = kRoomObjectTileAddress + tile_rel;
181 tile_data_ptr_ = pos;
182
183 // Enhanced bounds checking for tile data
184 if (pos < 0 || pos + 7 >= (int)rom_->size()) {
185 // Log error but don't crash
186 return;
187 }
188
189 // Read tile data with validation
190 uint16_t w0 = (uint16_t)(rom_data[pos] | (rom_data[pos + 1] << 8));
191 uint16_t w1 = (uint16_t)(rom_data[pos + 2] | (rom_data[pos + 3] << 8));
192 uint16_t w2 = (uint16_t)(rom_data[pos + 4] | (rom_data[pos + 5] << 8));
193 uint16_t w3 = (uint16_t)(rom_data[pos + 6] | (rom_data[pos + 7] << 8));
194
195 tiles_.clear();
196 tiles_.push_back(gfx::Tile16(gfx::WordToTileInfo(w0), gfx::WordToTileInfo(w1),
197 gfx::WordToTileInfo(w2), gfx::WordToTileInfo(w3)));
198 tile_count_ = 1;
199 tiles_loaded_ = true;

255int RoomObject::DetermineObjectType(uint8_t b1, uint8_t b3) {
256 // Type 3: Objects with ID >= 0xF00
257 // These have b3 >= 0xF8 (top nibble is 0xF)
258 if (b3 >= 0xF8) {
259 return 3;
260 }
261
262 // Type 2: Objects with ID >= 0x100
263 // These have b1 >= 0xFC (marker for Type2 encoding)
264 if (b1 >= 0xFC) {
265 return 2;
266 }
267
268 // Type 1: Standard objects (ID 0x00-0xFF)
269 return 1;
270}

272RoomObject RoomObject::DecodeObjectFromBytes(uint8_t b1, uint8_t b2, uint8_t b3,
273 uint8_t layer) {
274 uint8_t x = 0;
275 uint8_t y = 0;
276 uint8_t size = 0;
277 uint16_t id = 0;
278
279 // ZScream's approach: Check Type3 first, then decode as Type1,
280 // then override with Type2 if b1 >= 0xFC
281 // This is critical because Type1 objects can have b1 >= 0xFC when X is at max position
282
283 if (b3 >= 0xF8) {
284 // Type3: xxxxxxii yyyyyyii 11111iii
285 // X position: bits 2-7 of byte 1
286 x = (b1 & 0xFC) >> 2;
287
288 // Y position: bits 2-7 of byte 2
289 y = (b2 & 0xFC) >> 2;
290
291 // Size: Stored in same bits as ID lower bits
292 size = ((b1 & 0x03) << 2) | (b2 & 0x03);
293
294 // ID: Complex reconstruction (ZScream formula)
295 // Top 8 bits from byte 3 (shifted left by 4)
296 // OR'd with (0x80 + lower bits from b2 and b1)
297 id = ((b3 & 0xFF) << 4) | (0x80 + (((b2 & 0x03) << 2) + (b1 & 0x03)));
298 } else {
299 // Default decode as Type1: xxxxxxss yyyyyyss iiiiiiii
300 // X position: bits 2-7 of byte 1
301 x = (b1 & 0xFC) >> 2;
302
303 // Y position: bits 2-7 of byte 2
304 y = (b2 & 0xFC) >> 2;
305
306 // Size: bits 0-1 of byte 1 (high), bits 0-1 of byte 2 (low)
307 size = ((b1 & 0x03) << 2) | (b2 & 0x03);
308
309 // ID: byte 3 (0x00-0xFF)
310 id = b3;
311
312 // NOW check if this is actually Type2 and override
313 if (b1 >= 0xFC) {
314 // Type2: 111111xx xxxxyyyy yyiiiiii
315 // X position: bits 0-1 of byte 1 (high), bits 4-7 of byte 2 (low)
316 x = ((b1 & 0x03) << 4) | ((b2 & 0xF0) >> 4);
317
318 // Y position: bits 0-3 of byte 2 (high), bits 6-7 of byte 3 (low)
319 y = ((b2 & 0x0F) << 2) | ((b3 & 0xC0) >> 6);
320
321 // Size: 0 (Type2 objects don't use size parameter)
322 size = 0;
323
324 // ID: bits 0-5 of byte 3, OR with 0x100 to mark as Type2
325 id = (b3 & 0x3F) | 0x100;
326 }
327 }
328
329 return RoomObject(static_cast<int16_t>(id), x, y, size, layer);
330}

332RoomObject::ObjectBytes RoomObject::EncodeObjectToBytes() const {
333 ObjectBytes bytes;
334
335 // Determine type based on object ID
336 if (id_ >= 0xF00) {
337 // Type 3: xxxxxxii yyyyyyii 11111iii
338 bytes.b1 = (x_ << 2) | (id_ & 0x03);
339 bytes.b2 = (y_ << 2) | ((id_ >> 2) & 0x03);
340 bytes.b3 = (id_ >> 4) & 0xFF;
341 } else if (id_ >= 0x100) {
342 // Type 2: 111111xx xxxxyyyy yyiiiiii
343 bytes.b1 = 0xFC | ((x_ & 0x30) >> 4);
344 bytes.b2 = ((x_ & 0x0F) << 4) | ((y_ & 0x3C) >> 2);
345 bytes.b3 = ((y_ & 0x03) << 6) | (id_ & 0x3F);
346 } else {
347 // Type 1: xxxxxxss yyyyyyss iiiiiiii
348 // Clamp size to 0-15 range
349 uint8_t clamped_size = size_ > 15 ? 0 : size_;
350
351 bytes.b1 = (x_ << 2) | ((clamped_size >> 2) & 0x03);
352 bytes.b2 = (y_ << 2) | (clamped_size & 0x03);
353 bytes.b3 = static_cast<uint8_t>(id_);
354 }
355
356 return bytes;
357}
