126std::vector<uint8_t> ManualObjectRenderer::CreatePatternTile(uint8_t pattern_type,
127 uint8_t color_index) {
128 std::vector<uint8_t> tile_data(16 * 16);
129
130 switch (pattern_type) {
131 case 0: // Solid
132 std::fill(tile_data.begin(), tile_data.end(), color_index);
133 break;
134
135 case 1: // Checkerboard
136 for (int y = 0; y < 16; y++) {
137 for (int x = 0; x < 16; x++) {
138 tile_data[y * 16 + x] = ((x + y) % 2) ? color_index : 0;
139 }
140 }
141 break;
142
143 case 2: // Horizontal stripes
144 for (int y = 0; y < 16; y++) {
145 uint8_t color = (y % 4 < 2) ? color_index : 0;
146 for (int x = 0; x < 16; x++) {
147 tile_data[y * 16 + x] = color;
148 }
149 }
150 break;
151
152 case 3: // Vertical stripes
153 for (int x = 0; x < 16; x++) {
154 uint8_t color = (x % 4 < 2) ? color_index : 0;
155 for (int y = 0; y < 16; y++) {
156 tile_data[y * 16 + x] = color;
157 }
158 }
159 break;
160
161 default:
162 std::fill(tile_data.begin(), tile_data.end(), color_index);
163 break;
164 }
165
166 return tile_data;
