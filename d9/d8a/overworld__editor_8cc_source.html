506void OverworldEditor::DrawOverworldMaps() {
507 int xx = 0;
508 int yy = 0;
509 for (int i = 0; i < 0x40; i++) {
510 int world_index = i + (current_world_ * 0x40);
511
512 // Bounds checking to prevent crashes
513 if (world_index < 0 || world_index >= static_cast<int>(maps_bmp_.size())) {
514 continue; // Skip invalid map index
515 }
516
517 int scale = static_cast<int>(ow_map_canvas_.global_scale());
518 int map_x = (xx * kOverworldMapSize * scale);
519 int map_y = (yy * kOverworldMapSize * scale);
520
521 // Check if the map has a texture, if not, ensure it gets loaded
522 if (!maps_bmp_[world_index].texture() &&
523 maps_bmp_[world_index].is_active()) {
524 EnsureMapTexture(world_index);
525 }
526
527 // Only draw if the map has a texture or is the currently selected map
528 if (maps_bmp_[world_index].texture() || world_index == current_map_) {
529 ow_map_canvas_.DrawBitmap(maps_bmp_[world_index], map_x, map_y,
530 ow_map_canvas_.global_scale());
531 } else {
532 // Draw a placeholder for maps that haven't loaded yet
533 ImDrawList* draw_list = ImGui::GetWindowDrawList();
534 ImVec2 canvas_pos = ow_map_canvas_.zero_point();
535 ImVec2 placeholder_pos =
536 ImVec2(canvas_pos.x + map_x, canvas_pos.y + map_y);
537 ImVec2 placeholder_size =
538 ImVec2(kOverworldMapSize * scale, kOverworldMapSize * scale);
539
540 // Modern loading indicator with theme colors
541 draw_list->AddRectFilled(
542 placeholder_pos,
543 ImVec2(placeholder_pos.x + placeholder_size.x,
544 placeholder_pos.y + placeholder_size.y),
545 IM_COL32(32, 32, 32, 128)); // Dark gray with transparency
546
547 // Animated loading spinner
548 ImVec2 spinner_pos = ImVec2(
549 placeholder_pos.x + placeholder_size.x / 2,
550 placeholder_pos.y + placeholder_size.y / 2
551 );
552
553 const float spinner_radius = 8.0f;
554 const float rotation = static_cast<float>(ImGui::GetTime()) * 3.0f;
555 const float start_angle = rotation;
556 const float end_angle = rotation + IM_PI * 1.5f;
557
558 draw_list->PathArcTo(spinner_pos, spinner_radius, start_angle, end_angle, 12);
559 draw_list->PathStroke(IM_COL32(100, 180, 100, 255), 0, 2.5f);
560 }
561
562 xx++;
563 if (xx >= 8) {
564 yy++;
565 xx = 0;
566 }
567 }

570void OverworldEditor::DrawOverworldEdits() {
571 // Determine which overworld map the user is currently editing.
572 auto mouse_position = ow_map_canvas_.drawn_tile_position();
573
574 int map_x = mouse_position.x / kOverworldMapSize;
575 int map_y = mouse_position.y / kOverworldMapSize;
576 current_map_ = map_x + map_y * 8;
577 if (current_world_ == 1) {
578 current_map_ += 0x40;
579 } else if (current_world_ == 2) {
580 current_map_ += 0x80;
581 }
582
583 // Bounds checking to prevent crashes
584 if (current_map_ < 0 || current_map_ >= static_cast<int>(maps_bmp_.size())) {
585 return; // Invalid map index, skip drawing
586 }
587
588 // Validate tile16_blockset_ before calling GetTilemapData
589 if (!tile16_blockset_.atlas.is_active() ||
590 tile16_blockset_.atlas.vector().empty()) {
591 LOG_ERROR("OverworldEditor",
592 "Error: tile16_blockset_ is not properly initialized (active: %s, "
593 "size: %zu)",
594 tile16_blockset_.atlas.is_active() ? "true" : "false",
595 tile16_blockset_.atlas.vector().size());
596 return; // Skip drawing if blockset is invalid
597 }
598
599 // Render the updated map bitmap.
600 auto tile_data = gfx::GetTilemapData(tile16_blockset_, current_tile16_);
601 RenderUpdatedMapBitmap(mouse_position, tile_data);
602
603 // Calculate the correct superX and superY values
604 int superY = current_map_ / 8;
605 int superX = current_map_ % 8;
606 int mouse_x = mouse_position.x;
607 int mouse_y = mouse_position.y;
608 // Calculate the correct tile16_x and tile16_y positions
609 int tile16_x = (mouse_x % kOverworldMapSize) / (kOverworldMapSize / 32);
610 int tile16_y = (mouse_y % kOverworldMapSize) / (kOverworldMapSize / 32);
611
612 // Update the overworld_.map_tiles() based on tile16 ID and current world
613 auto& selected_world =
614 (current_world_ == 0) ? overworld_.mutable_map_tiles()->light_world
615 : (current_world_ == 1) ? overworld_.mutable_map_tiles()->dark_world
616 : overworld_.mutable_map_tiles()->special_world;
617
618 int index_x = superX * 32 + tile16_x;
619 int index_y = superY * 32 + tile16_y;
620
621 selected_world[index_x][index_y] = current_tile16_;
622}

632 current_map_, maps_bmp_.size());
633 return; // Invalid map index, skip rendering
634 }
635
636 // Calculate the tile index for x and y based on the click_position
637 int tile_index_x =
638 (static_cast<int>(click_position.x) % kOverworldMapSize) / kTile16Size;
639 int tile_index_y =
640 (static_cast<int>(click_position.y) % kOverworldMapSize) / kTile16Size;
641
642 // Calculate the pixel start position based on tile index and tile size
643 ImVec2 start_position;
644 start_position.x = static_cast<float>(tile_index_x * kTile16Size);
645 start_position.y = static_cast<float>(tile_index_y * kTile16Size);
646
647 // Update the bitmap's pixel data based on the start_position and tile_data
648 gfx::Bitmap& current_bitmap = maps_bmp_[current_map_];
649
650 // Validate bitmap state before writing
651 if (!current_bitmap.is_active() || current_bitmap.size() == 0) {
652 LOG_ERROR("OverworldEditor",
653 "ERROR: RenderUpdatedMapBitmap - Bitmap %d is not active or has no "
654 "data (active=%s, size=%zu)",
655 current_map_, current_bitmap.is_active() ? "true" : "false",
656 current_bitmap.size());
657 return;
658 }
659
660 for (int y = 0; y < kTile16Size; ++y) {
661 for (int x = 0; x < kTile16Size; ++x) {
662 int pixel_index =
663 (start_position.y + y) * kOverworldMapSize + (start_position.x + x);
664
665 // Bounds check for pixel index
666 if (pixel_index < 0 ||
667 pixel_index >= static_cast<int>(current_bitmap.size())) {
668 LOG_ERROR("OverworldEditor",
669 "ERROR: RenderUpdatedMapBitmap - pixel_index %d out of bounds "
670 "(bitmap size=%zu)",
671 pixel_index, current_bitmap.size());
672 continue;
673 }
674
675 // Bounds check for tile data
676 int tile_data_index = y * kTile16Size + x;
677 if (tile_data_index < 0 ||
678 tile_data_index >= static_cast<int>(tile_data.size())) {
679 LOG_ERROR("OverworldEditor",
680 "ERROR: RenderUpdatedMapBitmap - tile_data_index %d out of bounds "
681 "(tile_data size=%zu)",
682 tile_data_index, tile_data.size());
683 continue;
684 }
685
686 current_bitmap.WriteToPixel(pixel_index, tile_data[tile_data_index]);
687 }
688 }
689
690 current_bitmap.set_modified(true);
691
692 // Immediately update the texture to reflect changes
693 core::Renderer::Get().UpdateBitmap(&current_bitmap);
694}

946absl::Status OverworldEditor::CheckForCurrentMap() {

970 RETURN_IF_ERROR(overworld_.EnsureMapBuilt(current_map_));

972
973 const int current_highlighted_map = current_map_;
974
975 // Check if ZSCustomOverworld v3 is present
976 uint8_t asm_version = (*rom_)[zelda3::OverworldCustomASMHasBeenApplied];
977 bool use_v3_area_sizes = (asm_version >= 3);
978
979 // Get area size for v3+ ROMs, otherwise use legacy logic
980 if (use_v3_area_sizes) {

992 parent_map_y = highlight_parent / 8;

994 // Dark World (0x40-0x7F)
995 parent_map_x = (highlight_parent - 0x40) % 8;
996 parent_map_y = (highlight_parent - 0x40) / 8;

999 parent_map_x = (highlight_parent - 0x80) % 8;
1000 parent_map_y = (highlight_parent - 0x80) / 8;

1004 switch (area_size) {
1005 case AreaSizeEnum::LargeArea:

1007 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1008 parent_map_y * kOverworldMapSize,
1009 large_map_size, large_map_size);
1010 break;
1011 case AreaSizeEnum::WideArea:
1012 // 2x1 grid (1024x512) - horizontal
1013 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1014 parent_map_y * kOverworldMapSize,
1015 large_map_size, kOverworldMapSize);
1016 break;
1017 case AreaSizeEnum::TallArea:

1019 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1020 parent_map_y * kOverworldMapSize,
1021 kOverworldMapSize, large_map_size);
1022 break;
1023 case AreaSizeEnum::SmallArea:
1024 default:
1025 // 1x1 grid (512x512)
1026 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1027 parent_map_y * kOverworldMapSize,
1028 kOverworldMapSize, kOverworldMapSize);
1029 break;
1030 }
1031 } else {
1032 // Legacy logic for vanilla and v2 ROMs
1033 if (overworld_.overworld_map(current_map_)->is_large_map() ||
1034 overworld_.overworld_map(current_map_)->large_index() != 0) {
1035 const int highlight_parent =
1036 overworld_.overworld_map(current_highlighted_map)->parent();
1037 const int parent_map_x = highlight_parent % 8;
1038 const int parent_map_y = highlight_parent / 8;
1039 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1040 parent_map_y * kOverworldMapSize,
1041 large_map_size, large_map_size);
1042 } else {
1043 // Calculate map coordinates accounting for world offset
1044 int current_map_x;
1045 int current_map_y;
1046 if (current_world_ == 0) {
1047 // Light World (0x00-0x3F)
1048 current_map_x = current_highlighted_map % 8;
1049 current_map_y = current_highlighted_map / 8;
1050 } else if (current_world_ == 1) {
1051 // Dark World (0x40-0x7F)
1052 current_map_x = (current_highlighted_map - 0x40) % 8;
1053 current_map_y = (current_highlighted_map - 0x40) / 8;
1054 } else {
1055 // Special World (0x80-0x9F) - use display coordinates based on current_world_
1056 // The special world maps are displayed in the same 8x8 grid as LW/DW
1057 current_map_x = (current_highlighted_map - 0x80) % 8;
1058 current_map_y = (current_highlighted_map - 0x80) / 8;
1059 }
1060 ow_map_canvas_.DrawOutline(current_map_x * kOverworldMapSize,
1061 current_map_y * kOverworldMapSize,
1062 kOverworldMapSize, kOverworldMapSize);
1063 }
1064 }
1065
1066 // Ensure current map has texture created for rendering
1067 EnsureMapTexture(current_map_);
1068
1069 if (maps_bmp_[current_map_].modified()) {
1070 RefreshOverworldMap();
1071 RETURN_IF_ERROR(RefreshTile16Blockset());
1072
1073 // Ensure tile16 blockset is fully updated before rendering
1074 if (tile16_blockset_.atlas.is_active()) {
1075 Renderer::Get().UpdateBitmap(&tile16_blockset_.atlas);
1076 }
1077
1078 // Update map texture with the traditional direct update approach
1079 Renderer::Get().UpdateBitmap(&maps_bmp_[current_map_]);
1080 maps_bmp_[current_map_].set_modified(false);
1081 }
1082
1083 if (ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
1084 RETURN_IF_ERROR(RefreshTile16Blockset());
1085 }
1086
1087 // If double clicked, toggle the current map
1088 if (ImGui::IsMouseDoubleClicked(ImGuiMouseButton_Right)) {
1089 current_map_lock_ = !current_map_lock_;
1090 }
1091
1092 return absl::OkStatus();
1093}

1491 constexpr int kDarkWorldEssential =

1535 }

1792void OverworldEditor::RefreshMultiAreaMapsSafely(int map_index,
1793 zelda3::OverworldMap* map) {
1794 using zelda3::AreaSizeEnum;
1795
1796 // Skip if this is already a processed sibling to avoid double-processing
1797 static std::set<int> currently_processing;
1798 if (currently_processing.count(map_index)) {
1799 return;
1800 }
1801
1802 auto area_size = map->area_size();
1803 if (area_size == AreaSizeEnum::SmallArea) {
1804 return; // No siblings to coordinate
1805 }
1806
1807 LOG_DEBUG("OverworldEditor",
1808 "RefreshMultiAreaMapsSafely: Processing %s area map %d (parent: %d)",
1809 (area_size == AreaSizeEnum::LargeArea) ? "large"
1810 : (area_size == AreaSizeEnum::WideArea) ? "wide"
1811 : "tall",
1812 map_index, map->parent());
1813
1814 // Determine all maps that are part of this multi-area structure
1815 std::vector<int> sibling_maps;
1816 int parent_id = map->parent();
1817
1818 // Use the same logic as ZScream for area coordination
1819 switch (area_size) {
1820 case AreaSizeEnum::LargeArea: {
1821 // Large Area: 2x2 grid (4 maps total)
1822 // Parent is top-left (quadrant 0), siblings are:
1823 // +1 (top-right, quadrant 1), +8 (bottom-left, quadrant 2), +9 (bottom-right, quadrant 3)
1824 sibling_maps = {parent_id, parent_id + 1, parent_id + 8, parent_id + 9};
1825 LOG_DEBUG("OverworldEditor",
1826 "RefreshMultiAreaMapsSafely: Large area siblings: %d, %d, %d, %d",
1827 parent_id, parent_id + 1, parent_id + 8, parent_id + 9);
1828 break;
1829 }
1830
1831 case AreaSizeEnum::WideArea: {
1832 // Wide Area: 2x1 grid (2 maps total, horizontally adjacent)
1833 // Parent is left, sibling is +1 (right)
1834 sibling_maps = {parent_id, parent_id + 1};
1835 LOG_DEBUG("OverworldEditor",
1836 "RefreshMultiAreaMapsSafely: Wide area siblings: %d, %d",
1837 parent_id, parent_id + 1);
1838 break;
1839 }
1840
1841 case AreaSizeEnum::TallArea: {
1842 // Tall Area: 1x2 grid (2 maps total, vertically adjacent)
1843 // Parent is top, sibling is +8 (bottom)
1844 sibling_maps = {parent_id, parent_id + 8};
1845 LOG_DEBUG("OverworldEditor",
1846 "RefreshMultiAreaMapsSafely: Tall area siblings: %d, %d",
1847 parent_id, parent_id + 8);
1848 break;
1849 }
1850
1851 default:
1852 LOG_WARN("OverworldEditor",
1853 "RefreshMultiAreaMapsSafely: Unknown area size %d for map %d",
1854 static_cast<int>(area_size), map_index);
1855 return;
1856 }
1857
1858 // Mark all siblings as being processed to prevent recursion
1859 for (int sibling : sibling_maps) {
1860 currently_processing.insert(sibling);
1861 }
1862
1863 // Only refresh siblings that are visible/current and need updating
1864 for (int sibling : sibling_maps) {
1865 if (sibling == map_index) {
1866 continue; // Skip self (already processed above)
1867 }
1868
1869 // Bounds check
1870 if (sibling < 0 || sibling >= zelda3::kNumOverworldMaps) {
1871 continue;
1872 }
1873
1874 // Only refresh if it's visible or current
1875 bool is_current_map = (sibling == current_map_);
1876 bool is_current_world = (sibling / 0x40 == current_world_);
1877 bool needs_refresh = maps_bmp_[sibling].modified();
1878
1879 if ((is_current_map || is_current_world) && needs_refresh) {
1880 LOG_DEBUG("OverworldEditor",
1881 "RefreshMultiAreaMapsSafely: Refreshing %s area sibling map %d "
1882 "(parent: %d)",
1883 (area_size == AreaSizeEnum::LargeArea) ? "large"
1884 : (area_size == AreaSizeEnum::WideArea) ? "wide"
1885 : "tall",
1886 sibling, parent_id);
1887
1888 // Direct refresh without calling RefreshChildMapOnDemand to avoid recursion
1889 auto* sibling_map = overworld_.mutable_overworld_map(sibling);
1890 if (sibling_map && maps_bmp_[sibling].modified()) {
1891 sibling_map->LoadAreaGraphics();
1892
1893 auto status = sibling_map->BuildTileset();
1894 if (status.ok()) {
1895 status = sibling_map->BuildTiles16Gfx(*overworld_.mutable_tiles16(),
1896 overworld_.tiles16().size());
1897 if (status.ok()) {
1898 // Load palette for the sibling map
1899 status = sibling_map->LoadPalette();
1900 if (status.ok()) {
1901 status = sibling_map->BuildBitmap(
1902 overworld_.GetMapTiles(current_world_));
1903 if (status.ok()) {
1904 maps_bmp_[sibling].set_data(sibling_map->bitmap_data());
1905
1906 // SAFETY: Only set palette if bitmap has a valid surface
1907 if (maps_bmp_[sibling].is_active() && maps_bmp_[sibling].surface()) {
1908 maps_bmp_[sibling].SetPalette(overworld_.current_area_palette());
1909 }
1910 maps_bmp_[sibling].set_modified(false);
1911
1912 // Update texture if it exists
1913 if (maps_bmp_[sibling].texture()) {
1914 core::Renderer::Get().UpdateBitmap(&maps_bmp_[sibling]);
1915 } else {
1916 EnsureMapTexture(sibling);
1917 }
1918 }
1919 }
1920 }
1921 }
1922
1923 if (!status.ok()) {
1924 LOG_ERROR("OverworldEditor",
1925 "RefreshMultiAreaMapsSafely: Failed to refresh sibling map %d: "
1926 "%s",
1927 sibling, status.message().data());
1928 }
1929 }
1930 } else if (!is_current_map && !is_current_world) {
1931 // Mark non-visible siblings for deferred refresh
1932 maps_bmp_[sibling].set_modified(true);
1933 }
1934 }
1935
1936 // Clear processing set after completion
1937 for (int sibling : sibling_maps) {
1938 currently_processing.erase(sibling);
1939 }
1940}

2482absl::Status OverworldEditor::ApplyZSCustomOverworldASM(int target_version) {
2483 // Feature flag deprecated - ROM version gating is sufficient
2484 // User explicitly clicked upgrade button, so respect their request
2485
2486 // Validate target version
2487 if (target_version < 2 || target_version > 3) {
2488 return absl::InvalidArgumentError(absl::StrFormat(
2489 "Invalid target version: %d. Must be 2 or 3.", target_version));
2490 }
2491
2492 // Check current ROM version
2493 uint8_t current_version = (*rom_)[zelda3::OverworldCustomASMHasBeenApplied];
2494 if (current_version != 0xFF && current_version >= target_version) {
2495 return absl::AlreadyExistsError(absl::StrFormat(
2496 "ROM is already version %d or higher", current_version));
2497 }
2498
2499 LOG_INFO("OverworldEditor", "Applying ZSCustomOverworld ASM v%d to ROM...",
2500 target_version);
2501
2502 // Initialize Asar wrapper
2503 auto asar_wrapper = std::make_unique<core::AsarWrapper>();
2504 RETURN_IF_ERROR(asar_wrapper->Initialize());
2505
2506 // Create backup of ROM data
2507 std::vector<uint8_t> original_rom_data = rom_->vector();
2508 std::vector<uint8_t> working_rom_data = original_rom_data;
2509
2510 try {
2511 // Determine which ASM file to apply and use GetResourcePath for proper resolution
2512 std::string asm_file_name = (target_version == 3)
2513 ? "asm/yaze.asm" // Master file with v3
2514 : "asm/ZSCustomOverworld.asm"; // v2 standalone
2515
2516 // Use GetResourcePath to handle app bundles and various deployment scenarios
2517 std::string asm_file_path = util::GetResourcePath(asm_file_name);
2518
2519 LOG_INFO("OverworldEditor", "Using ASM file: %s", asm_file_path.c_str());
2520
2521 // Verify file exists
2522 if (!std::filesystem::exists(asm_file_path)) {
2523 return absl::NotFoundError(absl::StrFormat(
2524 "ASM file not found at: %s\n\n"
2525 "Expected location: assets/%s\n"
2526 "Make sure the assets directory is accessible.",
2527 asm_file_path, asm_file_name));
2528 }
2529
2530 // Apply the ASM patch
2531 auto patch_result =
2532 asar_wrapper->ApplyPatch(asm_file_path, working_rom_data);
2533 if (!patch_result.ok()) {
2534 return absl::InternalError(absl::StrFormat(
2535 "Failed to apply ASM patch: %s", patch_result.status().message()));
2536 }
2537
2538 const auto& result = patch_result.value();
2539 if (!result.success) {
2540 std::string error_details = "ASM patch failed with errors:\n";
2541 for (const auto& error : result.errors) {
2542 error_details += " - " + error + "\n";
2543 }
2544 if (!result.warnings.empty()) {
2545 error_details += "Warnings:\n";
2546 for (const auto& warning : result.warnings) {
2547 error_details += " - " + warning + "\n";
2548 }
2549 }
2550 return absl::InternalError(error_details);
2551 }
2552
2553 // Update ROM with patched data
2554 RETURN_IF_ERROR(rom_->LoadFromData(working_rom_data, false));
2555
2556 // Update version marker and feature flags
2557 RETURN_IF_ERROR(UpdateROMVersionMarkers(target_version));
2558
2559 // Log symbols found during patching
2560 LOG_INFO("OverworldEditor", "ASM patch applied successfully. Found %zu symbols:",
2561 result.symbols.size());
2562 for (const auto& symbol : result.symbols) {
2563 LOG_INFO("OverworldEditor", " %s @ $%06X", symbol.name.c_str(),
2564 symbol.address);
2565 }
2566
2567 // Refresh overworld data to reflect changes
2568 RETURN_IF_ERROR(overworld_.Load(rom_));
2569
2570 LOG_INFO("OverworldEditor", "ZSCustomOverworld v%d successfully applied to ROM",
2571 target_version);
2572 return absl::OkStatus();
2573
2574 } catch (const std::exception& e) {
2575 // Restore original ROM data on any exception
2576 auto restore_result = rom_->LoadFromData(original_rom_data, false);
2577 if (!restore_result.ok()) {
2578 LOG_ERROR("OverworldEditor", "Failed to restore ROM data: %s",
2579 restore_result.message().data());
2580 }
2581 return absl::InternalError(
2582 absl::StrFormat("Exception during ASM application: %s", e.what()));
2583 }
