882void OverworldEditor::RenderUpdatedMapBitmap(
883 const ImVec2& click_position, const std::vector<uint8_t>& tile_data) {
884
885 // Bounds checking to prevent crashes
886 if (current_map_ < 0 || current_map_ >= static_cast<int>(maps_bmp_.size())) {
887 LOG_ERROR("OverworldEditor",
888 "ERROR: RenderUpdatedMapBitmap - Invalid current_map_ %d "
889 "(maps_bmp_.size()=%zu)",
890 current_map_, maps_bmp_.size());
891 return; // Invalid map index, skip rendering
892 }
893
894 // Calculate the tile index for x and y based on the click_position
895 int tile_index_x =
896 (static_cast<int>(click_position.x) % kOverworldMapSize) / kTile16Size;
897 int tile_index_y =
898 (static_cast<int>(click_position.y) % kOverworldMapSize) / kTile16Size;
899
900 // Calculate the pixel start position based on tile index and tile size
901 ImVec2 start_position;
902 start_position.x = static_cast<float>(tile_index_x * kTile16Size);
903 start_position.y = static_cast<float>(tile_index_y * kTile16Size);
904
905 // Update the bitmap's pixel data based on the start_position and tile_data
906 gfx::Bitmap& current_bitmap = maps_bmp_[current_map_];
907
908 // Validate bitmap state before writing
909 if (!current_bitmap.is_active() || current_bitmap.size() == 0) {
910 LOG_ERROR("OverworldEditor",
911 "ERROR: RenderUpdatedMapBitmap - Bitmap %d is not active or has no "
912 "data (active=%s, size=%zu)",
913 current_map_, current_bitmap.is_active() ? "true" : "false",
914 current_bitmap.size());
915 return;
916 }
917
918 for (int y = 0; y < kTile16Size; ++y) {
919 for (int x = 0; x < kTile16Size; ++x) {
920 int pixel_index =
921 (start_position.y + y) * kOverworldMapSize + (start_position.x + x);
922
923 // Bounds check for pixel index
924 if (pixel_index < 0 ||
925 pixel_index >= static_cast<int>(current_bitmap.size())) {
926 LOG_ERROR("OverworldEditor",
927 "ERROR: RenderUpdatedMapBitmap - pixel_index %d out of bounds "
928 "(bitmap size=%zu)",
929 pixel_index, current_bitmap.size());
930 continue;
931 }
932
933 // Bounds check for tile data
934 int tile_data_index = y * kTile16Size + x;
935 if (tile_data_index < 0 ||
936 tile_data_index >= static_cast<int>(tile_data.size())) {
937 LOG_ERROR("OverworldEditor",
938 "ERROR: RenderUpdatedMapBitmap - tile_data_index %d out of bounds "
939 "(tile_data size=%zu)",
940 tile_data_index, tile_data.size());
941 continue;
942 }
943
944 current_bitmap.WriteToPixel(pixel_index, tile_data[tile_data_index]);
945 }
946 }
947
948 current_bitmap.set_modified(true);
949
950 // Immediately update the texture to reflect changes
951 core::Renderer::Get().UpdateBitmap(&current_bitmap);
952}

1235

1238 using zelda3::AreaSizeEnum;
1239 auto area_size = overworld_.overworld_map(current_map_)->area_size();

1241 overworld_.overworld_map(current_highlighted_map)->parent();

1244 int parent_map_x;

1248 parent_map_x = highlight_parent % 8;
1249 parent_map_y = highlight_parent / 8;

1251 // Dark World (0x40-0x7F)
1252 parent_map_x = (highlight_parent - 0x40) % 8;
1253 parent_map_y = (highlight_parent - 0x40) / 8;
1254 } else {
1255 // Special World (0x80-0x9F)
1256 parent_map_x = (highlight_parent - 0x80) % 8;
1257 parent_map_y = (highlight_parent - 0x80) / 8;
1258 }
1259
1260 // Draw outline based on area size
1261 switch (area_size) {
1262 case AreaSizeEnum::LargeArea:
1263 // 2x2 grid (1024x1024)
1264 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1265 parent_map_y * kOverworldMapSize,
1266 large_map_size, large_map_size);
1267 break;
1268 case AreaSizeEnum::WideArea:
1269 // 2x1 grid (1024x512) - horizontal
1270 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1271 parent_map_y * kOverworldMapSize,
1272 large_map_size, kOverworldMapSize);
1273 break;
1274 case AreaSizeEnum::TallArea:
1275 // 1x2 grid (512x1024) - vertical
1276 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1277 parent_map_y * kOverworldMapSize,
1278 kOverworldMapSize, large_map_size);
1279 break;
1280 case AreaSizeEnum::SmallArea:
1281 default:
1282 // 1x1 grid (512x512)
1283 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1284 parent_map_y * kOverworldMapSize,
1285 kOverworldMapSize, kOverworldMapSize);
1286 break;
1287 }
1288 } else {
1289 // Legacy logic for vanilla and v2 ROMs
1290 if (overworld_.overworld_map(current_map_)->is_large_map() ||
1291 overworld_.overworld_map(current_map_)->large_index() != 0) {
1292 const int highlight_parent =
1293 overworld_.overworld_map(current_highlighted_map)->parent();
1294 const int parent_map_x = highlight_parent % 8;
1295 const int parent_map_y = highlight_parent / 8;
1296 ow_map_canvas_.DrawOutline(parent_map_x * kOverworldMapSize,
1297 parent_map_y * kOverworldMapSize,
1298 large_map_size, large_map_size);
1299 } else {
1300 // Calculate map coordinates accounting for world offset
1301 int current_map_x;
1302 int current_map_y;
1303 if (current_world_ == 0) {
1304 // Light World (0x00-0x3F)
1305 current_map_x = current_highlighted_map % 8;
1306 current_map_y = current_highlighted_map / 8;
1307 } else if (current_world_ == 1) {
1308 // Dark World (0x40-0x7F)
1309 current_map_x = (current_highlighted_map - 0x40) % 8;
1310 current_map_y = (current_highlighted_map - 0x40) / 8;
1311 } else {
1312 // Special World (0x80-0x9F) - use display coordinates based on current_world_
1313 // The special world maps are displayed in the same 8x8 grid as LW/DW
1314 current_map_x = (current_highlighted_map - 0x80) % 8;
1315 current_map_y = (current_highlighted_map - 0x80) / 8;
1316 }

1318 current_map_y * kOverworldMapSize,
1319 kOverworldMapSize, kOverworldMapSize);
1320 }
1321 }

1966void OverworldEditor::ProcessDeferredTextures() {
1967 std::lock_guard<std::mutex> lock(deferred_textures_mutex_);
1968
1969 if (deferred_map_textures_.empty()) {
1970 return;
1971 }
1972
1973 // Priority-based loading: process more textures for visible maps
1974 const int textures_per_frame = 8; // Increased from 2 to 8 for faster loading
1975 int processed = 0;
1976
1977 // First pass: prioritize textures for the current world
1978 auto it = deferred_map_textures_.begin();
1979 while (it != deferred_map_textures_.end() && processed < textures_per_frame) {
1980 if (*it && !(*it)->texture()) {
1981 // Check if this texture belongs to the current world
1982 int map_index = -1;
1983 for (int i = 0; i < zelda3::kNumOverworldMaps; ++i) {
1984 if (&maps_bmp_[i] == *it) {
1985 map_index = i;
1986 break;
1987 }
1988 }
1989
1990 bool is_current_world = false;
1991 if (map_index >= 0) {
1992 int map_world = map_index / 0x40; // 64 maps per world
1993 is_current_world = (map_world == current_world_);
1994 }
1995
1996 // Prioritize current world maps, but also process others if we have capacity
1997 if (is_current_world || processed < textures_per_frame / 2) {
1998 Renderer::Get().RenderBitmap(*it);
1999 processed++;
2000 it = deferred_map_textures_.erase(
2001 it); // Remove immediately after processing
2002 } else {
2003 ++it;
2004 }
2005 } else {
2006 ++it;
2007 }
2008 }
2009
2010 // Second pass: process remaining textures if we still have capacity
2011 if (processed < textures_per_frame) {
2012 it = deferred_map_textures_.begin();
2013 while (it != deferred_map_textures_.end() &&
2014 processed < textures_per_frame) {
2015 if (*it && !(*it)->texture()) {
2016 Renderer::Get().RenderBitmap(*it);
2017 processed++;
2018 it = deferred_map_textures_.erase(it);
2019 } else {
2020 ++it;
2021 }
2022 }
2023 }
2024
2025 // Third pass: process deferred map refreshes for visible maps
2026 if (processed < textures_per_frame) {
2027 for (int i = 0;
2028 i < zelda3::kNumOverworldMaps && processed < textures_per_frame; ++i) {
2029 if (maps_bmp_[i].modified() && maps_bmp_[i].is_active()) {
2030 // Check if this map is visible
2031 bool is_visible = (i == current_map_) || (i / 0x40 == current_world_);
2032 if (is_visible) {
2033 RefreshOverworldMapOnDemand(i);
2034 processed++;
2035 }
2036 }
2037 }
2038 }

2212void OverworldEditor::RefreshMultiAreaMapsSafely(int map_index,
2213 zelda3::OverworldMap* map) {
2214 using zelda3::AreaSizeEnum;
2215
2216 // Skip if this is already a processed sibling to avoid double-processing
2217 static std::set<int> currently_processing;
2218 if (currently_processing.count(map_index)) {
2219 return;
2220 }
2221
2222 auto area_size = map->area_size();
2223 if (area_size == AreaSizeEnum::SmallArea) {
2224 return; // No siblings to coordinate
2225 }
2226
2227 LOG_DEBUG("OverworldEditor",
2228 "RefreshMultiAreaMapsSafely: Processing %s area map %d (parent: %d)",
2229 (area_size == AreaSizeEnum::LargeArea) ? "large"
2230 : (area_size == AreaSizeEnum::WideArea) ? "wide"
2231 : "tall",
2232 map_index, map->parent());
2233
2234 // Determine all maps that are part of this multi-area structure
2235 std::vector<int> sibling_maps;
2236 int parent_id = map->parent();
2237
2238 // Use the same logic as ZScream for area coordination
2239 switch (area_size) {
2240 case AreaSizeEnum::LargeArea: {
2241 // Large Area: 2x2 grid (4 maps total)
2242 // Parent is top-left (quadrant 0), siblings are:
2243 // +1 (top-right, quadrant 1), +8 (bottom-left, quadrant 2), +9 (bottom-right, quadrant 3)
2244 sibling_maps = {parent_id, parent_id + 1, parent_id + 8, parent_id + 9};
2245 LOG_DEBUG("OverworldEditor",
2246 "RefreshMultiAreaMapsSafely: Large area siblings: %d, %d, %d, %d",
2247 parent_id, parent_id + 1, parent_id + 8, parent_id + 9);
2248 break;
2249 }
2250
2251 case AreaSizeEnum::WideArea: {
2252 // Wide Area: 2x1 grid (2 maps total, horizontally adjacent)
2253 // Parent is left, sibling is +1 (right)
2254 sibling_maps = {parent_id, parent_id + 1};
2255 LOG_DEBUG("OverworldEditor",
2256 "RefreshMultiAreaMapsSafely: Wide area siblings: %d, %d",
2257 parent_id, parent_id + 1);
2258 break;
2259 }
2260
2261 case AreaSizeEnum::TallArea: {
2262 // Tall Area: 1x2 grid (2 maps total, vertically adjacent)
2263 // Parent is top, sibling is +8 (bottom)
2264 sibling_maps = {parent_id, parent_id + 8};
2265 LOG_DEBUG("OverworldEditor",
2266 "RefreshMultiAreaMapsSafely: Tall area siblings: %d, %d",
2267 parent_id, parent_id + 8);
2268 break;
2269 }
2270
2271 default:
2272 LOG_WARN("OverworldEditor",
2273 "RefreshMultiAreaMapsSafely: Unknown area size %d for map %d",
2274 static_cast<int>(area_size), map_index);
2275 return;
2276 }
2277
2278 // Mark all siblings as being processed to prevent recursion
2279 for (int sibling : sibling_maps) {
2280 currently_processing.insert(sibling);
2281 }
2282
2283 // Only refresh siblings that are visible/current and need updating
2284 for (int sibling : sibling_maps) {
2285 if (sibling == map_index) {
2286 continue; // Skip self (already processed above)
2287 }
2288
2289 // Bounds check
2290 if (sibling < 0 || sibling >= zelda3::kNumOverworldMaps) {
2291 continue;
2292 }
2293
2294 // Only refresh if it's visible or current
2295 bool is_current_map = (sibling == current_map_);
2296 bool is_current_world = (sibling / 0x40 == current_world_);
2297 bool needs_refresh = maps_bmp_[sibling].modified();
2298
2299 if ((is_current_map || is_current_world) && needs_refresh) {
2300 LOG_DEBUG("OverworldEditor",
2301 "RefreshMultiAreaMapsSafely: Refreshing %s area sibling map %d "
2302 "(parent: %d)",
2303 (area_size == AreaSizeEnum::LargeArea) ? "large"
2304 : (area_size == AreaSizeEnum::WideArea) ? "wide"
2305 : "tall",
2306 sibling, parent_id);
2307
2308 // Direct refresh without calling RefreshChildMapOnDemand to avoid recursion
2309 auto* sibling_map = overworld_.mutable_overworld_map(sibling);
2310 if (sibling_map && maps_bmp_[sibling].modified()) {
2311 sibling_map->LoadAreaGraphics();
2312
2313 auto status = sibling_map->BuildTileset();
2314 if (status.ok()) {
2315 status = sibling_map->BuildTiles16Gfx(*overworld_.mutable_tiles16(),
2316 overworld_.tiles16().size());
2317 if (status.ok()) {
2318 // Load palette for the sibling map
2319 status = sibling_map->LoadPalette();
2320 if (status.ok()) {
2321 status = sibling_map->BuildBitmap(
2322 overworld_.GetMapTiles(current_world_));
2323 if (status.ok()) {
2324 maps_bmp_[sibling].set_data(sibling_map->bitmap_data());
2325 maps_bmp_[sibling].SetPalette(
2326 overworld_.current_area_palette());
2327 maps_bmp_[sibling].set_modified(false);
2328
2329 // Update texture if it exists
2330 if (maps_bmp_[sibling].texture()) {
2331 core::Renderer::Get().UpdateBitmap(&maps_bmp_[sibling]);
2332 } else {
2333 EnsureMapTexture(sibling);
2334 }
2335 }
2336 }
2337 }
2338 }
2339
2340 if (!status.ok()) {
2341 LOG_ERROR("OverworldEditor",
2342 "RefreshMultiAreaMapsSafely: Failed to refresh sibling map %d: "
2343 "%s",
2344 sibling, status.message().data());
2345 }
2346 }
2347 } else if (!is_current_map && !is_current_world) {
2348 // Mark non-visible siblings for deferred refresh
2349 maps_bmp_[sibling].set_modified(true);
2350 }
2351 }
2352
2353 // Clear processing set after completion
2354 for (int sibling : sibling_maps) {
2355 currently_processing.erase(sibling);
2356 }
2357}
