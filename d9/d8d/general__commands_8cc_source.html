63absl::Status TryLoadProjectAndLabels(Rom& rom) {
64 // Try to find and load a project file in current directory
65 core::YazeProject project;
66 auto project_status = project.Open(".");

68 if (project_status.ok()) {
69 std::cout << "📂 Loaded project: " << project.name << "\n";
70
71 // Initialize embedded labels (all default Zelda3 resource names)
72 auto labels_status = project.InitializeEmbeddedLabels();
73 if (labels_status.ok()) {
74 std::cout << "✅ Embedded labels initialized (all Zelda3 resources available)\n";
75 }
76
77 // Load labels from project (either embedded or external)
78 if (!project.labels_filename.empty()) {
79 auto* label_mgr = rom.resource_label();
80 if (label_mgr && label_mgr->LoadLabels(project.labels_filename)) {
81 std::cout << "🏷️ Loaded custom labels from: " << project.labels_filename << "\n";
82 }
83 } else if (!project.resource_labels.empty() || project.use_embedded_labels) {
84 // Use labels embedded in project or default Zelda3 labels
85 auto* label_mgr = rom.resource_label();
86 if (label_mgr) {
87 label_mgr->labels_ = project.resource_labels;
88 label_mgr->labels_loaded_ = true;
89 std::cout << "🏷️ Using embedded Zelda3 labels (rooms, sprites, entrances, items, etc.)\n";
90 }
91 }
92 } else {
93 // No project found - use embedded defaults anyway
94 std::cout << "ℹ️ No project file found. Using embedded default Zelda3 labels.\n";
95 project.InitializeEmbeddedLabels();
96 }
97
98 return absl::OkStatus();
99}

125absl::StatusOr<DescribeOptions> ParseDescribeArgs(
126 const std::vector<std::string>& args) {
127 DescribeOptions options;
128 for (size_t i = 0; i < args.size(); ++i) {
129 const std::string& token = args[i];
130 std::string flag = token;
131 std::optional<std::string> inline_value;
132
133 if (absl::StartsWith(token, "--")) {
134 auto eq_pos = token.find('=');
135 if (eq_pos != std::string::npos) {
136 flag = token.substr(0, eq_pos);
137 inline_value = token.substr(eq_pos + 1);
138 }
139 }
140
141 auto require_value =
142 [&](absl::string_view flag_name) -> absl::StatusOr<std::string> {
143 if (inline_value.has_value()) {
144 return *inline_value;
145 }
146 if (i + 1 >= args.size()) {
147 return absl::InvalidArgumentError(
148 absl::StrFormat("Flag %s requires a value", flag_name));
149 }
150 return args[++i];
151 };
152
153 if (flag == "--resource") {
154 ASSIGN_OR_RETURN(auto value, require_value("--resource"));
155 options.resource = std::move(value);
156 } else if (flag == "--format") {
157 ASSIGN_OR_RETURN(auto value, require_value("--format"));
158 options.format = std::move(value);
159 } else if (flag == "--output") {
160 ASSIGN_OR_RETURN(auto value, require_value("--output"));
161 options.output_path = std::move(value);
162 } else if (flag == "--version") {
163 ASSIGN_OR_RETURN(auto value, require_value("--version"));
164 options.version = std::move(value);
165 } else if (flag == "--last-updated") {
166 ASSIGN_OR_RETURN(auto value, require_value("--last-updated"));
167 options.last_updated = std::move(value);
168 } else {
169 return absl::InvalidArgumentError(
170 absl::StrFormat("Unknown flag for agent describe: %s", token));
171 }
172 }
173
174 options.format = absl::AsciiStrToLower(options.format);
175 if (options.format != "json" && options.format != "yaml") {
176 return absl::InvalidArgumentError("--format must be either json or yaml");
177 }
178
179 return options;
180}

620absl::Status HandleDescribeCommand(const std::vector<std::string>& arg_vec) {
621 ASSIGN_OR_RETURN(auto options, ParseDescribeArgs(arg_vec));
622
623 const auto& catalog = ResourceCatalog::Instance();
624 std::optional<ResourceSchema> resource_schema;
625 if (options.resource.has_value()) {
626 auto resource_or = catalog.GetResource(*options.resource);
627 if (!resource_or.ok()) {
628 return resource_or.status();
629 }
630 resource_schema = resource_or.value();
631 }
632
633 std::string payload;
634 if (options.format == "json") {
635 if (resource_schema.has_value()) {
636 payload = catalog.SerializeResource(*resource_schema);
637 } else {
638 payload = catalog.SerializeResources(catalog.AllResources());
639 }
640 } else {
641 std::string last_updated =
642 options.last_updated.has_value()
643 ? *options.last_updated
644 : absl::FormatTime("%Y-%m-%d", absl::Now(), absl::LocalTimeZone());
645 if (resource_schema.has_value()) {
646 std::vector<ResourceSchema> schemas{*resource_schema};
647 payload = catalog.SerializeResourcesAsYaml(schemas, options.version,
648 last_updated);
649 } else {
650 payload = catalog.SerializeResourcesAsYaml(catalog.AllResources(),
651 options.version, last_updated);
652 }
653 }
654
655 if (options.output_path.has_value()) {

688absl::Status HandleSimpleChatCommand(const std::vector<std::string>& arg_vec,
689 Rom* rom, bool quiet) {
690 RETURN_IF_ERROR(EnsureRomLoaded(*rom, "agent simple-chat"));
691
692 auto _ = TryLoadProjectAndLabels(*rom);
693
694 std::optional<std::string> batch_file;
695 std::optional<std::string> single_message;
696 bool verbose = false;
697 std::optional<std::string> format_option;
698
699 for (size_t i = 0; i < arg_vec.size(); ++i) {
700 const std::string& arg = arg_vec[i];
701 if (absl::StartsWith(arg, "--file=")) {
702 batch_file = arg.substr(7);
703 } else if (arg == "--file" && i + 1 < arg_vec.size()) {
704 batch_file = arg_vec[++i];
705 } else if (absl::StartsWith(arg, "--format=")) {
706 format_option = arg.substr(9);
707 } else if (arg == "--format" && i + 1 < arg_vec.size()) {
708 format_option = arg_vec[++i];
709 } else if (arg == "--json") {
710 format_option = "json";
711 } else if (arg == "--markdown" || arg == "--md") {
712 format_option = "markdown";
713 } else if (arg == "--compact" || arg == "--raw") {
714 format_option = "compact";
715 } else if (arg == "--verbose" || arg == "-v") {
716 verbose = true;
717 } else if (!absl::StartsWith(arg, "--") && !single_message.has_value()) {
718 single_message = arg;
719 }
720 }
721
722 agent::AgentConfig config;
723 config.verbose = verbose;
724 if (format_option.has_value()) {
725 std::string normalized = absl::AsciiStrToLower(*format_option);
726 if (normalized == "json") {
727 config.output_format = AgentOutputFormat::kJson;
728 } else if (normalized == "markdown" || normalized == "md") {
729 config.output_format = AgentOutputFormat::kMarkdown;
730 } else if (normalized == "compact" || normalized == "raw") {
731 config.output_format = AgentOutputFormat::kCompact;
732 } else if (normalized == "text" || normalized == "friendly" ||
733 normalized == "pretty") {
734 config.output_format = AgentOutputFormat::kFriendly;
735 } else {
736 return absl::InvalidArgumentError(
737 absl::StrCat("Unsupported chat format: ", *format_option,
738 ". Supported formats: text, markdown, json, compact"));
739 }
740 }
741
742 SimpleChatSession session;
743 session.SetConfig(config);
744 session.SetRomContext(rom);
745
746 if (batch_file.has_value()) {
747 std::ifstream file(*batch_file);
748 if (!file.is_open()) {
749 return absl::NotFoundError(absl::StrCat("Failed to open file: ", *batch_file));
750 }
751 if (!quiet) {
752 std::cout << "Running batch session from: " << *batch_file << std::endl;
753 std::cout << "----------------------------------------\n\n";
754 }
755 std::string line;
756 int line_num = 0;
757 while (std::getline(file, line)) {
758 line_num++;
759 std::string trimmed_line = std::string(absl::StripAsciiWhitespace(line));
760 if (trimmed_line.empty() || absl::StartsWith(trimmed_line, "#")) {
761 continue;
762 }
763 if (!quiet) {
764 std::cout << "Input [" << line_num << "]: " << trimmed_line << std::endl;
765 }
766 std::string response;
767 auto status = session.SendAndWaitForResponse(trimmed_line, &response);
768 if (!status.ok()) {
769 std::cerr << "Error processing line " << line_num << ": " << status.message() << std::endl;
770 continue;
771 }
772 std::cout << response << "\n";
773 if (!quiet) {
774 std::cout << "\n";
775 }
776 }
777 return absl::OkStatus();
778 } else if (single_message.has_value()) {
779 std::string response;
780 auto status = session.SendAndWaitForResponse(*single_message, &response);
781 if (!status.ok()) {
782 return status;
783 }
784 std::cout << response << "\n";
785 return absl::OkStatus();
786 } else {
787 return session.RunInteractive();
788 }
789}

791absl::Status HandleAcceptCommand(const std::vector<std::string>& arg_vec,
792 Rom& rom) {
793 std::optional<std::string> proposal_id;
794 for (size_t i = 0; i < arg_vec.size(); ++i) {
795 const std::string& token = arg_vec[i];
796 if (absl::StartsWith(token, "--proposal-id=")) {
797 proposal_id = token.substr(14);
798 break;
799 }
800 if (token == "--proposal-id" && i + 1 < arg_vec.size()) {
801 proposal_id = arg_vec[i + 1];
802 break;
803 }
804 }
805
806 if (!proposal_id.has_value() || proposal_id->empty()) {
807 return absl::InvalidArgumentError(
808 "Usage: agent accept --proposal-id <proposal_id>");
809 }
810
811 auto& registry = ProposalRegistry::Instance();
812 ASSIGN_OR_RETURN(auto metadata, registry.GetProposal(*proposal_id));
813
814 if (metadata.status == ProposalRegistry::ProposalStatus::kAccepted) {
815 std::cout << "Proposal '" << *proposal_id << "' is already accepted."
816 << std::endl;
817 return absl::OkStatus();
818 }
819
820 if (metadata.sandbox_rom_path.empty()) {
821 return absl::FailedPreconditionError(absl::StrCat(
822 "Proposal '", *proposal_id,
823 "' is missing sandbox ROM metadata. Cannot accept."));
824 }
825
826 if (!std::filesystem::exists(metadata.sandbox_rom_path)) {
827 return absl::NotFoundError(absl::StrCat(
828 "Sandbox ROM not found at ", metadata.sandbox_rom_path.string()));
829 }
830
831 RETURN_IF_ERROR(
832 EnsureRomLoaded(rom, "agent accept --proposal-id <proposal_id>"));
833
834 Rom sandbox_rom;
835 auto sandbox_load_status = sandbox_rom.LoadFromFile(
836 metadata.sandbox_rom_path.string(), RomLoadOptions::CliDefaults());
837 if (!sandbox_load_status.ok()) {
838 return absl::InternalError(absl::StrCat(
839 "Failed to load sandbox ROM: ", sandbox_load_status.message()));
840 }
841
842 if (rom.size() != sandbox_rom.size()) {
843 rom.Expand(static_cast<int>(sandbox_rom.size()));
844 }
845
846 auto copy_status = rom.WriteVector(0, sandbox_rom.vector());
847 if (!copy_status.ok()) {
848 return absl::InternalError(absl::StrCat(
849 "Failed to copy sandbox ROM data: ", copy_status.message()));
850 }
851
852 auto save_status = rom.SaveToFile({.save_new = false});
853 if (!save_status.ok()) {
854 return absl::InternalError(absl::StrCat(
855 "Failed to save changes to main ROM: ", save_status.message()));
856 }
857
858 RETURN_IF_ERROR(registry.UpdateStatus(
859 *proposal_id, ProposalRegistry::ProposalStatus::kAccepted));
860 RETURN_IF_ERROR(registry.AppendLog(
861 *proposal_id,
862 absl::StrCat("Proposal accepted and applied to ", rom.filename())));
863
864 if (!metadata.sandbox_id.empty()) {
865 auto remove_status =
866 RomSandboxManager::Instance().RemoveSandbox(metadata.sandbox_id);
867 if (!remove_status.ok()) {
868 std::cerr << "Warning: Failed to remove sandbox '" << metadata.sandbox_id
869 << "': " << remove_status.message() << "\n";
870 }
871 }
872
873 std::cout << "✅ Proposal '" << *proposal_id << "' accepted and applied to '"
874 << rom.filename() << "'." << std::endl;
875 std::cout << " Source sandbox ROM: " << metadata.sandbox_rom_path
876 << std::endl;
877
878 return absl::OkStatus();
879}
