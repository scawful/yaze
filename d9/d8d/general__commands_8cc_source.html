59absl::Status TryLoadProjectAndLabels(Rom& rom) {
60 // Try to find and load a project file in current directory
61 core::YazeProject project;
62 auto project_status = project.Open(".");
63
64 if (project_status.ok()) {
65 std::cout << "📂 Loaded project: " << project.name << "\n";
66
67 // Initialize embedded labels (all default Zelda3 resource names)
68 auto labels_status = project.InitializeEmbeddedLabels();
69 if (labels_status.ok()) {
70 std::cout << "✅ Embedded labels initialized (all Zelda3 resources available)\n";
71 }
72
73 // Load labels from project (either embedded or external)
74 if (!project.labels_filename.empty()) {
75 auto* label_mgr = rom.resource_label();
76 if (label_mgr && label_mgr->LoadLabels(project.labels_filename)) {
77 std::cout << "🏷️ Loaded custom labels from: " << project.labels_filename << "\n";
78 }
79 } else if (!project.resource_labels.empty() || project.use_embedded_labels) {
80 // Use labels embedded in project or default Zelda3 labels
81 auto* label_mgr = rom.resource_label();
82 if (label_mgr) {
83 label_mgr->labels_ = project.resource_labels;
84 label_mgr->labels_loaded_ = true;
85 std::cout << "🏷️ Using embedded Zelda3 labels (rooms, sprites, entrances, items, etc.)\n";
86 }
87 }
88 } else {
89 // No project found - use embedded defaults anyway
90 std::cout << "ℹ️ No project file found. Using embedded default Zelda3 labels.\n";
91 project.InitializeEmbeddedLabels();
92 }
93
94 return absl::OkStatus();
95}

97absl::Status EnsureRomLoaded(Rom& rom, const std::string& command) {
98 if (rom.is_loaded()) {
99 return absl::OkStatus();

101
102 std::string rom_path = absl::GetFlag(FLAGS_rom);
103 if (rom_path.empty()) {

121absl::StatusOr<DescribeOptions> ParseDescribeArgs(
122 const std::vector<std::string>& args) {
123 DescribeOptions options;
124 for (size_t i = 0; i < args.size(); ++i) {
125 const std::string& token = args[i];
126 std::string flag = token;
127 std::optional<std::string> inline_value;
128
129 if (absl::StartsWith(token, "--")) {
130 auto eq_pos = token.find('=');
131 if (eq_pos != std::string::npos) {
132 flag = token.substr(0, eq_pos);
133 inline_value = token.substr(eq_pos + 1);
134 }
135 }
136
137 auto require_value =
138 [&](absl::string_view flag_name) -> absl::StatusOr<std::string> {
139 if (inline_value.has_value()) {
140 return *inline_value;
141 }
142 if (i + 1 >= args.size()) {
143 return absl::InvalidArgumentError(
144 absl::StrFormat("Flag %s requires a value", flag_name));
145 }
146 return args[++i];
147 };
148
149 if (flag == "--resource") {
150 ASSIGN_OR_RETURN(auto value, require_value("--resource"));
151 options.resource = std::move(value);
152 } else if (flag == "--format") {
153 ASSIGN_OR_RETURN(auto value, require_value("--format"));
154 options.format = std::move(value);
155 } else if (flag == "--output") {
156 ASSIGN_OR_RETURN(auto value, require_value("--output"));
157 options.output_path = std::move(value);
158 } else if (flag == "--version") {
159 ASSIGN_OR_RETURN(auto value, require_value("--version"));
160 options.version = std::move(value);
161 } else if (flag == "--last-updated") {
162 ASSIGN_OR_RETURN(auto value, require_value("--last-updated"));
163 options.last_updated = std::move(value);
164 } else {
165 return absl::InvalidArgumentError(
166 absl::StrFormat("Unknown flag for agent describe: %s", token));
167 }
168 }
169
170 options.format = absl::AsciiStrToLower(options.format);
171 if (options.format != "json" && options.format != "yaml") {
172 return absl::InvalidArgumentError("--format must be either json or yaml");
173 }
174
175 return options;
176}

180absl::Status HandleRunCommand(const std::vector<std::string>& arg_vec,
181 Rom& rom) {
182 if (arg_vec.size() < 2 || arg_vec[0] != "--prompt") {
183 return absl::InvalidArgumentError("Usage: agent run --prompt <prompt>");
184 }
185 std::string prompt = arg_vec[1];
186
187 RETURN_IF_ERROR(EnsureRomLoaded(rom, "agent run --prompt \"<prompt>\""));
188
189 // 1. Create a sandbox ROM to apply changes to
190 auto sandbox_or =
191 RomSandboxManager::Instance().CreateSandbox(rom, "agent-run");
192 if (!sandbox_or.ok()) {
193 return sandbox_or.status();
194 }
195 auto sandbox = sandbox_or.value();
196
197 // 2. Get commands from the AI service
198 auto ai_service = CreateAIService(); // Use service factory
199 auto response_or = ai_service->GenerateResponse(prompt);
200 if (!response_or.ok()) {
201 return response_or.status();
202 }
203 AgentResponse response = std::move(response_or.value());
204 const std::vector<std::string>& commands = response.commands;
205
206 // 3. Generate a structured proposal from the commands
207 Tile16ProposalGenerator generator;
208 auto proposal_or = generator.GenerateFromCommands(
209 prompt, commands, "ollama", &rom); // Pass original ROM to get old tiles
210 if (!proposal_or.ok()) {
211 return proposal_or.status();
212 }
213 auto proposal = proposal_or.value();
214
215 // 4. Apply the proposal to the sandbox ROM for preview
216 Rom sandbox_rom;
217 auto load_status = sandbox_rom.LoadFromFile(sandbox.rom_path.string());
218 if (!load_status.ok()) {
219 return absl::InternalError(
220 absl::StrCat("Failed to load sandbox ROM: ", load_status.message()));
221 }
222
223 auto apply_status = generator.ApplyProposal(proposal, &sandbox_rom);
224 if (!apply_status.ok()) {
225 return absl::InternalError(absl::StrCat(
226 "Failed to apply proposal to sandbox ROM: ", apply_status.message()));
227 }
228
229 // 5. Save the sandbox ROM to persist the changes for diffing
230 auto save_status = sandbox_rom.SaveToFile({.save_new = false});
231 if (!save_status.ok()) {
232 return absl::InternalError(
233 absl::StrCat("Failed to save sandbox ROM: ", save_status.message()));
234 }
235
236 // 6. Persist the proposal metadata and artifacts.
237 auto& registry = ProposalRegistry::Instance();
238
239 int executed_commands = 0;
240 for (const auto& command : commands) {
241 if (command.empty() || command[0] == '#') {
242 continue;
243 }
244 ++executed_commands;
245 }
246
247 std::string description = absl::StrFormat(
248 "Tile16 overworld edits (%d change%s)", proposal.changes.size(),
249 proposal.changes.size() == 1 ? "" : "s");
250
251 ASSIGN_OR_RETURN(
252 auto metadata,
253 registry.CreateProposal(sandbox.id, sandbox.rom_path, prompt, description));
254
255 proposal.id = metadata.id;
256
257 std::ostringstream diff_stream;
258 diff_stream << "Tile16 Proposal ID: " << metadata.id << "\n";
259 diff_stream << "Sandbox ID: " << sandbox.id << "\n";
260 diff_stream << "Sandbox ROM: " << sandbox.rom_path << "\n\n";
261 diff_stream << "Changes (" << proposal.changes.size() << "):\n";
262 for (const auto& change : proposal.changes) {
263 diff_stream << " - " << change.ToString() << "\n";
264 }
265
266 RETURN_IF_ERROR(registry.RecordDiff(metadata.id, diff_stream.str()));
267
268 RETURN_IF_ERROR(registry.AppendLog(
269 metadata.id, absl::StrCat("Prompt: ", prompt)));
270
271 if (!response.text_response.empty()) {
272 RETURN_IF_ERROR(registry.AppendLog(
273 metadata.id, absl::StrCat("AI Response: ", response.text_response)));
274 }
275
276 if (!response.reasoning.empty()) {
277 RETURN_IF_ERROR(registry.AppendLog(
278 metadata.id, absl::StrCat("Reasoning: ", response.reasoning)));
279 }
280
281 if (!response.tool_calls.empty()) {
282 RETURN_IF_ERROR(registry.AppendLog(
283 metadata.id,
284 absl::StrCat("Tool Calls: ", response.tool_calls.size())));
285 }
286
287 for (const auto& command : commands) {
288 if (command.empty() || command[0] == '#') {
289 continue;
290 }
291 RETURN_IF_ERROR(registry.AppendLog(
292 metadata.id, absl::StrCat("Command: ", command)));
293 }
294
295 RETURN_IF_ERROR(registry.AppendLog(
296 metadata.id,
297 absl::StrCat("Sandbox ROM saved to ", sandbox.rom_path.string())));
298
299 RETURN_IF_ERROR(
300 registry.UpdateCommandStats(metadata.id, executed_commands));
301 RETURN_IF_ERROR(registry.AppendLog(
302 metadata.id,
303 absl::StrCat("Commands executed: ", executed_commands)));
304
305 std::filesystem::path proposal_dir = metadata.log_path.parent_path();
306 std::filesystem::path proposal_path = proposal_dir / "proposal.json";
307 auto save_proposal_status =
308 generator.SaveProposal(proposal, proposal_path.string());
309 if (!save_proposal_status.ok()) {
310 return absl::InternalError(absl::StrCat("Failed to save proposal file: ",
311 save_proposal_status.message()));
312 }
313
314 RETURN_IF_ERROR(registry.AppendLog(
315 metadata.id,
316 absl::StrCat("Saved proposal JSON to ", proposal_path.string())));
317
318 std::cout
319 << "✅ Agent successfully planned and executed changes in a sandbox."
320 << std::endl;
321 std::cout << " Proposal ID: " << metadata.id << std::endl;
322 std::cout << " Sandbox ROM: " << sandbox.rom_path << std::endl;
323 std::cout << " Proposal dir: " << proposal_dir << std::endl;
324 std::cout << " Diff file: " << metadata.diff_path << std::endl;
325 std::cout << " Log file: " << metadata.log_path << std::endl;
326 std::cout << "\nTo review the changes, run:\n";
327 std::cout << " z3ed agent diff --proposal-id " << metadata.id << std::endl;
328 std::cout << "\nTo accept the changes, run:\n";
329 std::cout << " z3ed agent accept --proposal-id " << metadata.id << std::endl;
330
331 return absl::OkStatus();
332}

553absl::Status HandleDescribeCommand(const std::vector<std::string>& arg_vec) {
554 ASSIGN_OR_RETURN(auto options, ParseDescribeArgs(arg_vec));
555
556 const auto& catalog = ResourceCatalog::Instance();
557 std::optional<ResourceSchema> resource_schema;
558 if (options.resource.has_value()) {
559 auto resource_or = catalog.GetResource(*options.resource);
560 if (!resource_or.ok()) {
561 return resource_or.status();
562 }
563 resource_schema = resource_or.value();
564 }
565
566 std::string payload;
567 if (options.format == "json") {
568 if (resource_schema.has_value()) {
569 payload = catalog.SerializeResource(*resource_schema);
570 } else {
571 payload = catalog.SerializeResources(catalog.AllResources());
572 }
573 } else {
574 std::string last_updated =
575 options.last_updated.has_value()
576 ? *options.last_updated
577 : absl::FormatTime("%Y-%m-%d", absl::Now(), absl::LocalTimeZone());
578 if (resource_schema.has_value()) {
579 std::vector<ResourceSchema> schemas{*resource_schema};
580 payload = catalog.SerializeResourcesAsYaml(schemas, options.version,
581 last_updated);
582 } else {
583 payload = catalog.SerializeResourcesAsYaml(catalog.AllResources(),
584 options.version, last_updated);
585 }
586 }
587
588 if (options.output_path.has_value()) {
589 std::ofstream out(*options.output_path, std::ios::binary | std::ios::trunc);
590 if (!out.is_open()) {
591 return absl::InternalError(absl::StrFormat(
592 "Failed to open %s for writing", *options.output_path));
593 }
594 out << payload;
595 out.close();
596 if (!out) {
597 return absl::InternalError(absl::StrFormat("Failed to write schema to %s",
598 *options.output_path));
599 }
600 std::cout << absl::StrFormat("Wrote %s schema to %s", options.format,
601 *options.output_path)
602 << std::endl;
603 return absl::OkStatus();
604 }
605
606 std::cout << payload << std::endl;
607 return absl::OkStatus();
608}

621absl::Status HandleSimpleChatCommand(const std::vector<std::string>& arg_vec,
622 Rom& rom) {
623 RETURN_IF_ERROR(EnsureRomLoaded(rom, "agent simple-chat"));
624
625 // Try to load project and labels automatically
626 auto _ = TryLoadProjectAndLabels(rom); // Ignore errors - we'll use defaults
627
628 // Parse flags and positional arguments
629 std::optional<std::string> batch_file;
630 std::optional<std::string> single_message;
631 bool non_interactive = false;
632 bool verbose = false;
633 bool show_reasoning = true;
634 int max_tool_iterations = 4;
635 int max_retry_attempts = 3;
636
637 for (size_t i = 0; i < arg_vec.size(); ++i) {
638 const std::string& arg = arg_vec[i];
639
640 if (absl::StartsWith(arg, "--file=")) {
641 batch_file = arg.substr(7);
642 } else if (arg == "--file" && i + 1 < arg_vec.size()) {
643 batch_file = arg_vec[i + 1];
644 ++i;
645 } else if (arg == "--non-interactive" || arg == "-n") {
646 non_interactive = true;
647 } else if (arg == "--verbose" || arg == "-v") {
648 verbose = true;
649 } else if (arg == "--no-reasoning") {
650 show_reasoning = false;
651 } else if (absl::StartsWith(arg, "--max-tool-iterations=")) {
652 absl::SimpleAtoi(arg.substr(22), &max_tool_iterations);
653 } else if (arg == "--max-tool-iterations" && i + 1 < arg_vec.size()) {
654 absl::SimpleAtoi(arg_vec[i + 1], &max_tool_iterations);
655 ++i;
656 } else if (absl::StartsWith(arg, "--max-retries=")) {
657 absl::SimpleAtoi(arg.substr(14), &max_retry_attempts);
658 } else if (arg == "--max-retries" && i + 1 < arg_vec.size()) {
659 absl::SimpleAtoi(arg_vec[i + 1], &max_retry_attempts);
660 ++i;
661 } else if (!absl::StartsWith(arg, "--") && !single_message.has_value()) {
662 // Treat first non-flag argument as the message
663 single_message = arg;
664 }
665 }
666
667 // Configure agent
668 agent::AgentConfig config;
669 config.verbose = verbose;
670 config.show_reasoning = show_reasoning;
671 config.max_tool_iterations = max_tool_iterations;
672 config.max_retry_attempts = max_retry_attempts;
673
674 SimpleChatSession session;
675 session.SetConfig(config);
676 session.SetRomContext(&rom);
677
678 // Priority: batch file > single message > interactive/piped
679 if (batch_file.has_value()) {
680 return session.RunBatch(*batch_file);
681 } else if (single_message.has_value()) {
682 // Single message mode - send message and print response
683 std::string response;
684 auto status = session.SendAndWaitForResponse(*single_message, &response);
685 if (!status.ok()) {
686 return status;
687 }
688 std::cout << response << "\n";
689 return absl::OkStatus();
690 } else {
691 // Interactive or piped input mode
692 return session.RunInteractive();
693 }
694}

696absl::Status HandleAcceptCommand(const std::vector<std::string>& arg_vec,
697 Rom& rom) {
698 std::optional<std::string> proposal_id;
699 for (size_t i = 0; i < arg_vec.size(); ++i) {
700 const std::string& token = arg_vec[i];
701 if (absl::StartsWith(token, "--proposal-id=")) {
702 proposal_id = token.substr(14);
703 break;
704 }
705 if (token == "--proposal-id" && i + 1 < arg_vec.size()) {
706 proposal_id = arg_vec[i + 1];
707 break;
708 }
709 }
710
711 if (!proposal_id.has_value() || proposal_id->empty()) {
712 return absl::InvalidArgumentError(
713 "Usage: agent accept --proposal-id <proposal_id>");
714 }
715
716 auto& registry = ProposalRegistry::Instance();
717 ASSIGN_OR_RETURN(auto metadata, registry.GetProposal(*proposal_id));
718
719 if (metadata.status == ProposalRegistry::ProposalStatus::kAccepted) {
720 std::cout << "Proposal '" << *proposal_id << "' is already accepted."
721 << std::endl;
722 return absl::OkStatus();
723 }
724
725 if (metadata.sandbox_rom_path.empty()) {
726 return absl::FailedPreconditionError(absl::StrCat(
727 "Proposal '", *proposal_id,
728 "' is missing sandbox ROM metadata. Cannot accept."));
729 }
730
731 if (!std::filesystem::exists(metadata.sandbox_rom_path)) {
732 return absl::NotFoundError(absl::StrCat(
733 "Sandbox ROM not found at ", metadata.sandbox_rom_path.string()));
734 }
735
736 RETURN_IF_ERROR(
737 EnsureRomLoaded(rom, "agent accept --proposal-id <proposal_id>"));
738
739 Rom sandbox_rom;
740 auto sandbox_load_status = sandbox_rom.LoadFromFile(
741 metadata.sandbox_rom_path.string(), RomLoadOptions::CliDefaults());
742 if (!sandbox_load_status.ok()) {
743 return absl::InternalError(absl::StrCat(
744 "Failed to load sandbox ROM: ", sandbox_load_status.message()));
745 }
746
747 if (rom.size() != sandbox_rom.size()) {
748 rom.Expand(static_cast<int>(sandbox_rom.size()));
749 }
750
751 auto copy_status = rom.WriteVector(0, sandbox_rom.vector());
752 if (!copy_status.ok()) {
753 return absl::InternalError(absl::StrCat(
754 "Failed to copy sandbox ROM data: ", copy_status.message()));
755 }
756
757 auto save_status = rom.SaveToFile({.save_new = false});
758 if (!save_status.ok()) {
759 return absl::InternalError(absl::StrCat(
760 "Failed to save changes to main ROM: ", save_status.message()));
761 }
762
763 RETURN_IF_ERROR(registry.UpdateStatus(
764 *proposal_id, ProposalRegistry::ProposalStatus::kAccepted));
765 RETURN_IF_ERROR(registry.AppendLog(
766 *proposal_id,
767 absl::StrCat("Proposal accepted and applied to ", rom.filename())));
768
769 if (!metadata.sandbox_id.empty()) {
770 auto remove_status =
771 RomSandboxManager::Instance().RemoveSandbox(metadata.sandbox_id);
772 if (!remove_status.ok()) {
773 std::cerr << "Warning: Failed to remove sandbox '" << metadata.sandbox_id
774 << "': " << remove_status.message() << "\n";
775 }
776 }
777
778 std::cout << "✅ Proposal '" << *proposal_id << "' accepted and applied to '"
779 << rom.filename() << "'." << std::endl;
780 std::cout << " Source sandbox ROM: " << metadata.sandbox_rom_path
781 << std::endl;
782
783 return absl::OkStatus();
784}
