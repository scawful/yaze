51absl::StatusOr<DescribeOptions> ParseDescribeArgs(
52 const std::vector<std::string>& args) {
53 DescribeOptions options;
54 for (size_t i = 0; i < args.size(); ++i) {
55 const std::string& token = args[i];
56 std::string flag = token;
57 std::optional<std::string> inline_value;
58
59 if (absl::StartsWith(token, "--")) {
60 auto eq_pos = token.find('=');
61 if (eq_pos != std::string::npos) {
62 flag = token.substr(0, eq_pos);
63 inline_value = token.substr(eq_pos + 1);
64 }
65 }
66
67 auto require_value =
68 [&](absl::string_view flag_name) -> absl::StatusOr<std::string> {
69 if (inline_value.has_value()) {
70 return *inline_value;
71 }
72 if (i + 1 >= args.size()) {
73 return absl::InvalidArgumentError(
74 absl::StrFormat("Flag %s requires a value", flag_name));
75 }
76 return args[++i];
77 };
78
79 if (flag == "--resource") {
80 ASSIGN_OR_RETURN(auto value, require_value("--resource"));
81 options.resource = std::move(value);
82 } else if (flag == "--format") {
83 ASSIGN_OR_RETURN(auto value, require_value("--format"));
84 options.format = std::move(value);
85 } else if (flag == "--output") {
86 ASSIGN_OR_RETURN(auto value, require_value("--output"));
87 options.output_path = std::move(value);
88 } else if (flag == "--version") {
89 ASSIGN_OR_RETURN(auto value, require_value("--version"));
90 options.version = std::move(value);
91 } else if (flag == "--last-updated") {
92 ASSIGN_OR_RETURN(auto value, require_value("--last-updated"));
93 options.last_updated = std::move(value);
94 } else {
95 return absl::InvalidArgumentError(
96 absl::StrFormat("Unknown flag for agent describe: %s", token));
97 }
98 }
99
100 options.format = absl::AsciiStrToLower(options.format);
101 if (options.format != "json" && options.format != "yaml") {
102 return absl::InvalidArgumentError("--format must be either json or yaml");
103 }
104
105 return options;
106}

110absl::Status HandleRunCommand(const std::vector<std::string>& arg_vec,
111 Rom& rom) {
112 if (arg_vec.size() < 2 || arg_vec[0] != "--prompt") {
113 return absl::InvalidArgumentError("Usage: agent run --prompt <prompt>");
114 }
115 std::string prompt = arg_vec[1];
116
117 if (!rom.is_loaded()) {
118 std::string rom_path = absl::GetFlag(FLAGS_rom);
119 if (rom_path.empty()) {
120 return absl::FailedPreconditionError(
121 "No ROM loaded. Use --rom=<path> to specify ROM file.\n"
122 "Example: z3ed agent run --rom=zelda3.sfc --prompt \"Your prompt "
123 "here\"");
124 }
125
126 auto status = rom.LoadFromFile(rom_path);
127 if (!status.ok()) {
128 return absl::FailedPreconditionError(absl::StrFormat(
129 "Failed to load ROM from '%s': %s", rom_path, status.message()));
130 }
131 }
132
133 // 1. Create a sandbox ROM to apply changes to
134 auto sandbox_or =
135 RomSandboxManager::Instance().CreateSandbox(rom, "agent-run");
136 if (!sandbox_or.ok()) {
137 return sandbox_or.status();
138 }
139 auto sandbox = sandbox_or.value();
140
141 // 2. Get commands from the AI service
142 auto ai_service = CreateAIService(); // Use service factory
143 auto response_or = ai_service->GenerateResponse(prompt);
144 if (!response_or.ok()) {
145 return response_or.status();
146 }
147 std::vector<std::string> commands = response_or.value().commands;
148
149 // 3. Generate a structured proposal from the commands
150 Tile16ProposalGenerator generator;
151 auto proposal_or = generator.GenerateFromCommands(
152 prompt, commands, "ollama", &rom); // Pass original ROM to get old tiles
153 if (!proposal_or.ok()) {
154 return proposal_or.status();
155 }
156 auto proposal = proposal_or.value();
157
158 // 4. Apply the proposal to the sandbox ROM for preview
159 Rom sandbox_rom;
160 auto load_status = sandbox_rom.LoadFromFile(sandbox.rom_path.string());
161 if (!load_status.ok()) {
162 return absl::InternalError(
163 absl::StrCat("Failed to load sandbox ROM: ", load_status.message()));
164 }
165
166 auto apply_status = generator.ApplyProposal(proposal, &sandbox_rom);
167 if (!apply_status.ok()) {
168 return absl::InternalError(absl::StrCat(
169 "Failed to apply proposal to sandbox ROM: ", apply_status.message()));
170 }
171
172 // 5. Save the sandbox ROM to persist the changes for diffing
173 auto save_status = sandbox_rom.SaveToFile({.save_new = false});
174 if (!save_status.ok()) {
175 return absl::InternalError(
176 absl::StrCat("Failed to save sandbox ROM: ", save_status.message()));
177 }
178
179 // 6. Save the proposal metadata for later use (accept/reject)
180 // For now, we'll just use the proposal generator's save function.
181 // A better approach would be to integrate with ProposalRegistry.
182 auto proposal_path =
183 RomSandboxManager::Instance().RootDirectory() / (proposal.id + ".json");
184 auto save_proposal_status =
185 generator.SaveProposal(proposal, proposal_path.string());
186 if (!save_proposal_status.ok()) {
187 return absl::InternalError(absl::StrCat("Failed to save proposal file: ",
188 save_proposal_status.message()));
189 }
190
191 std::cout
192 << "✅ Agent successfully planned and executed changes in a sandbox."
193 << std::endl;
194 std::cout << " Proposal ID: " << proposal.id << std::endl;
195 std::cout << " Sandbox ROM: " << sandbox.rom_path << std::endl;
196 std::cout << " Proposal file: " << proposal_path << std::endl;
197 std::cout << "\nTo review the changes, run:\n";
198 std::cout << " z3ed agent diff --proposal-id " << proposal.id << std::endl;
199 std::cout << "\nTo accept the changes, run:\n";
200 std::cout << " z3ed agent accept --proposal-id " << proposal.id << std::endl;
201
202 return absl::OkStatus();
203}

416absl::Status HandleDescribeCommand(const std::vector<std::string>& arg_vec) {
417 ASSIGN_OR_RETURN(auto options, ParseDescribeArgs(arg_vec));
418
419 const auto& catalog = ResourceCatalog::Instance();
420 std::optional<ResourceSchema> resource_schema;
421 if (options.resource.has_value()) {
422 auto resource_or = catalog.GetResource(*options.resource);
423 if (!resource_or.ok()) {
424 return resource_or.status();
425 }
426 resource_schema = resource_or.value();
427 }
428
429 std::string payload;
430 if (options.format == "json") {
431 if (resource_schema.has_value()) {
432 payload = catalog.SerializeResource(*resource_schema);
433 } else {
434 payload = catalog.SerializeResources(catalog.AllResources());
435 }
436 } else {
437 std::string last_updated =
438 options.last_updated.has_value()
439 ? *options.last_updated
440 : absl::FormatTime("%Y-%m-%d", absl::Now(), absl::LocalTimeZone());
441 if (resource_schema.has_value()) {
442 std::vector<ResourceSchema> schemas{*resource_schema};
443 payload = catalog.SerializeResourcesAsYaml(schemas, options.version,
444 last_updated);
445 } else {
446 payload = catalog.SerializeResourcesAsYaml(catalog.AllResources(),
447 options.version, last_updated);
448 }
449 }
450
451 if (options.output_path.has_value()) {
452 std::ofstream out(*options.output_path, std::ios::binary | std::ios::trunc);
453 if (!out.is_open()) {
454 return absl::InternalError(absl::StrFormat(
455 "Failed to open %s for writing", *options.output_path));
456 }
457 out << payload;
458 out.close();
459 if (!out) {
460 return absl::InternalError(absl::StrFormat("Failed to write schema to %s",
461 *options.output_path));
462 }
463 std::cout << absl::StrFormat("Wrote %s schema to %s", options.format,
464 *options.output_path)
465 << std::endl;
466 return absl::OkStatus();
467 }
468
469 std::cout << payload << std::endl;
470 return absl::OkStatus();
471}
