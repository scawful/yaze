229Component UnifiedLayout::CreateHexViewerPanel() {
230 static int offset = 0;
231 const int lines_to_show = 20;
232
233 return Renderer([&] {
234 if (!rom_context_) {
235 return vbox({
236 text("🔍 Hex Viewer") | bold | center,
237 separator(),
238 text("No ROM loaded") | center | color(Color::Red),
239 separator(),
240 text("Load a ROM to view hex data") | center | dim
241 }) | border;
242 }
243
244 std::vector<Element> rows;
245 for (int i = 0; i < lines_to_show; ++i) {
246 int current_offset = offset + (i * 16);
247 if (current_offset >= static_cast<int>(rom_context_->size())) {
248 break;
249 }
250
251 Elements row;
252 row.push_back(text(absl::StrFormat("0x%08X: ", current_offset)) | color(Color::Yellow));
253
254 for (int j = 0; j < 16; ++j) {
255 if (current_offset + j < static_cast<int>(rom_context_->size())) {
256 row.push_back(text(absl::StrFormat("%02X ", rom_context_->vector()[current_offset + j])));
257 } else {
258 row.push_back(text(" "));
259 }
260 }
261
262 row.push_back(separator());
263
264 for (int j = 0; j < 16; ++j) {
265 if (current_offset + j < static_cast<int>(rom_context_->size())) {
266 char c = rom_context_->vector()[current_offset + j];
267 row.push_back(text(std::isprint(c) ? std::string(1, c) : "."));
268 } else {
269 row.push_back(text(" "));
270 }
271 }
272
273 rows.push_back(hbox(row));
274 }
275
276 return vbox({
277 text("🔍 Hex Viewer") | bold | center,
278 separator(),
279 vbox(rows) | frame | flex,
280 separator(),
281 text(absl::StrFormat("Offset: 0x%08X", offset)) | color(Color::Cyan),
282 separator(),
283 text("↑/↓: Scroll | q: Back") | dim | center
284 }) | border;
285 });
286}

475bool UnifiedLayout::HandleGlobalEvents(const Event& event) {
476 // Global shortcuts
477 if (event == Event::Character('q')) {
478 screen_.Exit();
479 return true;
480 }
481
482 if (event == Event::Character('h')) {
483 SwitchMainPanel(PanelType::kHelp);
484 return true;
485 }
486
487 if (event == Event::Character('c')) {
488 ToggleChat();
489 return true;
490 }
491
492 if (event == Event::Character('s')) {
493 ToggleStatus();
494 return true;
495 }
496
497 if (event == Event::Character('t')) {
498 SwitchMainPanel(PanelType::kRomTools);
499 return true;
500 }
501
502 if (event == Event::Character('m')) {
503 SwitchMainPanel(PanelType::kMainMenu);
504 return true;
