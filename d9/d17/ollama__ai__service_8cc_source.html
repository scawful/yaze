75 return absl::InternalError(absl::StrFormat(

77 res->status, res->body));

79
80 // Check if requested model is available
81 nlohmann::json models_json = nlohmann::json::parse(res->body);
82 bool model_found = false;

89 model_found = true;
90 break;
91 }
92 }
93 }
94 }
95
96 if (!model_found) {
97 return absl::NotFoundError(absl::StrFormat(
98 "Model '%s' not found on Ollama server.\n"
99 "Pull it with: ollama pull %s\n"
100 "Available models: ollama list",
101 config_.model, config_.model));
102 }
103
104 return absl::OkStatus();
105 } catch (const std::exception& e) {
106 return absl::InternalError(absl::StrCat(

167absl::StatusOr<std::vector<std::string>> OllamaAIService::GetCommands(
168 const std::string& prompt) {
169#if !YAZE_HAS_HTTPLIB || !YAZE_HAS_JSON
170 return absl::UnimplementedError(
171 "Ollama service requires httplib and JSON support. "
172 "Install vcpkg dependencies or use bundled libraries.");
173#else
174
175 // Build request payload
176 nlohmann::json request_body = {
177 {"model", config_.model},
178 {"prompt", config_.system_prompt + "\n\nUSER REQUEST: " + prompt},
179 {"stream", false},
180 {"options", {
181 {"temperature", config_.temperature},
182 {"num_predict", config_.max_tokens}
183 }},
184 {"format", "json"} // Force JSON output
185 };
186
187 try {
188 httplib::Client cli(config_.base_url);
189 cli.set_read_timeout(60); // Longer timeout for inference
190
191 auto res = cli.Post("/api/generate", request_body.dump(), "application/json");
192
193 if (!res) {
194 return absl::UnavailableError(
195 "Failed to connect to Ollama. Is 'ollama serve' running?\n"
196 "Start with: ollama serve");
197 }
198
199 if (res->status != 200) {
200 return absl::InternalError(absl::StrFormat(
201 "Ollama API error: HTTP %d\nResponse: %s",
202 res->status, res->body));
203 }
204
205 // Parse response to extract generated text
206 auto generated_text_or = ParseOllamaResponse(res->body);
207 if (!generated_text_or.ok()) {
208 return generated_text_or.status();
209 }
210 std::string generated_text = generated_text_or.value();
211
212 // Parse the command array from generated text
213 nlohmann::json commands_json;
214 try {
215 commands_json = nlohmann::json::parse(generated_text);
216 } catch (const nlohmann::json::exception& e) {
217 // Sometimes the LLM includes extra text - try to extract JSON array
218 size_t start = generated_text.find('[');
219 size_t end = generated_text.rfind(']');
220
221 if (start != std::string::npos && end != std::string::npos && end > start) {
222 std::string json_only = generated_text.substr(start, end - start + 1);
223 try {
224 commands_json = nlohmann::json::parse(json_only);
225 } catch (const nlohmann::json::exception&) {
226 return absl::InvalidArgumentError(
227 "LLM did not return valid JSON. Response:\n" + generated_text);
228 }
229 } else {
230 return absl::InvalidArgumentError(
231 "LLM did not return a JSON array. Response:\n" + generated_text);
232 }
233 }
234
235 if (!commands_json.is_array()) {
236 return absl::InvalidArgumentError(
237 "LLM did not return a JSON array. Response:\n" + generated_text);
238 }
239
240 std::vector<std::string> commands;
241 for (const auto& cmd : commands_json) {
242 if (cmd.is_string()) {
243 commands.push_back(cmd.get<std::string>());
244 }
245 }
246
247 if (commands.empty()) {
248 return absl::InvalidArgumentError(
249 "LLM returned empty command list. Prompt may be unclear.\n"
250 "Try rephrasing your request to be more specific.");
251 }
252
253 return commands;
254
255 } catch (const std::exception& e) {
256 return absl::InternalError(absl::StrCat(
257 "Ollama request failed: ", e.what()));
258 }
259#endif
260}
