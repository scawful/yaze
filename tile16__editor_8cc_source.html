185absl::Status Tile16Editor::DrawToCurrentTile16(ImVec2 click_position) {
186 constexpr int tile8_size = 8;
187 constexpr int tile16_size = 16;
188
189 // Bounds check for current_tile8_
190 if (current_tile8_ < 0 || current_tile8_ >= static_cast<int>(current_gfx_individual_.size())) {
191 return absl::OutOfRangeError(absl::StrFormat("Invalid tile8 index: %d", current_tile8_));
192 }
193
194 if (!current_gfx_individual_[current_tile8_].is_active()) {
195 return absl::FailedPreconditionError("Source tile8 bitmap not active");
196 }
197
198 if (!current_tile16_bmp_.is_active()) {
199 return absl::FailedPreconditionError("Target tile16 bitmap not active");
200 }
201
202 // Calculate the tile index for x and y based on the click_position
203 // Adjusting for Tile16 (16x16) which contains 4 Tile8 (8x8)
204 int tile_index_x = static_cast<int>(click_position.x) / tile8_size;
205 int tile_index_y = static_cast<int>(click_position.y) / tile8_size;
206
207 // Ensure we're within the bounds of the Tile16 (0-1 for both x and y)
208 tile_index_x = std::min(1, std::max(0, tile_index_x));
209 tile_index_y = std::min(1, std::max(0, tile_index_y));
210
211 // Calculate the pixel start position within the Tile16
212 // Each Tile8 is 8x8 pixels, so we multiply by 8 to get the pixel offset
213 int start_x = tile_index_x * tile8_size;
214 int start_y = tile_index_y * tile8_size;
215
216 // Get source tile data
217 const auto& source_tile = current_gfx_individual_[current_tile8_];
218 if (source_tile.size() < 64) { // 8x8 = 64 pixels
219 return absl::FailedPreconditionError("Source tile data too small");
220 }
221
222 // Draw the Tile8 to the correct position within the Tile16
223 for (int y = 0; y < tile8_size; ++y) {
224 for (int x = 0; x < tile8_size; ++x) {
225 // Calculate the pixel position in the Tile16 bitmap
226 int pixel_x = start_x + x;
227 int pixel_y = start_y + y;
228 int pixel_index = pixel_y * tile16_size + pixel_x;
229
230 // Bounds check for tile16 bitmap
231 if (pixel_index < 0 || pixel_index >= static_cast<int>(current_tile16_bmp_.size())) {
232 continue;
233 }
234
235 // Calculate the pixel position in the Tile8 bitmap
236 int gfx_pixel_index = y * tile8_size + x;
237
238 // Apply flipping if needed
239 if (x_flip) {
240 gfx_pixel_index = y * tile8_size + (tile8_size - 1 - x);
241 }
242 if (y_flip) {
243 gfx_pixel_index = (tile8_size - 1 - y) * tile8_size + x;
244 }
245 if (x_flip && y_flip) {
246 gfx_pixel_index =
247 (tile8_size - 1 - y) * tile8_size + (tile8_size - 1 - x);
248 }
249
250 // Bounds check for source tile
251 if (gfx_pixel_index >= 0 && gfx_pixel_index < static_cast<int>(source_tile.size())) {
252 // Write the pixel to the Tile16 bitmap
253 current_tile16_bmp_.WriteToPixel(pixel_index, source_tile.data()[gfx_pixel_index]);
254 }
255 }
256 }
257
258 current_tile16_bmp_.set_modified(true);
259 return absl::OkStatus();

460absl::Status Tile16Editor::LoadTile8() {
461 if (!current_gfx_bmp_.is_active() || current_gfx_bmp_.data() == nullptr) {
462 return absl::FailedPreconditionError("Current graphics bitmap not initialized");
463 }
464
465 const auto& ow_main_pal_group = rom()->palette_group().overworld_main;
466 if (ow_main_pal_group.size() == 0) {
467 return absl::FailedPreconditionError("Overworld palette group not loaded");
468 }
469
470 current_gfx_individual_.clear();
471 current_gfx_individual_.reserve(1024);
472
473 // Process tiles sequentially to avoid race conditions
474 for (int index = 0; index < 1024; index++) {
475 std::array<uint8_t, 0x40> tile_data{};
476
477 // Calculate the position in the current gfx data
478 int num_columns = current_gfx_bmp_.width() / 8;
479 if (num_columns <= 0) {
480 continue; // Skip invalid tiles
481 }
482
483 // Copy the pixel data for the current tile into the vector
484 for (int ty = 0; ty < 8; ty++) {
485 for (int tx = 0; tx < 8; tx++) {
486 // Calculate the position in the tile data vector
487 int position = tx + (ty * 8);
488
489 // Calculate the position in the current gfx data
490 int x = (index % num_columns) * 8 + tx;
491 int y = (index / num_columns) * 8 + ty;
492 int gfx_position = x + (y * current_gfx_bmp_.width());
493
494 // Bounds check
495 if (gfx_position >= 0 && gfx_position < static_cast<int>(current_gfx_bmp_.size())) {
496 uint8_t value = current_gfx_bmp_.data()[gfx_position];
497
498 // Handle palette adjustment
499 if (value & 0x80) {
500 value -= 0x88;
501 }
502
503 tile_data[position] = value;
504 }
505 }
506 }
507
508 // Create the tile bitmap
509 current_gfx_individual_.emplace_back();
510 auto &tile_bitmap = current_gfx_individual_.back();
511
512 try {
513 tile_bitmap.Create(8, 8, 8, tile_data);
514 if (current_palette_ < ow_main_pal_group.size()) {
515 tile_bitmap.SetPaletteWithTransparent(ow_main_pal_group[0], current_palette_);
516 }
517 Renderer::Get().RenderBitmap(&tile_bitmap);
518 } catch (const std::exception& e) {
519 // Log error but continue with other tiles
520 util::logf("Error creating tile %d: %s", index, e.what());
521 continue;
522 }
523 }
524
525 return absl::OkStatus();
526}

645absl::Status Tile16Editor::SaveTile16ToScratchSpace(int slot) {
646 if (slot < 0 || slot >= 4) {
647 return absl::InvalidArgumentError("Invalid scratch space slot");
648 }
649
