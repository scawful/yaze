355absl::Status Tile16Editor::RegenerateTile16BitmapFromROM() {
356 // Get the current tile16 data from ROM
357 auto* tile_data = GetCurrentTile16Data();
358 if (!tile_data) {
359 return absl::FailedPreconditionError("Cannot access current tile16 data");
360 }
361
362 // Create a new 16x16 bitmap for the tile16
363 std::vector<uint8_t> tile16_pixels(kTile16PixelCount, 0);
364
365 // Process each quadrant (2x2 grid of 8x8 tiles)
366 for (int quadrant = 0; quadrant < 4; ++quadrant) {
367 gfx::TileInfo* tile_info = nullptr;
368 int quadrant_x = quadrant % 2;
369 int quadrant_y = quadrant / 2;
370
371 // Get the tile info for this quadrant
372 switch (quadrant) {
373 case 0:
374 tile_info = &tile_data->tile0_;
375 break;
376 case 1:
377 tile_info = &tile_data->tile1_;
378 break;
379 case 2:
380 tile_info = &tile_data->tile2_;
381 break;
382 case 3:
383 tile_info = &tile_data->tile3_;
384 break;
385 }
386
387 if (!tile_info)
388 continue;
389
390 // Get the tile8 ID and properties
391 int tile8_id = tile_info->id_;
392 bool x_flip = tile_info->horizontal_mirror_;
393 bool y_flip = tile_info->vertical_mirror_;
394 uint8_t palette = tile_info->palette_;
395
396 // Get the source tile8 bitmap
397 if (tile8_id >= 0 &&
398 tile8_id < static_cast<int>(current_gfx_individual_.size()) &&
399 current_gfx_individual_[tile8_id].is_active()) {
400
401 const auto& source_tile8 = current_gfx_individual_[tile8_id];
402
403 // Copy the 8x8 tile into the appropriate quadrant of the 16x16 tile
404 for (int ty = 0; ty < kTile8Size; ++ty) {
405 for (int tx = 0; tx < kTile8Size; ++tx) {
406 // Apply flip transformations
407 int src_x = x_flip ? (kTile8Size - 1 - tx) : tx;
408 int src_y = y_flip ? (kTile8Size - 1 - ty) : ty;
409 int src_index = src_y * kTile8Size + src_x;
410
411 // Calculate destination in tile16
412 int dst_x = (quadrant_x * kTile8Size) + tx;
413 int dst_y = (quadrant_y * kTile8Size) + ty;
414 int dst_index = dst_y * kTile16Size + dst_x;
415
416 // Copy pixel with bounds checking
417 if (src_index >= 0 &&
418 src_index < static_cast<int>(source_tile8.size()) &&
419 dst_index >= 0 && dst_index < kTile16PixelCount) {
420 uint8_t pixel = source_tile8.data()[src_index];
421 // Apply palette offset if needed
422 tile16_pixels[dst_index] = pixel;
423 }
424 }
425 }
426 }
427 }
428
429 // Update the current tile16 bitmap with the regenerated data
430 current_tile16_bmp_.Create(kTile16Size, kTile16Size, 8, tile16_pixels);
431
432 // Set the appropriate palette
433 const auto& ow_main_pal_group = rom()->palette_group().overworld_main;
434 if (ow_main_pal_group.size() > 0) {
435 current_tile16_bmp_.SetPalette(ow_main_pal_group[0]);
436 }
437
438 // Render the updated bitmap
439 core::Renderer::Get().RenderBitmap(&current_tile16_bmp_);
440
441 util::logf("Regenerated Tile16 bitmap for tile %d from ROM data",
442 current_tile16_);
443 return absl::OkStatus();
444}

446absl::Status Tile16Editor::DrawToCurrentTile16(ImVec2 click_position,
447 const gfx::Bitmap* source_tile) {
448 constexpr int kTile8Size = 8;
449 constexpr int kTile16Size = 16;
450
451 // Save undo state before making changes
452 auto now = std::chrono::steady_clock::now();
453 auto time_since_last_edit =
454 std::chrono::duration_cast<std::chrono::milliseconds>(now -
455 last_edit_time_)
456 .count();
457
458 if (time_since_last_edit > 100) { // 100ms threshold
459 SaveUndoState();
460 last_edit_time_ = now;
461 }
462
463 // Validate inputs
464 if (current_tile8_ < 0 ||
465 current_tile8_ >= static_cast<int>(current_gfx_individual_.size())) {
466 return absl::OutOfRangeError(
467 absl::StrFormat("Invalid tile8 index: %d", current_tile8_));
468 }
469
470 if (!current_gfx_individual_[current_tile8_].is_active()) {
471 return absl::FailedPreconditionError("Source tile8 bitmap not active");
472 }
473
474 if (!current_tile16_bmp_.is_active()) {
475 return absl::FailedPreconditionError("Target tile16 bitmap not active");
476 }
477
478 // Determine which quadrant was clicked - handle the 8x scale factor properly
479 int quadrant_x = (click_position.x >= kTile8Size) ? 1 : 0;
480 int quadrant_y = (click_position.y >= kTile8Size) ? 1 : 0;
481
482 int start_x = quadrant_x * kTile8Size;
483 int start_y = quadrant_y * kTile8Size;
484
485 // Get source tile8 data - use provided tile if available, otherwise use current tile8
486 const gfx::Bitmap* tile_to_use =
487 source_tile ? source_tile : &current_gfx_individual_[current_tile8_];
488 if (tile_to_use->size() < 64) {
489 return absl::FailedPreconditionError("Source tile data too small");
490 }
491
492 // Copy tile8 into tile16 quadrant with proper transformations
493 for (int tile_y = 0; tile_y < kTile8Size; ++tile_y) {
494 for (int tile_x = 0; tile_x < kTile8Size; ++tile_x) {
495 // Apply flip transformations to source coordinates only if using original tile
496 // If a pre-flipped tile is provided, use direct coordinates
497 int src_x;
498 int src_y;
499 if (source_tile) {
500 // Pre-flipped tile provided, use direct coordinates
501 src_x = tile_x;
502 src_y = tile_y;
503 } else {
504 // Original tile, apply flip transformations
505 src_x = x_flip ? (kTile8Size - 1 - tile_x) : tile_x;
506 src_y = y_flip ? (kTile8Size - 1 - tile_y) : tile_y;
507 }
508 int src_index = src_y * kTile8Size + src_x;
509
510 // Calculate destination in tile16
511 int dst_x = start_x + tile_x;
512 int dst_y = start_y + tile_y;
513 int dst_index = dst_y * kTile16Size + dst_x;
514
515 // Bounds check and copy pixel
516 if (src_index >= 0 && src_index < static_cast<int>(tile_to_use->size()) &&
517 dst_index >= 0 &&
518 dst_index < static_cast<int>(current_tile16_bmp_.size())) {
519
520 uint8_t pixel_value = tile_to_use->data()[src_index];
521
522 // Keep original pixel values - palette selection is handled by TileInfo metadata
523 // not by modifying pixel data directly
524 current_tile16_bmp_.WriteToPixel(dst_index, pixel_value);
525 }
526 }
527 }
528
529 // Recreate the bitmap since we modified pixel data
530 auto tile16_data = current_tile16_bmp_.vector();
531 current_tile16_bmp_.Create(kTile16Size, kTile16Size, 8, tile16_data);
532
533 // Re-apply the correct palette to the tile16 bitmap after pixel updates
534 const auto& ow_main_pal_group = rom()->palette_group().overworld_main;
535 if (ow_main_pal_group.size() > 0) {
536 current_tile16_bmp_.SetPalette(ow_main_pal_group[0]);
537 }
538
539 core::Renderer::Get().RenderBitmap(&current_tile16_bmp_);
540
541 // Update ROM data when painting to tile16
542 auto* tile_data = GetCurrentTile16Data();
543 if (tile_data) {
544 // Update the quadrant's TileInfo based on current settings
545 int quadrant_index = quadrant_x + (quadrant_y * 2);
546 if (quadrant_index >= 0 && quadrant_index < 4) {
547
548 // Create new TileInfo with current settings
549 gfx::TileInfo new_tile_info(static_cast<uint16_t>(current_tile8_),
550 current_palette_, y_flip, x_flip,
551 priority_tile);
552
553 // Get pointer to the correct quadrant TileInfo
554 gfx::TileInfo* quadrant_tile = nullptr;
555 switch (quadrant_index) {
556 case 0:
557 quadrant_tile = &tile_data->tile0_;
558 break;
559 case 1:
560 quadrant_tile = &tile_data->tile1_;
561 break;
562 case 2:
563 quadrant_tile = &tile_data->tile2_;
564 break;
565 case 3:
566 quadrant_tile = &tile_data->tile3_;
567 break;
568 }
569
570 if (quadrant_tile && !(*quadrant_tile == new_tile_info)) {
571 *quadrant_tile = new_tile_info;
572 // Update the tiles_info array as well
573 tile_data->tiles_info[quadrant_index] = new_tile_info;
574
575 util::logf(
576 "Updated ROM Tile16 %d, quadrant %d: Tile8=%d, Pal=%d, XFlip=%d, "
577 "YFlip=%d, Priority=%d",
578 current_tile16_, quadrant_index, current_tile8_, current_palette_,
579 x_flip, y_flip, priority_tile);
580 }
581 }
582 }
583
584 if (live_preview_enabled_) {
585 RETURN_IF_ERROR(UpdateOverworldTilemap());
586 }
587
588 return absl::OkStatus();
589}

869absl::Status Tile16Editor::LoadTile8() {
870 if (!current_gfx_bmp_.is_active() || current_gfx_bmp_.data() == nullptr) {
871 return absl::FailedPreconditionError(
872 "Current graphics bitmap not initialized");
873 }
874
875 const auto& ow_main_pal_group = rom()->palette_group().overworld_main;
876 if (ow_main_pal_group.size() == 0) {
877 return absl::FailedPreconditionError("Overworld palette group not loaded");
878 }
879
880 current_gfx_individual_.clear();
881
882 // Calculate how many 8x8 tiles we can fit based on the current graphics bitmap size
883 // SNES graphics are typically 128 pixels wide (16 tiles of 8 pixels each)
884 const int tiles_per_row = current_gfx_bmp_.width() / 8;
885 const int total_rows = current_gfx_bmp_.height() / 8;
886 const int total_tiles = tiles_per_row * total_rows;
887
888 current_gfx_individual_.reserve(total_tiles);
889
890 // Extract individual 8x8 tiles from the graphics bitmap
891 for (int tile_y = 0; tile_y < total_rows; ++tile_y) {
892 for (int tile_x = 0; tile_x < tiles_per_row; ++tile_x) {
893 std::vector<uint8_t> tile_data(64); // 8x8 = 64 pixels
894
895 // Extract tile data from the main graphics bitmap
896 for (int py = 0; py < 8; ++py) {
897 for (int px = 0; px < 8; ++px) {
898 int src_x = tile_x * 8 + px;
899 int src_y = tile_y * 8 + py;
900 int src_index = src_y * current_gfx_bmp_.width() + src_x;
901 int dst_index = py * 8 + px;
902
903 if (src_index < static_cast<int>(current_gfx_bmp_.size()) &&
904 dst_index < 64) {
905 uint8_t pixel_value = current_gfx_bmp_.data()[src_index];
906
907 // Apply normalization based on settings
908 if (auto_normalize_pixels_) {
909 pixel_value &= palette_normalization_mask_;
910 }
911
912 tile_data[dst_index] = pixel_value;
913 }
914 }
915 }
916
917 // Create the individual tile bitmap
918 current_gfx_individual_.emplace_back();
919 auto& tile_bitmap = current_gfx_individual_.back();
920
921 try {
922 tile_bitmap.Create(8, 8, 8, tile_data);
923 // Use the helper method to set proper palette
924 RETURN_IF_ERROR(UpdateTile8Palette(current_gfx_individual_.size() - 1));
925 } catch (const std::exception& e) {
926 // Use optimized logging for potential hot spots
927 util::logf("Error creating tile at (%d,%d): %s", tile_x, tile_y,
928 e.what());
929 // Create an empty bitmap as fallback
930 tile_bitmap.Create(8, 8, 8, std::vector<uint8_t>(64, 0));
931 }
932 }
933 }
934
935 return absl::OkStatus();
936}

938absl::Status Tile16Editor::SetCurrentTile(int tile_id) {
939 if (tile_id < 0 || tile_id >= zelda3::kNumTile16Individual) {
940 return absl::OutOfRangeError(
941 absl::StrFormat("Invalid tile16 id: %d", tile_id));
942 }
943
944 if (!tile16_blockset_) {
945 return absl::FailedPreconditionError("Tile16 blockset not initialized");
946 }
947
948 current_tile16_ = tile_id;
949
950 // Extract tile data using the same method as GetTilemapData
951 auto tile_data = gfx::GetTilemapData(*tile16_blockset_, tile_id);
952
953 if (tile_data.empty()) {
954 // If GetTilemapData fails, manually extract from the atlas
955 const int kTilesPerRow = 8; // Standard tile16 blockset layout
956 int tile_x = (tile_id % kTilesPerRow) * kTile16Size;
957 int tile_y = (tile_id / kTilesPerRow) * kTile16Size;
958
959 tile_data.resize(kTile16PixelCount);
960
961 // Manual extraction without the buggy offset increment
962 for (int ty = 0; ty < kTile16Size; ty++) {
963 for (int tx = 0; tx < kTile16Size; tx++) {
964 int pixel_x = tile_x + tx;
965 int pixel_y = tile_y + ty;
966 int src_index = (pixel_y * tile16_blockset_->atlas.width()) + pixel_x;
967 int dst_index = ty * kTile16Size + tx;
968
969 if (src_index < static_cast<int>(tile16_blockset_->atlas.size()) &&
970 dst_index < static_cast<int>(tile_data.size())) {
971 uint8_t pixel_value = tile16_blockset_->atlas.data()[src_index];
972 // Normalize pixel values to valid palette range
973 pixel_value &= 0x0F; // Keep only lower 4 bits for palette index
974 tile_data[dst_index] = pixel_value;
975 }
976 }
977 }
978 } else {
979 // Normalize the extracted data based on settings
980 if (auto_normalize_pixels_) {
981 for (auto& pixel : tile_data) {
982 pixel &= palette_normalization_mask_;
983 }
984 }
985 }
986
987 // Create the bitmap with the extracted data
988 current_tile16_bmp_.Create(kTile16Size, kTile16Size, 8, tile_data);
989
990 // Set the correct palette - tile16 should have independent palette
991 const auto& ow_main_pal_group = rom()->palette_group().overworld_main;
992 if (ow_main_pal_group.size() > 0) {
993 // Use SetPalette instead of SetPaletteWithTransparent for tile16
994 current_tile16_bmp_.SetPalette(ow_main_pal_group[0]);
995 }
996
997 // Render the bitmap
998 core::Renderer::Get().RenderBitmap(&current_tile16_bmp_);
999
1000 // Simple success logging
1001 util::logf("SetCurrentTile: loaded tile %d successfully", tile_id);
1002
1003 return absl::OkStatus();
1004}

1292// Undo/Redo system
1293void Tile16Editor::SaveUndoState() {
1294 if (!current_tile16_bmp_.is_active()) {
1295 return;
1296 }
1297
1298 UndoState state;
1299 state.tile_id = current_tile16_;
1300 state.tile_bitmap.Create(16, 16, 8, current_tile16_bmp_.vector());
1301 state.tile_bitmap.SetPalette(current_tile16_bmp_.palette());
1302 state.palette = current_palette_;
1303 state.x_flip = x_flip;
1304 state.y_flip = y_flip;
1305 state.priority = priority_tile;
1306
1307 undo_stack_.push_back(std::move(state));
1308
1309 // Limit undo stack size
1310 if (undo_stack_.size() > kMaxUndoStates_) {
1311 undo_stack_.erase(undo_stack_.begin());
1312 }

1417 }

1420 return absl::OutOfRangeError("Current tile16 ID out of range");
1421 }
1422
1423 // Update the tilemap with our modified bitmap
1424 tile16_blockset_->tile_bitmaps[current_tile16_] = current_tile16_bmp_;
