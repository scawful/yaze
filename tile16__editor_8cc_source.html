154absl::Status Tile16Editor::DrawToCurrentTile16(ImVec2 click_position) {
155 constexpr int tile8_size = 8;
156 constexpr int tile16_size = 16;
157
158 // Calculate the tile index for x and y based on the click_position
159 // Adjusting for Tile16 (16x16) which contains 4 Tile8 (8x8)
160 int tile_index_x = static_cast<int>(click_position.x) / tile8_size;
161 int tile_index_y = static_cast<int>(click_position.y) / tile8_size;
162 std::cout << "Tile Index X: " << tile_index_x << std::endl;
163 std::cout << "Tile Index Y: " << tile_index_y << std::endl;
164
165 // Calculate the pixel start position within the Tile16
166 ImVec2 start_position;
167 start_position.x = tile_index_x * 0x40;
168 start_position.y = tile_index_y * 0x40;
169 std::cout << "Start Position X: " << start_position.x << std::endl;
170 std::cout << "Start Position Y: " << start_position.y << std::endl;
171
172 // Draw the Tile8 to the correct position within the Tile16
173 for (int y = 0; y < tile8_size; ++y) {
174 for (int x = 0; x < tile8_size; ++x) {
175 int pixel_index =
176 (start_position.y + y) * tile16_size + ((start_position.x) + x);
177 int gfx_pixel_index = y * tile8_size + x;
178 current_tile16_bmp_.WriteToPixel(
179 pixel_index,
180 current_gfx_individual_[current_tile8_].data()[gfx_pixel_index]);
181 }
182 }
183
184 return absl::OkStatus();
185}

276absl::Status Tile16Editor::LoadTile8() {
277 auto ow_main_pal_group = rom()->palette_group().overworld_main;
278 current_gfx_individual_.reserve(1024);
279
280 std::vector<std::future<std::array<uint8_t, 0x40>>> futures;
281
282 for (int index = 0; index < 1024; index++) {
283 auto task_function = [&]() {
284 std::array<uint8_t, 0x40> tile_data;
285 // Copy the pixel data for the current tile into the vector
286 for (int ty = 0; ty < 8; ty++) {
287 for (int tx = 0; tx < 8; tx++) {
288 // Current Gfx Data is 16 sheets of 8x8 tiles ordered 16 wide by 4
289 // tall
290
291 // Calculate the position in the tile data vector
292 int position = tx + (ty * 0x08);
293
294 // Calculate the position in the current gfx data
295 int num_columns = current_gfx_bmp_.width() / 8;
296 int x = (index % num_columns) * 8 + tx;
297 int y = (index / num_columns) * 8 + ty;
298 int gfx_position = x + (y * 0x100);
299
300 // Get the pixel value from the current gfx data
301 uint8_t value = current_gfx_bmp_.data()[gfx_position];
302
303 if (value & 0x80) {
304 value -= 0x88;
305 }
306
307 tile_data[position] = value;
308 }
309 }
310 return tile_data;
311 };
312 futures.emplace_back(std::async(std::launch::async, task_function));
313 }
314
315 for (auto &future : futures) {
316 future.wait();
317 auto tile_data = future.get();
318 current_gfx_individual_.emplace_back();
319 auto &tile_bitmap = current_gfx_individual_.back();
320 tile_bitmap.Create(0x08, 0x08, 0x08, tile_data);
321 RETURN_IF_ERROR(tile_bitmap.ApplyPaletteWithTransparent(
322 ow_main_pal_group[0], current_palette_));
323 Renderer::GetInstance().RenderBitmap(&tile_bitmap);
324 }
325
326 map_blockset_loaded_ = true;
327
328 return absl::OkStatus();
329}
