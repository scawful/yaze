39 while (SDL_PollEvent(&event)) {

41 switch (event.type) {
42 case SDL_KEYDOWN:
43 case SDL_KEYUP: {
44 io.KeyShift = ((SDL_GetModState() & KMOD_SHIFT) != 0);
45 io.KeyCtrl = ((SDL_GetModState() & KMOD_CTRL) != 0);
46 io.KeyAlt = ((SDL_GetModState() & KMOD_ALT) != 0);
47 io.KeySuper = ((SDL_GetModState() & KMOD_GUI) != 0);
48 break;
49 }
50 case SDL_WINDOWEVENT:
51 switch (event.window.event) {
52 case SDL_WINDOWEVENT_CLOSE:
53 active_ = false;
54 break;
55 case SDL_WINDOWEVENT_SIZE_CHANGED:
56 io.DisplaySize.x = static_cast<float>(event.window.data1);

58 break;
59 default:
60 break;
61 }
62 break;
63 default:
64 break;
65 }
66 }

68 int mouseX;
69 int mouseY;
70 const int buttons = SDL_GetMouseState(&mouseX, &mouseY);

72 io.DeltaTime = 1.0f / 60.0f;
73 io.MousePos = ImVec2(static_cast<float>(mouseX), static_cast<float>(mouseY));
74 io.MouseDown[0] = buttons & SDL_BUTTON(SDL_BUTTON_LEFT);
75 io.MouseDown[1] = buttons & SDL_BUTTON(SDL_BUTTON_RIGHT);

190absl::Status Controller::LoadAudioDevice() {
191 SDL_AudioSpec want, have;
192 SDL_memset(&want, 0, sizeof(want));
193 want.freq = audio_frequency_;
194 want.format = AUDIO_S16;
195 want.channels = 2;
196 want.samples = 2048;
197 want.callback = NULL; // Uses the queue
198 audio_device_ = SDL_OpenAudioDevice(NULL, 0, &want, &have, 0);
199 if (audio_device_ == 0) {
200 return absl::InternalError(
201 absl::StrFormat("Failed to open audio: %s\n", SDL_GetError()));
202 }
203 audio_buffer_ = std::make_shared<int16_t>(audio_frequency_ / 50 * 4);
204 SDL_PauseAudioDevice(audio_device_, 0);
205 editor_manager_.emulator().set_audio_buffer(audio_buffer_.get());
206 editor_manager_.emulator().set_audio_device_id(audio_device_);
207 return absl::OkStatus();
208}
