20RoomSize CalculateRoomSize(Rom *rom, int room_id) {
21 // Calculate the size of the room based on how many objects are used per room
22 // Some notes from hacker Zarby89
23 // vanilla rooms are using in average ~0x80 bytes
24 // a "normal" person who wants more details than vanilla will use around 0x100
25 // bytes per rooms you could fit 128 rooms like that in 1 bank
26 // F8000 I don't remember if that's PC or snes tho
27 // Check last rooms
28 // F8000+(roomid*3)
29 // So we want to search the rom() object at this addressed based on the room
30 // ID since it's the roomid * 3 we will by pulling 3 bytes at a time We can do
31 // this with the rom()->ReadByteVector(addr, size)
32 // Existing room size address calculation...
33 RoomSize room_size;
34 room_size.room_size_pointer = 0;
35 room_size.room_size = 0;
36
37 auto room_size_address = 0xF8000 + (room_id * 3);
38 util::logf("Room #%#03X Addresss: %#06X", room_id, room_size_address);
39
40 // Reading bytes for long address construction
41 uint8_t low = rom->data()[room_size_address];
42 uint8_t high = rom->data()[room_size_address + 1];
43 uint8_t bank = rom->data()[room_size_address + 2];
44
45 // Constructing the long address
46 int long_address = (bank << 16) | (high << 8) | low;
47 util::logf("%#06X", long_address);
48 room_size.room_size_pointer = long_address;
49
50 if (long_address == 0x0A8000) {
51 // Blank room disregard in size calculation
52 util::logf("Size of Room #%#03X: 0 bytes", room_id);
53 room_size.room_size = 0;
54 } else {
55 // use the long address to calculate the size of the room
56 // we will use the room_id_ to calculate the next room's address
57 // and subtract the two to get the size of the room
58
59 int next_room_address = 0xF8000 + ((room_id + 1) * 3);
60 util::logf("Next Room Address: %#06X", next_room_address);
61
62 // Reading bytes for long address construction
63 uint8_t next_low = rom->data()[next_room_address];
64 uint8_t next_high = rom->data()[next_room_address + 1];
65 uint8_t next_bank = rom->data()[next_room_address + 2];
66
67 // Constructing the long address
68 int next_long_address = (next_bank << 16) | (next_high << 8) | next_low;
69 util::logf("%#06X", next_long_address);
70
71 // Calculate the size of the room
72 int actual_room_size = next_long_address - long_address;
73 room_size.room_size = actual_room_size;
74 util::logf("Size of Room #%#03X: %d bytes", room_id, actual_room_size);
75 }
76
77 return room_size;
78}

237 return;
238 }
239
240 auto gfx_buffer_data = rom()->mutable_graphics_buffer();
241 if (!gfx_buffer_data || gfx_buffer_data->empty()) {
242 return;

244
245 // Copy room graphics to buffer
246 int sheet_pos = 0;
247 for (int i = 0; i < 16; i++) {
248 // Validate block index
249 if (blocks_[i] < 0 || blocks_[i] > 255) {
250 sheet_pos += kGfxBufferRoomOffset;

259 sheet_pos += kGfxBufferRoomOffset;

261 }
262
263 while (data < kGfxBufferRoomOffset) {
264 int buffer_index = data + block_offset;
265 if (buffer_index >= 0 && buffer_index < static_cast<int>(gfx_buffer_data->size())) {
266 uint8_t map_byte = (*gfx_buffer_data)[buffer_index];
267 if (i < 4) {
268 map_byte += kGfxBufferRoomSpriteLastLineOffset;
269 }
270
271 // Validate current_gfx16_ access
272 int gfx_index = data + sheet_pos;
273 if (gfx_index >= 0 && gfx_index < static_cast<int>(sizeof(current_gfx16_))) {
274 current_gfx16_[gfx_index] = map_byte;
275 }
276 }
277 data++;
278 }
279
280 sheet_pos += kGfxBufferRoomOffset;

329 // Validate animated_frame_ bounds

346 // Validate buffer access for first operation
347 int first_offset = data + (92 * 2048) + (512 * animated_frame_);
348 if (first_offset >= 0 && first_offset < static_cast<int>(gfx_buffer_data->size())) {

354 current_gfx16_[gfx_offset] = map_byte;

425void Room::ParseObjectsFromLocation(int objects_location) {
426 auto rom_data = rom()->vector();
427
428 z3_staircases_.clear();
429 int nbr_of_staircase = 0;
430
431 int pos = objects_location;
432 uint8_t b1 = 0;
433 uint8_t b2 = 0;
434 uint8_t b3 = 0;
435 uint8_t posX = 0;
436 uint8_t posY = 0;
437 uint8_t sizeX = 0;
438 uint8_t sizeY = 0;
439 uint8_t sizeXY = 0;
440 short oid = 0;
441 int layer = 0;
442 bool door = false;
443 bool end_read = false;
444
445 // Enhanced parsing loop with bounds checking
446 while (!end_read && pos < (int)rom_->size()) {
447 // Check if we have enough bytes to read
448 if (pos + 1 >= (int)rom_->size()) {
449 break;
450 }
451
452 b1 = rom_data[pos];
453 b2 = rom_data[pos + 1];
454
455 if (b1 == 0xFF && b2 == 0xFF) {
456 pos += 2; // Jump to next layer
457 layer++;
458 door = false;
459 if (layer == 3) {
460 break;
461 }
462 continue;
463 }
464
465 if (b1 == 0xF0 && b2 == 0xFF) {
466 pos += 2; // Jump to door section
467 door = true;
468 continue;
469 }
470
471 // Check if we have enough bytes for object data
472 if (pos + 2 >= (int)rom_->size()) {
473 break;
474 }
475
476 b3 = rom_data[pos + 2];
477 if (door) {
478 pos += 2;
479 } else {
480 pos += 3;
481 }
482
483 if (!door) {
484 // Parse object with enhanced validation
485 if (b3 >= 0xF8) {
486 oid = static_cast<short>((b3 << 4) |
487 0x80 + (((b2 & 0x03) << 2) + ((b1 & 0x03))));
488 posX = static_cast<uint8_t>((b1 & 0xFC) >> 2);
489 posY = static_cast<uint8_t>((b2 & 0xFC) >> 2);
490 sizeXY = static_cast<uint8_t>((((b1 & 0x03) << 2) + (b2 & 0x03)));
491 } else {
492 oid = b3;
493 posX = static_cast<uint8_t>((b1 & 0xFC) >> 2);
494 posY = static_cast<uint8_t>((b2 & 0xFC) >> 2);
495 sizeX = static_cast<uint8_t>((b1 & 0x03));
496 sizeY = static_cast<uint8_t>((b2 & 0x03));
497 sizeXY = static_cast<uint8_t>(((sizeX << 2) + sizeY));
498 }
499
500 if (b1 >= 0xFC) {
501 oid = static_cast<short>((b3 & 0x3F) + 0x100);
502 posX = static_cast<uint8_t>(((b2 & 0xF0) >> 4) + ((b1 & 0x3) << 4));
503 posY = static_cast<uint8_t>(((b2 & 0x0F) << 2) + ((b3 & 0xC0) >> 6));
504 sizeXY = 0;
505 }
506
507 // Validate object ID before creating object
508 if (oid >= 0 && oid <= 0x3FF) {
509 RoomObject r(oid, posX, posY, sizeXY, static_cast<uint8_t>(layer));
510 r.set_rom(rom_);
511 tile_objects_.push_back(r);
512
513 // Handle special object types
514 HandleSpecialObjects(oid, posX, posY, nbr_of_staircase);
515 }
516 } else {
517 // Handle door objects (placeholder for future implementation)
518 // tile_objects_.push_back(z3_object_door(static_cast<short>((b2 << 8) + b1),
519 // 0, 0, 0, static_cast<uint8_t>(layer)));
520 }
521 }
