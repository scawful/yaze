60void Room::CalculateRoomSize() {
61 // Calculate the size of the room based on how many objects are used per room
62 // Some notes from hacker Zarby89
63 // vanilla rooms are using in average ~0x80 bytes
64 // a "normal" person who wants more details than vanilla will use around 0x100
65 // bytes per rooms you could fit 128 rooms like that in 1 bank
66 // F8000 I don't remember if that's PC or snes tho
67 // Check last rooms
68 // F8000+(roomid*3)
69 // So we want to search the rom() object at this addressed based on the room
70 // ID since it's the roomid * 3 we will by pulling 3 bytes at a time We can do
71 // this with the rom()->ReadByteVector(addr, size)
72 try {
73 // Existing room size address calculation...
74 auto room_size_address = 0xF8000 + (room_id_ * 3);
75
76 std::cout << "Room #" << room_id_ << " Address: " << std::hex
77 << room_size_address << std::endl;
78
79 // Reading bytes for long address construction
80 uint8_t low = rom()->data()[room_size_address];
81 uint8_t high = rom()->data()[room_size_address + 1];
82 uint8_t bank = rom()->data()[room_size_address + 2];
83
84 // Constructing the long address
85 int long_address = (bank << 16) | (high << 8) | low;
86 std::cout << std::hex << std::setfill('0') << std::setw(6) << long_address
87 << std::endl;
88 room_size_pointer_ = long_address;
89
90 if (long_address == 0x0A8000) {
91 // Blank room disregard in size calculation
92 std::cout << "Size of Room #" << room_id_ << ": 0 bytes" << std::endl;
93 room_size_ = 0;
94 } else {
95 // use the long address to calculate the size of the room
96 // we will use the room_id_ to calculate the next room's address
97 // and subtract the two to get the size of the room
98
99 int next_room_address = 0xF8000 + ((room_id_ + 1) * 3);
100
101 std::cout << "Next Room Address: " << std::hex << next_room_address
102 << std::endl;
103
104 // Reading bytes for long address construction
105 uint8_t next_low = rom()->data()[next_room_address];
106 uint8_t next_high = rom()->data()[next_room_address + 1];
107 uint8_t next_bank = rom()->data()[next_room_address + 2];
108
109 // Constructing the long address
110 int next_long_address = (next_bank << 16) | (next_high << 8) | next_low;
111
112 std::cout << std::hex << std::setfill('0') << std::setw(6)
113 << next_long_address << std::endl;
114
115 // Calculate the size of the room
116 int room_size = next_long_address - long_address;
117 room_size_ = room_size;
118 std::cout << "Size of Room #" << room_id_ << ": " << std::dec << room_size
119 << " bytes" << std::endl;
120 }
121 } catch (const std::exception &e) {
122 std::cout << "Error: " << e.what() << std::endl;
123 }
