111absl::Status HandleTestCommand(const std::vector<std::string>& arg_vec) {
112 if (arg_vec.size() < 2 || arg_vec[0] != "--test") {
113 return absl::InvalidArgumentError("Usage: agent test --test <test_name>");
114 }
115
116#ifdef _WIN32
117 // Windows doesn't support fork/exec, so users must run tests directly
118 return absl::UnimplementedError(
119 "GUI test command is not supported on Windows. "
120 "Please run yaze_test.exe directly with --enable-ui-tests flag.");
121#else
122 // Unix-like systems (macOS, Linux) support fork/exec for process spawning
123 std::string test_name = arg_vec[1];
124
125 // Get the executable path using platform-specific methods
126 char exe_path[1024];
127#ifdef __APPLE__
128 uint32_t size = sizeof(exe_path);
129 if (_NSGetExecutablePath(exe_path, &size) != 0) {
130 return absl::InternalError("Could not get executable path");
131 }
132#elif defined(__linux__)
133 ssize_t len = readlink("/proc/self/exe", exe_path, sizeof(exe_path) - 1);
134 if (len == -1) {
135 return absl::InternalError("Could not get executable path");
136 }
137 exe_path[len] = '\0';
138#else
139 return absl::UnimplementedError(
140 "GUI test command is not supported on this platform. "
141 "Please run yaze_test directly with --enable-ui-tests flag.");
142#endif
143
144 // Extract directory from executable path
145 std::string exe_dir = std::string(exe_path);
146 exe_dir = exe_dir.substr(0, exe_dir.find_last_of("/"));
147 std::string yaze_test_path = exe_dir + "/yaze_test";
148
149 // Prepare command arguments for execv
150 std::vector<std::string> command_args;
151 command_args.push_back(yaze_test_path);
152 command_args.push_back("--enable-ui-tests");
153 command_args.push_back("--test=" + test_name);
154
155 std::vector<char*> argv;
156 for (const auto& arg : command_args) {
157 argv.push_back((char*)arg.c_str());
158 }
159 argv.push_back(nullptr);
160
161 // Fork and execute the test process
162 pid_t pid = fork();
163 if (pid == -1) {
164 return absl::InternalError("Failed to fork process");
165 }
166
167 if (pid == 0) {
168 // Child process: execute the test binary
169 execv(yaze_test_path.c_str(), argv.data());
170 // If execv returns, it must have failed
171 _exit(EXIT_FAILURE); // Use _exit in child process after failed exec
172 } else {
173 // Parent process: wait for child to complete
174 int status;
175 if (waitpid(pid, &status, 0) == -1) {
176 return absl::InternalError("Failed to wait for child process");
177 }
178
179 if (WIFEXITED(status)) {
180 int exit_code = WEXITSTATUS(status);
181 if (exit_code == 0) {
182 return absl::OkStatus();
183 } else {
184 return absl::InternalError(
185 absl::StrFormat("yaze_test exited with code %d", exit_code));
186 }
187 } else if (WIFSIGNALED(status)) {
188 return absl::InternalError(
189 absl::StrFormat("yaze_test terminated by signal %d", WTERMSIG(status)));
190 } else {
191 return absl::InternalError("yaze_test terminated abnormally");
192 }
193 }
194#endif

230absl::Status Agent::Run(const std::vector<std::string>& arg_vec) {

233 }

245 return HandleTestCommand(subcommand_args);

247 return HandleLearnCommand();

249 return HandleCommitCommand(rom_);

251 return HandleRevertCommand(rom_);
