93absl::Status HandleMusicInfoCommand(
94 const std::vector<std::string>& arg_vec, Rom* rom_context) {
95 if (!rom_context || !rom_context->is_loaded()) {
96 return absl::FailedPreconditionError("ROM not loaded");
97 }
98
99 int track_id = -1;
100 std::string format = "json";
101
102 for (size_t i = 0; i < arg_vec.size(); ++i) {
103 const std::string& token = arg_vec[i];
104 if (token == "--id" || token == "--track") {
105 if (i + 1 < arg_vec.size()) {
106 std::string id_str = arg_vec[++i];
107 if (absl::StartsWith(id_str, "0x") || absl::StartsWith(id_str, "0X")) {
108 track_id = std::stoi(id_str, nullptr, 16);
109 } else {
110 absl::SimpleAtoi(id_str, &track_id);
111 }
112 }
113 } else if (absl::StartsWith(token, "--id=") || absl::StartsWith(token, "--track=")) {
114 std::string id_str = token.substr(token.find('=') + 1);
115 if (absl::StartsWith(id_str, "0x") || absl::StartsWith(id_str, "0X")) {
116 track_id = std::stoi(id_str, nullptr, 16);
117 } else {
118 absl::SimpleAtoi(id_str, &track_id);
119 }
120 } else if (token == "--format") {
121 if (i + 1 < arg_vec.size()) {
122 format = arg_vec[++i];
123 }
124 } else if (absl::StartsWith(token, "--format=")) {
125 format = token.substr(9);
126 }
127 }
128
129 if (track_id < 0) {
130 return absl::InvalidArgumentError(
131 "Usage: music-info --id <track_id> [--format json|text]");
132 }

151 std::cout << "{\n";
