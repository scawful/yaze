126void Arena::UpdateTexture(SDL_Texture* texture, SDL_Surface* surface) {
127 if (!texture || !surface) {
128 SDL_Log("Invalid texture or surface passed to UpdateTexture");
129 return;
130 }
131
132 if (surface->pixels == nullptr) {
133 SDL_Log("Surface pixels are nullptr");
134 return;
135 }
136
137 // Additional safety checks to prevent crashes
138 if (surface->w <= 0 || surface->h <= 0) {
139 SDL_Log("Invalid surface dimensions: %dx%d", surface->w, surface->h);
140 return;
141 }
142
143 if (!surface->format) {
144 SDL_Log("Surface format is nullptr");
145 return;
146 }
147
148 // Convert surface to RGBA8888 format for texture compatibility
149 auto converted_surface =
150 std::unique_ptr<SDL_Surface, util::SDL_Surface_Deleter>(
151 SDL_ConvertSurfaceFormat(surface, SDL_PIXELFORMAT_RGBA8888, 0),
152 util::SDL_Surface_Deleter());
153
154 if (!converted_surface) {
155 SDL_Log("SDL_ConvertSurfaceFormat failed: %s", SDL_GetError());
156 return;
157 }
158
159 // Additional validation for converted surface
160 if (!converted_surface->pixels) {
161 SDL_Log("Converted surface pixels are nullptr");
162 return;
163 }
164
165 if (converted_surface->w <= 0 || converted_surface->h <= 0) {
166 SDL_Log("Invalid converted surface dimensions: %dx%d", converted_surface->w, converted_surface->h);
167 return;
168 }
169
170 // Validate texture before locking
171 int texture_w, texture_h;
172 if (SDL_QueryTexture(texture, nullptr, nullptr, &texture_w, &texture_h) != 0) {
173 SDL_Log("SDL_QueryTexture failed: %s", SDL_GetError());
174 return;
175 }
176
177 if (texture_w != converted_surface->w || texture_h != converted_surface->h) {
178 SDL_Log("Texture/surface size mismatch: texture=%dx%d, surface=%dx%d",
179 texture_w, texture_h, converted_surface->w, converted_surface->h);
180 return;
181 }
182
183 // Lock texture for direct pixel access
184 void* pixels;
185 int pitch;
186 if (SDL_LockTexture(texture, nullptr, &pixels, &pitch) != 0) {
187 SDL_Log("SDL_LockTexture failed: %s", SDL_GetError());
188 return;
189 }
190
191 // Additional safety check for locked pixels
192 if (!pixels) {
193 SDL_Log("Locked texture pixels are nullptr");
194 SDL_UnlockTexture(texture);
195 return;
196 }
197
198 // Validate copy size to prevent buffer overrun
199 size_t copy_size = converted_surface->h * converted_surface->pitch;
200 size_t max_texture_size = texture_h * pitch;
201
202 if (copy_size > max_texture_size) {
203 SDL_Log("Copy size (%zu) exceeds texture capacity (%zu)", copy_size, max_texture_size);
204 SDL_UnlockTexture(texture);
205 return;
206 }
207
208 // Copy pixel data efficiently with bounds checking
209 memcpy(pixels, converted_surface->pixels, copy_size);
210
211 SDL_UnlockTexture(texture);
212}

214SDL_Surface* Arena::AllocateSurface(int width, int height, int depth,
215 int format) {
216 // Try to reuse existing surface of same size and format from pool
217 for (auto it = surface_pool_.available_surfaces_.begin();
218 it != surface_pool_.available_surfaces_.end(); ++it) {
219 auto& info = surface_pool_.surface_info_[*it];
220 if (std::get<0>(info) == width && std::get<1>(info) == height &&
221 std::get<2>(info) == depth && std::get<3>(info) == format) {
222 SDL_Surface* surface = *it;
223 surface_pool_.available_surfaces_.erase(it);
224
225 // Clear the surface pixels before reusing for safety
226 if (surface && surface->pixels) {
227 memset(surface->pixels, 0, surface->h * surface->pitch);
228 }
229
230 // Store in hash map with automatic cleanup
231 surfaces_[surface] =
232 std::unique_ptr<SDL_Surface, util::SDL_Surface_Deleter>(surface);
233 return surface;
234 }
235 }
236
237 // Create new surface if none available in pool

320void Arena::UpdateTextureRegion(SDL_Texture* texture, SDL_Surface* surface, SDL_Rect* rect) {
321 if (!texture || !surface) {
322 return;
323 }
324
325 if (surface->pixels == nullptr) {
326 return;
327 }
328
329 // Convert surface to RGBA8888 format for texture compatibility
330 auto converted_surface =
331 std::unique_ptr<SDL_Surface, util::SDL_Surface_Deleter>(
332 SDL_ConvertSurfaceFormat(surface, SDL_PIXELFORMAT_RGBA8888, 0),
333 util::SDL_Surface_Deleter());
334
335 if (!converted_surface) {
336 return;
337 }
338
339 // Lock texture for direct pixel access
340 void* pixels;
341 int pitch;
342 if (SDL_LockTexture(texture, rect, &pixels, &pitch) != 0) {
343 return;
344 }
345
346 // Copy pixel data efficiently with bounds checking
347 if (rect) {
348 // Validate rect bounds against surface dimensions
349 int max_x = std::min(rect->x + rect->w, converted_surface->w);
350 int max_y = std::min(rect->y + rect->h, converted_surface->h);
351 int safe_x = std::max(0, rect->x);
352 int safe_y = std::max(0, rect->y);
353 int safe_w = max_x - safe_x;
354 int safe_h = max_y - safe_y;
355
356
357 if (safe_w > 0 && safe_h > 0) {
358 // Copy only the safe region
359 int src_offset = safe_y * converted_surface->pitch + safe_x * 4; // 4 bytes per RGBA pixel
360 int dst_offset = 0;
361 for (int y = 0; y < safe_h; y++) {
362 // Additional safety check for each row
363 if (src_offset + safe_w * 4 <= converted_surface->h * converted_surface->pitch) {
364 memcpy(static_cast<char*>(pixels) + dst_offset,
365 static_cast<char*>(converted_surface->pixels) + src_offset,
366 safe_w * 4);
367 }
368 src_offset += converted_surface->pitch;
369 dst_offset += pitch;
370 }
371 }
372 } else {
373 // Copy entire surface
374 memcpy(pixels, converted_surface->pixels,
375 converted_surface->h * converted_surface->pitch);
376 }
377
378 SDL_UnlockTexture(texture);
379}

458std::vector<gfx::Bitmap*> Arena::GetNextDeferredTextureBatch(
459 int high_priority_limit, int low_priority_limit) {
460 std::lock_guard<std::mutex> lock(deferred_mutex_);
461
462 std::vector<gfx::Bitmap*> batch;
463
464 if (deferred_textures_.empty()) {
465 return batch;
466 }
467
468 // Sort by priority (lower number = higher priority)
469 std::sort(deferred_textures_.begin(), deferred_textures_.end(),
470 [](const DeferredTexture& a, const DeferredTexture& b) {
471 return a.priority < b.priority;
472 });
473
474 // Phase 1: Collect high-priority items (priority 0-10)
475 auto it = deferred_textures_.begin();
476 while (it != deferred_textures_.end() && batch.size() < static_cast<size_t>(high_priority_limit)) {
477 if (it->bitmap && it->priority <= 10 && !it->bitmap->texture()) {
478 batch.push_back(it->bitmap);
479 it = deferred_textures_.erase(it);
480 } else {
481 ++it;
482 }
483 }
484
485 // Phase 2: Collect low-priority items (priority 11+) if we have capacity
486 if (batch.size() < static_cast<size_t>(high_priority_limit)) {
487 it = deferred_textures_.begin();
488 int low_count = 0;
489 while (it != deferred_textures_.end() && low_count < low_priority_limit) {
490 if (it->bitmap && it->priority > 10 && !it->bitmap->texture()) {
491 batch.push_back(it->bitmap);
492 low_count++;
493 it = deferred_textures_.erase(it);
494 } else {
495 ++it;
496 }
497 }
498 }
499
500 return batch;
501}
