125void Arena::UpdateTexture(SDL_Texture* texture, SDL_Surface* surface) {
126 if (!texture || !surface) {
127 SDL_Log("Invalid texture or surface passed to UpdateTexture");
128 return;
129 }
130
131 if (surface->pixels == nullptr) {
132 SDL_Log("Surface pixels are nullptr");
133 return;
134 }
135
136 // Additional safety checks to prevent crashes
137 if (surface->w <= 0 || surface->h <= 0) {
138 SDL_Log("Invalid surface dimensions: %dx%d", surface->w, surface->h);
139 return;
140 }
141
142 if (!surface->format) {
143 SDL_Log("Surface format is nullptr");
144 return;
145 }
146
147 // Convert surface to RGBA8888 format for texture compatibility
148 auto converted_surface =
149 std::unique_ptr<SDL_Surface, util::SDL_Surface_Deleter>(
150 SDL_ConvertSurfaceFormat(surface, SDL_PIXELFORMAT_RGBA8888, 0),
151 util::SDL_Surface_Deleter());
152
153 if (!converted_surface) {
154 SDL_Log("SDL_ConvertSurfaceFormat failed: %s", SDL_GetError());
155 return;
156 }
157
158 // Additional validation for converted surface
159 if (!converted_surface->pixels) {
160 SDL_Log("Converted surface pixels are nullptr");
161 return;
162 }
163
164 if (converted_surface->w <= 0 || converted_surface->h <= 0) {
165 SDL_Log("Invalid converted surface dimensions: %dx%d", converted_surface->w, converted_surface->h);
166 return;
167 }
168
169 // Validate texture before locking
170 int texture_w, texture_h;
171 if (SDL_QueryTexture(texture, nullptr, nullptr, &texture_w, &texture_h) != 0) {
172 SDL_Log("SDL_QueryTexture failed: %s", SDL_GetError());
173 return;
174 }
175
176 if (texture_w != converted_surface->w || texture_h != converted_surface->h) {
177 SDL_Log("Texture/surface size mismatch: texture=%dx%d, surface=%dx%d",
178 texture_w, texture_h, converted_surface->w, converted_surface->h);
179 return;
180 }
181
182 // Lock texture for direct pixel access
183 void* pixels;
184 int pitch;
185 if (SDL_LockTexture(texture, nullptr, &pixels, &pitch) != 0) {
186 SDL_Log("SDL_LockTexture failed: %s", SDL_GetError());
187 return;
188 }
189
190 // Additional safety check for locked pixels
191 if (!pixels) {
192 SDL_Log("Locked texture pixels are nullptr");
193 SDL_UnlockTexture(texture);
194 return;
195 }
196
197 // Validate copy size to prevent buffer overrun
198 size_t copy_size = converted_surface->h * converted_surface->pitch;
199 size_t max_texture_size = texture_h * pitch;
200
201 if (copy_size > max_texture_size) {
202 SDL_Log("Copy size (%zu) exceeds texture capacity (%zu)", copy_size, max_texture_size);
203 SDL_UnlockTexture(texture);
204 return;
205 }
206
207 // Copy pixel data efficiently with bounds checking
208 memcpy(pixels, converted_surface->pixels, copy_size);
209
210 SDL_UnlockTexture(texture);
211}

319void Arena::UpdateTextureRegion(SDL_Texture* texture, SDL_Surface* surface, SDL_Rect* rect) {
320 if (!texture || !surface) {
321 return;
322 }
323
324 if (surface->pixels == nullptr) {
325 return;
326 }
327
328 // Convert surface to RGBA8888 format for texture compatibility
329 auto converted_surface =
330 std::unique_ptr<SDL_Surface, util::SDL_Surface_Deleter>(
331 SDL_ConvertSurfaceFormat(surface, SDL_PIXELFORMAT_RGBA8888, 0),
332 util::SDL_Surface_Deleter());
333
334 if (!converted_surface) {
335 return;
336 }
337
338 // Lock texture for direct pixel access
339 void* pixels;
340 int pitch;
341 if (SDL_LockTexture(texture, rect, &pixels, &pitch) != 0) {
342 return;
343 }
344
345 // Copy pixel data efficiently with bounds checking
346 if (rect) {
347 // Validate rect bounds against surface dimensions
348 int max_x = std::min(rect->x + rect->w, converted_surface->w);
349 int max_y = std::min(rect->y + rect->h, converted_surface->h);
350 int safe_x = std::max(0, rect->x);
351 int safe_y = std::max(0, rect->y);
352 int safe_w = max_x - safe_x;
353 int safe_h = max_y - safe_y;
354
355
356 if (safe_w > 0 && safe_h > 0) {
357 // Copy only the safe region
358 int src_offset = safe_y * converted_surface->pitch + safe_x * 4; // 4 bytes per RGBA pixel
359 int dst_offset = 0;
360 for (int y = 0; y < safe_h; y++) {
361 // Additional safety check for each row
362 if (src_offset + safe_w * 4 <= converted_surface->h * converted_surface->pitch) {
363 memcpy(static_cast<char*>(pixels) + dst_offset,
364 static_cast<char*>(converted_surface->pixels) + src_offset,
365 safe_w * 4);
366 }
367 src_offset += converted_surface->pitch;
368 dst_offset += pitch;
369 }
370 }
371 } else {
372 // Copy entire surface
373 memcpy(pixels, converted_surface->pixels,
374 converted_surface->h * converted_surface->pitch);
375 }
376
377 SDL_UnlockTexture(texture);
378}
