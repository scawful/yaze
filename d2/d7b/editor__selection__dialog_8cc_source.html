208void EditorSelectionDialog::MarkRecentlyUsed(EditorType type) {
209 // Remove if already in list
210 auto it = std::find(recent_editors_.begin(), recent_editors_.end(), type);
211 if (it != recent_editors_.end()) {
212 recent_editors_.erase(it);
213 }
214
215 // Add to front
216 recent_editors_.insert(recent_editors_.begin(), type);
217
218 // Limit size
219 if (recent_editors_.size() > kMaxRecentEditors) {
220 recent_editors_.resize(kMaxRecentEditors);
221 }
222
223 SaveRecentEditors();
224}

226void EditorSelectionDialog::LoadRecentEditors() {
227 try {
228 auto data = util::LoadConfigFile("recent_editors.txt");
229 if (!data.empty()) {
230 std::istringstream ss(data);
231 std::string line;
232 while (std::getline(ss, line) &&
233 recent_editors_.size() < kMaxRecentEditors) {
234 int type_int = std::stoi(line);
235 if (type_int >= 0 && type_int < static_cast<int>(EditorType::kSettings)) {
236 recent_editors_.push_back(static_cast<EditorType>(type_int));
237 }
238 }
239 }
240 } catch (...) {
241 // Ignore errors, just start with empty recent list
242 }
243}
