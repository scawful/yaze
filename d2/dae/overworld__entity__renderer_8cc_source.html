24void OverworldEntityRenderer::DrawEntrances(ImVec2 canvas_p0, ImVec2 scrolling,
25 int current_world,
26 int current_mode) {
27 int i = 0;
28 for (auto& each : overworld_->entrances()) {
29 if (each.map_id_ < 0x40 + (current_world * 0x40) &&
30 each.map_id_ >= (current_world * 0x40) && !each.deleted) {
31 // Use theme-aware color with proper transparency
32 ImVec4 entrance_color = GetEntranceColor();
33 if (each.is_hole_) {
34 // Holes are more opaque for visibility
35 entrance_color.w = 0.78f; // 200/255 alpha
36 }
37 canvas_->DrawRect(each.x_, each.y_, 16, 16, entrance_color);
38 std::string str = util::HexByte(each.entrance_id_);
39
40 if (current_mode == 1) { // EditingMode::ENTRANCES
41 HandleEntityDragging(&each, canvas_p0, scrolling, is_dragging_entity_,
42 dragged_entity_, current_entity_);
43
44 if (IsMouseHoveringOverEntity(each, canvas_p0, scrolling) &&
45 ImGui::IsMouseDoubleClicked(ImGuiMouseButton_Left)) {
46 jump_to_tab_ = each.entrance_id_;
47 }
48
49 if (IsMouseHoveringOverEntity(each, canvas_p0, scrolling) &&
50 ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
51 current_entrance_id_ = i;
52 current_entrance_ = each;
53 }
54 }
55
56 canvas_->DrawText(str, each.x_, each.y_);
57 }
58 i++;
59 }
60
61 if (DrawEntranceInserterPopup()) {
62 // Get the deleted entrance ID and insert it at the mouse position
63 auto deleted_entrance_id = overworld_->deleted_entrances().back();
64 overworld_->deleted_entrances().pop_back();
65 auto& entrance = overworld_->entrances()[deleted_entrance_id];
66 entrance.map_id_ = current_map_;
67 entrance.entrance_id_ = deleted_entrance_id;
68 entrance.x_ = canvas_->hover_mouse_pos().x;
69 entrance.y_ = canvas_->hover_mouse_pos().y;
70 entrance.deleted = false;
71 }
72
73 if (current_mode == 1) { // EditingMode::ENTRANCES
74 const auto is_hovering =
75 IsMouseHoveringOverEntity(current_entrance_, canvas_p0, scrolling);
76
77 if (!is_hovering && ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
78 ImGui::OpenPopup("Entrance Inserter");
79 } else {
80 if (DrawOverworldEntrancePopup(
81 overworld_->entrances()[current_entrance_id_])) {
82 overworld_->entrances()[current_entrance_id_] = current_entrance_;
83 }
84
85 if (overworld_->entrances()[current_entrance_id_].deleted) {
86 overworld_->mutable_deleted_entrances()->emplace_back(
87 current_entrance_id_);
88 }
89 }

195void OverworldEntityRenderer::DrawSprites(int current_world, int game_state,
196 int current_mode) {
197 int i = 0;
198 for (auto& sprite : *overworld_->mutable_sprites(game_state)) {
199 // Filter sprites by current world - only show sprites for the current world
200 if (!sprite.deleted() && sprite.map_id() < 0x40 + (current_world * 0x40) &&
201 sprite.map_id() >= (current_world * 0x40)) {
202 // Sprites are already stored with global coordinates (realX, realY from
203 // ROM loading) So we can use sprite.x_ and sprite.y_ directly
204 int sprite_x = sprite.x_;
205 int sprite_y = sprite.y_;
206
207 // Temporarily update sprite coordinates for entity interaction
208 int original_x = sprite.x_;
209 int original_y = sprite.y_;
210
211 canvas_->DrawRect(sprite_x, sprite_y, 16, 16, GetSpriteColor());
212 if (current_mode == 4) { // EditingMode::SPRITES
213 HandleEntityDragging(&sprite, canvas_->zero_point(),
214 canvas_->scrolling(), is_dragging_entity_,
215 dragged_entity_, current_entity_);
216 if (IsMouseHoveringOverEntity(sprite, canvas_->zero_point(),
217 canvas_->scrolling()) &&
218 ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
219 current_sprite_id_ = i;
220 current_sprite_ = sprite;
221 }
222 }
223 if (core::FeatureFlags::get().overworld.kDrawOverworldSprites) {
224 if ((*sprite_previews_)[sprite.id()].is_active()) {
225 canvas_->DrawBitmap((*sprite_previews_)[sprite.id()], sprite_x,
226 sprite_y, 2.0f);
227 }
228 }
229
230 canvas_->DrawText(absl::StrFormat("%s", sprite.name()), sprite_x,
231 sprite_y);
232
233 // Restore original coordinates
234 sprite.x_ = original_x;
235 sprite.y_ = original_y;
236 }
237 i++;
238 }
239
240 DrawSpriteInserterPopup();
241 if (current_mode == 4) { // EditingMode::SPRITES
242 const auto hovering = IsMouseHoveringOverEntity(
243 overworld_->mutable_sprites(game_state)->at(current_sprite_id_),
244 canvas_->zero_point(), canvas_->scrolling());
245
246 if (!hovering && ImGui::IsMouseClicked(ImGuiMouseButton_Right)) {
247 ImGui::OpenPopup("Sprite Inserter");
248 } else {
249 if (DrawSpriteEditorPopup(overworld_->mutable_sprites(game_state)
250 ->at(current_sprite_id_))) {
251 overworld_->mutable_sprites(game_state)->at(current_sprite_id_) =
252 current_sprite_;
253 }
254 }
255 }
