169void RenderSnesPpu(Emulator* emu) {
170 if (!emu) return;
171
172 auto& theme_manager = ThemeManager::Get();
173 const auto& theme = theme_manager.GetCurrentTheme();
174
175 ImGui::PushStyleColor(ImGuiCol_ChildBg, ConvertColorToImVec4(theme.editor_background));
176 ImGui::BeginChild("##SNES_PPU", ImVec2(0, 0), true,
177 ImGuiWindowFlags_NoScrollbar | ImGuiWindowFlags_NoScrollWithMouse);
178
179 ImVec2 canvas_size = ImGui::GetContentRegionAvail();
180 ImVec2 snes_size = ImVec2(512, 480);
181
182 if (emu->is_snes_initialized() && emu->ppu_texture()) {
183 // Center the SNES display
184 float aspect = snes_size.x / snes_size.y;
185 float display_w = canvas_size.x;
186 float display_h = display_w / aspect;
187
188 if (display_h > canvas_size.y) {
189 display_h = canvas_size.y;
190 display_w = display_h * aspect;
191 }
192
193 float pos_x = (canvas_size.x - display_w) * 0.5f;
194 float pos_y = (canvas_size.y - display_h) * 0.5f;
195
196 ImGui::SetCursorPos(ImVec2(pos_x, pos_y));
197
198 // Render PPU texture
199 ImGui::Image((ImTextureID)(intptr_t)emu->ppu_texture(),
200 ImVec2(display_w, display_h),
201 ImVec2(0, 0), ImVec2(1, 1));
202 } else {
203 // Not initialized - show placeholder
204 ImVec2 text_size = ImGui::CalcTextSize("SNES PPU Output\n512x480");
205 ImGui::SetCursorPos(ImVec2((canvas_size.x - text_size.x) * 0.5f,
206 (canvas_size.y - text_size.y) * 0.5f));
207 ImGui::TextColored(ConvertColorToImVec4(theme.text_disabled),
208 ICON_MD_VIDEOGAME_ASSET "\nSNES PPU Output\n512x480");
209 }
210
211 ImGui::EndChild();
212 ImGui::PopStyleColor();
213}
