33absl::StatusOr<Rom> LoadRomFromFlag() {
34 std::string rom_path = absl::GetFlag(FLAGS_rom);
35 if (rom_path.empty()) {
36 return absl::FailedPreconditionError(
37 "No ROM loaded. Use --rom=<path> to specify ROM file.");
38 }
39
40 Rom rom;
41 auto status = rom.LoadFromFile(rom_path);
42 if (!status.ok()) {
43 return absl::FailedPreconditionError(absl::StrFormat(
44 "Failed to load ROM from '%s': %s", rom_path, status.message()));
45 }
46
47 return rom;
48}

50absl::StatusOr<Rom> LoadRomFromPathOrFlag(
51 const std::optional<std::string>& override_path) {
52 if (override_path.has_value()) {
53 Rom rom;
54 auto status = rom.LoadFromFile(*override_path);
55 if (!status.ok()) {
56 return absl::FailedPreconditionError(absl::StrFormat(
57 "Failed to load ROM from '%s': %s", *override_path,
58 status.message()));
59 }
60 return rom;
61 }
62 return LoadRomFromFlag();
63}

83 return absl::InvalidArgumentError("--format requires a value.");
