174// Configure Google Test filters based on test mode
175void ConfigureTestFilters(const TestConfig& config) {
176 std::vector<std::string> filters;
177
178 switch (config.mode) {
179 case TestMode::kUnit:
180 filters.push_back("UnitTest.*");
181 break;
182 case TestMode::kIntegration:
183 filters.push_back("IntegrationTest.*");
184 break;
185 case TestMode::kE2E:
186 filters.push_back("E2ETest.*");
187 break;
188 case TestMode::kRomDependent:
189 filters.push_back("*RomDependent*");
190 break;
191 case TestMode::kZSCustomOverworld:
192 filters.push_back("*ZSCustomOverworld*");
193 break;
194 case TestMode::kCore:
195 filters.push_back("*Core*");
196 filters.push_back("*Asar*");
197 filters.push_back("*Rom*");
198 break;
199 case TestMode::kGraphics:
200 filters.push_back("*Graphics*");
201 filters.push_back("*Gfx*");
202 filters.push_back("*Palette*");
203 filters.push_back("*Tile*");
204 break;
205 case TestMode::kEditor:
206 filters.push_back("*Editor*");
207 break;
208 case TestMode::kDeprecated:
209 filters.push_back("*Deprecated*");
210 break;
211 case TestMode::kSpecific:
212 if (!config.test_pattern.empty()) {
213 filters.push_back(config.test_pattern);
214 }
215 break;
216 case TestMode::kAll:
217 default:
218 // No filters - run all tests
219 break;
220 }
221
222 if (!filters.empty()) {
223 std::string filter_string;
224 for (size_t i = 0; i < filters.size(); i++) {
225 if (i > 0) filter_string += ":";
226 filter_string += filters[i];
227 }
228
229 ::testing::GTEST_FLAG(filter) = filter_string;
230
231 if (config.verbose) {
232 std::cout << "Test filter: " << filter_string << std::endl;
233 }
234 }
