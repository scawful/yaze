72void BackgroundRenderer::RenderGridBackground(ImDrawList* draw_list, const ImVec2& window_pos,
73 const ImVec2& window_size, const Color& grid_color) {
74 if (!draw_list || grid_settings_.grid_size <= 0) return;
75
76 // Grid parameters with optional animation
77 float grid_size = grid_settings_.grid_size;
78 float offset_x = 0.0f;
79 float offset_y = 0.0f;
80
81 // Apply animation if enabled
82 if (grid_settings_.enable_animation) {
83 float animation_offset = animation_time_ * grid_settings_.animation_speed * 10.0f;
84 offset_x = fmodf(animation_offset, grid_size);
85 offset_y = fmodf(animation_offset * 0.7f, grid_size); // Different speed for interesting effect
86 }
87
88 // Window center for radial calculations
89 ImVec2 center = ImVec2(window_pos.x + window_size.x * 0.5f,
90 window_pos.y + window_size.y * 0.5f);
91 float max_distance = sqrtf(window_size.x * window_size.x + window_size.y * window_size.y) * 0.5f;
92
93 // Apply breathing effect to color if enabled
94 Color themed_grid_color = grid_color;
95 themed_grid_color.alpha = grid_settings_.opacity;
96
97 if (grid_settings_.enable_breathing) {
98 float breathing_factor = 1.0f + grid_settings_.breathing_intensity *
99 sinf(animation_time_ * grid_settings_.breathing_speed);
100 themed_grid_color.red = std::min(1.0f, themed_grid_color.red * breathing_factor);
101 themed_grid_color.green = std::min(1.0f, themed_grid_color.green * breathing_factor);
102 themed_grid_color.blue = std::min(1.0f, themed_grid_color.blue * breathing_factor);
103 }
104
105 if (grid_settings_.enable_dots) {
106 // Render grid as dots
107 for (float x = window_pos.x - offset_x; x < window_pos.x + window_size.x + grid_size; x += grid_size) {
108 for (float y = window_pos.y - offset_y; y < window_pos.y + window_size.y + grid_size; y += grid_size) {
109 ImVec2 dot_pos(x, y);
110
111 // Calculate radial fade
112 float fade_factor = 1.0f;
113 if (grid_settings_.radial_fade) {
114 float distance = sqrtf((dot_pos.x - center.x) * (dot_pos.x - center.x) +
115 (dot_pos.y - center.y) * (dot_pos.y - center.y));
116 fade_factor = 1.0f - std::min(distance / grid_settings_.fade_distance, 1.0f);
117 fade_factor = fade_factor * fade_factor; // Square for smoother falloff
118 }
119
120 if (fade_factor > 0.01f) {
121 ImU32 dot_color = BlendColorWithFade(themed_grid_color, fade_factor);
122 DrawGridDot(draw_list, dot_pos, dot_color, grid_settings_.dot_size);
123 }
124 }
125 }
126 } else {
127 // Render grid as lines
128 // Vertical lines
129 for (float x = window_pos.x - offset_x; x < window_pos.x + window_size.x + grid_size; x += grid_size) {
130 ImVec2 line_start(x, window_pos.y);
131 ImVec2 line_end(x, window_pos.y + window_size.y);
132
133 // Calculate average fade for this line
134 float avg_fade = 0.0f;
135 if (grid_settings_.radial_fade) {
136 for (float y = window_pos.y; y < window_pos.y + window_size.y; y += grid_size * 0.5f) {
137 float distance = sqrtf((x - center.x) * (x - center.x) + (y - center.y) * (y - center.y));
138 float fade = 1.0f - std::min(distance / grid_settings_.fade_distance, 1.0f);
139 avg_fade += fade * fade;
140 }
141 avg_fade /= (window_size.y / (grid_size * 0.5f));
142 } else {
143 avg_fade = 1.0f;
144 }
145
146 if (avg_fade > 0.01f) {
147 ImU32 line_color = BlendColorWithFade(themed_grid_color, avg_fade);
148 DrawGridLine(draw_list, line_start, line_end, line_color, grid_settings_.line_thickness);
149 }
150 }
151
152 // Horizontal lines
153 for (float y = window_pos.y - offset_y; y < window_pos.y + window_size.y + grid_size; y += grid_size) {
154 ImVec2 line_start(window_pos.x, y);
155 ImVec2 line_end(window_pos.x + window_size.x, y);
156
157 // Calculate average fade for this line
158 float avg_fade = 0.0f;
159 if (grid_settings_.radial_fade) {
160 for (float x = window_pos.x; x < window_pos.x + window_size.x; x += grid_size * 0.5f) {
161 float distance = sqrtf((x - center.x) * (x - center.x) + (y - center.y) * (y - center.y));
162 float fade = 1.0f - std::min(distance / grid_settings_.fade_distance, 1.0f);
163 avg_fade += fade * fade;
164 }
165 avg_fade /= (window_size.x / (grid_size * 0.5f));
166 } else {
167 avg_fade = 1.0f;
168 }
169
170 if (avg_fade > 0.01f) {
171 ImU32 line_color = BlendColorWithFade(themed_grid_color, avg_fade);
172 DrawGridLine(draw_list, line_start, line_end, line_color, grid_settings_.line_thickness);

178void BackgroundRenderer::RenderRadialGradient(ImDrawList* draw_list, const ImVec2& center,
179 float radius, const Color& inner_color, const Color& outer_color) {
180 if (!draw_list) return;
181
182 const int segments = 32;
183 const int rings = 8;
184
185 for (int ring = 0; ring < rings; ++ring) {
186 float ring_radius = radius * (ring + 1) / rings;
187 float inner_ring_radius = radius * ring / rings;
188
189 // Interpolate colors for this ring
190 float t = static_cast<float>(ring) / rings;
191 Color ring_color = {
192 inner_color.red * (1.0f - t) + outer_color.red * t,
193 inner_color.green * (1.0f - t) + outer_color.green * t,
194 inner_color.blue * (1.0f - t) + outer_color.blue * t,
195 inner_color.alpha * (1.0f - t) + outer_color.alpha * t
196 };
197
198 ImU32 color = ImGui::ColorConvertFloat4ToU32(ConvertColorToImVec4(ring_color));
199
200 if (ring == 0) {
201 // Center circle
202 draw_list->AddCircleFilled(center, ring_radius, color, segments);
203 } else {
204 // Ring
205 for (int i = 0; i < segments; ++i) {
206 float angle1 = (2.0f * M_PI * i) / segments;
207 float angle2 = (2.0f * M_PI * (i + 1)) / segments;
208
209 ImVec2 p1_inner = ImVec2(center.x + cosf(angle1) * inner_ring_radius,
210 center.y + sinf(angle1) * inner_ring_radius);
211 ImVec2 p2_inner = ImVec2(center.x + cosf(angle2) * inner_ring_radius,
212 center.y + sinf(angle2) * inner_ring_radius);
213 ImVec2 p1_outer = ImVec2(center.x + cosf(angle1) * ring_radius,
214 center.y + sinf(angle1) * ring_radius);
215 ImVec2 p2_outer = ImVec2(center.x + cosf(angle2) * ring_radius,
216 center.y + sinf(angle2) * ring_radius);
217
218 draw_list->AddQuadFilled(p1_inner, p2_inner, p2_outer, p1_outer, color);
219 }
220 }
221 }

309float BackgroundRenderer::CalculateRadialFade(const ImVec2& pos, const ImVec2& center, float max_distance) const {
310 float distance = sqrtf((pos.x - center.x) * (pos.x - center.x) +
311 (pos.y - center.y) * (pos.y - center.y));
312 float fade = 1.0f - std::min(distance / max_distance, 1.0f);
313 return fade * fade; // Square for smoother falloff
314}
