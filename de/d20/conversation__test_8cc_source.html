26absl::Status LoadRomForAgent(Rom& rom) {
27 if (rom.is_loaded()) {
28 return ::absl::OkStatus();
29 }
30
31 std::string rom_path = ::absl::GetFlag(FLAGS_rom);
32 if (rom_path.empty()) {
33 return ::absl::InvalidArgumentError(
34 "No ROM loaded. Pass --rom=<path> to z3ed agent test-conversation.");
35 }
36
37 auto status = rom.LoadFromFile(rom_path);
38 if (!status.ok()) {
39 return ::absl::FailedPreconditionError(
40 ::absl::StrCat("Failed to load ROM from '", rom_path,
41 "': ", status.message()));
42 }
43
44 return ::absl::OkStatus();
45}

201bool ValidateResponse(const ChatMessage& response,
202 const ConversationTestCase& test_case) {
203 bool passed = true;
204
205 // Check for expected keywords
206 for (const auto& keyword : test_case.expected_keywords) {
207 if (response.message.find(keyword) == std::string::npos) {
208 std::cout << "⚠️ Warning: Expected keyword '" << keyword
209 << "' not found in response\n";
210 // Don't fail test, just warn
211 }
212 }
213
214 // Check for tool calls (if we have table data, tools were likely called)
215 if (test_case.expect_tool_calls && !response.table_data.has_value()) {
216 std::cout << "⚠️ Warning: Expected tool calls but no table data found\n";
217 }
218
219 // Check for commands
220 if (test_case.expect_commands) {
221 bool has_commands = response.message.find("overworld") != std::string::npos ||
222 response.message.find("dungeon") != std::string::npos ||
223 response.message.find("set-tile") != std::string::npos;
224 if (!has_commands) {
225 std::cout << "⚠️ Warning: Expected commands but none found\n";
226 }
227 }
228
229 return passed;
230}

262 << " message" << (history.size() == 1 ? "" : "s") << ")\n";
263 for (const auto& message : history) {

266 std::cout << " [" << sender << "] " << message.message << "\n";
267 }
268 std::cout << "\n";
269 }
270
271 if (all_passed) {
272 std::cout << "✅ Test PASSED: " << test_case.name << "\n";
273 return absl::OkStatus();
274 }
275
276 std::cout << "⚠️ Test completed with warnings: " << test_case.name << "\n";
277 return absl::InternalError(
278 absl::StrCat("Conversation test failed validation: ", test_case.name));

281absl::Status LoadTestCasesFromFile(const std::string& file_path,
282 std::vector<ConversationTestCase>* test_cases) {
283 std::ifstream file(file_path);
284 if (!file.is_open()) {
285 return absl::NotFoundError(
286 absl::StrCat("Could not open test file: ", file_path));
287 }
288
289 nlohmann::json test_json;
290 try {
291 file >> test_json;
292 } catch (const nlohmann::json::parse_error& e) {
293 return absl::InvalidArgumentError(
294 absl::StrCat("Failed to parse test file: ", e.what()));
295 }
296
297 if (!test_json.is_array()) {
298 return absl::InvalidArgumentError(
299 "Test file must contain a JSON array of test cases");
300 }
301
302 for (const auto& test_obj : test_json) {
303 ConversationTestCase test_case;
304 test_case.name = test_obj.value("name", "unnamed_test");
305 test_case.description = test_obj.value("description", "");
306
307 if (test_obj.contains("prompts") && test_obj["prompts"].is_array()) {
308 for (const auto& prompt : test_obj["prompts"]) {
309 if (prompt.is_string()) {
310 test_case.user_prompts.push_back(prompt.get<std::string>());
311 }
312 }
313 }
314
315 if (test_obj.contains("expected_keywords") &&
316 test_obj["expected_keywords"].is_array()) {
317 for (const auto& keyword : test_obj["expected_keywords"]) {
318 if (keyword.is_string()) {
319 test_case.expected_keywords.push_back(keyword.get<std::string>());
320 }
321 }
322 }
323
324 test_case.expect_tool_calls = test_obj.value("expect_tool_calls", false);
325 test_case.expect_commands = test_obj.value("expect_commands", false);
326
327 test_cases->push_back(test_case);
328 }
329
330 return absl::OkStatus();
331}
